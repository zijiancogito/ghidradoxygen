132 inline bool Merge::compareHighByBlock(const HighVariable *a,const HighVariable *b)
133 
134 {
135  int4 result = a->wholecover.compareTo(b->wholecover);
136  if ( result == 0 ) {
137  Varnode *v1 = a->getInstance( 0 );
138  Varnode *v2 = b->getInstance( 0 );
139 
140  if ( v1->getAddr() == v2->getAddr() ) {
141  PcodeOp *def1 = v1->getDef();
142  PcodeOp *def2 = v2->getDef();
143  if ( def1 == (PcodeOp *) 0 ) {
144  return true;
145  }
146  else if ( def2 == (PcodeOp *) 0 ) {
147  return false;
148  }
149  return (def1->getAddr() < def2->getAddr());
150  }
151  return (v1->getAddr() < v2->getAddr());
152  }
153  return (result < 0);
154 }

321  private void splitToMaxWidthLines(ArrayList<String> res, String line) {
322  int maxchar;
323  if ((maxWidth == 0) || (indentWidth == 0)) {
324  maxchar = 40;
325  }
326  else {
327  maxchar = maxWidth / indentWidth;
328  }
329  String[] toklist = line.split("[ \t]+");
330  StringBuffer buf = new StringBuffer();
331  int cursize = 0;
332  boolean atleastone = false;
333  int i = 0;
334  while (i < toklist.length) {
335  if (!atleastone) {
336  buf.append(' ');
337  buf.append(toklist[i]);
338  atleastone = true;
339  cursize += toklist[i].length() + 1;
340  i += 1;
341  continue;
342  }
343  if (cursize + toklist[i].length() >= maxchar) {
344  String finishLine = buf.toString();
345  res.add(finishLine);
346  cursize = 5;
347  atleastone = false;
348  buf = new StringBuffer();
349  buf.append(" ");
350  }
351  else {
352  buf.append(' ');
353  buf.append(toklist[i]);
354  cursize += toklist[i].length() + 1;
355  i += 1;
356  }
357  }
358  String finalLine = buf.toString();
359  if (finalLine.length() != 0) {
360  res.add(finalLine);
361  }
362  }

365  if (docroot == null) {

414  private SearchLocation findNextTokenGoingForward(
415  java.util.function.Function<String, SearchMatch> matcher, String searchString,
416  FieldLocation currentLocation) {
417 
418  int row = currentLocation.getIndex().intValue();
419  for (int i = row; i < fieldList.length; i++) {
420  ClangTextField field = (ClangTextField) fieldList[i];
421  String textLine =
422  getTextLineFromOffset((i == row) ? currentLocation : null, field, true);
423 
424  SearchMatch match = matcher.apply(textLine);
425  if (match != SearchMatch.NO_MATCH) {
426  if (i == row) {
427  String fullLine = field.getText();
428  match.start += fullLine.length() - textLine.length();
429  }
430  FieldNumberColumnPair pair = getFieldIndexFromOffset(match.start, field);
431  FieldLocation fieldLocation =
432  new FieldLocation(i, pair.getFieldNumber(), 0, pair.getColumn());
433 
434  return new FieldBasedSearchLocation(fieldLocation, match.start, match.end - 1,
435  searchString, true);
436  }
437  }
438  return null;
439  }

464  public SearchLocation findNextTokenForSearchRegex(String searchString,
465  FieldLocation currentLocation, boolean forwardSearch) {
466 
467  Pattern pattern = null;
468  try {
469  pattern = Pattern.compile(searchString, Pattern.CASE_INSENSITIVE | Pattern.DOTALL);
470  }
471  catch (PatternSyntaxException e) {
472  Msg.showError(this, decompilerPanel, "Regular Expression Syntax Error", e.getMessage());
473  return null;
474  }
475 
476  Pattern finalPattern = pattern;
477  if (forwardSearch) {
478 
479  java.util.function.Function<String, SearchMatch> function = textLine -> {
480 
481  Matcher matcher = finalPattern.matcher(textLine);
482  if (matcher.find()) {
483  int start = matcher.start();
484  int end = matcher.end();
485  return new SearchMatch(start, end);
486  }
487 
488  return SearchMatch.NO_MATCH;
489  };
490 
491  return findNextTokenGoingForward(function, searchString, currentLocation);
492  }
493 
494  java.util.function.Function<String, SearchMatch> reverse = textLine -> {
495 
496  Matcher matcher = finalPattern.matcher(textLine);
497 
498  if (!matcher.find()) {
499  return SearchMatch.NO_MATCH;
500  }
501 
502  int start = matcher.start();
503  int end = matcher.end();
504 
505  // Since the matcher can only match from the start to end of line, we need
506  // to find all matches and then take the last match
507  while (matcher.find()) {
508  start = matcher.start();
509  end = matcher.end();
510  }
511 
512  return new SearchMatch(start, end);
513  };
514 
515  return findNextTokenGoingBackward(reverse, searchString, currentLocation);
516  }

518  public SearchLocation findNextTokenForSearch(String searchString, FieldLocation currentLocation,
519  boolean forwardSearch) {
520 
521  if (forwardSearch) {
522 
523  java.util.function.Function<String, SearchMatch> function = textLine -> {
524 
525  int index = StringUtilities.indexOfIgnoreCase(textLine, searchString);
526  if (index == -1) {
527  return SearchMatch.NO_MATCH;
528  }
529  return new SearchMatch(index, index + searchString.length());
530  };
531 
532  return findNextTokenGoingForward(function, searchString, currentLocation);
533  }
534 
535  java.util.function.Function<String, SearchMatch> function = textLine -> {
536 
537  int index = StringUtilities.lastIndexOfIgnoreCase(textLine, searchString);
538  if (index == -1) {
539  return SearchMatch.NO_MATCH;
540  }
541  return new SearchMatch(index, index + searchString.length());

547  private String getTextLineFromOffset(FieldLocation location, ClangTextField textField,
548  boolean forwardSearch) {
549  if (location == null) { // the cursor location is not on this line; use all of the text
550  return textField.getText();
551  }
552 
553  if (textField.getText().isEmpty()) { // the cursor is on blank line
554  return "";
555  }
556 
557  String partialText = textField.getText();
558 
559  if (forwardSearch) {
560 
561  // Protects against the location column being out of range (this can
562  // happen if we're searching forward and the cursor is past the last token).
563  if (location.getCol() + 1 >= partialText.length()) {
564  return "";
565  }
566  return partialText.substring(location.getCol() + 1);
567  }
568 
569  // backwards search
570  return partialText.substring(0, location.getCol());
571  }

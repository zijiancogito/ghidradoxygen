189 bool IfaceStatus::runCommand(void)
190 
191 {
192  string line; // Next line from input stream
193 
194  if (!sorted) {
195  sort(comlist.begin(),comlist.end(),compare_ifacecommand);
196  sorted = true;
197  }
198  readLine(line);
199  if (line.empty()) return false;
200  saveHistory(line);
201 
202  vector<string> fullcommand;
203  vector<IfaceCommand *>::const_iterator first = comlist.begin();
204  vector<IfaceCommand *>::const_iterator last = comlist.end();
205  istringstream is(line);
206  int4 match;
207 
208  match = expandCom(fullcommand, is,first,last); // Try to expand the command
209  if (match == 0) {
210  *optr << "ERROR: Invalid command" << endl;
211  return false;
212  }
213  else if ( fullcommand.size() == 0 ) // Nothing useful typed
214  return false;
215  else if (match>1) {
216  if ( (*first)->numWords() != fullcommand.size()) { // Check for complete but not unique
217  *optr << "ERROR: Incomplete command" << endl;
218  return false;
219  }
220  }
221  else if (match<0)
222  *optr << "ERROR: Incomplete command" << endl;
223 
224  (*first)->execute(is); // Try to execute the (first) command
225  return true; // Indicate a command was executed
226 }

228 void IfaceStatus::restrict(vector<IfaceCommand *>::const_iterator &first,
229  vector<IfaceCommand *>::const_iterator &last,

232 {

248 static bool maxmatch(string &res,const string &op1,const string &op2)
249 
250 { // Set res to maximum characters in common
251  // at the beginning of op1 and op2
252  int4 len;
253  bool equal;
254 
255  if (op1.size() == op2.size()) {
256  len = op1.size();
257  equal = true;
258  }
259  else {
260  equal = false;
261  len = ( op1.size() < op2.size() ) ? op1.size() : op2.size();
262  }
263 
264  res.erase();
265  for(int4 i=0;i<len;++i) {
266  if (op1[i] == op2[i])
267  res += op1[i];
268  else
269  return false;
270  }
271  return equal;
272 }

274 int4 IfaceStatus::expandCom(vector<string> &expand,istream &s,
275  vector<IfaceCommand *>::const_iterator &first,
276  vector<IfaceCommand *>::const_iterator &last)
277 
278 { // Expand tokens on stream to full command
279  // Return range of possible commands
280  // If command is complete with extra arguments
281  // return (dont process) remaining args
282  // Return number of matching commands
283 
284  int4 pos; // Which word are we currently expanding
285  string tok;
286  bool res;
287 
288  expand.clear(); // Make sure command list is empty
289  res = true;
290  if (first == last) // If subrange is empty, return 0
291  return 0;
292  for(pos=0;;++pos) {
293  s >> ws; // Skip whitespace
294  if (first == (last-1)) { // If subrange is unique
295  if (s.eof()) // If no more input
296  for(;pos<(*first)->numWords();++pos) // Automatically provide missing words
297  expand.push_back( (*first)->getCommandWord(pos) );
298  if ((*first)->numWords() == pos) // If all words are matched
299  return 1; // Finished
300  }
301  if (!res) { // Last word was ambiguous
302  if (!s.eof())
303  return (last-first);
304  return (first-last); // Negative number to indicate last word incomplete
305  }
306  if (s.eof()) { // if no other words
307  if (expand.empty())
308  return (first-last);
309  return (last-first); // return number of matches
310  }
311  s >> tok; // Get next token
312  expand.push_back(tok);
313  restrict(first,last,expand);
314  if (first == last) // If subrange is empty, return 0
315  return 0;
316  res = maxmatch(tok, (*first)->getCommandWord(pos), (*(last-1))->getCommandWord(pos));
317  expand.back() = tok;
318  }
319 }

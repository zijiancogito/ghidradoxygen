29 void SleighBase::buildXrefs(void)
30 
31 {
32  SymbolScope *glb = symtab.getGlobalScope();
33  SymbolTree::const_iterator iter;
34  SleighSymbol *sym;
35  int4 errors = 0;
36  ostringstream s;
37 
38  for(iter=glb->begin();iter!=glb->end();++iter) {
39  sym = *iter;
40  if (sym->getType() == SleighSymbol::varnode_symbol) {
41  pair<VarnodeData,string> ins(((VarnodeSymbol *)sym)->getFixedVarnode(),sym->getName());
42  pair<map<VarnodeData,string>::iterator,bool> res = varnode_xref.insert(ins);
43  if (!res.second) {
44  s << "Duplicate (offset,size) pair for registers: ";
45  s << sym->getName() << " and " << (*(res.first)).second << '\n';
46  errors += 1;
47  }
48  }
49  else if (sym->getType() == SleighSymbol::userop_symbol) {
50  int4 index = ((UserOpSymbol *)sym)->getIndex();
51  while(userop.size() <= index)
52  userop.push_back("");
53  userop[index] = sym->getName();
54  }
55  else if (sym->getType() == SleighSymbol::context_symbol) {
56  ContextSymbol *csym = (ContextSymbol *)sym;
57  ContextField *field = (ContextField *)csym->getPatternValue();
58  int4 startbit = field->getStartBit();
59  int4 endbit = field->getEndBit();
60  registerContext(csym->getName(),startbit,endbit);
61  }
62  }
63  if (errors > 0)
64  throw SleighError(s.str());
65 }

105 string SleighBase::getRegisterName(AddrSpace *base,uintb off,int4 size) const
106 
107 {
108  VarnodeData sym;
109  sym.space = base;
110  sym.offset = off;
111  sym.size = size;
112  map<VarnodeData,string>::const_iterator iter = varnode_xref.upper_bound(sym); // First point greater than offset
113  if (iter == varnode_xref.begin()) return "";
114  iter--;
115  const VarnodeData &point((*iter).first);
116  if (point.space != base) return "";
117  uintb offbase = point.offset;
118  if (point.offset+point.size >= off+size)
119  return (*iter).second;
120 
121  while(iter != varnode_xref.begin()) {
122  --iter;
123  const VarnodeData &point((*iter).first);
124  if ((point.space != base)||(point.offset != offbase)) return "";
125  if (point.offset+point.size >= off+size)
126  return (*iter).second;
127  }
128  return "";
129 }

145 void SleighBase::saveXml(ostream &s) const
146 
147 {
148  s << "<sleigh";
149  a_v_b(s,"bigendian",isBigEndian());
150  a_v_i(s,"align",alignment);
151  a_v_u(s,"uniqbase",getUniqueBase());
152  if (maxdelayslotbytes > 0)
153  a_v_u(s,"maxdelay",maxdelayslotbytes);
154  if (unique_allocatemask != 0)
155  a_v_u(s,"uniqmask",unique_allocatemask);
156  if (numSections != 0)
157  a_v_u(s,"numsections",numSections);
158  s << ">\n";
159 
160  s << "<spaces";
161  a_v(s,"defaultspace",getDefaultSpace()->getName());
162  s << ">\n";
163  for(int4 i=0;i<numSpaces();++i) {
164  AddrSpace *spc = getSpace(i);
165  if ((spc->getType()==IPTR_CONSTANT) ||
166  (spc->getType()==IPTR_FSPEC)||
167  (spc->getType()==IPTR_IOP)||
168  (spc->getType()==IPTR_JOIN))
169  continue;
170  spc->saveXml(s);
171  }
172  s << "</spaces>\n";
173  symtab.saveXml(s);
174  s << "</sleigh>\n";
175 }

180 void SleighBase::restoreXml(const Element *el)
181 
182 {
183  maxdelayslotbytes = 0;
184  unique_allocatemask = 0;
185  numSections = 0;
186  setBigEndian(xml_readbool(el->getAttributeValue("bigendian")));
187  {

192  {
193  istringstream s(el->getAttributeValue("uniqbase"));
194  s.unsetf(ios::dec | ios::hex | ios::oct);
195  uintm ubase;
196  s >> ubase;
197  setUniqueBase(ubase);
198  }
199  int4 numattr = el->getNumAttributes();
200  for(int4 i=0;i<numattr;++i) {
201  const string &attrname( el->getAttributeName(i) );
202  if (attrname == "maxdelay") {
203  istringstream s1(el->getAttributeValue(i));
204  s1.unsetf(ios::dec | ios::hex | ios::oct);
205  s1 >> maxdelayslotbytes;
206  }
207  else if (attrname == "uniqmask") {
208  istringstream s2(el->getAttributeValue(i));
209  s2.unsetf(ios::dec | ios::hex | ios::oct);
210  s2 >> unique_allocatemask;
211  }
212  else if (attrname == "numsections") {
213  istringstream s3(el->getAttributeValue(i));
214  s3.unsetf(ios::dec | ios::hex | ios::oct);
215  s3 >> numSections;
216  }
217  }
218  const List &list(el->getChildren());
219  List::const_iterator iter;
220  iter = list.begin();
221  while((*iter)->getName() == "floatformat") {
222  floatformats.push_back(FloatFormat());
223  floatformats.back().restoreXml(*iter);
224  ++iter;
225  }
226  restoreXmlSpaces(*iter,this);
227  iter++;
228  symtab.restoreXml(*iter,this);
229  root = (SubtableSymbol *)symtab.getGlobalScope()->findSymbol("instruction");
230  buildXrefs();
231 }

229 uintb ContextDatabase::getTrackedValue(const VarnodeData &mem,const Address &point) const
230 
231 {
232  const TrackedSet &tset(getTrackedSet(point));
233  uintb endoff = mem.offset + mem.size - 1;
234  uintb tendoff;
235  for(int4 i=0;i<tset.size();++i) {
236  const TrackedContext &tcont( tset[i] );
237  // tcont must contain -mem-
238  if (tcont.loc.space != mem.space) continue;
239  if (tcont.loc.offset > mem.offset) continue;
240  tendoff = tcont.loc.offset + tcont.loc.size - 1;
241  if (tendoff < endoff) continue;
242  uintb res = tcont.val;
243  // If we have proper containment, trim value based on endianness
244  if (tcont.loc.space->isBigEndian()) {
245  if (endoff != tendoff)
246  res >>= (8* (tendoff - mem.offset));
247  }
248  else {
249  if (mem.offset != tcont.loc.offset)
250  res >>= (8* (mem.offset-tcont.loc.offset));
251  }
252  res &= calc_mask( mem.size ); // Final trim based on size
253  return res;
254  }
255  return (uintb)0;
256 }

436 void ContextInternal::getRegionForSet(vector<uintm *> &res,const Address &addr1,const Address &addr2,
437  int4 num,uintm mask)
438 {
439  database.split(addr1);
440  partmap<Address,FreeArray>::iterator aiter,biter;
441 
442  aiter = database.begin(addr1);
443  if (!addr2.isInvalid()) {
444  database.split(addr2);
445  biter = database.begin(addr2);
446  }
447  else
448  biter = database.end();

458 void ContextInternal::getRegionToChangePoint(vector<uintm *> &res,const Address &addr,int4 num,uintm mask)
459 
460 {
461  database.split(addr);
462  partmap<Address,FreeArray>::iterator aiter,biter;
463  uintm *maskArray,*vecArray;
464 
465  aiter = database.begin(addr);
466  biter = database.end();
467  if (aiter == biter) return;
468  vecArray = (*aiter).second.array;
469  res.push_back(vecArray);
470  maskArray = (*aiter).second.mask;
471  maskArray[num] |= mask;
472  ++aiter;
473  while(aiter != biter) {
474  vecArray = (*aiter).second.array;
475  maskArray = (*aiter).second.mask;
476  if ((maskArray[num] & mask) != 0) break; // Reached point where this value was definitively set before
477  res.push_back(vecArray);
478  ++aiter;
479  }

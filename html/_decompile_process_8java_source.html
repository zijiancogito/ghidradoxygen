114  private void setup() throws IOException {
115  if (disposestate != DisposeState.NOT_DISPOSED) {
116  throw new IOException("Decompiler has been disposed");
117  }
118  if (nativeProcess != null) {
119  // Something bad happened to the process or the interface
120  // and now we try to restart
121  nativeProcess.destroy(); // Make sure previous bad process is killed
122  nativeProcess = null;
123  }
124  if (exepath == null) {
125  throw new IOException("Could not find decompiler executable");
126  }
127  try {
128  nativeProcess = runtime.exec(exepath);
129 
130  nativeIn = nativeProcess.getInputStream();
131  nativeOut = nativeProcess.getOutputStream();
132  statusGood = true;
133  }
134  catch (IOException e) {
135  disposestate = DisposeState.DISPOSED_ON_STARTUP_FAILURE;
136  statusGood = false;
137  Msg.showError(this, null, "Problem launching decompiler",
138  "Please report this stack trace to the Ghidra Team", e);
139  throw e;

143  private int readToBurst() throws IOException {
144  if (nativeIn == null) {
145  // we've been disposed!
146  // (not sure if throwing an exception the best)
147  throw new IOException("Decompiler disposed!");
148  }
149 
150  int cur;
151  for (;;) {
152  do {
153  cur = nativeIn.read();
154  }
155  while (cur > 0);
156  if (cur == -1) {
157  break;
158  }
159  do {
160  cur = nativeIn.read();
161  }
162  while (cur == 0);
163  if (cur == 1) {
164  cur = nativeIn.read();
165  if (cur == -1) {
166  break;
167  }
168  return cur;
169  }
170  if (cur == -1) {
171  break;
172  }
173  }
174  throw new IOException("Decompiler process died");
175  }

177  private void readToResponse() throws IOException, DecompileException {
178  nativeOut.flush(); // Make sure decompiler has access to all the info it has been sent
179  int type;
180  do {
181  type = readToBurst();
182  }
183  while ((type & 1) == 1);
184  if (type == 10) {
185  generateException();
186  }
187  if (type == 6) {
188  return;
189  }
190  throw new IOException("Ghidra/decompiler alignment error");

211  return cur;

213  }
214  if (cur == -1) {
215  break;
216  }
217  }

245  private void writeBytes(PackedBytes out) throws IOException {
246  write(string_start);

253  sz >>>= 6;

275  private LimitedByteBuffer readResponse() throws IOException, DecompileException {
276  readToResponse();
277  int type = readToBurst();
278  String name;
279  LimitedByteBuffer retbuf = null;
280  LimitedByteBuffer buf = null;
281 
282  while (type != 7) {
283  switch (type) {
284  case 4:
285  name = readQueryString();
286  try {
287  if (name.length() < 4) {
288  throw new Exception("Bad decompiler query: " + name);
289  }
290  switch (name.charAt(3)) {
291  case 'B':
292  getBytes(); // getBytes
293  break;
294  case 'C':
295  if (name.equals("getComments")) {
296  getComments();
297  }
298  else if (name.equals("getCallFixup")) {
299  getPcodeInject(InjectPayload.CALLFIXUP_TYPE);
300  }
301  else if (name.equals("getCallotherFixup")) {
302  getPcodeInject(InjectPayload.CALLOTHERFIXUP_TYPE);
303  }
304  else if (name.equals("getCallMech")) {
305  getPcodeInject(InjectPayload.CALLMECHANISM_TYPE);
306  }
307  else {
308  getCPoolRef();
309  }
310  break;
311  case 'E':
312  getExternalRefXML(); // getExternalRefXML
313  break;
314  case 'M':
315  getMappedSymbolsXML(); // getMappedSymbolsXML
316  break;
317  case 'P':
318  getPcodePacked(); // getPacked
319  break;
320  case 'R':
321  if (name.equals("getRegister")) {
322  getRegister();
323  }
324  else {
325  getRegisterName();
326  }
327  break;
328  case 'S':
329  getSymbol(); // getSymbol
330  break;
331  case 'T':
332  if (name.equals("getType")) {
333  getType();
334  }
335  else {
336  getTrackedRegisters();
337  }
338  break;
339  case 'U':
340  getUserOpName(); // getUserOpName
341  break;
342  case 'X':
343  getPcodeInject(InjectPayload.EXECUTABLEPCODE_TYPE);
344  break;
345  default:
346  throw new Exception("Unsupported decompiler query '" + name + "'");
347  }
348  }
349  catch (Exception e) { // Catch ANY exception query generates
350  // and pass it down to decompiler
351  write(exception_start);
352  String extype = e.getClass().getName();
353  String msg = e.getMessage();
354  if (msg == null) {
355  msg = "";
356  }
357  writeString(extype);
358  writeString(msg);
359  write(exception_end);
360 
361  if (disposestate == DisposeState.NOT_DISPOSED) {
362  Msg.error(this, "Unexpected Exception: " + e.getMessage(), e);
363  }
364  }
365  nativeOut.flush(); // Make sure decompiler receives response
366  readToBurst(); // Read query terminator
367  break;
368  case 6:
369  throw new IOException("GHIDRA/decompiler out of alignment");
370  case 10:
371  generateException();
372  break;
373  case 14: // Start of the main decompiler output
374  if (buf != null) {
375  throw new IOException("Nested decompiler output");
376  }
377  // Allocate storage buffer for the result, which is generally not tiny. So we
378  // start with any initial allocation of 1024 bytes, also give an absolute upper bound
379  // determined by maxResultSizeMBYtes
380  buf = new LimitedByteBuffer(1024, maxResultSizeMBYtes << 20);
381  break;
382  case 15: // This is the end of the main decompiler output
383  if (buf == null) {
384  throw new IOException("Mismatched string header");
385  }
386  retbuf = buf;
387  buf = null; // Reset the main buffer as a native message may follow
388  break;
389  case 16: // Beginning of any native message from the decompiler
390 // if (buf!=null)
391 // throw new IOException("Nested decompiler output");
392  // if buf is non-null, then res was interrupted
393  // so we just throw out the partial result
394  buf = new LimitedByteBuffer(64, 1 << 20);
395  break;
396  case 17: // End of the native message from the decompiler
397  if (buf == null) {
398  throw new IOException("Mismatched message header");
399  }
400  callback.setNativeMessage(buf.toString());
401  buf = null;
402  break;
403  default:
404  throw new IOException("GHIDRA/decompiler alignment error");
405 
406  }
407  if (buf == null) {
408  type = readToBurst();
409  }
410  else {
411  type = readToBuffer(buf);
412  }
413  }
414  return retbuf;
415  }

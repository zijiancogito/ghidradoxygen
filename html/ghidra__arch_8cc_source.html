62 int4 ArchitectureGhidra::readToAnyBurst(istream &s)
63 
64 {
65  int4 c;
66 
67  for(;;) {
68  do {
69  c = s.get();
70  } while(c>0);
71  while(c==0) {
72  c = s.get();
73  }
74  if (c==1) {
75  c = s.get();
76  return c;
77  }
78  if (c<0) // If pipe closed, our parent process is probably dead
79  exit(1); // So we exit to avoid a runaway process
80  }
81 }

88 void ArchitectureGhidra::readStringStream(istream &s,string &res)
89 
90 {
91  int4 c;
92 
93  int4 type = readToAnyBurst(s);
94  if (type != 14) throw JavaError("alignment","Expecting string");
95  c = s.get();
96  while(c > 0) {
97  res += (char)c;
98  c = s.get();
99  }
100  while(c==0) {
101  c = s.get();
102  }
103  if (c==1) {
104  c = s.get();
105  if (c == 15) return;
106  }
107  if (c<0) // If pipe closed, our parent process is probably dead
108  exit(1); // So we exit to avoid a runaway process
109  throw JavaError("alignment","Expecting string terminator");
110 }

138 uint1 *ArchitectureGhidra::readPackedStream(istream &s)
139 
140 {
141  int4 type = readToAnyBurst(s);
142  if (type == 14) {
143  uint4 size = 0;
144  int4 c = s.get();
145  size ^= (c-0x20);
146  c = s.get();
147  size ^= ((c-0x20)<<6);
148  c = s.get();
149  size ^= ((c-0x20)<<12);
150  c = s.get();
151  size ^= ((c-0x20)<<18);
152  uint1 *res = new uint1[ size ];
153  s.read((char *)res,size);
154  type = readToAnyBurst(s);
155  if (type != 15)
156  throw JavaError("alignment","Expecting packed string end");
157  return res;
158  }
159  if ((type&1)==1)
160  return (uint1 *)0;
161  throw JavaError("alignment","Expecting string or end of query response");
162 }

179 
180 {

183  if (type==10) {

582 void ArchitectureGhidra::getBytes(uint1 *buf,int4 size,const Address &inaddr)
583 
584 {
585  sout.write("\000\000\001\004",4);
586  writeStringStream(sout,"getBytes");
587  sout.write("\000\000\001\016",4); // Beginning of string header
588  inaddr.saveXml(sout,size);
589  sout.write("\000\000\001\017",4);
590  sout.write("\000\000\001\005",4);
591  sout.flush();
592 
593  readToResponse(sin);
594  int4 type = readToAnyBurst(sin);
595  if (type == 12) {
596  uint1 *dblbuf = new uint1[size * 2];
597  sin.read((char *)dblbuf,size*2);
598  for (int4 i=0; i < size; i++) {
599  buf[i] = ((dblbuf[i*2]-'A') << 4) | (dblbuf[i*2 + 1]-'A');
600  }
601  delete [] dblbuf;
602  }
603  else if ((type&1)==1) {
604  ostringstream errmsg;
605  errmsg << "GHIDRA has no data in the loadimage at " << inaddr.getShortcut();
606  inaddr.printRaw(errmsg);
607  throw DataUnavailError(errmsg.str());
608  }
609  else
610  throw JavaError("alignment","Expecting bytes or end of query response");
611  type = readToAnyBurst(sin);
612  if (type != 13)
613  throw JavaError("alignment","Expecting byte alignment end");
614  readResponseEnd(sin);
615 }

321 
322 {
323  vector<LoopBody *> containlist;
324 
325  for(int4 i=0;i<body.size();++i) {
326  FlowBlock *curblock = body[i];
327  if (curblock != head) {
328  LoopBody *subloop = LoopBody::find(curblock,looporder);
329  if (subloop != (LoopBody *)0) {
330  containlist.push_back(subloop);
331  subloop->depth += 1;
332  }
333  }
334  }
335  // Note the following code works even though the depth fields may shift during subsequent calls to this routine
336  // Once a LoopBody calls this routine
337  // the depth of any contained loop will permanently be bigger than this LoopBody
338  // because any other loop will either
339  // increment the depth of both this LoopBody and any loop that it contains OR
340  // increment neither the LoopBody nor a loop it contains OR
341  // NOT increment the LoopBody but DO increment a contained loop
342  // So when the immediate container a of loop b calls this routine
343  // a has a depth greater than any containing LoopBody that has already run

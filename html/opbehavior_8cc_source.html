178 uintb OpBehaviorIntSless::evaluateBinary(int4 sizeout,int4 sizein,uintb in1,uintb in2) const
179 
180 {
181  uintb res,mask,bit1,bit2;
182 
183  if (sizein<=0)
184  res = 0;
185  else {
186  mask = 0x80;
187  mask <<= 8*(sizein-1);
188  bit1 = in1 & mask; // Get the sign bits
189  bit2 = in2 & mask;
190  if (bit1 != bit2)
191  res = (bit1 != 0) ? 1 : 0;
192  else
193  res = (in1 < in2) ? 1 : 0;
194  }
195  return res;

254 uintb OpBehaviorIntSext::recoverInputUnary(int4 sizeout,uintb out,int4 sizein) const
255 
256 {
257  uintb masklong = calc_mask(sizeout);
258  uintb maskshort = calc_mask(sizein);
259 
260  if ((out & (maskshort ^ (maskshort>>1))) == 0) { // Positive input
261  if ((out & maskshort) != out)
262  throw EvaluationError("Output is not in range of sext operation");
263  }
264  else { // Negative input
265  if ((out & (masklong^maskshort)) != (masklong^maskshort))
266  throw EvaluationError("Output is not in range of sext operation");
267  }
268  return (out&maskshort);

416 uintb OpBehaviorIntSright::evaluateBinary(int4 sizeout,int4 sizein,uintb in1,uintb in2) const
417 
418 {
419  uintb res;
420  if (signbit_negative(in1,sizein)) {
421  res = in1 >> in2;
422  uintb mask = calc_mask(sizein);
423  mask = (mask >> in2) ^ mask;

432 uintb OpBehaviorIntSright::recoverInputBinary(int4 slot,int4 sizeout,uintb out,int4 sizein,uintb in) const
433 
434 {
435  if (slot!=0)
436  return OpBehavior::recoverInputBinary(slot,sizeout,out,sizein,in);
437 
438  int4 sa = in;
439  uintb testval = out>>(sizein*8-sa-1);
440  int4 count=0;
441  for(int4 i=0;i<=sa;++i) {
442  if ((testval&1)!=0) count += 1;
443  testval >>= 1;
444  }
445  if (count != sa+1)
446  throw EvaluationError("Output is not in range of right shift operation");
447  return out<<sa;
448 }

465 uintb OpBehaviorIntSdiv::evaluateBinary(int4 sizeout,int4 sizein,uintb in1,uintb in2) const
466 
467 {

469  throw EvaluationError("Divide by 0");
470  intb num = in1; // Convert to signed
471  intb denom = in2;
472  sign_extend(num,8*sizein-1);
473  sign_extend(denom,8*sizein-1);
474  intb sres = num/denom; // Do the signed division
475  zero_extend(sres,8*sizeout-1); // Cut to appropriate size
476  return (uintb)sres; // Recast as unsigned

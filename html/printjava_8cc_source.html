113 bool PrintJava::isArrayType(const Datatype *ct)
114 
115 {
116  if (ct->getMetatype() != TYPE_PTR) // Java arrays are always Ghidra pointer types
117  return false;
118  ct = ((TypePointer *)ct)->getPtrTo();
119  switch(ct->getMetatype()) {
120  case TYPE_UINT: // Pointer to unsigned is placeholder for class reference, not an array
121  if (ct->isCharPrint())
122  return true;
123  break;
124  case TYPE_INT:
125  case TYPE_BOOL:
126  case TYPE_FLOAT: // Pointer to primitive type is an array
127  case TYPE_PTR: // Pointer to class reference is an array
128  return true;
129  default:
130  break;
131  }
132  return false;
133 }

208 void PrintJava::opStore(const PcodeOp *op)
209 
210 {

212  pushOp(&assignment,op); // This is an assignment
213  if (needZeroArray(op->getIn(1))) {

227 void PrintJava::opCallind(const PcodeOp *op)
228 
229 {
230  pushOp(&function_call,op);
231  // implied vn's pushed on in reverse order for efficiency
232  // see PrintLanguage::pushVnImplied
233  int4 startparam = isSet(hide_thisparam) && op->hasThisPointer() ? 2 : 1;
234  if (op->numInput()>startparam + 1) { // Multiple parameters
235  pushVnImplied(op->getIn(0),op,mods);
236  for(int4 i=startparam;i<op->numInput()-1;++i)
237  pushOp(&comma,op);
238  for(int4 i=op->numInput()-1;i>=startparam;--i)
239  pushVnImplied(op->getIn(i),op,mods);
240  }
241  else if (op->numInput()==startparam + 1) { // One parameter
242  pushVnImplied(op->getIn(startparam),op,mods);
243  pushVnImplied(op->getIn(0),op,mods);
244  }
245  else { // A void function
246  pushVnImplied(op->getIn(0),op,mods);
247  pushAtom(Atom("",blanktoken,EmitXml::no_color));
248  }

251 void PrintJava::opCpoolRefOp(const PcodeOp *op)
252 
253 {
254  const Varnode *outvn = op->getOut();
255  const Varnode *vn0 = op->getIn(0);
256  vector<uintb> refs;
257  for(int4 i=1;i<op->numInput();++i)
258  refs.push_back(op->getIn(i)->getOffset());
259  const CPoolRecord *rec = glb->cpool->getRecord(refs);
260  if (rec == (const CPoolRecord *)0) {
261  pushAtom(Atom("UNKNOWNREF",syntax,EmitXml::const_color,op,outvn));
262  }
263  else {
264  switch(rec->getTag()) {
265  case CPoolRecord::string_literal:
266  {
267  ostringstream str;
268  int4 len = rec->getByteDataLength();
269  if (len > 2048)
270  len = 2048;
271  str << '\"';
272  escapeCharacterData(str,rec->getByteData(),len,1,false);
273  if (len == rec->getByteDataLength())
274  str << '\"';
275  else {
276  str << "...\"";
277  }
278  pushAtom(Atom(str.str(),vartoken,EmitXml::const_color,op,outvn));
279  break;
280  }
281  case CPoolRecord::class_reference:
282  pushAtom(Atom(rec->getToken(),vartoken,EmitXml::type_color,op,outvn));
283  break;
284  case CPoolRecord::instance_of:
285  {
286  Datatype *dt = rec->getType();
287  while(dt->getMetatype() == TYPE_PTR) {
288  dt = ((TypePointer *)dt)->getPtrTo();
289  }
290  pushOp(&instanceof,op);
291  pushVnImplied(vn0,op,mods);
292  pushAtom(Atom(dt->getName(),syntax,EmitXml::type_color,op,outvn));
293  break;
294  }
295  case CPoolRecord::primitive: // Should be eliminated
296  case CPoolRecord::pointer_method:
297  case CPoolRecord::pointer_field:
298  case CPoolRecord::array_length:
299  case CPoolRecord::check_cast:
300  default:
301  {
302  Datatype *ct = rec->getType();
303  EmitXml::syntax_highlight color = EmitXml::var_color;
304  if (ct->getMetatype() == TYPE_PTR) {
305  ct = ((TypePointer *)ct)->getPtrTo();
306  if (ct->getMetatype() == TYPE_CODE)
307  color = EmitXml::funcname_color;
308  }
309  if (vn0->isConstant()) { // If this is NOT relative to an object reference
310  pushAtom(Atom(rec->getToken(),vartoken,color,op,outvn));
311  }
312  else {
313  pushOp(&object_member,op);
314  pushVnImplied(vn0,op,mods);
315  pushAtom(Atom(rec->getToken(),syntax,color,op,outvn));
316  }
317  }
318  }
319  }

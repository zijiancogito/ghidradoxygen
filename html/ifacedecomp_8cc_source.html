599 void IfcDecompile::execute(istream &s)
600 
601 { // Do decompilation of current function
602  int4 res;
603 
604  if (dcp->fd == (Funcdata *)0)
605  throw IfaceExecutionError("No function selected");
606 
607  if (dcp->fd->hasNoCode()) {
608  *status->optr << "No code for " << dcp->fd->getName() << endl;
609  return;
610  }
611  if (dcp->fd->isProcStarted()) { // Free up old decompile
612  *status->optr << "Clearing old decompilation" << endl;
613  dcp->conf->clearAnalysis(dcp->fd);
614  }
615 
616  *status->optr << "Decompiling " << dcp->fd->getName() << endl;
617  dcp->conf->allacts.getCurrent()->reset(*dcp->fd);
618  res = dcp->conf->allacts.getCurrent()->perform( *dcp->fd );
619  if (res<0) {
620  *status->optr << "Break at ";

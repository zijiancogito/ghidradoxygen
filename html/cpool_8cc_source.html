20 void CPoolRecord::saveXml(ostream &s) const
21 
22 {
23  s << "<cpoolrec";
24  if (tag == pointer_method)
25  a_v(s,"tag","method");
26  else if (tag == pointer_field)
27  a_v(s,"tag","field");
28  else if (tag == instance_of)
29  a_v(s,"tag","instanceof");
30  else if (tag == array_length)
31  a_v(s,"tag","arraylength");
32  else if (tag == check_cast)
33  a_v(s,"tag","checkcast");
34  else if (tag == string_literal)
35  a_v(s,"tag","string");
36  else if (tag == class_reference)
37  a_v(s,"tag","classref");
38  else
39  a_v(s,"tag","primitive");
40  if (hasThisPointer())
41  a_v_b(s,"hasthis",true);
42  if (isConstructor())
43  a_v_b(s,"constructor",true);
44  if (isDestructor())
45  a_v_b(s,"destructor",true);
46  s << ">\n";
47  if (tag == primitive) {
48  s << " <value>0x";
49  s << hex << value;
50  s << "</value>\n";
51  }
52  if (byteData != (uint1 *)0) {
53  s << " <data length=\"" << dec << byteDataLen << "\">\n";
54  int4 wrap = 0;
55  for(int4 i=0;i<byteDataLen;++i) {
56  s << setfill('0') << setw(2) << hex << byteData[i] << ' ';
57  wrap += 1;
58  if (wrap > 15) {
59  s << '\n';
60  wrap = 0;
61  }
62  }
63  s << " </data>\n";
64  }
65  else {
66  s << " <token>";
67  xml_escape(s,token.c_str());
68  s << " </token>\n";
69  }
70  type->saveXml(s);
71  s << "</cpoolrec>\n";
72 }

77 void CPoolRecord::restoreXml(const Element *el,TypeFactory &typegrp)
78 
79 {
80  tag = primitive; // Default tag
81  value = 0;
82  flags = 0;
83  int4 num = el->getNumAttributes();
84  for(int4 i=0;i<num;++i) {
85  const string &attr(el->getAttributeName(i));
86  if (attr == "tag") {
87  const string &tagstring(el->getAttributeValue(i));
88  if (tagstring == "method")
89  tag = pointer_method;
90  else if (tagstring == "field")
91  tag = pointer_field;
92  else if (tagstring == "instanceof")
93  tag = instance_of;
94  else if (tagstring == "arraylength")
95  tag = array_length;
96  else if (tagstring == "checkcast")
97  tag = check_cast;
98  else if (tagstring == "string")
99  tag = string_literal;
100  else if (tagstring == "classref")
101  tag = class_reference;
102  }
103  else if (attr == "hasthis") {
104  if (xml_readbool(el->getAttributeValue(i)))
105  flags |= CPoolRecord::has_thisptr;
106  }
107  else if (attr == "constructor") {
108  if (xml_readbool(el->getAttributeValue(i)))
109  flags |= CPoolRecord::is_constructor;
110  }
111  else if (attr == "destructor") {
112  if (xml_readbool(el->getAttributeValue(i)))
113  flags |= CPoolRecord::is_destructor;
114  }
115  }
116  const List &list(el->getChildren());
117  List::const_iterator iter;
118 
119  iter = list.begin();
120  const Element *subel;
121  if (tag == primitive) { // First tag must be value
122  subel = *iter;
123  istringstream s1(subel->getContent());
124  s1.unsetf(ios::dec | ios::hex | ios::oct);
125  s1 >> value;
126  ++iter;
127  }
128  subel = *iter;
129  ++iter;
130  if (subel->getName() == "token")
131  token = subel->getContent();
132  else {
133  istringstream s2(subel->getAttributeValue("length"));
134  s2.unsetf(ios::dec | ios::hex | ios::oct);
135  s2 >> byteDataLen;
136  istringstream s3(subel->getContent());
137  byteData = new uint1[byteDataLen];
138  for(int4 i=0;i<byteDataLen;++i) {
139  uint4 val;
140  s3 >> ws >> hex >> val;
141  byteData[i] = (uint1)val;
142  }
143  }
144  if (tag == string_literal && (byteData == (uint1 *)0))
145  throw LowlevelError("Bad constant pool record: missing <data>");
146  subel = *iter;
147  if (flags != 0) {
148  bool hasThisPtr = ((flags & has_thisptr)!=0);
149  bool isConstructor = ((flags & is_constructor)!=0);
150  bool isDestructor = ((flags & is_destructor)!=0);
151  type = typegrp.restoreXmlTypeWithCodeFlags(subel,hasThisPtr,isConstructor,isDestructor);
152  }
153  else
154  type = typegrp.restoreXmlType(subel);
155 }

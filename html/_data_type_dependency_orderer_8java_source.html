237  private void createAcyclicDependencyLists() {
238  whoDependsOnMe.clear();
239  whoIDependOn.clear();
240  noDependentsQueue.clear();
241  structList.clear();
242  orderedDependentsList.clear();
243  procSet.clear();
244  procSet.addAll(inputSet);
245  doneSet.clear();
246 
247  //Set up dependency graph edges
248  while (!procSet.isEmpty()) {
249  Entry entry = procSet.iterator().next();
250  DataType dataType = entry.dataType;
251  //Msg.debug(this, "SET_SIZE: " + procSet.size());
252  //Msg.debug(this, "DTYPE_IN: " + dataType.getName());
253  if (dataType instanceof Pointer) {
254  addDependent(entry, ((Pointer) dataType).getDataType());
255  }
256  else if (dataType instanceof Array) {
257  addDependent(entry, ((Array) dataType).getDataType());
258  }
259  else if (dataType instanceof TypeDef) {
260  addDependent(entry, ((TypeDef) dataType).getDataType());
261  }
262  else if (dataType instanceof Structure) {
263  DataTypeComponent dtcomps[] = ((Structure) dataType).getComponents();
264  for (DataTypeComponent dtcomp : dtcomps) {
265  addDependent(entry, dtcomp.getDataType());
266  }
267  }
268  else if (dataType instanceof Composite) {
269  DataTypeComponent dtcomps[] = ((Composite) dataType).getComponents();
270  for (DataTypeComponent dtcomp : dtcomps) {
271  addDependent(entry, dtcomp.getDataType());
272  }
273  }
274  else if (dataType instanceof FunctionDefinition) {
275  ParameterDefinition paramDefs[] = ((FunctionDefinition) dataType).getArguments();
276  addDependent(entry, ((FunctionDefinition) dataType).getReturnType());
277  for (ParameterDefinition paramDef : paramDefs) {
278  addDependent(entry, paramDef.getDataType());
279  }
280  }
281  else { //Includes BuiltIn types, etc.
282  addDependent(entry);
283 // orderedDependentsList.add(dataType); //These weren't originally on the list, but should cause not harm.
284 // Msg.debug(this,
285 // "NO_LIST ITEM: " + dataType.getName() + "\n Size: " + dataType.getLength());
286  }
287  doneSet.add(entry);
288  procSet.remove(entry);
289  }
290 
291 // Msg.debug(this, dumpDebug());
292 
293  //Create dependentsStack by traversing and trimming edges of now-acyclic dependency graph
294  //
295  //Locate starting types: those that are not dependent on other types.
296  if (whoDependsOnMe.isEmpty()) {
297  throw new AssertException("Cannot create dependency graph on data types.");
298  }
299  for (Entry entry : doneSet) {
300  if (!whoIDependOn.containsKey(entry) || (whoIDependOn.get(entry).size() == 0)) {
301  noDependentsQueue.add(entry);
302  whoIDependOn.remove(entry);
303  }
304  }
305 
306  // Place noDependents types into the depentsStack and remove edges to types that are
307  // dependent on them, checking to see if these dependents are no longer dependent on any
308  // other types, and thus placing them in the noDependents stack for their processing.
309  while (!noDependentsQueue.isEmpty()) {
310  Entry entry = noDependentsQueue.remove();
311  //dependency stack of all types.
312  //Msg.debug(this, "ORDERED_LIST_SIZE: " + orderedDependentsList.size() + " -- TYPE: " +
313  // dataType.getName());
314  orderedDependentsList.add(entry.dataType);
315  //dependency stack of struct or typedef to struct types for which zero-sized structs should first be used. See _____TODO:method
316  if ((entry.dataType instanceof Structure) || ((entry.dataType instanceof TypeDef) &&
317  (((TypeDef) entry.dataType).getBaseDataType() instanceof Structure))) {
318  structList.add(entry.dataType);
319  }
320  removeMyDependentsEdgesToMe(entry);
321  }
322  if (!whoDependsOnMe.isEmpty() || !whoIDependOn.isEmpty()) {
323  throw new AssertException(
324  "Cycles still exist in the data type dependency graph. Debug follows.\n" +
325  dumpDebug());
326  }
327  }

329  private void addDependent(Entry entry, DataType subType) {
330  //subtype might be null, but don't expect entry to be null, but error?
331  if ((entry == null) || (subType == null)) {
332  return;
333  }
334  Entry subEntry = createEntry(subType);
335  if (!doneSet.contains(subEntry)) {
336  procSet.add(subEntry);
337  }
338  if (entry.dataType instanceof Pointer) { //avoid cycles with structures/composites
339  if ((subType instanceof Structure) || (subType instanceof TypeDef) &&
340  (((TypeDef) subType).getBaseDataType() instanceof Structure)) {
341  return;
342  }
343  }
344  Set<Entry> dependents = whoDependsOnMe.get(subEntry);
345  if (dependents == null) {
346  dependents = new HashSet<Entry>();
347  whoDependsOnMe.put(subEntry, dependents);
348  }
349  dependents.add(entry); //ignores duplicates
350  Set<Entry> support = whoIDependOn.get(entry);
351  if (support == null) {
352  support = new HashSet<Entry>();
353  whoIDependOn.put(entry, support);
354  }
355  support.add(subEntry); //ignores duplicates
356  }

371  private void removeMyDependentsEdgesToMe(Entry entry) {
372  Set<Entry> myDependents = whoDependsOnMe.get(entry);
373  if (myDependents != null) {
374  Iterator<Entry> myDependentsIter = myDependents.iterator();
375  while (myDependentsIter.hasNext()) {
376  Entry myDependent = myDependentsIter.next();
377  //get reverse information to delete forward information.
378  Set<Entry> supportSet = whoIDependOn.get(myDependent);
379  supportSet.remove(entry);
380  if (supportSet.size() == 0) {
381  noDependentsQueue.add(myDependent);
382  whoIDependOn.remove(myDependent);
383  }
384  }
385  myDependents.clear();
386  whoDependsOnMe.remove(entry);
387  }

143 void LoadImageXml::pad(void)
144 
145 {
146  map<Address,vector<uint1> >::iterator iter,lastiter;
147 
148  // Search for completely redundant chunks
149  if (chunk.empty()) return;
150  lastiter = chunk.begin();
151  iter = lastiter;
152  ++iter;
153  while(iter!=chunk.end()) {
154  if ((*lastiter).first.getSpace() == (*iter).first.getSpace()) {
155  uintb end1 = (*lastiter).first.getOffset() + (*lastiter).second.size() - 1;
156  uintb end2 = (*iter).first.getOffset() + (*iter).second.size() - 1;
157  if (end1 >= end2) {
158  chunk.erase(iter);
159  iter = lastiter;
160  ++iter;
161  continue;
162  }
163  }
164  lastiter = iter;
165  ++iter;
166  }
167 
168  iter = chunk.begin();
169  while(iter!=chunk.end()) {
170  Address endaddr = (*iter).first + (*iter).second.size();
171  if (endaddr < (*iter).first) {
172  ++iter;
173  continue; // All the way to end of space
174  }
175  ++iter;
176  int4 maxsize = 512;
177  uintb room = endaddr.getSpace()->getHighest() - endaddr.getOffset() + 1;
178  if ((uintb)maxsize > room)
179  maxsize = (int4)room;
180  if ((iter!=chunk.end())&&((*iter).first.getSpace()==endaddr.getSpace())) {
181  if (endaddr.getOffset() >= (*iter).first.getOffset()) continue;
182  room = (*iter).first.getOffset() - endaddr.getOffset();
183  if ((uintb)maxsize > room)
184  maxsize = (int4)room;
185  }
186  vector<uint1> &vec( chunk[endaddr] );
187  for(int4 i=0;i<maxsize;++i)
188  vec.push_back(0);
189  }
190 }

192 void LoadImageXml::loadFill(uint1 *ptr,int4 size,const Address &addr)
193 
194 {
195  map<Address,vector<uint1> >::const_iterator iter;
196  Address curaddr;
197  bool emptyhit = false;
198 
199  curaddr = addr;
200  iter = chunk.upper_bound(curaddr); // First one greater than
201  if (iter != chunk.begin())
202  --iter; // Last one less or equal
203  while((size>0)&&(iter!=chunk.end())) {
204  const vector<uint1> &chnk((*iter).second);
205  int4 chnksize = chnk.size();
206  int4 over = curaddr.overlap(0,(*iter).first,chnksize);
207  if (over!=-1) {
208  if (chnksize-over > size)
209  chnksize = over+size;
210  for(int4 i=over;i<chnksize;++i)
211  *ptr++ = chnk[i];
212  size -= (chnksize-over);
213  curaddr = curaddr + (chnksize-over);
214  ++iter;
215  }
216  else {
217  emptyhit = true;
218  break;
219  }
220  }
221  if ((size>0)||emptyhit) {
222  ostringstream errmsg;
223  errmsg << "Bytes at ";
224  curaddr.printRaw(errmsg);

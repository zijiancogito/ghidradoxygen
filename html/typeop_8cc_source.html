227 bool TypeOp::markExplicitUnsigned(PcodeOp *op,int4 slot) const
228 
229 {
230  if ((addlflags & inherits_sign)==0) return false;
231  if ((slot==1) && ((addlflags & inherits_sign_zero)!=0)) return false;
232  Varnode *vn = op->getIn(slot);
233  if (!vn->isConstant()) return false;
234  Datatype *dt = vn->getHigh()->getType();
235  type_metatype meta = dt->getMetatype();
236  if ((meta != TYPE_UINT)&&(meta != TYPE_UNKNOWN)) return false;
237  if (dt->isCharPrint()) return false;
238  if (dt->isEnumType()) return false;
239  if ((op->numInput() == 2) && ((addlflags & inherits_sign_zero)==0)) {
240  Varnode *firstvn = op->getIn(1-slot);
241  meta = firstvn->getHigh()->getType()->getMetatype();
242  if ((meta == TYPE_UINT)||(meta == TYPE_UNKNOWN))
243  return false; // Other side of the operation will force the unsigned
244  }
245  // Check if type is going to get forced anyway
246  Varnode *outvn = op->getOut();
247  if (outvn != (Varnode *)0) {
248  if (outvn->isExplicit()) return false;
249  PcodeOp *lone = outvn->loneDescend();
250  if (lone != (PcodeOp *)0) {
251  if (!lone->inheritsSign()) return false;
252  }
253  }
254 
255  vn->setUnsignedPrint();
256  return true;
257 }

362 Datatype *TypeOpLoad::getInputCast(const PcodeOp *op,int4 slot,const CastStrategy *castStrategy) const
363 
364 {
365  if (slot!=1) return (Datatype *)0;
366  Datatype *reqtype = op->getOut()->getHigh()->getType(); // Cast load pointer to match output
367  const Varnode *invn = op->getIn(1);
368  Datatype *curtype = invn->getHigh()->getType();
369  AddrSpace *spc = Address::getSpaceFromConst(op->getIn(0)->getAddr());
370  // Its possible that the input type is not a pointer to the output type
371  // (or even a pointer) due to cycle trimming in the type propagation algorithms
372  if (curtype->getMetatype() == TYPE_PTR)
373  curtype = ((TypePointer *)curtype)->getPtrTo();
374  else
375  return tlst->getTypePointer(invn->getSize(),reqtype,spc->getWordSize());
376  if ((curtype != reqtype)&&(curtype->getSize() == reqtype->getSize())) {
377  // If we have a non-standard in = ptr a out = b (a!=b)
378  // We may want to postpone casting BEFORE the load in favor of casting AFTER the load
379  type_metatype curmeta = curtype->getMetatype();
380  if ((curmeta!=TYPE_STRUCT)&&(curmeta!=TYPE_ARRAY)&&(curmeta!=TYPE_SPACEBASE)) {
381  // if the input is a pointer to a primitive type
382  if ((!invn->isImplied())||(!invn->isWritten())||(invn->getDef()->code() != CPUI_CAST))
383  return (Datatype *)0; // Postpone cast to output
384  // If we reach here, the input is a CAST to the wrong type
385  // We fallthru (returning the proper input case) so that the bad cast can either be
386  // adjusted or we recast
387  }
388  }
389  reqtype = castStrategy->castStandard(reqtype,curtype,false,true);
390  if (reqtype == (Datatype *)0) return reqtype;
391  return tlst->getTypePointer(invn->getSize(),reqtype,spc->getWordSize());
392 }

541 Datatype *TypeOpCall::getInputLocal(const PcodeOp *op,int4 slot) const
542 
543 {
544  const FuncCallSpecs *fc;
545  const Varnode *vn;
546  Datatype *ct;
547 
548  vn = op->getIn(0);
549  if ((slot==0)||(vn->getSpace()->getType()!=IPTR_FSPEC))// Do we have a prototype to look at
550  return TypeOp::getInputLocal(op,slot);
551 
552  // Get types of call input parameters
553  fc = FuncCallSpecs::getFspecFromConst(vn->getAddr());
554  // Its false to assume that the parameter symbol corresponds
555  // to the varnode in the same slot, but this is easiest until
556  // we get giant sized parameters working properly
557  if ((fc->numParams() == op->numInput()-1)||(fc->isDotdotdot())) {
558  ProtoParameter *param = fc->getParam(slot-1);
559  if ((param != (ProtoParameter *)0)&&(param->isTypeLocked())) {
560  ct = param->getType();
561  if ((ct->getMetatype() != TYPE_VOID)&&
562  (ct->getSize() <= op->getIn(slot)->getSize())) // parameter may not match varnode
563  return ct;
564  }
565  }
566  return TypeOp::getInputLocal(op,slot);

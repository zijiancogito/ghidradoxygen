233 void AddrSpace::printRaw(ostream &s,uintb offset) const
234 
235 {
236  int4 sz = getAddrSize();
237  if (sz > 4) {
238  if ((offset>>32) == 0)

240  else if ((offset>>48) == 0)

282 uintb AddrSpace::read(const string &s,int4 &size) const
283 
284 {
285  const char *enddata;
286  char *tmpdata;
287  int4 expsize;
288  string::size_type append;
289  string frontpart;
290  uintb offset;
291 
292  append = s.find_first_of(":+");
293  try {
294  if (append == string::npos) {
295  const VarnodeData &point(trans->getRegister(s));
296  offset = point.offset;
297  size = point.size;
298  }
299  else {
300  frontpart = s.substr(0,append);
301  const VarnodeData &point(trans->getRegister(frontpart));
302  offset = point.offset;
303  size = point.size;
304  }
305  }
306  catch(LowlevelError &err) { // Name doesn't exist
307  offset = strtoul(s.c_str(),&tmpdata,0);
308  offset = addressToByte(offset,wordsize);
309  enddata = (const char *) tmpdata;
310  if (enddata - s.c_str() == s.size()) { // If no size or offset override
311  size = getAddrSize(); // Return "natural" size
312  return offset;
313  }
314  size = getAddrSize();
315  }
316  if (append != string::npos) {
317  enddata = s.c_str()+append;
318  expsize = get_offset_size( enddata, offset );
319  if (expsize!=-1) {
320  size = expsize;
321  return offset;
322  }
323  }
324  return offset;
325 }

539 uintb JoinSpace::restoreXmlAttributes(const Element *el,uint4 &size) const
540 
541 {
542  vector<VarnodeData> pieces;
543  int4 numAttribs = el->getNumAttributes();
544  uint4 sizesum = 0;
545  uint4 logicalsize = 0;
546  for(int4 i=0;i<numAttribs;++i) {
547  string attrName = el->getAttributeName(i);
548  if (0!=attrName.compare(0,5,"piece")) {
549  if (attrName == "logicalsize") {
550  istringstream s3(el->getAttributeValue(i));
551  s3.unsetf(ios::dec | ios::hex | ios::oct);
552  s3 >> logicalsize;
553  }
554  continue;
555  }
556  int4 pos = (int4)(attrName[5] - '1');
557  while(pieces.size() <= pos)
558  pieces.push_back(VarnodeData());
559  VarnodeData &vdat( pieces[pos] );
560 
561  string attrVal = el->getAttributeValue(i);
562  string::size_type offpos = attrVal.find(':');
563  if (offpos == string::npos) {
564  const Translate *trans = getTrans();
565  const VarnodeData &point(trans->getRegister(attrVal));
566  vdat = point;
567  }
568  else {
569  string::size_type szpos = attrVal.find(':',offpos+1);
570  if (szpos==string::npos)
571  throw LowlevelError("join address piece attribute is malformed");
572  string spcname = attrVal.substr(0,offpos);
573  vdat.space = getManager()->getSpaceByName(spcname);
574  istringstream s1(attrVal.substr(offpos+1,szpos));
575  s1.unsetf(ios::dec | ios::hex | ios::oct);
576  s1 >> vdat.offset;
577  istringstream s2(attrVal.substr(szpos+1));
578  s2.unsetf(ios::dec | ios::hex | ios::oct);
579  s2 >> vdat.size;
580  }
581  sizesum += vdat.size;
582  }
583  JoinRecord *rec = getManager()->findAddJoin(pieces,logicalsize);

610 
611 {

614  int4 i=0;
615  while(i < s.size()) {
616  pieces.push_back(VarnodeData()); // Prepare to read next VarnodeData
617  string token;
618  while((i<s.size())&&(s[i]!=',')) {
619  token += s[i];

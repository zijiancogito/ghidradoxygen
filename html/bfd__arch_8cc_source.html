82 void BfdArchitecture::resolveArchitecture(void)
83 
84 {
85  archid = getTarget();
86  if (archid.find(':')==string::npos) {
87  archid = loader->getArchType();
88  // kludge to distinguish windows binaries from linux/gcc
89  if (archid.find("efi-app-ia32") != string::npos)
90  archid = "x86:LE:32:default:windows";
91  else if (archid.find("pe-i386") != string::npos)
92  archid = "x86:LE:32:default:windows";
93  else if (archid.find("pei-i386") != string::npos)
94  archid = "x86:LE:32:default:windows";
95  else if (archid.find("pei-x86-64") != string::npos)
96  archid = "x86:LE:64:default:windows";
97  else if (archid.find("sparc") != string::npos)
98  archid = "sparc:BE:32:default:default";
99  else if (archid.find("elf64") != string::npos)
100  archid = "x86:LE:64:default:gcc";
101  else if (archid.find("elf") != string::npos)
102  archid = "x86:LE:32:default:gcc";
103  else if (archid.find("mach-o") != string::npos)
104  archid = "PowerPC:BE:32:default:macosx";
105  else
106  throw LowlevelError("Cannot convert bfd target to sleigh target: "+archid);
107  }
108  SleighArchitecture::resolveArchitecture();
109 }

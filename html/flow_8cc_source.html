87 
88 {
89  PcodeOp *retop;
90  list<PcodeOp *>::const_iterator iter = op->getInsertIter();
91  ++iter;
92  if (iter != obank.endDead()) {
93  retop = *iter;
94  if (!retop->isInstructionStart()) // If within same instruction
95  return retop; // Then this is the fall thru
96  }
97  // Find address of instruction containing this op
98  map<Address,VisitStat>::const_iterator miter;
99  miter = visited.upper_bound(op->getAddr());
100  if (miter == visited.begin()) return (PcodeOp *)0;
101  --miter;
102  if ((*miter).first + (*miter).second.size <= op->getAddr())
103  return (PcodeOp *)0;
104  return target( (*miter).first + (*miter).second.size);

115 {

147 PcodeOp *FlowInfo::findRelTarget(PcodeOp *op,Address &res) const
148 
149 {
150  const Address &addr(op->getIn(0)->getAddr());
151  uintm id = op->getTime() + addr.getOffset();
152  SeqNum seqnum(op->getAddr(),id);
153  PcodeOp *retop = obank.findOp(seqnum);
154  if (retop != (PcodeOp *)0) // Is this a "properly" internal branch
155  return retop;
156 
157  // Now we check if the relative branch is really to the next instruction
158  SeqNum seqnum1(op->getAddr(),id-1);
159  retop = obank.findOp(seqnum1); // We go back one sequence number
160  if (retop != (PcodeOp *)0) {
161  // If the PcodeOp exists here then branch was indeed to next instruction
162  map<Address,VisitStat>::const_iterator miter;
163  miter = visited.upper_bound(retop->getAddr());
164  if (miter != visited.begin()) {
165  --miter;
166  res = (*miter).first + (*miter).second.size;
167  if (op->getAddr() < res)
168  return (PcodeOp *)0; // Indicate that res has the fallthru address
169  }
170  }
171  ostringstream errmsg;
172  errmsg << "Bad relative branch at instruction : (";
173  errmsg << op->getAddr().getSpace()->getName() << ',';
174  op->getAddr().printRaw(errmsg);
175  errmsg << ')';
176  throw LowlevelError(errmsg.str());
177 }

249 PcodeOp *FlowInfo::xrefControlFlow(list<PcodeOp *>::const_iterator oiter,bool &startbasic,bool &isfallthru,FuncCallSpecs *fc)
250 
251 {
252  PcodeOp *op = (PcodeOp *)0;
253  isfallthru = false;
254  uintm maxtime=0; // Deepest internal relative branch
255  while(oiter != obank.endDead()) {
256  op = *oiter++;
257  if (startbasic) {
258  data.opSetFlag(op,PcodeOp::startbasic);
259  startbasic = false;
260  }
261  switch(op->code()) {
262  case CPUI_CBRANCH:
263  {
264  const Address &destaddr( op->getIn(0)->getAddr() );
265  if (destaddr.isConstant()) {
266  Address fallThruAddr;
267  PcodeOp *destop = findRelTarget(op,fallThruAddr);
268  if (destop != (PcodeOp *)0) {
269  data.opSetFlag(destop,PcodeOp::startbasic); // Make sure the target op is a basic block start
270  uintm newtime = destop->getTime();
271  if (newtime > maxtime)
272  maxtime = newtime;
273  }
274  else
275  isfallthru = true; // Relative branch is to end of instruction
276  }
277  else
278  newAddress(op,destaddr); // Generate branch address
279  startbasic = true;
280  }
281  break;
282  case CPUI_BRANCH:
283  {
284  const Address &destaddr( op->getIn(0)->getAddr() );
285  if (destaddr.isConstant()) {
286  Address fallThruAddr;
287  PcodeOp *destop = findRelTarget(op,fallThruAddr);
288  if (destop != (PcodeOp *)0) {
289  data.opSetFlag(destop,PcodeOp::startbasic); // Make sure the target op is a basic block start
290  uintm newtime = destop->getTime();
291  if (newtime > maxtime)
292  maxtime = newtime;
293  }
294  else
295  isfallthru = true; // Relative branch is to end of instruction
296  }
297  else
298  newAddress(op,destaddr); // Generate branch address
299  if (op->getTime() >= maxtime) {
300  deleteRemainingOps(oiter);
301  oiter = obank.endDead();
302  }
303  startbasic = true;
304  }
305  break;
306  case CPUI_BRANCHIND:
307  tablelist.push_back(op); // Put off trying to recover the table
308  if (op->getTime() >= maxtime) {
309  deleteRemainingOps(oiter);
310  oiter = obank.endDead();
311  }
312  startbasic = true;
313  break;
314  case CPUI_RETURN:
315  if (op->getTime() >= maxtime) {
316  deleteRemainingOps(oiter);
317  oiter = obank.endDead();
318  }
319  startbasic = true;
320  break;
321  case CPUI_CALL:
322  if (setupCallSpecs(op,fc))
323  --oiter; // Backup one op, to pickup halt
324  break;
325  case CPUI_CALLIND:
326  if (setupCallindSpecs(op,true,fc))
327  --oiter; // Backup one op, to pickup halt
328  break;
329  case CPUI_CALLOTHER:
330  {
331  InjectedUserOp *userop = dynamic_cast<InjectedUserOp *>(glb->userops.getOp(op->getIn(0)->getOffset()));
332  if (userop != (InjectedUserOp *)0)
333  injectlist.push_back(op);
334  break;
335  }
336  default:
337  break;
338  }
339  }
340  if (isfallthru) // We have seen an explicit relative branch to end of instruction
341  startbasic = true; // So we know next instruction starts a basicblock
342  else { // If we haven't seen a relative branch, calculate fallthru by looking at last op
343  if (op == (PcodeOp *)0)
344  isfallthru = true; // No ops at all, mean a fallthru
345  else {
346  switch(op->code()) {
347  case CPUI_BRANCH:
348  case CPUI_BRANCHIND:
349  case CPUI_RETURN:
350  break; // If the last instruction is a branch, then no fallthru
351  default:
352  isfallthru = true; // otherwise it is a fallthru
353  break;
354  }
355  }
356  }
357  return op;
358 }

369 bool FlowInfo::processInstruction(const Address &curaddr,bool &startbasic)
370 
371 {
372  bool emptyflag;
373  bool isfallthru = true;
374  // JumpTable *jt;
375  list<PcodeOp *>::const_iterator oiter;
376  int4 step;
377  uint4 flowoverride;
378 
379  if (insn_count >= insn_max) {
380  if ((flags & error_toomanyinstructions)!=0)
381  throw LowlevelError("Flow exceeded maximum allowable instructions");
382  else {
383  step = 1;
384  artificialHalt(curaddr,PcodeOp::badinstruction);
385  data.warning("Too many instructions -- Truncating flow here",curaddr);
386  if (!hasTooManyInstructions()) {
387  flags |= toomanyinstructions_present;
388  data.warningHeader("Exceeded maximum allowable instructions: Some flow is truncated");
389  }
390  }
391  }
392  insn_count += 1;
393 
394  if (obank.empty())
395  emptyflag = true;
396  else {
397  emptyflag = false;
398  oiter = obank.endDead();
399  --oiter;
400  }
401  if (flowoverride_present)
402  flowoverride = data.getOverride().getFlowOverride(curaddr);
403  else
404  flowoverride = Override::NONE;
405 
406  try {
407  step = glb->translate->oneInstruction(emitter,curaddr); // Generate ops for instruction
408  }
409  catch(UnimplError &err) { // Instruction is unimplemented
410  if ((flags & ignore_unimplemented)!=0) {
411  step = err.instruction_length;
412  if (!hasUnimplemented()) {
413  flags |= unimplemented_present;
414  data.warningHeader("Control flow ignored unimplemented instructions");
415  }
416  }
417  else if ((flags & error_unimplemented)!=0)
418  throw err; // rethrow
419  else {
420  // Add infinite loop instruction
421  step = 1; // Pretend size 1
422  artificialHalt(curaddr,PcodeOp::unimplemented);
423  data.warning("Unimplemented instruction - Truncating control flow here",curaddr);
424  if (!hasUnimplemented()) {
425  flags |= unimplemented_present;
426  data.warningHeader("Control flow encountered unimplemented instructions");
427  }
428  }
429  }
430  catch(BadDataError &err) {
431  if ((flags & error_unimplemented)!=0)
432  throw err; // rethrow
433  else {
434  // Add infinite loop instruction
435  step = 1; // Pretend size 1
436  artificialHalt(curaddr,PcodeOp::badinstruction);
437  data.warning("Bad instruction - Truncating control flow here",curaddr);
438  if (!hasBadData()) {
439  flags |= baddata_present;
440  data.warningHeader("Control flow encountered bad instruction data");
441  }
442  }
443  }
444  VisitStat &stat(visited[curaddr]); // Mark that we visited this instruction
445  stat.size = step; // Record size of instruction
446 
447  if (curaddr < minaddr) // Update minimum and maximum address
448  minaddr = curaddr;
449  if (maxaddr < curaddr+step) // Keep track of biggest and smallest address
450  maxaddr = curaddr+step;
451 
452  if (emptyflag) // Make sure oiter points at first new op
453  oiter = obank.beginDead();
454  else
455  ++oiter;
456 
457  if (oiter != obank.endDead()) {
458  stat.seqnum = (*oiter)->getSeqNum();
459  data.opSetFlag(*oiter,PcodeOp::startmark); // Mark the first op in the instruction
460  if (flowoverride != Override::NONE)
461  data.overrideFlow(curaddr,flowoverride);
462  xrefControlFlow(oiter,startbasic,isfallthru,(FuncCallSpecs *)0);
463  }
464 
465  if (isfallthru)
466  addrlist.push_back(curaddr+step);
467  return isfallthru;
468 }

475 bool FlowInfo::setFallthruBound(Address &bound)
476 
477 {
478  map<Address,VisitStat>::const_iterator iter;
479  const Address &addr( addrlist.back() );
480 
481  iter = visited.upper_bound(addr); // First range greater than addr
482  if (iter!=visited.begin()) {
483  --iter; // Last range less than or equal to us
484  if (addr == (*iter).first) { // If we have already visited this address
485  addrlist.pop_back(); // Throw it away
486  PcodeOp *op = target(addr); // But make sure the address
487  data.opSetFlag(op,PcodeOp::startbasic); // starts a basic block
488  return false;
489  }
490  if (addr < (*iter).first + (*iter).second.size)
491  reinterpreted(addr);
492  ++iter;
493  }
494  if (iter!=visited.end()) // Whats the maximum distance we can go
495  bound = (*iter).first;
496  else
497  bound = eaddr;
498  return true;
499 }

531 void FlowInfo::fallthru(void)
532 
533 {
534  Address bound;
535 
536  if (!setFallthruBound(bound)) return;
537 
538  Address curaddr;
539  bool startbasic = true;
540  bool fallthruflag;
541 
542  for(;;) {
543  curaddr = addrlist.back();
544  addrlist.pop_back();
545  fallthruflag = processInstruction(curaddr,startbasic);
546  if (!fallthruflag) break;
547  if (addrlist.empty()) break;
548  if (bound <= addrlist.back()) {
549  if (bound == eaddr) {
550  handleOutOfBounds(eaddr,addrlist.back());
551  unprocessed.push_back(addrlist.back());
552  addrlist.pop_back();
553  return;
554  }
555  if (bound == addrlist.back()) { // Hit the bound exactly
556  if (startbasic) {
557  PcodeOp *op = target(addrlist.back());
558  data.opSetFlag(op,PcodeOp::startbasic);
559  }
560  addrlist.pop_back();
561  break;
562  }
563  if (!setFallthruBound(bound)) return; // Reset bound
564  }
565  }
566 }

749 void FlowInfo::generateOps(void)
750 
751 {
752  vector<PcodeOp *> notreached; // indirect ops that are not reachable
753  int4 notreachcnt = 0;
754  clearProperties();
755  addrlist.push_back(data.getAddress());
756  while(!addrlist.empty()) // Recovering as much as possible except jumptables
757  fallthru();
758  do {
759  bool collapsed_jumptable = false;
760  while(!tablelist.empty()) { // For each jumptable found
761  PcodeOp *op = tablelist.back();
762  tablelist.pop_back();
763  int4 failuremode;
764  JumpTable *jt = data.recoverJumpTable(op,this,failuremode); // Recover it
765  if (jt == (JumpTable *)0) { // Could not recover jumptable
766  if ((failuremode == 3) && (!tablelist.empty()) && (!isInArray(notreached,op))) {
767  // If the indirect op was not reachable with current flow AND there is more flow to generate,
768  // AND we haven't tried to recover this table before
769  notreached.push_back(op); // Save this op so we can try to recovery table again later
770  }
771  else if (!isFlowForInline()) // Unless this flow is being inlined for something else
772  truncateIndirectJump(op,failuremode); // Treat the indirect jump as a call
773  }
774  else {
775  int4 num = jt->numEntries();
776  for(int4 i=0;i<num;++i)
777  newAddress(op,jt->getAddressByIndex(i));
778  if (jt->isPossibleMultistage())
779  collapsed_jumptable = true;
780  while(!addrlist.empty()) // Try to fill in as much more as possible
781  fallthru();
782  }
783  }
784 
785  checkContainedCall(); // Check for PIC constructions
786  if (collapsed_jumptable)
787  checkMultistageJumptables();
788  while(notreachcnt < notreached.size()) {
789  tablelist.push_back(notreached[notreachcnt]);
790  notreachcnt += 1;

878 void FlowInfo::collectEdges(void)
879 
880 {
881  list<PcodeOp *>::const_iterator iter,iterend,iter1,iter2;
882  PcodeOp *op,*targ_op;
883  JumpTable *jt;
884  bool nextstart;
885  int4 i,num;
886 
887  if (bblocks.getSize() != 0)
888  throw RecovError("Basic blocks already calculated\n");
889 
890  iter = obank.beginDead();
891  iterend = obank.endDead();
892  while(iter!=iterend) {
893  op = *iter++;
894  if (iter==iterend)
895  nextstart = true;
896  else
897  nextstart = (*iter)->isBlockStart();
898  switch(op->code()) {
899  case CPUI_BRANCH:
900  targ_op = branchTarget(op);
901  block_edge1.push_back(op);
902  // block_edge2.push_back(op->Input(0)->getAddr().Iop());
903  block_edge2.push_back(targ_op);
904  break;
905  case CPUI_BRANCHIND:
906  jt = data.findJumpTable(op);
907  if (jt == (JumpTable *)0) break;
908  // If we are in this routine and there is no table
909  // Then we must be doing partial flow analysis
910  // so assume there are no branches out
911  num = jt->numEntries();
912  for(i=0;i<num;++i) {
913  targ_op = target(jt->getAddressByIndex(i));
914  if (targ_op->isMark()) continue; // Already a link between these blocks
915  targ_op->setMark();
916  block_edge1.push_back(op);
917  block_edge2.push_back(targ_op);
918  }
919  iter1 = block_edge1.end(); // Clean up our marks
920  iter2 = block_edge2.end();
921  while(iter1 != block_edge1.begin()) {
922  --iter1;
923  --iter2;
924  if ((*iter1)==op)
925  (*iter2)->clearMark();
926  else
927  break;
928  }
929  break;
930  case CPUI_RETURN:
931  break;
932  case CPUI_CBRANCH:
933  targ_op = fallthruOp(op); // Put in fallthru edge
934  block_edge1.push_back(op);
935  block_edge2.push_back(targ_op);
936  targ_op = branchTarget(op);
937  block_edge1.push_back(op);
938  block_edge2.push_back(targ_op);
939  break;
940  default:
941  if (nextstart) { // Put in fallthru edge if new basic block
942  targ_op = fallthruOp(op);
943  block_edge1.push_back(op);
944  block_edge2.push_back(targ_op);
945  }
946  break;
947  }
948  }
949 }

872  istate.reserve(list.size());
873  for(int4 i=0;i<list.size();++i) {
874  tmpbl = list[i];
875  tmpbl->index = -1; // reverse post-order starts at 0
876  tmpbl->visitcount = -1;
877  tmpbl->copymap = tmpbl;
878  if (tmpbl->sizeIn()==0) // Keep track of all potential roots of the tree
879  rootlist.push_back(tmpbl);
880  }
881  if (rootlist.size() > 1) { // Make sure orighead is visited last, (so it is first in the reverse post order)
882  tmpbl = rootlist[rootlist.size()-1];
883  rootlist[rootlist.size()-1] = rootlist[0];
884  rootlist[0] = tmpbl;
885  }
886  else if (rootlist.size() == 0) { // If there's no obvious starting block
887  rootlist.push_back(list[0]); // Assume first block is entry point
888  }
889  origrootpos = rootlist.size()-1; // Position of original head in rootlist
890 
891  for(int4 repeat=0;repeat<2;++repeat) {
892  bool extraroots = false;
893  int4 rpostcount = list.size();
894  int4 rootindex = 0;
895  clearEdgeFlags(~((uint4)0)); // Clear all edge flags
896  while(preorder.size() < list.size()) {
897  FlowBlock *startbl = (FlowBlock *)0;
898  while(rootindex<rootlist.size()) { // Go thru blocks with no in edges
899  startbl = rootlist[rootindex];
900  rootindex += 1;
901  if (startbl->visitcount == -1) break;
902  // If we reach here, startbl isn't really a root (root from previous pass)
903  for(int4 i=rootindex;i<rootlist.size();++i)
904  rootlist[i-1] = rootlist[i];
905  rootlist.pop_back(); // Remove it
906  rootindex -= 1;
907  startbl = (FlowBlock *)0;
908  }
909  if (startbl == (FlowBlock *)0) { // If we didn't find one, just take next unvisited
910  extraroots = true;
911  for(int4 i=0;i<list.size();++i) {
912  startbl = list[i];
913  if (startbl->visitcount == -1) break;
914  }
915  rootlist.push_back(startbl); // We have to treat this block as another root
916  rootindex += 1; // Update root traversal state
917  }
918 
919  state.push_back(startbl);
920  istate.push_back(0);
921  startbl->visitcount = preorder.size();
922  preorder.push_back(startbl);
923  startbl->numdesc = 1;
924 
925  while(!state.empty()) {
926  FlowBlock *curbl = state.back();
927  if (curbl->sizeOut() <= istate.back()) { // We've visited all children of this node
928  state.pop_back();
929  istate.pop_back();
930  rpostcount -= 1;
931  curbl->index = rpostcount;
932  rpostorder[rpostcount] = curbl;
933  if (!state.empty())
934  state.back()->numdesc += curbl->numdesc;
935  }
936  else {

938  istate.back() += 1; // Next visit to this state should try next child
939  if (curbl->isIrreducibleOut(edgenum)) continue; // Pretend irreducible edges don't exist

941 
942  if (childbl->visitcount == -1) { // If we haven't visited this node before
943  curbl->setOutEdgeFlag(edgenum,f_tree_edge);
944  state.push_back(childbl);
945  istate.push_back(0);
946  childbl->visitcount = preorder.size();
947  preorder.push_back(childbl);
948  childbl->numdesc = 1;
949  }
950  else if (childbl->index == -1) // childbl is already on stack
951  curbl->setOutEdgeFlag(edgenum,f_back_edge|f_loop_edge);
952  else if (curbl->visitcount < childbl->visitcount) // childbl processing is already done
953  curbl->setOutEdgeFlag(edgenum,f_forward_edge);
954  else
955  curbl->setOutEdgeFlag(edgenum,f_cross_edge);
956  }
957  }
958  }
959  if (!extraroots) break;
960  if (repeat==1)
961  throw LowlevelError("Could not generate spanning tree");
962 
963  // We had extra roots we did not know about so we have to regenerate the post order so entry block comes first
964  tmpbl = rootlist[rootlist.size()-1];
965  rootlist[rootlist.size()-1] = rootlist[origrootpos]; // Move entry block to last position in rootlist
966  rootlist[origrootpos] = tmpbl;
967 
968  for(int4 i=0;i<list.size();++i) {
969  tmpbl = list[i];
970  tmpbl->index = -1; // reverse post-order starts at 0
971  tmpbl->visitcount = -1;
972  tmpbl->copymap = tmpbl;
973  }
974  preorder.clear();
975  state.clear();
976  istate.clear();
977  }
978 
979  if (rootlist.size() > 1) { // Make sure orighead is at the front of the rootlist as well
980  tmpbl = rootlist[rootlist.size()-1];
981  rootlist[rootlist.size()-1] = rootlist[0];
982  rootlist[0] = tmpbl;
983  }
984 
985  list = rpostorder;
986 }

997 bool BlockGraph::findIrreducible(const vector<FlowBlock *> &preorder,int4 &irreduciblecount)
998 
999 {
1000  vector<FlowBlock *> reachunder; // The current reachunder set being built (also with mark set on each block)
1001  bool needrebuild = false;
1002  int4 xi = preorder.size()-1;
1003  while(xi >= 0) { // For each vertex in reverse pre-order
1004  FlowBlock *x = preorder[xi];
1005  xi -= 1;
1006  int4 sizein = x->sizeIn();
1007  for(int4 i=0;i<sizein;++i) {
1008  if (!x->isBackEdgeIn(i)) continue; // For each back-edge into x
1009  FlowBlock *y = x->getIn(i);
1010  if (y==x) continue; // Reachunder set does not include the loop head
1011  reachunder.push_back(y->copymap); // Add FIND(y) to reachunder
1012  y->copymap->setMark();
1013  }
1014  int4 q = 0;
1015  while(q < reachunder.size()) {
1016  FlowBlock *t = reachunder[q];
1017  q += 1;
1018  int4 sizein_t = t->sizeIn();
1019  for(int4 i=0;i<sizein_t;++i) {
1020  if (t->isIrreducibleIn(i)) continue; // Pretend irreducible edges don't exist
1021  // All back-edges into t have already been collapsed, so this is
1022  FlowBlock *y = t->getIn(i); // For each forward, tree, or cross edge
1023  FlowBlock *yprime = y->copymap; // y' = FIND(y)
1024  if ((x->visitcount > yprime->visitcount)||( x->visitcount + x->numdesc <= yprime->visitcount)) {
1025  // The original Tarjan algorithm reports reducibility failure at this point
1026  irreduciblecount += 1;
1027  int4 edgeout = t->getInRevIndex(i);
1028  y->setOutEdgeFlag(edgeout,f_irreducible);
1029  if (t->isTreeEdgeIn(i))
1030  needrebuild = true; // If a tree edge is irreducible, we need to rebuild the spanning tree
1031  else // Otherwise we can pretend the edge was already marked irreducible
1032  y->clearOutEdgeFlag(edgeout,f_cross_edge|f_forward_edge);
1033  }
1034  else if ((!yprime->isMark())&&(yprime != x)) { // if yprime is not in reachunder and not equal to x
1035  reachunder.push_back(yprime);
1036  yprime->setMark();
1037  }
1038  }
1039  }
1040  // Collapse reachunder into a single node, labeled as x
1041  for(int4 i=0;i<reachunder.size();++i) {
1042  FlowBlock *s = reachunder[i];
1043  s->clearMark();
1044  s->copymap = x;
1045  }
1046  reachunder.clear();
1047  }
1048  return needrebuild;
1049 }

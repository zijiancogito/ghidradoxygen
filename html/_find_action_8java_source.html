48  @Override
49  public boolean isEnabledForContext(ActionContext context) {
50  if (!(context instanceof DecompilerActionContext)) {
51  return false;
52  }
53  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
54  if (decompilerActionContext.isDecompiling()) {
55  // Let this through here and handle it in actionPerformed(). This lets us alert
56  // the user that they have to wait until the decompile is finished. If we are not
57  // enabled at this point, then the keybinding will be propagated to the global
58  // actions, which is not what we want.
59  return true;
60  }
61  return true;
62  }

64  @Override
65  public void actionPerformed(ActionContext context) {
66  // Note: we intentionally do this check here and not in isEnabledForContext() so
67  // that global events do not get triggered.
68  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
69  if (decompilerActionContext.isDecompiling()) {
70  Msg.showInfo(getClass(), context.getComponentProvider().getComponent(),
71  "Decompiler Action Blocked",

123  ClangTextField textLine = (ClangTextField) lines.get(lineNumber);
124 
125  int fieldNumber = 0; // always 0, as the field is the entire line and it is the only field
126  int rowCount = textLine.getNumRows();
127  int row = rowCount - 1; // 0-based
128  int column = textLine.getNumCols(row);
129  FieldLocation fieldLocation = new FieldLocation(lineNumber, fieldNumber, row, column);
130  return new DecompilerCursorPosition(fieldLocation);

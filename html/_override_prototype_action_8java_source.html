112  @Override
113  public void actionPerformed(ActionContext context) {
114  // Note: we intentionally do this check here and not in isEnabledForContext() so
115  // that global events do not get triggered.
116  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
117  if (decompilerActionContext.isDecompiling()) {
118  Msg.showInfo(getClass(), context.getComponentProvider().getComponent(),
119  "Decompiler Action Blocked",
120  "You cannot perform Decompiler actions while the Decompiler is busy");
121  return;
122  }
123 
124  Function func = controller.getFunction();
125  Program program = func.getProgram();
126  PcodeOp op = getCallOp(controller);
127  Function calledfunc = getCalledFunction(op);
128  boolean varargs = false;
129  if (calledfunc != null)
130  varargs = calledfunc.hasVarArgs();
131  if ((op.getOpcode() == PcodeOp.CALL) && !varargs) {
132  if (OptionDialog.showOptionDialog(
133  controller.getDecompilerPanel(),
134  "Warning : Localized Override",
135  "Incorrect information entered here may hide other good information.\n"
136  + "For direct calls, it is usually better to alter the prototype on the function\n"
137  + "itself, rather than overriding the local call. Proceed anyway?", "Proceed") != 1)
138  return;
139  }
140  Address addr = op.getSeqnum().getTarget();
141  String name = "func"; // Default if we don't have a real name
142  String conv = program.getCompilerSpec().getDefaultCallingConvention().getName();
143  if (calledfunc != null) {
144  name = calledfunc.getName();
145  conv = calledfunc.getCallingConventionName();
146  }
147 
148  String signature = generateSignature(op, name);
149  ProtoOverrideDialog dialog = new ProtoOverrideDialog(tool, func, signature, conv);
150  // dialog.setHelpLocation( new HelpLocation( getOwner(), "Edit_Function_Signature" ) );
151  tool.showDialog(dialog);
152  FunctionDefinition fdef = dialog.getFunctionDefinition();
153  if (fdef == null)
154  return;
155  int transaction = program.startTransaction("Override Signature");
156  boolean commit = false;
157  try {
158  HighFunctionDBUtil.writeOverride(func, addr, fdef);

164  }

174  public static PcodeOp getCallOp(DecompilerController controller) {
175  DecompilerPanel decompilerPanel = controller.getDecompilerPanel();
176  ClangToken tokenAtCursor = decompilerPanel.getTokenAtCursor();
177  if (tokenAtCursor == null) {
178  return null;
179  }
180  if (tokenAtCursor instanceof ClangFuncNameToken) {
181  return ((ClangFuncNameToken) tokenAtCursor).getPcodeOp();
182  }
183 
184  Address addr = tokenAtCursor.getMinAddress();
185  if (addr == null)
186  return null;
187  Instruction instr = controller.getProgram().getListing().getInstructionAt(addr);
188  if (instr == null)
189  return null;
190  if (!instr.getFlowType().isCall())
191  return null;
192  ClangFunction cfunc = tokenAtCursor.getClangFunction();
193  if (cfunc == null)
194  return null;
195  HighFunction hfunc = cfunc.getHighFunction();
196  Iterator<PcodeOpAST> iter = hfunc.getPcodeOps(addr);
197  while (iter.hasNext()) {
198  PcodeOpAST op = iter.next();
199  if ((op.getOpcode() == PcodeOp.CALL) || (op.getOpcode() == PcodeOp.CALLIND))
200  return op;
201  }
202 
203  return null;
204  }

206  private Function getCalledFunction(PcodeOp op) {
207  if (op.getOpcode() != PcodeOp.CALL)
208  return null;
209  Address addr = op.getInput(0).getAddress();
210  Program program = controller.getProgram();
211  return program.getFunctionManager().getFunctionAt(addr);
212  }

214  private String generateSignature(PcodeOp op, String name) {
215  StringBuffer buf = new StringBuffer();
216  Varnode vn = op.getOutput();
217  DataType dt = null;
218  if (vn != null) {
219  dt = vn.getHigh().getDataType();
220  }
221  if (dt != null)
222  buf.append(dt.getDisplayName());
223  else
224  buf.append(DataType.VOID.getDisplayName());
225 
226  buf.append(' ').append(name).append('(');
227  for (int i = 1; i < op.getNumInputs(); ++i) {
228  vn = op.getInput(i);
229  dt = null;
230  if (vn != null)
231  dt = vn.getHigh().getDataType();
232  if (dt != null)
233  buf.append(dt.getDisplayName());
234  else
235  buf.append("BAD");
236  if (i != op.getNumInputs() - 1)
237  buf.append(',');
238  }
239  buf.append(')');
240  return buf.toString();
241  }

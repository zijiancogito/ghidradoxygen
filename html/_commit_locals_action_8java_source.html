46  @Override
47  public boolean isEnabledForContext(ActionContext context) {
48  if (!(context instanceof DecompilerActionContext)) {
49  return false;
50  }
51 
52  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
53  if (decompilerActionContext.isDecompiling()) {
54  // Let this through here and handle it in actionPerformed(). This lets us alert
55  // the user that they have to wait until the decompile is finished. If we are not
56  // enabled at this point, then the keybinding will be propagated to the global
57  // actions, which is not what we want.
58  return true;
59  }
60 
61  Function function = controller.getFunction();
62  if (function == null || function instanceof UndefinedFunction) {
63  return false;
64  }
65  return getHighFunction() != null;
66  }
67 
68  @Override
69  public void actionPerformed(ActionContext context) {
70  // Note: we intentionally do this check here and not in isEnabledForContext() so
71  // that global events do not get triggered.
72  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
73  if (decompilerActionContext.isDecompiling()) {
74  Msg.showInfo(getClass(), context.getComponentProvider().getComponent(),
75  "Decompiler Action Blocked",
76  "You cannot perform Decompiler actions while the Decompiler is busy");
77  return;
78  }
79 
80  Program program = controller.getProgram();
81  int transaction = program.startTransaction("Commit Params/Return");
82  try {
83  HighFunction hfunc = getHighFunction();
84  HighFunctionDBUtil.commitLocalsToDatabase(hfunc, SourceType.USER_DEFINED);
85  }
86  finally {
87  program.endTransaction(transaction, true);
88  }
89  }
90 
91  private HighFunction getHighFunction() {
92  DecompilerPanel decompilerPanel = controller.getDecompilerPanel();
93  ClangToken tokenAtCursor = decompilerPanel.getTokenAtCursor();
94  if (tokenAtCursor == null) {
95  return null;
96  }
97  ClangFunction clfunc = tokenAtCursor.getClangFunction();
98  if (clfunc == null) {
99  return null;
100  }
101  return clfunc.getHighFunction();
102  }

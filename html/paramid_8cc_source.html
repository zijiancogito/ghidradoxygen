19 // direct reads is for all opcodes, with special for these:
20 // BRANCH is direct read on input0. No direct write.
21 // CBRANCH is direct read on input0 and input1. No direct write.
22 // BRANCHIND is direct read on input0 (like call but no params). No direct write.
23 // CALL is direct read on input0 (putative/presumptive param flag on params--other inputs). Special (non-direct) write of output.
24 // CALLIND same as on CALL. Special (non-direct) write of output.
25 // CALLOTHER is direct read on ALL PARAMETERS (input0 and up)--is specified in sleigh. Direct write if output exists.
26 // INDIRECT is least powerful input and output of all.
27 // MULTIEQUALS is flow through but must test for and not flow through loop paths (whether from param forward our return backward directions).
28 //

30 #define MAXDEPTH 10
31 void ParamMeasure::walkforward( WalkState &state, PcodeOp *ignoreop, Varnode *vn )
32 
33 {
34  state.depth += 1;
35  if (state.depth >= MAXDEPTH) {
36  state.depth -= 1;
37  return;
38  }
39  list<PcodeOp *>::const_iterator iter = vn->beginDescend();
40  while( rank != state.terminalrank && iter != vn->endDescend() ) {
41  PcodeOp *op = *iter;
42  if( op != ignoreop ) {
43  OpCode oc = op->getOpcode()->getOpcode();
44  switch( oc ) {
45  case CPUI_BRANCH:
46  case CPUI_BRANCHIND:
47  if( op->getSlot(vn) == 0 ) updaterank( DIRECTREAD, state.best );
48  break;
49  case CPUI_CBRANCH:
50  if( op->getSlot(vn) < 2 ) updaterank( DIRECTREAD, state.best );
51  break;
52  case CPUI_CALL:
53  case CPUI_CALLIND:
54  if( op->getSlot(vn) == 0 ) updaterank( DIRECTREAD, state.best );
55  else {
56  numcalls++;
57  updaterank( SUBFNPARAM, state.best );
58  }
59  break;
60  case CPUI_CALLOTHER:
61  updaterank( DIRECTREAD, state.best );
62  break;
63  case CPUI_RETURN:
64  updaterank( THISFNRETURN, state.best );
65  break;
66  case CPUI_INDIRECT:
67  updaterank( INDIRECT, state.best );
68  break;
69  case CPUI_MULTIEQUAL:
70  // The only op for which there can be a loop in the graph is with the MULTIEQUAL (not for CALL, etc.).
71  // Walk forward only if the path is not part of a loop.
72  if( !op->getParent()->isLoopIn(op->getSlot(vn)) ) walkforward( state, (PcodeOp *)0, op->getOut() );
73  break;
74  default:
75  updaterank( DIRECTREAD, state.best );
76  break;
77  }
78  }
79  iter++;
80  }
81  state.depth -= 1;
82 }

84 void ParamMeasure::walkbackward( WalkState &state, PcodeOp *ignoreop, Varnode *vn )
85 
86 {
87  if( vn->isInput() ) {
88  updaterank( THISFNPARAM, state.best );
89  return;
90  }
91  else if( !vn->isWritten() ) {
92  updaterank( THISFNPARAM, state.best ); //TODO: not sure about this.
93  return;
94  }
95 
96  PcodeOp *op = vn->getDef();
97  OpCode oc = op->getOpcode()->getOpcode();
98  switch( oc ) {
99  case CPUI_BRANCH:
100  case CPUI_BRANCHIND:
101  case CPUI_CBRANCH:
102  case CPUI_CALL:
103  case CPUI_CALLIND:
104  break;
105  case CPUI_CALLOTHER:
106  if( op->getOut() != (Varnode *) 0 ) updaterank( DIRECTREAD, state.best );
107  break;
108  case CPUI_RETURN:
109  updaterank( SUBFNRETURN, state.best );
110  break;
111  case CPUI_INDIRECT:
112  updaterank( INDIRECT, state.best );
113  break;
114  case CPUI_MULTIEQUAL:
115  // The only op for which there can be a loop in the graph is with the MULTIEQUAL (not for CALL, etc.).
116  // Walk backward only if the path is not part of a loop.
117  for( int4 slot = 0; slot < op->numInput() && rank != state.terminalrank; slot++ )
118  if( !op->getParent()->isLoopIn(slot) ) walkbackward( state, op, op->getIn(slot) );
119  break;
120  default:
121  //Might be DIRECTWRITEWITHOUTREAD, but we do not know yet.
122  //So now try to walk forward to see if there is at least one path
123  // forward (other than the path we took to get here walking backward)
124  // in which there is not a direct read of this write.
125  ParamMeasure pmfw( vn->getAddr(), vn->getSize(), vn->getType(), INPUT );
126  pmfw.calculateRank( false, vn, ignoreop );
127  if( pmfw.getMeasure() == DIRECTREAD )
128  updaterank( DIRECTWRITEWITHREAD, state.best );
129  else
130  updaterank( DIRECTWRITEWITHOUTREAD, state.best );
131  break;
132  }
133 }

135 void ParamMeasure::calculateRank(bool best,Varnode *basevn,PcodeOp *ignoreop)
136 
137 {
138  WalkState state;
139  state.best = best;
140  state.depth = 0;
141  if( best ) {
142  rank = WORSTRANK;
143  state.terminalrank = (io == INPUT) ? DIRECTREAD : DIRECTWRITEWITHOUTREAD;
144  } else {
145  rank = BESTRANK;
146  state.terminalrank = INDIRECT;
147  }
148  numcalls = 0;
149  if (io == INPUT)
150  walkforward(state, ignoreop, basevn);
151  else
152  walkbackward(state, ignoreop, basevn);
153 }

179 ParamIDAnalysis::ParamIDAnalysis( Funcdata *fd_in, bool justproto )
180 
181 {
182  fd = fd_in;
183  if (justproto) { // We only provide info on the recovered prototype
184  const FuncProto &fproto( fd->getFuncProto() );
185  int4 num = fproto.numParams();
186  for(int4 i=0;i<num;++i) {
187  ProtoParameter *param = fproto.getParam(i);
188  InputParamMeasures.push_back( ParamMeasure(param->getAddress(),param->getSize(),
189  param->getType(),ParamMeasure::INPUT) );
190  Varnode *vn = fd->findVarnodeInput(param->getSize(),param->getAddress());
191  if (vn != (Varnode *)0)
192  InputParamMeasures.back().calculateRank(true,vn,(PcodeOp *)0);
193  }
194 
195  ProtoParameter *outparam = fproto.getOutput();
196  if (!outparam->getAddress().isInvalid()) { // If we don't have a void type
197  OutputParamMeasures.push_back( ParamMeasure( outparam->getAddress(),outparam->getSize(),
198  outparam->getType(),ParamMeasure::OUTPUT) );
199  list<PcodeOp *>::const_iterator rtn_iter = fd->beginOp( CPUI_RETURN );
200  while( rtn_iter != fd->endOp( CPUI_RETURN ) ) {
201  PcodeOp *rtn_op = *rtn_iter;
202  // For RETURN op, input0 is address location of indirect return, input1,
203  // if it exists, is the Varnode returned, output = not sure.
204  if( rtn_op->numInput() == 2 ) {
205  Varnode *ovn = rtn_op->getIn(1);
206  if( ovn != (Varnode *)0 ) { //Not a void return
207  OutputParamMeasures.back().calculateRank(true, ovn, rtn_op );
208  break;
209  }
210  }
211  rtn_iter++;
212  }
213  }
214  }
215  else {
216  // Need to list input varnodes that are outside of the model
217  VarnodeDefSet::const_iterator iter,enditer;
218  iter = fd->beginDef(Varnode::input);
219  enditer = fd->endDef(Varnode::input);
220  while(iter != enditer) {
221  Varnode *invn = *iter;
222  ++iter;
223  InputParamMeasures.push_back( ParamMeasure(invn->getAddr(),invn->getSize(),
224  invn->getType(),ParamMeasure::INPUT) );
225  InputParamMeasures.back().calculateRank(true, invn, (PcodeOp *)0 );
226  }
227  }
228 }

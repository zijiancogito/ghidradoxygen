77  private boolean matches(Program otherProgram, ProgramLocation otherLocation) {
78  if (functionToDecompile == null) {
79  return false;
80  }
81  if (program != otherProgram) {
82  return false;
83  }
84  if (otherLocation.getAddress() == null) {
85  return true;

93  @Override
94  public void monitoredRun(TaskMonitor monitor) {
95  monitor.setIndeterminate(true);
96  Function function = findFunction(monitor);
97  if (function == null) {
98  return;
99  }
100  CodeUnit cu = program.getListing().getCodeUnitAt(function.getEntryPoint());
101  if (cu instanceof Data && ((Data) cu).isDefined()) {
102  return;
103  }
104  monitor.setMessage("Decompiling function: " + function.getName() + "...");
105  functionToDecompile = function;
106  try {
107  decompileResults =
108  decompilerManager.decompile(program, functionToDecompile, debugFile, monitor);
109  }
110  catch (DecompileException e) {

112  }

114  }

136  private Function findFunction(TaskMonitor monitor) {
137  if (program == null || location == null) {
138  return null;
139  }
140  Address address = location.getAddress();
141  if (address == null) {
142  return null;
143  }
144 
145  if (monitor.isCancelled()) {
146  return null;
147  }
148 
149  Function function = program.getFunctionManager().getFunctionContaining(address);
150  if (function != null) {
151  return function;
152  }
153 
154  // couldn't find the function, make an undefined one
155  function = UndefinedFunction.findFunction(program, address, monitor);
156  if (function != null) {
157  // Make sure there isn't a real function at the location found
158  // function may not contain currentAddress in its body.
159  // This will cause provider to re-decompile when
160  // clicking around the currentAddress :(
161  Function realFunction =
162  program.getFunctionManager().getFunctionAt(function.getEntryPoint());
163  if (realFunction != null) {
164  return realFunction;
165  }
166  }
167 
168  return function;
169  }

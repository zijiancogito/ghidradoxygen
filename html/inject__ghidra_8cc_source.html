19 
20 {
21  s << "<context>\n";
22  baseaddr.saveXml(s);
23  calladdr.saveXml(s);

47 void InjectPayloadGhidra::inject(InjectContext &con,PcodeEmit &emit) const
48 
49 {
50  Document *doc;
51  ArchitectureGhidra *ghidra = (ArchitectureGhidra *)con.glb;
52  try {
53  doc = ghidra->getPcodeInject(name,type,con);
54  }
55  catch(JavaError &err) {
56  throw LowlevelError("Error getting pcode snippet: " + err.explain);
57  }
58  catch(XmlError &err) {
59  throw LowlevelError("Error in pcode snippet xml: "+err.explain);
60  }
61  if (doc == (Document *)0) {
62  throw LowlevelError("Could not retrieve pcode snippet: "+name);
63  }
64  const Element *el = doc->getRoot();
65  const List &list(el->getChildren());
66  List::const_iterator iter;
67  for(iter=list.begin();iter!=list.end();++iter)
68  emit.restoreXmlOp(*iter,ghidra->translate);
69  delete doc;
70 }

149 int4 PcodeInjectLibraryGhidra::allocateInject(const string &sourceName,const string &name,int4 type)
150 
151 {
152  int4 injectid = injection.size();
153  InjectPayload *payload;
154  switch(type) {
155  case InjectPayload::CALLFIXUP_TYPE:
156  payload = new InjectCallfixupGhidra(sourceName,name);
157  break;
158  case InjectPayload::CALLOTHERFIXUP_TYPE:
159  payload = new InjectCallotherGhidra(sourceName,name);
160  break;
161  case InjectPayload::CALLMECHANISM_TYPE:
162  payload = new InjectPayloadGhidra(sourceName,name,InjectPayload::CALLMECHANISM_TYPE);
163  break;
164  case InjectPayload::EXECUTABLEPCODE_TYPE:
165  payload = new ExecutablePcodeGhidra(contextCache.glb,sourceName,name);
166  break;
167  default:
168  throw LowlevelError("Bad injection type");
169  }
170  injection.push_back(payload);
171  return injectid;
172 }

174 void PcodeInjectLibraryGhidra::registerInject(int4 injectid)
175 
176 {
177  InjectPayload *payload = injection[injectid];
178  switch(payload->getType()) {
179  case InjectPayload::CALLFIXUP_TYPE:
180  registerCallFixup(payload->getName(), injectid);
181  break;
182  case InjectPayload::CALLOTHERFIXUP_TYPE:
183  registerCallOtherFixup(payload->getName(), injectid);
184  break;
185  case InjectPayload::CALLMECHANISM_TYPE:
186  registerCallMechanism(payload->getName(), injectid);
187  break;
188  case InjectPayload::EXECUTABLEPCODE_TYPE:
189  registerExeScript(payload->getName(), injectid);
190  break;
191  default:
192  throw LowlevelError("Unknown p-code inject type");
193  }
194 }

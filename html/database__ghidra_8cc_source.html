63 Scope *ScopeGhidra::reresolveScope(const vector<string> &path) const
64 
65 {
66  if (path.size()==1) return cache;
67  // Get pointer to ourselves (which is not const)
68  Scope *curscope = glb->symboltab->getGlobalScope();
69  int4 i;
70  for(i=1;i<path.size();++i) {
71  Scope *nextscope = curscope->resolveScope(path[i]);
72  if (nextscope == (Scope *)0) break;
73  curscope = nextscope;
74  }
75  while(i != path.size()) {
76  curscope = createNewScope(path[i],curscope);
77  i += 1;
78  }
79  return curscope;
80 }

90 void ScopeGhidra::processHole(const Element *el) const
91 
92 {
93  Range range;
94  range.restoreXml(el,ghidra);
95  holes.insertRange(range.getSpace(),range.getFirst(),range.getLast());
96  uint4 flags = 0;
97  for(int4 i=0;i<el->getNumAttributes();++i) {
98  if ((el->getAttributeName(i)=="readonly")&&
99  xml_readbool(el->getAttributeValue(i)))
100  flags |= Varnode::readonly;
101  else if ((el->getAttributeName(i)=="volatile")&&
102  xml_readbool(el->getAttributeValue(i)))
103  flags |= Varnode::volatil;
104  }
105  if (flags != 0) {
106  ghidra->symboltab->setPropertyRange(flags,range);
107  cacheDirty = true;
108  }

117 Symbol *ScopeGhidra::dump2Cache(Document *doc) const
118 
119 {
120  const Element *el = doc->getRoot();
121  Symbol *sym = (Symbol *)0;
122 
123  if (el->getName() == "hole") {
124  processHole(el);
125  return sym;
126  }
127 
128  List::const_iterator iter = el->getChildren().begin();
129  // The first subnode must be scope information
130  el = *iter;
131  vector<string> path;
132  const List &list2(el->getChildren());
133  List::const_iterator iter2;
134  for(iter2=list2.begin();iter2!=list2.end();++iter2)
135  path.push_back( (*iter2)->getContent() );
136 
137  Scope *scope = reresolveScope(path);
138  ++iter; // The second part is a <mapsym>
139  el = *iter;
140  try {
141  sym = scope->addMapSym(el);
142  }
143  catch(RecovError &err) {
144  // Duplicate name error (when trying to create the new function's scope)
145  // Check for function object where the full size is not stored in the cache
146  // Entries for functions always start at the entry address
147  // of the function in order to deal with non-contiguous functions
148  // But ghidra will still return the function for queries at addresses in the
149  // interior of the function
150  const Element *symel = *el->getChildren().begin();
151  if (symel->getName() == "function") { // Make sure new record is for a function
152  const Element *baseAddrEl = *symel->getChildren().begin();
153  Address baseaddr = Address::restoreXml( baseAddrEl, glb ); // Decode address from record
154  vector<Symbol *> symList;
155  scope->queryByName(symel->getAttributeValue("name"),symList); // Lookup symbols with duplicate name
156  for(int4 i=0;i<symList.size();++i) {
157  FunctionSymbol *funcSym = dynamic_cast<FunctionSymbol *>(symList[i]);
158  if (funcSym != (FunctionSymbol *)0) { // If duplicate symbol is for function
159  if (funcSym->getFunction()->getAddress() == baseaddr) { // and the address matches
160  sym = funcSym; // use the old symbol
161  break;
162  }
163  }
164  }
165  }
166  if (sym == (Symbol *)0) {
167  ostringstream s;
168  s << err.explain << ": entry didn't cache";
169  throw LowlevelError(s.str());
170  }
171  }
172  if (sym != (Symbol *)0) {
173  SymbolEntry *entry = sym->getFirstWholeMap();
174  if (entry != (SymbolEntry *)0) {
175  if (scope != cache) { // We have a namespace cache
176  // With a global namespace, mark the address range as a "hole", so the same query won't
177  // go up again. With this limitation, a function can only refer to a single global symbol at a
178  // a given address
179  AddrSpace *spc = entry->getAddr().getSpace();
180  uintb first = entry->getAddr().getOffset();
181  uintb last = first+entry->getSize()-1;
182  holes.insertRange(spc,first,last);
183  }
184 // // Add a range to the namespace, so that -map_scope-
185 // // can pick up references to this symbol
186 // ghidra->symboltab->addRange(scope,spc,first,last);
187 
188 // The decompiler does not currently maintain properties itself
189 // So we need to cache any properties returned by the query
190  uint4 props = sym->getFlags() & (Varnode::readonly | Varnode::volatil);
191  if (props != 0) {
192  Range rng(entry->getAddr().getSpace(),entry->getFirst(),entry->getLast());
193  ghidra->symboltab->setPropertyRange(props,rng);
194  cacheDirty = true;
195  }
196 // uint4 flags = ghidra->symboltab->getProperty(entry->getAddr());
197 // if (flags != 0)
198 // scope->setAttribute(sym,flags);
199  }
200  }
201  return sym;
202 }

210 Symbol *ScopeGhidra::removeQuery(const Address &addr) const
211 
212 {
213  Document *doc;
214  Symbol *sym = (Symbol *)0;
215 
216  // Don't send up queries on constants or uniques
217  // if (addr.getSpace()->getType() != IPTR_PROCESSOR) return sym;
218  // if (!cache->inScope(addr,1,Address())) return (Symbol *)0;
219 
220  // There is an efficiency/functionality trade-off problem here.
221  // We don't want to send every address query up to Ghidra
222  // as this is inefficient. Previously we only sent up if the
223  // the address was in the discovery scope, but there is the
224  // possibility that there is a global symbol that is not in
225  // the discovery scope. Now we send up if there is ANY
226  // part of the space the address is in that is part of the
227  // discovery scope.
228  if (addr.getSpace()->getIndex() >= spacerange.size())
229  return (Symbol *)0;
230  if (spacerange[addr.getSpace()->getIndex()] == 0)
231  return (Symbol *)0;
232 
233  // Have we queried this address before
234  if (holes.inRange(addr,1)) return (Symbol *)0;
235  doc = ghidra->getMappedSymbolsXML(addr); // Query GHIDRA about this address
236  if (doc != (Document *)0) {
237  sym = dump2Cache(doc); // Add it to the cache
238  delete doc;
239  }
240  return sym;
241 }

264 SymbolEntry *ScopeGhidra::findAddr(const Address &addr,
265  const Address &usepoint) const
266 {
267  SymbolEntry *entry;
268  entry = cache->findAddr(addr,usepoint);
269  if (entry == (SymbolEntry *)0) { // Didn't find symbol
270  entry = cache->findContainer(addr,1,Address());
271  if (entry != (SymbolEntry *)0)
272  return (SymbolEntry *)0; // Address is already queried, but symbol doesn't start at our address
273  Symbol *sym = removeQuery(addr); // Query server
274  if (sym != (Symbol *)0)
275  entry = sym->getMapEntry(addr);
276  // entry may be null for certain queries, ghidra may return symbol of size <8 with
277  // address equal to START of function, even though the query was for an address INTERNAL to the function
278  }
279  if ((entry != (SymbolEntry *)0)&&(entry->getAddr()==addr))
280  return entry;
281  return (SymbolEntry *)0;
282 }

284 SymbolEntry *ScopeGhidra::findContainer(const Address &addr,int4 size,
285  const Address &usepoint) const
286 {
287  SymbolEntry *entry;
288  entry = cache->findClosestFit(addr,size,usepoint);
289  if (entry == (SymbolEntry *)0) {
290  Symbol *sym = removeQuery(addr);
291  if (sym != (Symbol *)0)
292  entry = sym->getMapEntry(addr);
293  // entry may be null for certain queries, ghidra may return symbol of size <8 with
294  // address equal to START of function, even though the query was for an address INTERNAL to the function
295  }
296  if (entry != (SymbolEntry *)0) {
297  // Entry contains addr, does it contain addr+size
298  uintb last = entry->getAddr().getOffset() + entry->getSize() -1;
299  if (last >= addr.getOffset() + size-1)
300  return entry;
301  }
302  return (SymbolEntry *)0;

358 Funcdata *ScopeGhidra::resolveExternalRefFunction(ExternRefSymbol *sym) const
359 
360 {
361  Funcdata *fd = (Funcdata *)0;
362  const Scope *basescope = ghidra->symboltab->mapScope(this,sym->getRefAddr(),Address());
363  // Truncate search at this scope, we don't want
364  // the usual remote_query if the function isn't in cache
365  // this won't recover external functions, but will just
366  // return the externalref symbol again
367  stackFunction(basescope,this,sym->getRefAddr(),&fd);
368  if (fd == (Funcdata *)0)
369  fd = cache->findFunction(sym->getRefAddr());
370  if (fd == (Funcdata *)0) {
371  // If the function isn't in cache, we use the special
372  // getExternalRefXML interface to recover the external function
373  Document *doc;
374  SymbolEntry *entry = sym->getFirstWholeMap();
375  doc = ghidra->getExternalRefXML(entry->getAddr());
376  if (doc != (Document *)0) {
377  FunctionSymbol *sym;
378  // Make sure referenced function is cached
379  sym = dynamic_cast<FunctionSymbol *>(dump2Cache(doc));
380  delete doc;
381  if (sym != (FunctionSymbol *)0)
382  fd = sym->getFunction();
383  }
384  }
385  return fd;
386 }

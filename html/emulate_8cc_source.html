141 void Emulate::executeCurrentOp(void)
142 
143 {
144  if (currentBehave == (OpBehavior *)0) { // Presumably a NO-OP
145  fallthruOp();
146  return;
147  }
148  if (currentBehave->isSpecial()) {
149  switch(currentBehave->getOpcode()) {
150  case CPUI_LOAD:
151  executeLoad();
152  fallthruOp();
153  break;
154  case CPUI_STORE:
155  executeStore();
156  fallthruOp();
157  break;
158  case CPUI_BRANCH:
159  executeBranch();
160  break;
161  case CPUI_CBRANCH:
162  if (executeCbranch())
163  executeBranch();
164  else
165  fallthruOp();
166  break;
167  case CPUI_BRANCHIND:
168  executeBranchind();
169  break;
170  case CPUI_CALL:
171  executeCall();
172  break;
173  case CPUI_CALLIND:
174  executeCallind();
175  break;
176  case CPUI_CALLOTHER:
177  executeCallother();
178  break;
179  case CPUI_RETURN:
180  executeBranchind();
181  break;
182  case CPUI_MULTIEQUAL:
183  executeMultiequal();
184  fallthruOp();
185  break;
186  case CPUI_INDIRECT:
187  executeIndirect();
188  fallthruOp();
189  break;
190  case CPUI_SEGMENTOP:
191  executeSegmentOp();
192  fallthruOp();
193  break;
194  case CPUI_CPOOLREF:
195  executeCpoolRef();
196  fallthruOp();
197  break;
198  case CPUI_NEW:
199  executeNew();
200  fallthruOp();
201  break;
202  default:
203  throw LowlevelError("Bad special op");
204  }
205  }
206  else if (currentBehave->isUnary()) { // Unary operation
207  executeUnary();
208  fallthruOp();
209  }
210  else { // Binary operation
211  executeBinary();
212  fallthruOp(); // All binary ops are fallthrus
213  }
214 }

389 void EmulatePcodeCache::fallthruOp(void)
390 
391 {
392  instruction_start = false;
393  current_op += 1;
394  if (current_op >= opcache.size()) {
395  current_address = current_address + instruction_length;
396  createInstruction(current_address);

402 void EmulatePcodeCache::executeBranch(void)
403 
404 {
405  const Address &destaddr( currentOp->getInput(0)->getAddr() );
406  if (destaddr.isConstant()) {
407  uintm id = destaddr.getOffset();
408  id = id + (uintm)current_op;
409  current_op = id;
410  if (current_op == opcache.size())
411  fallthruOp();
412  else if ((current_op < 0)||(current_op >= opcache.size()))
413  throw LowlevelError("Bad intra-instruction branch");
414  }
415  else
416  setExecuteAddress(destaddr);
417 }

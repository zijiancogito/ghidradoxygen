106 Varnode *ConditionMarker::findMatch(PcodeOp *op)
107 
108 {
109  PcodeOp *curop;
110  // FlowBlock *bl = op->getParent();
111  state = 0;
112  Varnode *curvn = op->getIn(1);
113  multion = false;
114  binon = false;
115 
116  matchflip = op->isBooleanFlip();
117 
118  for(;;) {
119  if (curvn->isMark()) return curvn;
120  bool popstate = true;
121  if (curvn->isWritten()) {
122  curop = curvn->getDef();
123  if (curop->code() == CPUI_BOOL_NEGATE) {
124  curvn = curop->getIn(0);
125  if (!binon) // Only flip if we haven't seen binop yet, as binops get compared directly
126  matchflip = !matchflip;
127  popstate = false;
128  }
129 // else if (curop->code() == CPUI_MULTIEQUAL) {
130 // if ((curop->getParent()==bl)&&(!multion)) {
131 // opstate[state] = curop;
132 // slotstate[state] = 0;
133 // flipstate[state] = matchflip;
134 // state += 1;
135 // curvn = curop->Input(0);
136 // multion = true;
137 // popstate = false;
138 // }
139 // }
140  else if (curop->isBoolOutput()&&(curop->getEvalType()==PcodeOp::binary)) {
141  if (!binon) {
142  opstate[state] = curop;
143  slotstate[state] = 0;
144  flipstate[state] = matchflip;
145  state += 1;
146  curvn = curop->getIn(0);
147  binon = true;
148  popstate = false;
149  }
150  }
151  }
152  if (popstate) {
153  while(state > 0) {
154  curop = opstate[state-1];
155  matchflip = flipstate[state-1];
156  slotstate[state-1] += 1;
157  if (slotstate[state-1] < curop->numInput()) {
158  curvn = curop->getIn(slotstate[state-1]);
159  break;
160  }
161  state -= 1;
162  if (opstate[state]->code() == CPUI_MULTIEQUAL)
163  multion = false;
164  else
165  binon = false;
166  }
167  if (state==0) break;
168  }
169  }
170  return (Varnode *)0;

193 bool ConditionMarker::varnodeComplement(Varnode *a,Varnode *b)
194 
195 {
196  if (a->isConstant() && b->isConstant()) {
197  uintb vala = a->getOffset();
198  uintb valb = b->getOffset();
199  if ((vala==0)&&(valb==1)) return true;
200  if ((vala==1)&&(valb==0)) return true;
201  return false;
202  }
203  PcodeOp *negop;
204  if (a->isWritten()) {
205  negop = a->getDef();
206  if (negop->code() == CPUI_BOOL_NEGATE)
207  if (negop->getIn(0) == b)
208  return true;
209  }
210  if (b->isWritten()) {
211  negop = b->getDef();
212  if (negop->code() == CPUI_BOOL_NEGATE)
213  if (negop->getIn(0) == a)
214  return true;
215  }
216  return false;

226 bool ConditionMarker::sameOpComplement(PcodeOp *bin1op,PcodeOp *bin2op)
227 
228 {
229  OpCode opcode = bin1op->code();
230  if ((opcode == CPUI_INT_SLESS)||(opcode==CPUI_INT_LESS)) {
231  // Basically we test for the scenario like: x < 9 8 < x
232  int4 constslot = 0;
233  if (bin1op->getIn(1)->isConstant())
234  constslot = 1;
235  if (!bin1op->getIn(constslot)->isConstant()) return false;
236  if (!bin2op->getIn(1-constslot)->isConstant()) return false;
237  if (!varnodeSame(bin1op->getIn(1-constslot),bin2op->getIn(constslot))) return false;
238  uintb val1 = bin1op->getIn(constslot)->getOffset();
239  uintb val2 = bin2op->getIn(1-constslot)->getOffset();
240  if (constslot!=0) {
241  uintb tmp = val2;
242  val2 = val1;
243  val1 = tmp;
244  }
245  if (val1 + 1 != val2) return false;
246  if ((val2 == 0)&&(opcode==CPUI_INT_LESS)) return false; // Corner case for unsigned
247  if (opcode==CPUI_INT_SLESS) { // Corner case for signed
248  int4 sz = bin1op->getIn(constslot)->getSize();
249  if (signbit_negative(val2,sz) && (!signbit_negative(val1,sz)))
250  return false;
251  }
252  return true;
253  }
254  return false;
255 }

262 bool ConditionMarker::andOrComplement(PcodeOp *bin1op,PcodeOp *bin2op)
263 
264 {
265  if (bin1op->code() == CPUI_BOOL_AND) {
266  if (bin2op->code() != CPUI_BOOL_OR) return false;
267  }
268  else if (bin1op->code() == CPUI_BOOL_OR) {
269  if (bin2op->code() != CPUI_BOOL_AND) return false;
270  }
271  else
272  return false;
273 
274  // Reaching here, one is AND and one is OR
275  if (varnodeComplement( bin1op->getIn(0), bin2op->getIn(0))) {
276  if (varnodeComplement( bin1op->getIn(1), bin2op->getIn(1)))
277  return true;
278  }
279  else if (varnodeComplement( bin1op->getIn(0), bin2op->getIn(1))) {
280  if (varnodeComplement( bin1op->getIn(1), bin2op->getIn(0)))
281  return true;
282  }
283  return false;

293 bool ConditionMarker::finalJudgement(Varnode *vn)
294 
295 {
296  if (initop->isBooleanFlip())
297  matchflip = !matchflip;
298  if ((vn == basevn)&&(!binon)) // No binary operation involved
299  return true;
300  if (boolvn != (Varnode *)0)
301  matchflip = !matchflip;
302  if ((vn == boolvn)&&(!binon)) // Negations involved
303  return true;
304  if ((binaryop == (PcodeOp *)0)||(!binon))
305  return false; // Conditions don't match
306 
307  // Both conditions used binary op
308  PcodeOp *binary2op = (PcodeOp *)0;
309  for(int4 i=0;i<state;++i) { // Find the binary op
310  binary2op = opstate[i];
311  if (binary2op->isBoolOutput()) break;
312  }
313  // Check if the binary ops are exactly the same
314  if (binaryop->code() == binary2op->code()) {
315  if (varnodeSame(binaryop->getIn(0),binary2op->getIn(0)) &&
316  varnodeSame(binaryop->getIn(1),binary2op->getIn(1)))
317  return true;
318  if (sameOpComplement(binaryop,binary2op)) {
319  matchflip = !matchflip;
320  return true;
321  }
322  return false;
323  }
324  // If not, check if the binary ops are complements of one another
325  matchflip = !matchflip;
326  if (andOrComplement(binaryop,binary2op))
327  return true;
328  int4 slot1 = 0;
329  int4 slot2 = 0;
330  bool reorder;
331  if (binaryop->code() != get_booleanflip(binary2op->code(),reorder))
332  return false;
333  if (reorder) slot2 = 1;
334  if (!varnodeSame(binaryop->getIn(slot1),binary2op->getIn(slot2)))
335  return false;
336  if (!varnodeSame(binaryop->getIn(1-slot1),binary2op->getIn(1-slot2)))
337  return false;
338  return true;
339 }

342 
343 {
344  setupInitOp(initop);
345  Varnode *matchvn = findMatch(op);
346  if (matchvn == (Varnode *)0) return false;
347  if (!finalJudgement(matchvn)) return false;
348

351  multislot = -1;

357  }
358  }

450 bool ConditionalExecution::testMultiRead(Varnode *vn,PcodeOp *op)
451 
452 {
453  if (op->getParent() == iblock) {
454  if (!directsplit) {
455  if (op->code() == CPUI_COPY) // The COPY is tested separately
456  return true; // If the COPY's output reads can be altered, then -vn- can be altered
457  return false;
458  }
459  }
460  if (op->code() == CPUI_RETURN) {
461  if ((op->numInput() < 2)||(op->getIn(1) != vn)) return false; // Only test for flow thru to return value
462  returnop.push_back(op); // mark that CPUI_RETURN needs special handling
463  }

472 bool ConditionalExecution::testOpRead(Varnode *vn,PcodeOp *op)
473 
474 {
475  if (op->getParent() == iblock) return true;
476  if ((op->code() == CPUI_RETURN)&&(!directsplit)) {
477  if ((op->numInput() < 2)||(op->getIn(1) != vn)) return false; // Only test for flow thru to return value
478  PcodeOp *copyop = vn->getDef();
479  if (copyop->code() == CPUI_COPY) {
480  // Ordinarily, if -vn- is produced by a COPY we want to return false here because the propagation
481  // hasn't had time to happen here. But if the flow is into a RETURN this can't propagate, so
482  // we allow this as a read that can be altered. (We have to move the COPY)
483  Varnode *invn = copyop->getIn(0);
484  if (!invn->isWritten()) return false;
485  PcodeOp *upop = invn->getDef();
486  if ((upop->getParent() == iblock)&&(upop->code() != CPUI_MULTIEQUAL))
487  return false;
488  returnop.push_back(op);
489  return true;
490  }
491  }
492  return false;
493 }

500 void ConditionalExecution::predefineDirectMulti(PcodeOp *op)
501 
502 {
503  PcodeOp *newop = fd->newOp(posta_block->sizeIn()+1,posta_block->getStart());
504  Varnode *outvn = op->getOut();
505  Varnode *newoutvn;
506  newoutvn = fd->newVarnodeOut(outvn->getSize(),outvn->getAddr(),newop);
507  fd->opSetOpcode(newop,CPUI_MULTIEQUAL);
508  Varnode *vn;
509  int4 inslot = iblock->getOutRevIndex(posta_outslot);
510  for(int4 i=0;i<posta_block->sizeIn();++i) {
511  if (i==inslot)
512  vn = op->getIn(1-camethruposta_slot);
513  else
514  vn = newoutvn;
515  fd->opSetInput(newop,vn,i);
516  }
517  fd->opSetInput(newop,op->getIn(camethruposta_slot),posta_block->sizeIn());
518  fd->opInsertBegin(newop,posta_block);
519 
520  // Cache this new data flow holder
521  replacement[posta_block->getIndex()] = newoutvn;

527 void ConditionalExecution::adjustDirectMulti(void)
528 
529 {
530  list<PcodeOp *>::const_iterator iter;
531  PcodeOp *op;
532  iter = posta_block->beginOp();
533  int4 inslot = iblock->getOutRevIndex(posta_outslot);
534  while(iter != posta_block->endOp()) {
535  op = *iter++;
536  if (op->code() != CPUI_MULTIEQUAL) continue;

587 Varnode *ConditionalExecution::getReplacementRead(PcodeOp *op,BlockBasic *bl)
588 
589 {
590  map<int4,Varnode *>::const_iterator iter;
591  iter = replacement.find(bl->getIndex());
592  if (iter != replacement.end())
593  return (*iter).second;
594  BlockBasic *curbl = bl;
595  // Flow must eventually come through iblock
596  while(curbl->getImmedDom() != iblock) {
597  curbl = (BlockBasic *)curbl->getImmedDom(); // Get immediate dominator
598  if (curbl == (FlowBlock *)0)
599  throw LowlevelError("Conditional execution: Could not find dominator");
600  }
601  iter = replacement.find(curbl->getIndex());
602  if (iter != replacement.end()) {
603  replacement[bl->getIndex()] = (*iter).second;
604  return (*iter).second;
605  }
606  Varnode *res;
607  if (curbl->sizeIn()==1) {
608  // Since dominator is iblock, In(0) must be iblock
609  // Figure what side of -iblock- we came through
610  int4 slot = (curbl->getInRevIndex(0) == posta_outslot) ? camethruposta_slot : 1-camethruposta_slot;
611  res = op->getIn(slot);
612  }
613  else
614  res = getNewMulti(op,curbl);
615  replacement[curbl->getIndex()] = res;
616  if (curbl != bl)
617  replacement[bl->getIndex()] = res;
618  return res;
619 }

624 void ConditionalExecution::doReplacement(PcodeOp *op)
625 
626 {
627  if (op->code() == CPUI_COPY) {
628  if (op->getOut()->hasNoDescend()) // Verify that this has been dealt with by fixReturnOp
629  return;
630  // It could be a COPY internal to iblock, we need to remove it like any other op
631  }
632  replacement.clear();
633  if (directsplit)
634  predefineDirectMulti(op);
635  Varnode *vn = op->getOut();
636  list<PcodeOp *>::const_iterator iter = vn->beginDescend();
637  while(iter != vn->endDescend()) {
638  PcodeOp *readop = *iter;
639  int4 slot = readop->getSlot(vn);
640  BlockBasic *bl = readop->getParent();
641  Varnode *rvn;
642  if (bl == iblock) {
643  if (directsplit)
644  fd->opSetInput(readop,op->getIn(1-camethruposta_slot),slot); // We know op is MULTIEQUAL
645  else
646  fd->opUnsetInput(readop,slot);
647  }
648  else {
649  if (readop->code() == CPUI_MULTIEQUAL) {
650  BlockBasic *inbl = (BlockBasic *)bl->getIn(slot);
651  if (inbl == iblock) {
652  int4 s = (bl->getInRevIndex(slot) == posta_outslot) ? camethruposta_slot : 1-camethruposta_slot;
653  rvn = op->getIn(s);
654  }
655  else
656  rvn = getReplacementRead(op,inbl);
657  }
658  else
659  rvn = getReplacementRead(op,bl);
660  fd->opSetInput(readop,rvn,slot);
661  }
662  // The last descendant is now gone
663  iter = vn->beginDescend();
664  }
665 }

691 bool ConditionalExecution::testRemovability(PcodeOp *op)
692 
693 {
694  list<PcodeOp *>::const_iterator iter;
695  PcodeOp *readop;
696  Varnode *vn;
697 
698  if (op->code() == CPUI_MULTIEQUAL) {
699  vn = op->getOut();
700  for(iter=vn->beginDescend();iter!=vn->endDescend();++iter) {
701  readop = *iter;
702  if (!testMultiRead(vn,readop))
703  return false;
704  }
705  }
706  else {
707  if (op->isFlowBreak() || op->isCall()) return false;
708  if ((op->code()==CPUI_LOAD)||(op->code()==CPUI_STORE))
709  return false;
710  if (op->code()==CPUI_INDIRECT) return false;
711 
712  vn = op->getOut();
713  if (vn != (Varnode *)0) {
714  bool hasnodescend = true;
715  for(iter=vn->beginDescend();iter!=vn->endDescend();++iter) {
716  readop = *iter;
717  if (!testOpRead(vn,readop))
718  return false;
719  hasnodescend = false;
720  }
721  if (hasnodescend && (!heritageyes[vn->getSpace()->getIndex()])) // Check if heritage is performed for this varnode's space
722  return false;
723  }
724  }
725  return true;
726 }

886 bool RuleOrPredicate::MultiPredicate::discoverZeroSlot(Varnode *vn)
887 
888 {

899  if (!zerovn->isConstant()) continue;

1031 int4 RuleOrPredicate::applyOp(PcodeOp *op,Funcdata &data)
1032 
1033 {
1034  MultiPredicate branch0;
1035  MultiPredicate branch1;
1036  bool test0 = branch0.discoverZeroSlot(op->getIn(0));
1037  bool test1 = branch1.discoverZeroSlot(op->getIn(1));
1038  if ((test0==false) && (test1==false)) return 0;
1039  if (!test0) // branch1 has MULTIEQUAL form, but branch0 does not
1040  return checkSingle(op->getIn(0),branch1,op,data);
1041  else if (!test1) // branch0 has MULTIEQUAL form, but branch1 does not
1042  return checkSingle(op->getIn(1),branch0,op,data);
1043  if (!branch0.discoverCbranch()) return 0;
1044  if (!branch1.discoverCbranch()) return 0;
1045  if (branch0.condBlock == branch1.condBlock) {
1046  if (branch0.zeroBlock == branch1.zeroBlock) return 0; // zero sets must be along different paths
1047  }
1048  else { // Make sure cbranches have shared condition and the different zero sets have complementary paths
1049  ConditionMarker condmarker;
1050  if (!condmarker.verifyCondition(branch0.cbranch,branch1.cbranch)) return 0;
1051  if (condmarker.getMultiSlot() != -1) return 0;
1052  branch0.discoverPathIsTrue();
1053  branch1.discoverPathIsTrue();
1054  bool finalBool = branch0.zeroPathIsTrue == branch1.zeroPathIsTrue;
1055  if (condmarker.getFlip())
1056  finalBool = !finalBool;
1057  if (finalBool) return 0; // One path hits both zero sets, they must be on different paths
1058  }
1059  int4 order = branch0.op->compareOrder(branch1.op);
1060  if (order == 0) return 0; // can this happen?
1061  BlockBasic *finalBlock;
1062  bool slot0SetsBranch0; // True if non-zero setting of branch0 flows throw slot0
1063  if (order < 0) { // branch1 happens after
1064  finalBlock = branch1.op->getParent();
1065  slot0SetsBranch0 = branch1.zeroSlot == 0;
1066  }
1067  else { // branch0 happens after
1068  finalBlock = branch0.op->getParent();
1069  slot0SetsBranch0 = branch0.zeroSlot == 1;
1070  }
1071  PcodeOp *newMulti = data.newOp(2,finalBlock->getStart());
1072  data.opSetOpcode(newMulti,CPUI_MULTIEQUAL);
1073  if (slot0SetsBranch0) {
1074  data.opSetInput(newMulti,branch0.otherVn,0);
1075  data.opSetInput(newMulti,branch1.otherVn,1);
1076  }
1077  else {
1078  data.opSetInput(newMulti,branch1.otherVn,0);
1079  data.opSetInput(newMulti,branch0.otherVn,1);
1080  }
1081  Varnode *newvn = data.newUniqueOut(branch0.otherVn->getSize(),newMulti);
1082  data.opInsertBegin(newMulti,finalBlock);
1083  data.opRemoveInput(op,1);
1084  data.opSetInput(op,newvn,0);
1085  data.opSetOpcode(op,CPUI_COPY);
1086  return 1;

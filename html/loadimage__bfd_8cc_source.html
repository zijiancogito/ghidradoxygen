98 asection *LoadImageBfd::findSection(uintb offset,uintb &secsize) const
99 
100 { // Return section containing offset, or closest greater section
101  asection *p;
102  uintb start,stop;
103 
104  for(p = thebfd->sections; p != (asection *)NULL; p = p->next) {
105  start = p->vma;
106  secsize = (p->size!=0) ? p->size : p->rawsize;
107  stop = start + secsize;
108  if ((offset>=start)&&(offset<stop))
109  return p;
110  }
111  asection *champ = (asection *)0;
112  for(p = thebfd->sections; p != (asection *)NULL; p = p->next) {
113  if (p->vma > offset) {
114  if (champ == (asection *)0)
115  champ = p;
116  else if (p->vma < champ->vma)
117  champ = p;
118  }
119  }
120  return champ;
121 }

123 void LoadImageBfd::loadFill(uint1 *ptr,int4 size,const Address &addr)
124 
125 {
126  asection *p;
127  uintb secsize;
128  uintb curaddr,offset;
129  bfd_size_type readsize;
130  int4 cursize;
131 
132  if (addr.getSpace() != spaceid)
133  throw DataUnavailError("Trying to get loadimage bytes from space: "+addr.getSpace()->getName());
134  curaddr = addr.getOffset();
135  if ((curaddr>=bufoffset)&&(curaddr+size<bufoffset+bufsize)) { // Requested bytes were previously buffered
136  uint1 *bufptr = buffer + (curaddr-bufoffset);
137  memcpy(ptr,bufptr,size);
138  return;
139  }
140  bufoffset = curaddr; // Load buffer with bytes from new address
141  offset = 0;
142  cursize = bufsize; // Read an entire buffer
143 
144  while(cursize>0) {
145  p = findSection(curaddr,secsize);
146  if (p == (asection *)0) {
147  if (offset==0) // Initial address not mapped
148  break;
149  memset(buffer+offset,0,cursize); // Fill out the rest of the buffer with 0
150  memcpy(ptr,buffer,size);
151  return;
152  }
153  if (p->vma > curaddr) { // No section matches
154  if (offset==0) // Initial address not mapped
155  break;
156  readsize = p->vma - curaddr;
157  if (readsize > cursize)
158  readsize = cursize;
159  memset(buffer+offset,0,readsize); // Fill in with zeroes to next section
160  }
161  else {
162  readsize = cursize;
163  if (curaddr+readsize>p->vma+secsize) // Adjust to biggest possible read
164  readsize = (bfd_size_type)(p->vma+secsize-curaddr);
165  bfd_get_section_contents(thebfd,p,buffer+offset,(file_ptr)(curaddr-p->vma),readsize);
166  }
167  offset += readsize;
168  cursize -= readsize;
169  curaddr += readsize;
170  }
171  if (cursize > 0) {
172  ostringstream errmsg;
173  errmsg << "Unable to load " << dec << cursize << " bytes at " << addr.getShortcut();
174  addr.printRaw(errmsg);
175  throw DataUnavailError(errmsg.str());
176  }
177  memcpy(ptr,buffer,size); // Copy requested bytes from the buffer
178 }

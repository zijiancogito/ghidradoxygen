112  public static Address getStorageAddress(ClangToken tokenAtCursor,
113  DecompilerController controller) {
114  Varnode vnode = tokenAtCursor.getVarnode();
115  Address storageAddress = null;
116  if (vnode != null) {
117  storageAddress = vnode.getAddress();
118  }
119  // op could be a PTRSUB, need to dig it out...
120  else if (tokenAtCursor instanceof ClangVariableToken) {
121  PcodeOp op = ((ClangVariableToken) tokenAtCursor).getPcodeOp();
122  if (op == null) {
123  return null;
124  }
125  if (op.getOpcode() == PcodeOp.PTRSUB) {
126  vnode = op.getInput(0);
127  if (vnode.isRegister()) {
128  AddressSpace stackspace =
129  controller.getProgram().getAddressFactory().getStackSpace();
130  if (stackspace != null) {
131  Address caddr = op.getInput(1).getAddress();
132  storageAddress = stackspace.getAddress(caddr.getOffset());
133  }
134  }
135  else {
136  Address caddr = op.getInput(1).getAddress();
137  storageAddress =
138  controller.getLocation().getAddress().getNewAddress(caddr.getOffset());
139  }
140  }
141  }
142  return storageAddress;
143  }

145  public static HighVariable forgeHighVariable(Address addr, DecompilerController controller) {
146  HighVariable res = null;
147  Program program = controller.getProgram();
148  HighFunction hfunc = controller.getDecompileData().getHighFunction();
149  if (addr.isStackAddress()) {
150  LocalSymbolMap lsym = hfunc.getLocalSymbolMap();
151  HighSymbol hsym = lsym.findLocal(addr, null);
152  if (hsym != null)
153  res = hsym.getHighVariable();
154  }
155  else {
156  Data data = program.getListing().getDataAt(addr);
157  if (data != null) {
158  DataType dt = data.getDataType();
159  try {
160  res =
161  new HighGlobal(data.getLabel(), dt, new Varnode(addr, dt.getLength()),
162  null, hfunc);
163  }
164  catch (InvalidInputException e) {
165  Msg.error(RenameVariableAction.class, e.getMessage());
166  }
167  }
168  }
169  return res;
170  }

172  @Override
173  public void actionPerformed(ActionContext context) {
174  // Note: we intentionally do this check here and not in isEnabledForContext() so
175  // that global events do not get triggered.
176  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
177  if (decompilerActionContext.isDecompiling()) {
178  Msg.showInfo(getClass(), context.getComponentProvider().getComponent(),
179  "Decompiler Action Blocked",
180  "You cannot perform Decompiler actions while the Decompiler is busy");
181  return;
182  }
183 
184  DecompilerPanel decompilerPanel = controller.getDecompilerPanel();
185  final ClangToken tokenAtCursor = decompilerPanel.getTokenAtCursor();
186  HighVariable variable = tokenAtCursor.getHighVariable();
187 
188  if (variable == null) {
189  if (tokenAtCursor instanceof ClangVariableToken) {
190  Address addr = getStorageAddress(tokenAtCursor, controller);
191  variable = forgeHighVariable(addr, controller);
192  }
193  }
194 // TODO: Constant equates do not work properly with decompiler
195 // if (variable instanceof HighConstant) {
196 // nameTask =
197 // new RenameConstantTask(tool, tokenAtCursor.getText(), (HighConstant) variable,
198 // controller.getProgram());
199 // }
200 // else
201  if (variable instanceof HighLocal) {
202  nameTask =
203  new RenameVariableTask(tool, variable.getName(), controller.getHighFunction(),
204  variable, tokenAtCursor.getVarnode(), SourceType.USER_DEFINED);
205  }
206  else if (variable instanceof HighGlobal) {
207  Address addr = variable.getRepresentative().getAddress();
208  if (addr == null || !addr.isMemoryAddress()) {
209  Msg.showError(this, tool.getToolFrame(), "Rename Failed",
210  "Memory storage not found for global variable");
211  return;
212  }
213  nameTask =
214  new RenameGlobalVariableTask(tool, tokenAtCursor.getText(), addr,
215  controller.getProgram());
216  }
217  else if (tokenAtCursor instanceof ClangFieldToken) {
218  Structure dt = getStructDataType(tokenAtCursor);
219  if (dt == null) {
220  Msg.showError(this, tool.getToolFrame(), "Rename Failed",
221  "Could not find structure datatype");
222  return;
223  }
224  int offset = getDataTypeOffset(tokenAtCursor);
225  if (offset < 0 || offset >= dt.getLength()) {
226  Msg.showError(this, tool.getToolFrame(), "Rename Failed",
227  "Could not resolve field within structure");
228  return;
229  }
230  nameTask = new RenameStructureFieldTask(tool, tokenAtCursor.getText(), dt, offset);
231  }
232  else {
233  Msg.showError(this, tool.getToolFrame(), "Rename Failed",
234  "Selected variable does not support renaming");
235  return;
236  }
237 
238  boolean dialogres = nameTask.runDialog();
239  if (dialogres) {
240  Program program = controller.getProgram();
241  int transaction = program.startTransaction(nameTask.getTransactionName());
242  boolean commit = false;
243  try {
244  nameTask.commit();
245  commit = true;
246  }
247  catch (DuplicateNameException e) {
248  Msg.showError(this, tool.getToolFrame(), "Rename Failed", e.getMessage());
249  }
250  catch (InvalidInputException e) {
251  Msg.showError(this, tool.getToolFrame(), "Rename Failed", e.getMessage());
252  }
253  finally {
254  program.endTransaction(transaction, commit);
255  }
256  }
257  }

264  public static Structure getStructDataType(ClangToken tok) {
265  // We already know tok is a ClangFieldToken
266  ClangFieldToken fieldtok = (ClangFieldToken) tok;
267  DataType dt = fieldtok.getDataType();
268  if (dt == null)
269  return null;
270  if (dt instanceof TypeDef)
271  dt = ((TypeDef) dt).getBaseDataType();
272  if (dt instanceof Structure)
273  return (Structure) dt;
274  return null;
275  }

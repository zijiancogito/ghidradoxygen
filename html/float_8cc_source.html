80 double FloatFormat::createFloat(bool sign,uintb signif,int4 exp)
81 
82 {
83  signif >>= 1; // Throw away 1 bit of precision we will
84  // lose anyway, to make sure highbit is 0
85  int4 precis = 8*sizeof(uintb) - 1; // fullword - 1 we threw away
86  double res = (double)signif;
87  int4 expchange = exp - precis + 1; // change in exponent is precis
88  // -1 integer bit
89  res = ldexp(res,expchange);
90  if (sign)
91  res = res * -1.0;
92  return res;
93 }

102 FloatFormat::floatclass FloatFormat::extractExpSig(double x,bool *sgn,uintb *signif,int4 *exp)
103 
104 {
105  int4 e;
106 
107  *sgn = (signbit(x) != 0);
108  if (x == 0.0) return zero;
109  if (isinf(x)!=0) return infinity;
110  if (isnan(x)!=0) return nan;
111  if (*sgn)
112  x = -x;
113  double norm = frexp(x,&e); // norm is between 1/2 and 1
114  norm = ldexp(norm,8*sizeof(uintb)-1); // norm between 2^62 and 2^63
115 
116  *signif = (uintb)norm; // Convert to normalized integer
117  *signif <<= 1;
118 
119  e -= 1; // Consider normalization between 1 and 2
120  *exp = e;
121  return normalized;
122 }

233 double FloatFormat::getHostFloat(uintb encoding,floatclass *type) const
234 
235 {
236  bool sgn = extractSign(encoding);
237  uintb frac = extractFractionalCode(encoding);
238  int4 exp = extractExponentCode(encoding);
239  bool normal = true;
240 
241  if (exp == 0) {
242  if ( frac == 0 ) { // Floating point zero
243  *type = zero;
244  // FIXME: add on sign-bit for +0 or -0 allowed by standard
245  return sgn ? -0.0 : +0.0;
246  }
247  *type = denormalized;
248  // Number is denormalized
249  normal = false;
250  }
251  else if (exp == maxexponent) {
252  if ( frac == 0 ) { // Floating point infinity
253  *type = infinity;
254  // FIXME: add on sign-bit for +inf or -inf allowed by standard
255  return sgn ? -INFINITY : +INFINITY;
256  }
257  *type = nan;
258  // encoding is "Not a Number" NaN
259  return sgn ? -NAN : +NAN; // Sign is usually ignored
260  }
261  else
262  *type = normalized;
263 
264  // Get "true" exponent and fractional
265  exp -= bias;
266  if (normal && jbitimplied) {
267  frac >>= 1; // Make room for 1 jbit
268  uintb highbit = 1;
269  highbit <<= 8*sizeof(uintb)-1;
270  frac |= highbit; // Stick bit in at top
271  }
272  return createFloat(sgn,frac,exp);
273 }

277 uintb FloatFormat::getEncoding(double host) const
278 
279 {
280  floatclass type;
281  bool sgn;
282  uintb signif;
283  int4 exp;
284 
285  type = extractExpSig(host,&sgn,&signif,&exp);
286  if (type == zero)
287  return getZeroEncoding(sgn);
288  else if (type == infinity)
289  return getInfinityEncoding(sgn);
290  else if (type == nan)
291  return getNaNEncoding(sgn);
292 
293  // convert exponent and fractional to their encodings
294  exp += bias;
295  if (exp < 0) // Exponent is too small to represent
296  return getZeroEncoding(sgn);
297  if (exp > maxexponent) // Exponent is too big to represent
298  return getInfinityEncoding(sgn);
299  if (jbitimplied && (exp !=0))
300  signif <<= 1; // Cut off top bit (which should be 1)
301 
302  uintb res = 0;
303  res = setFractionalCode(res,signif);
304  res = setExponentCode(res,(uintb)exp);
305  return setSign(res,sgn);
306 }

311 uintb FloatFormat::convertEncoding(uintb encoding,const FloatFormat *formin) const
312 
313 {
314  bool sgn = formin->extractSign(encoding);
315  uintb frac = formin->extractFractionalCode(encoding);
316  int4 exp = formin->extractExponentCode(encoding);
317 
318  if (exp == formin->maxexponent) { // NaN or INFINITY encoding
319  exp = maxexponent;
320  }
321  else {
322  exp -= formin->bias;
323  exp += bias;
324  if (exp < 0)
325  return getZeroEncoding(sgn);
326  if (exp > maxexponent)
327  return getInfinityEncoding(sgn);
328  }
329  if (jbitimplied && !formin->jbitimplied)
330  frac <<= 1; // Cut off top bit (which should be 1)
331  else if (formin->jbitimplied && !jbitimplied) {
332  frac >>= 1; // Make room for 1 jbit
333  uintb highbit = 1;
334  highbit <<= 8*sizeof(uintb)-1;
335  frac |= highbit; // Stick bit in at top
336  }
337 
338  uintb res = 0;
339  res = setFractionalCode(res,frac);
340  res = setExponentCode(res,(uintb)exp);
341  return setSign(res,sgn);
342 }

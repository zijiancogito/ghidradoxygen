119 
120 {
121  for(;;) {
122  typestack.push_back(ct);
123  if (ct->getName().size() != 0) // This can be a base type
124  break;
125  if (ct->getMetatype()==TYPE_PTR)
126  ct = ((TypePointer *)ct)->getPtrTo();
127  else if (ct->getMetatype()==TYPE_ARRAY)
128  ct = ((TypeArray *)ct)->getBase();
129  else if (ct->getMetatype()==TYPE_CODE) {
130  const FuncProto *proto = ((TypeCode *)ct)->getPrototype();
131  if (proto != (const FuncProto *)0)
132  ct = proto->getOutputType();

144 void PrintC::pushPrototypeInputs(const FuncProto *proto)
145 
146 {
147  int4 sz = proto->numParams();
148 
149  if ((sz == 0)&&(!proto->isDotdotdot()))
150  pushAtom(Atom("void",syntax,EmitXml::keyword_color));
151  else {
152  for(int4 i=0;i<sz-1;++i)
153  pushOp(&comma,(const PcodeOp *)0); // Print a comma for each parameter (above 1)
154  if (proto->isDotdotdot()&&(sz!=0)) // Print comma for dotdotdot (if it is not by itself)
155  pushOp(&comma,(const PcodeOp *)0);
156  for(int4 i=0;i<sz;++i) {
157  ProtoParameter *param = proto->getParam(i);
158  pushTypeStart(param->getType(),true);
159  pushAtom(Atom("",blanktoken,EmitXml::no_color));
160  pushTypeEnd(param->getType());
161  }
162  if (proto->isDotdotdot()) {
163  if (sz != 0)
164  pushAtom(Atom("...",syntax,EmitXml::no_color));
165  else {
166  // In ANSI C, a prototype with empty parens means the parameters are unspecified (not void)
167  // In C++, empty parens mean void, we use the ANSI C convention
168  pushAtom(Atom("",blanktoken,EmitXml::no_color)); // An empty list of parameters
169  }
170  }
171  }

177 void PrintC::pushTypeStart(const Datatype *ct,bool noident)
178 
179 {
180  // Find the root type (the one with an identifier) and layout
181  // the stack of types, so we can access in reverse order
182  vector<const Datatype *> typestack;
183  buildTypeStack(ct,typestack);
184 
185  ct = typestack.back(); // The base type
186  OpToken *tok;
187 
188  if (noident && (typestack.size()==1))
189  tok = &type_expr_nospace;
190  else
191  tok = &type_expr_space;
192 
193  if (ct->getName().size()==0) { // Check for anonymous type
194  // We could support a struct or enum declaration here
195  string name = genericTypeName(ct);
196  pushOp(tok,(const PcodeOp *)0);
197  pushAtom(Atom(name,typetoken,EmitXml::type_color,ct));
198  }
199  else {
200  pushOp(tok,(const PcodeOp *)0);
201  pushAtom(Atom(ct->getName(),typetoken,EmitXml::type_color,ct));
202  }
203  for(int4 i=typestack.size()-2;i>=0;--i) {
204  ct = typestack[i];
205  if (ct->getMetatype() == TYPE_PTR)
206  pushOp(&ptr_expr,(const PcodeOp *)0);
207  else if (ct->getMetatype() == TYPE_ARRAY)
208  pushOp(&array_expr,(const PcodeOp *)0);
209  else if (ct->getMetatype() == TYPE_CODE)
210  pushOp(&function_call,(const PcodeOp *)0);
211  else {
212  clear();
213  throw LowlevelError("Bad type expression");
214  }
215  }

226 void PrintC::pushTypeEnd(const Datatype *ct)
227 
228 {
229  pushMod();
230  setMod(force_dec);
231 
232  for(;;) {
233  if (ct->getName().size() != 0) // This is the base type
234  break;
235  if (ct->getMetatype()==TYPE_PTR)
236  ct = ((const TypePointer *)ct)->getPtrTo();
237  else if (ct->getMetatype()==TYPE_ARRAY) {
238  const TypeArray *ctarray = (const TypeArray *)ct;
239  ct = ctarray->getBase();
240  push_integer(ctarray->numElements(),4,false,
241  (const Varnode *)0,(const PcodeOp *)0);
242  }
243  else if (ct->getMetatype()==TYPE_CODE) {
244  const TypeCode *ctcode = (const TypeCode *)ct;
245  const FuncProto *proto = ctcode->getPrototype();
246  if (proto != (const FuncProto *)0) {
247  pushPrototypeInputs(proto);
248  ct = proto->getOutputType();
249  }
250  else
251  // An empty list of parameters
252  pushAtom(Atom("",blanktoken,EmitXml::no_color));
253  }
254  else
255  break; // Some other anonymous type
256  }
257 
258  popMod();

266 bool PrintC::checkArrayDeref(const Varnode *vn) const
267 
268 {
269  const PcodeOp *op;
270 
271  if (!vn->isImplied()) return false;
272  if (!vn->isWritten()) return false;
273  op = vn->getDef();
274  if (op->code()==CPUI_SEGMENTOP) {
275  vn = op->getIn(2);
276  if (!vn->isImplied()) return false;

279  }

377 void PrintC::opCbranch(const PcodeOp *op)
378 
379 {
380  // FIXME: This routine shouldn't emit directly
381  bool yesif = isSet(flat);
382  bool yesparen = !isSet(comma_separate);
383  bool booleanflip = op->isBooleanFlip();
384  uint4 m = mods;
385 
386  if (yesif) { // If not printing block structure
387  emit->tagOp("if",EmitXml::keyword_color,op);
388  emit->spaces(1);
389  if (op->isFallthruTrue()) { // and the fallthru is the true branch
390  booleanflip = !booleanflip; // print negation of condition
391  m |= falsebranch; // and print the false (non-fallthru) branch
392  }
393  }
394  int4 id;
395  if (yesparen)
396  id = emit->openParen('(');
397  else
398  id = emit->openGroup();
399  if (booleanflip) {
400  if (checkPrintNegation(op->getIn(1))) {
401  m |= PrintLanguage::negatetoken;
402  booleanflip = false;
403  }
404  }
405  if (booleanflip)
406  pushOp(&boolean_not,op);
407  pushVnImplied(op->getIn(1),op,m);
408  // Make sure stack is clear before emitting more
409  recurse();
410  if (yesparen)
411  emit->closeParen(')',id);
412  else
413  emit->closeGroup(id);
414 
415  if (yesif) {
416  emit->spaces(1);
417  emit->print("goto",EmitXml::keyword_color);
418  emit->spaces(1);
419  pushVnImplied(op->getIn(0),op,mods);
420  }
421 }

545 void PrintC::opReturn(const PcodeOp *op)
546 
547 {
548  string nm;
549  switch(op->getHaltType()) {
550  default: // The most common case, plain return
551  // FIXME: This routine shouldn't emit directly
552  emit->tagOp("return",EmitXml::keyword_color,op);
553  if (op->numInput()>1) {
554  emit->spaces(1);
555  pushVnImplied(op->getIn(1),op,mods);
556  }
557  return;
558  case PcodeOp::noreturn: // Previous instruction does not exit
559  case PcodeOp::halt: // Process halts
560  nm = "halt";
561  break;
562  case PcodeOp::badinstruction:
563  nm = "halt_baddata"; // CPU executes bad instruction
564  break;
565  case PcodeOp::unimplemented: // instruction is unimplemented
566  nm = "halt_unimplemented";
567  break;
568  case PcodeOp::missing: // Did not analyze this instruction
569  nm = "halt_missing";
570  break;
571  }
572  pushOp(&function_call,op);
573  pushAtom(Atom(nm,optoken,EmitXml::funcname_color,op));
574  pushAtom(Atom("",blanktoken,EmitXml::no_color));

728  }

959 void PrintC::push_integer(uintb val,int4 sz,bool sign,
960  const Varnode *vn,const PcodeOp *op)
961 {

971  if (pushEquate(val,sz,(EquateSymbol *)sym,vn,op))

993  }

996  }

1005  if (print_negsign)

1020  t << dec << val; // Just emit as decimal

1035  t << 'U'; // Force unsignedness explicitly

1600  ct = arrayof;
1601  succeeded = true;
1602  }

1605  castStrategy->isSubpieceCastEndian(outtype,ct,off,
1606  sym->getFirstWholeMap()->getAddr().getSpace()->isBigEndian())) {

1626  }

98 void PcodeCompile::force_size(VarnodeTpl *vt,const ConstTpl &size,const vector<OpTpl *> &ops)
99 
100 {
101  if ((vt->getSize().getType()!=ConstTpl::real)||(vt->getSize().getReal() != 0))
102  return; // Size already exists
103 
104  vt->setSize(size);
105  if (!vt->isLocalTemp()) return;
106  // If the variable is a local temporary
107  // The size may need to be propagated to the various
108  // uses of the variable
109  OpTpl *op;
110  VarnodeTpl *vn;
111 
112  for(int4 i=0;i<ops.size();++i) {
113  op = ops[i];
114  vn = op->getOut();
115  if ((vn!=(VarnodeTpl *)0)&&(vn->isLocalTemp())) {
116  if (vn->getOffset() == vt->getOffset()) {
117  if ((size.getType() == ConstTpl::real)&&(vn->getSize().getType() == ConstTpl::real)&&
118  (vn->getSize().getReal() != 0) && (vn->getSize().getReal() != size.getReal()))
119  throw SleighError("Localtemp size mismatch");
120  vn->setSize(size);
121  }
122  }
123  for(int4 j=0;j<op->numInput();++j) {
124  vn = op->getIn(j);
125  if (vn->isLocalTemp()&&(vn->getOffset()==vt->getOffset())) {
126  if ((size.getType() == ConstTpl::real)&&(vn->getSize().getType() == ConstTpl::real)&&
127  (vn->getSize().getReal() != 0) && (vn->getSize().getReal() != size.getReal()))
128  throw SleighError("Localtemp size mismatch");
129  vn->setSize(size);
130  }
131  }
132  }

135 void PcodeCompile::matchSize(int4 j,OpTpl *op,bool inputonly,const vector<OpTpl *> &ops)
136 
137 { // Find something to fill in zero size varnode
138  // j is the slot we are trying to fill (-1=output)
139  // Don't check output for non-zero if inputonly is true
140  VarnodeTpl *match = (VarnodeTpl *)0;
141  VarnodeTpl *vt;
142  int4 i,inputsize;
143 
144  vt = (j==-1) ? op->getOut() : op->getIn(j);
145  if (!inputonly) {
146  if (op->getOut() != (VarnodeTpl *)0)
147  if (!op->getOut()->isZeroSize())
148  match = op->getOut();
149  }
150  inputsize = op->numInput();
151  for(i=0;i<inputsize;++i) {
152  if (match != (VarnodeTpl *)0) break;
153  if (op->getIn(i)->isZeroSize()) continue;
154  match = op->getIn(i);
155  }
156  if (match != (VarnodeTpl *)0)
157  force_size(vt,match->getSize(),ops);
158 }

160 void PcodeCompile::fillinZero(OpTpl *op,const vector<OpTpl *> &ops)
161 
162 { // Try to get rid of zero size varnodes in op
163  // Right now this is written assuming operands for the constructor are
164  // are built before any other pcode in the constructor is generated
165 
166  int4 inputsize,i;
167 
168  switch(op->getOpcode()) {
169  case CPUI_COPY: // Instructions where all inputs and output are same size
170  case CPUI_INT_ADD:
171  case CPUI_INT_SUB:
172  case CPUI_INT_2COMP:
173  case CPUI_INT_NEGATE:
174  case CPUI_INT_XOR:
175  case CPUI_INT_AND:
176  case CPUI_INT_OR:
177  case CPUI_INT_MULT:
178  case CPUI_INT_DIV:
179  case CPUI_INT_SDIV:
180  case CPUI_INT_REM:
181  case CPUI_INT_SREM:
182  case CPUI_FLOAT_ADD:
183  case CPUI_FLOAT_DIV:
184  case CPUI_FLOAT_MULT:
185  case CPUI_FLOAT_SUB:
186  case CPUI_FLOAT_NEG:
187  case CPUI_FLOAT_ABS:
188  case CPUI_FLOAT_SQRT:
189  case CPUI_FLOAT_CEIL:
190  case CPUI_FLOAT_FLOOR:
191  case CPUI_FLOAT_ROUND:
192  if ((op->getOut()!=(VarnodeTpl *)0)&&(op->getOut()->isZeroSize()))
193  matchSize(-1,op,false,ops);
194  inputsize = op->numInput();
195  for(i=0;i<inputsize;++i)
196  if (op->getIn(i)->isZeroSize())
197  matchSize(i,op,false,ops);
198  break;
199  case CPUI_INT_EQUAL: // Instructions with bool output
200  case CPUI_INT_NOTEQUAL:
201  case CPUI_INT_SLESS:
202  case CPUI_INT_SLESSEQUAL:
203  case CPUI_INT_LESS:
204  case CPUI_INT_LESSEQUAL:
205  case CPUI_INT_CARRY:
206  case CPUI_INT_SCARRY:
207  case CPUI_INT_SBORROW:
208  case CPUI_FLOAT_EQUAL:
209  case CPUI_FLOAT_NOTEQUAL:
210  case CPUI_FLOAT_LESS:
211  case CPUI_FLOAT_LESSEQUAL:
212  case CPUI_FLOAT_NAN:
213  case CPUI_BOOL_NEGATE:
214  case CPUI_BOOL_XOR:
215  case CPUI_BOOL_AND:
216  case CPUI_BOOL_OR:
217  if (op->getOut()->isZeroSize())
218  force_size(op->getOut(),ConstTpl(ConstTpl::real,1),ops);
219  inputsize = op->numInput();
220  for(i=0;i<inputsize;++i)
221  if (op->getIn(i)->isZeroSize())
222  matchSize(i,op,true,ops);
223  break;
224  // The shift amount does not necessarily have to be the same size
225  // But if no size is specified, assume it is the same size
226  case CPUI_INT_LEFT:
227  case CPUI_INT_RIGHT:
228  case CPUI_INT_SRIGHT:
229  if (op->getOut()->isZeroSize()) {
230  if (!op->getIn(0)->isZeroSize())
231  force_size(op->getOut(),op->getIn(0)->getSize(),ops);
232  }
233  else if (op->getIn(0)->isZeroSize())
234  force_size(op->getIn(0),op->getOut()->getSize(),ops);
235  // fallthru to subpiece constant check
236  case CPUI_SUBPIECE:
237  if (op->getIn(1)->isZeroSize())
238  force_size(op->getIn(1),ConstTpl(ConstTpl::real,4),ops);
239  break;
240  case CPUI_CPOOLREF:
241  if (op->getOut()->isZeroSize() && (!op->getIn(0)->isZeroSize()))
242  force_size(op->getOut(),op->getIn(0)->getSize(),ops);
243  if (op->getIn(0)->isZeroSize() && (!op->getOut()->isZeroSize()))
244  force_size(op->getIn(0),op->getOut()->getSize(),ops);
245  for(i=1;i<op->numInput();++i) {
246  if (op->getIn(i)->isZeroSize())
247  force_size(op->getIn(i),ConstTpl(ConstTpl::real,sizeof(uintb)),ops);
248  }
249  break;
250  default:
251  break;
252  }
253 }

255 bool PcodeCompile::propagateSize(ConstructTpl *ct)
256 
257 { // Fill in size for varnodes with size 0
258  // Return first OpTpl with a size 0 varnode
259  // that cannot be filled in or NULL otherwise
260  vector<OpTpl *> zerovec,zerovec2;
261  vector<OpTpl *>::const_iterator iter;
262  int4 lastsize;
263 
264  for(iter=ct->getOpvec().begin();iter!=ct->getOpvec().end();++iter)
265  if ((*iter)->isZeroSize()) {
266  fillinZero(*iter,ct->getOpvec());
267  if ((*iter)->isZeroSize())
268  zerovec.push_back(*iter);
269  }
270  lastsize = zerovec.size() + 1;
271  while( zerovec.size() < lastsize ) {
272  lastsize = zerovec.size();
273  zerovec2.clear();
274  for(iter=zerovec.begin();iter!=zerovec.end();++iter) {
275  fillinZero(*iter,ct->getOpvec());
276  if ((*iter)->isZeroSize())
277  zerovec2.push_back( *iter );
278  }
279  zerovec = zerovec2;
280  }
281  if ( lastsize != 0 ) return false;
282  return true;
283 }

467  // The first varnode input to the load is a constant reference to the AddrSpace being loaded

563 VarnodeTpl *PcodeCompile::buildTruncatedVarnode(VarnodeTpl *basevn,uint4 bitoffset,uint4 numbits)
564 
565 { // Build a truncated form -basevn- that matches the bitrange [ -bitoffset-, -numbits- ] if possible
566  // using just ConstTpl mechanics, otherwise return null
567  uint4 byteoffset = bitoffset / 8; // Convert to byte units
568  uint4 numbytes = numbits / 8;
569  uintb fullsz = 0;
570  if (basevn->getSize().getType() == ConstTpl::real) {
571  // If we know the size of base, make sure the bit range is in bounds
572  fullsz = basevn->getSize().getReal();
573  if (fullsz == 0) return (VarnodeTpl *)0;
574  if (byteoffset + numbytes > fullsz)
575  throw SleighError("Requested bit range out of bounds");
576  }
577 
578  if ((bitoffset % 8) != 0) return (VarnodeTpl *)0;
579  if ((numbits % 8) != 0) return (VarnodeTpl *)0;
580 
581  if (basevn->getSpace().isUniqueSpace()) // Do we really want to prevent truncated uniques??
582  return (VarnodeTpl *)0;
583 
584  ConstTpl::const_type offset_type = basevn->getOffset().getType();
585  if ((offset_type != ConstTpl::real)&&(offset_type != ConstTpl::handle))
586  return (VarnodeTpl *)0;
587

589  if (offset_type == ConstTpl::handle) {
590  // We put in the correct adjustment to offset assuming things are little endian
591  // We defer the correct big endian calculation until after the consistency check
592  // because we need to know the subtable export sizes
593  specialoff = ConstTpl(ConstTpl::handle,basevn->getOffset().getHandleIndex(),
594  ConstTpl::v_offset_plus,byteoffset);
595  }
596  else {
597  if (basevn->getSize().getType() != ConstTpl::real)
598  throw SleighError("Could not construct requested bit range");
599  uintb plus;
600  if (defaultspace->isBigEndian())
601  plus = fullsz - (byteoffset + numbytes);
602  else
603  plus = byteoffset;
604  specialoff = ConstTpl(ConstTpl::real,basevn->getOffset().getReal() + plus);
605  }
606  VarnodeTpl *res = new VarnodeTpl(basevn->getSpace(),specialoff,ConstTpl(ConstTpl::real,numbytes));

610 vector<OpTpl *> *PcodeCompile::assignBitRange(VarnodeTpl *vn,uint4 bitoffset,uint4 numbits,ExprTree *rhs)
611 
612 { // Create an expression assigning the rhs to a bitrange within sym
613  string errmsg;
614  if (numbits == 0)
615  errmsg = "Size of bitrange is zero";
616  uint4 smallsize = (numbits+7)/8; // Size of input (output of rhs)
617  bool shiftneeded = (bitoffset != 0);
618  bool zextneeded = true;
619  uintb mask = (uintb)2;
620  mask = ~(((mask<<(numbits-1))-1) << bitoffset);
621 
622  if (vn->getSize().getType()==ConstTpl::real) {
623  // If we know the size of the bitranged varnode, we can
624  // do some immediate checks, and possibly simplify things
625  uint4 symsize = vn->getSize().getReal();
626  if (symsize > 0)
627  zextneeded = (symsize > smallsize);
628  symsize *= 8; // Convert to number of bits
629  if ((bitoffset>=symsize)||(bitoffset+numbits>symsize))
630  errmsg = "Assigned bitrange is bad";
631  else if ((bitoffset==0)&&(numbits==symsize))
632  errmsg = "Assigning to bitrange is superfluous";
633  }
634 
635  if (errmsg.size()>0) { // Was there an error condition
636  reportError(errmsg); // Report the error
637  delete vn; // Clean up
638  vector<OpTpl *> *resops = rhs->ops; // Passthru old expression
639  rhs->ops = (vector<OpTpl *> *)0;
640  delete rhs;
641  return resops;
642  }
643 
644  // We know what the size of the input has to be
645  force_size(rhs->outvn,ConstTpl(ConstTpl::real,smallsize),*rhs->ops);
646 
647  ExprTree *res;
648  VarnodeTpl *finalout = buildTruncatedVarnode(vn,bitoffset,numbits);
649  if (finalout != (VarnodeTpl *)0) {
650  res = createOpOutUnary(finalout,CPUI_COPY,rhs);
651  }
652  else {
653  if (bitoffset + numbits > 64)
654  errmsg = "Assigned bitrange extends past first 64 bits";
655  res = new ExprTree(vn);
656  appendOp(CPUI_INT_AND,res,mask,0);
657  if (zextneeded)
658  createOp(CPUI_INT_ZEXT,rhs);
659  if (shiftneeded)
660  appendOp(CPUI_INT_LEFT,rhs,bitoffset,4);
661 
662  finalout = new VarnodeTpl(*vn);
663  res = createOpOut(finalout,CPUI_INT_OR,res,rhs);
664  }
665  if (errmsg.size() > 0)
666  reportError(errmsg);
667  vector<OpTpl *> *resops = res->ops;
668  res->ops = (vector<OpTpl *> *)0;
669  delete res;
670  return resops;

673 ExprTree *PcodeCompile::createBitRange(SpecificSymbol *sym,uint4 bitoffset,uint4 numbits)
674 
675 { // Create an expression computing the indicated bitrange of sym
676  // The result is truncated to the smallest byte size that can
677  // contain the indicated number of bits. The result has the
678  // desired bits shifted all the way to the right
679  string errmsg;
680  if (numbits == 0)
681  errmsg = "Size of bitrange is zero";
682  VarnodeTpl *vn = sym->getVarnode();
683  uint4 finalsize = (numbits+7)/8; // Round up to neareast byte size
684  uint4 truncshift = 0;
685  bool maskneeded = ((numbits%8)!=0);
686  bool truncneeded = true;
687 
688  // Special case where we can set the size, without invoking
689  // a truncation operator
690  if ((errmsg.size()==0)&&(bitoffset==0)&&(!maskneeded)) {
691  if ((vn->getSpace().getType()==ConstTpl::handle)&&vn->isZeroSize()) {
692  vn->setSize(ConstTpl(ConstTpl::real,finalsize));
693  ExprTree *res = new ExprTree(vn);
694  // VarnodeTpl *cruft = buildTemporary();
695  // delete cruft;
696  return res;
697  }
698  }
699 
700  if (errmsg.size()==0) {
701  VarnodeTpl *truncvn = buildTruncatedVarnode(vn,bitoffset,numbits);
702  if (truncvn != (VarnodeTpl *)0) { // If we are able to construct a simple truncated varnode
703  ExprTree *res = new ExprTree(truncvn); // Return just the varnode as an expression
704  delete vn;
705  return res;
706  }
707  }
708 
709  if (vn->getSize().getType()==ConstTpl::real) {
710  // If we know the size of the input varnode, we can
711  // do some immediate checks, and possibly simplify things
712  uint4 insize = vn->getSize().getReal();
713  if (insize > 0) {
714  truncneeded = (finalsize < insize);
715  insize *= 8; // Convert to number of bits
716  if ((bitoffset >= insize)||(bitoffset+numbits > insize))
717  errmsg = "Bitrange is bad";
718  if (maskneeded && ((bitoffset+numbits)==insize))
719  maskneeded = false;
720  }
721  }
722 
723  uintb mask = (uintb)2;
724  mask = ((mask<<(numbits-1))-1);
725 
726  if (truncneeded && ((bitoffset % 8)==0)) {
727  truncshift = bitoffset/8;
728  bitoffset = 0;
729  }
730 
731  if ((bitoffset==0)&&(!truncneeded)&&(!maskneeded))
732  errmsg = "Superfluous bitrange";
733 
734  if (maskneeded && (finalsize > 8))
735  errmsg = "Illegal masked bitrange producing varnode larger than 64 bits: " + sym->getName();
736 
737  ExprTree *res = new ExprTree(vn);
738 
739  if (errmsg.size()>0) { // Check for error condition
740  reportError(errmsg);
741  return res;
742  }
743 
744  if (bitoffset !=0)
745  appendOp(CPUI_INT_RIGHT,res,bitoffset,4);
746  if (truncneeded)
747  appendOp(CPUI_SUBPIECE,res,truncshift,4);
748  if (maskneeded)
749  appendOp(CPUI_INT_AND,res,mask,finalsize);
750  force_size(res->outvn,ConstTpl(ConstTpl::real,finalsize),*res->ops);
751  return res;
752 }

110  private boolean loadFromCache(Program program, ProgramLocation location,
111  ViewerPosition viewerPosition) {
112  FunctionManager functionManager = program.getFunctionManager();
113  Function function = functionManager.getFunctionContaining(location.getAddress());
114 
115  if (function == null) { // cache can't handle null keys
116  return false;
117  }
118 
119  DecompileResults results = decompilerCache.getIfPresent(function);
120  if (results == null) {
121  return false;
122  }
123 
124  // cancel any pending decompile tasks, so that previous requests don't overwrite the latest request
125  decompilerMgr.cancelAll();
126  setDecompileData(
127  new DecompileData(program, function, location, results, null, null, viewerPosition));
128 
129  return true;
130  }

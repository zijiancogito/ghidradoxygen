81 int4 ParamEntry::justifiedContain(const Address &addr,int4 sz) const
82 
83 {
84  if (joinrec != (JoinRecord *)0) {
85  int4 res = 0;
86  for(int4 i=joinrec->numPieces()-1;i>=0;--i) { // Move from least significant to most
87  const VarnodeData &vdata(joinrec->getPiece(i));
88  int4 cur = vdata.getAddr().justifiedContain(vdata.size,addr,sz,false);
89  if (cur<0)
90  res += vdata.size; // We skipped this many less significant bytes
91  else {
92  return res + cur;
93  }
94  }
95  return -1; // Not contained at all
96  }
97  if (alignment==0) {
98  // Ordinary endian containment
99  Address entry(spaceid,addressbase);
100  return entry.justifiedContain(size,addr,sz,((flags&force_left_justify)!=0));
101  }
102  if (spaceid != addr.getSpace()) return -1;
103  if (addr.getOffset() < addressbase) return -1;
104  uintb endaddr = addr.getOffset() + sz - 1;
105  if (endaddr < addr.getOffset()) return -1; // Don't allow wrap around
106  if (endaddr > (addressbase+size-1)) return -1;
107  if (!isLeftJustified()) { // For right justified (big endian), endaddr must be aligned
108  int4 res = (int4)((endaddr+1) % alignment);
109  if (res==0) return 0;
110  return (alignment-res);
111  }
112  return (int4)(addr.getOffset() % alignment);

125 bool ParamEntry::getContainer(const Address &addr,int4 sz,VarnodeData &res) const
126 
127 {
128  Address endaddr = addr + (sz-1);
129  if (joinrec != (JoinRecord *)0) {
130  for(int4 i=joinrec->numPieces()-1;i>=0;--i) { // Move from least significant to most
131  const VarnodeData &vdata(joinrec->getPiece(i));
132  if ((addr.overlap(0,vdata.getAddr(),vdata.size) >=0)&&
133  (endaddr.overlap(0,vdata.getAddr(),vdata.size)>=0)) {
134  res = vdata;
135  return true;
136  }
137  }
138  return false; // Not contained at all
139  }
140  Address entry(spaceid,addressbase);
141  if (addr.overlap(0,entry,size)<0) return false;
142  if (endaddr.overlap(0,entry,size)<0) return false;
143  if (alignment==0) {
144  // Ordinary endian containment
145  res.space = spaceid;
146  res.offset = addressbase;
147  res.size = size;
148  return true;
149  }

151  res.space = spaceid;
152  res.offset = addr.getOffset() - al;

240 Address ParamEntry::getAddrBySlot(int4 &slotnum,int4 sz) const
241 
242 {
243  Address res; // Start with an invalid result
244  int4 spaceused;
245  if (sz < minsize) return res;
246  if (alignment == 0) { // If not an aligned entry (allowing multiple slots)
247  if (slotnum != 0) return res; // Can only allocate slot 0
248  if (sz > size) return res; // Check on maximum size
249  res = Address(spaceid,addressbase); // Get base address of the slot
250  spaceused = size;
251  if (((flags & smallsize_floatext)!=0)&&(sz != size)) { // Do we have an implied floating-point extension
252  AddrSpaceManager *manager = spaceid->getManager();
253  res = manager->constructFloatExtensionAddress(res,size,sz);
254  return res;
255  }
256  }
257  else {
258  int4 slotsused = sz / alignment; // How many slots does a -sz- byte object need
259  if ( (sz % alignment) != 0)
260  slotsused += 1;
261  if (slotnum + slotsused > numslots) // Check if there are enough slots left
262  return res;
263  spaceused = slotsused * alignment;
264  int4 index;
265  if (isReverseStack()) {
266  index = numslots;
267  index -= slotnum;
268  index -= slotsused;
269  }
270  else
271  index = slotnum;
272  res = Address(spaceid, addressbase + index * alignment);
273  slotnum += slotsused; // Inform caller of number of slots used
274  }
275  if (!isLeftJustified()) // Adjust for right justified (big endian)
276  res = res + (spaceused - sz);
277  return res;
278 }

286 
287 {
288  flags = 0;
289  type = TYPE_UNKNOWN;
290  size = minsize = -1; // Must be filled in
291  alignment = 0; // default
292  numslots = 1;

302  }

304  istringstream i2(el->getAttributeValue(i));

306  i2 >> alignment;
307  }
308  else if (attrname == "align") { // new style
309  istringstream i4(el->getAttributeValue(i));

315  i3.unsetf(ios::dec | ios::hex | ios::oct);
316  i3 >> size;
317  }

321  istringstream i5(el->getAttributeValue(i));

323  i5 >> group;
324  }

326  istringstream i6(el->getAttributeValue(i));
327  i6.unsetf(ios::dec | ios::hex | ios::oct);
328  i6 >> groupsize;
329  }

341  throw LowlevelError("Bad extension attribute");
342  }

345  }
346  if ((size==-1)||(minsize==-1))
347  throw LowlevelError("ParamEntry not fully specified");
348  if (alignment == size)
349  alignment = 0;
350  Address addr;
351  addr = Address::restoreXml( *el->getChildren().begin(),manage);
352  spaceid = addr.getSpace();
353  addressbase = addr.getOffset();
354  if (alignment != 0) {
355  if ((addressbase % alignment) != 0)
356  throw LowlevelError("Stack <pentry> address must match alignment");
357  numslots = size / alignment;
358  }
359  if (spaceid->isReverseJustified()) {
360  if (spaceid->isBigEndian())
361  flags |= force_left_justify;
362  else
363  throw LowlevelError("No support for right justification in little endian encoding");
364  }
365  if (!normalstack) {
366  flags |= reverse_stack;
367  if (alignment != 0) {
368  if ((size % alignment) != 0)
369  throw LowlevelError("For positive stack growth, <pentry> size must match alignment");
370  }
371  }
372  resolveJoin();
373 }

511 void ParamListStandard::buildTrialMap(ParamActive *active) const
512 
513 {
514  vector<const ParamEntry *> hitlist; // List of groups for which we have a representative
515  bool seenfloattrial = false;
516  bool seeninttrial = false;
517 
518  for(int4 i=0;i<active->getNumTrials();++i) {
519  ParamTrial &paramtrial(active->getTrial(i));
520  int4 entslot = findEntry(paramtrial.getAddress(),paramtrial.getSize());
521  // Note: if a trial is "definitely not used" but there is a matching entry,
522  // we still include it in the map
523  if (entslot == -1)
524  paramtrial.markNoUse();
525  else {
526  const ParamEntry *curentry = &(entry[entslot]);
527  paramtrial.setEntry( curentry, 0 ); // Keep track of entry recovered for this trial
528 
529  if (curentry->getType() == TYPE_FLOAT)
530  seenfloattrial = true;
531  else
532  seeninttrial = true;
533 
534  // Make sure we list that the entries group is marked
535  int4 grp = curentry->getGroup();
536  while(hitlist.size() <= grp)
537  hitlist.push_back((const ParamEntry *)0);
538  const ParamEntry *lastentry = hitlist[grp];
539  if (lastentry == (const ParamEntry *)0)
540  hitlist[grp] = curentry; // This is the first hit for this group
541  }
542  }
543 
544  // Created unreferenced (unref) ParamTrial for any group that we don't have a representive for
545  // if that group occurs before one where we do have a representative
546 
547  for(int4 i=0;i<hitlist.size();++i) {
548  const ParamEntry *curentry = hitlist[i];
549 
550  if (curentry == (const ParamEntry *)0) {
551  for(int4 j=0;j<entry.size();++j) {
552  curentry = &entry[j];
553  if (curentry->getGroup() == i) break; // Find first entry of the missing group
554  }
555  if ((!seenfloattrial)&&(curentry->getType()==TYPE_FLOAT))
556  continue; // Don't fill in unreferenced floats if we haven't seen any floats
557  if ((!seeninttrial)&&(curentry->getType()!=TYPE_FLOAT))
558  continue; // Don't fill in unreferenced int if all we have seen is floats
559  int4 sz = curentry->isExclusion() ? curentry->getSize() : curentry->getAlign();
560  int4 nextslot = 0;
561  Address addr = curentry->getAddrBySlot(nextslot,sz);
562  int4 trialpos = active->getNumTrials();
563  active->registerTrial(addr,sz);
564  ParamTrial &paramtrial(active->getTrial(trialpos));
565  paramtrial.markUnref();
566  paramtrial.setEntry(curentry,0);
567  }
568  else if (!curentry->isExclusion()) {
569  // For non-exclusion groups, we need to create a secondary hitlist to find holes within the group
570  vector<int4> slotlist;
571  for(int4 j=0;j<active->getNumTrials();++j) {
572  ParamTrial &paramtrial(active->getTrial(j));
573  if (paramtrial.getEntry() != curentry) continue;
574  int4 slot = curentry->getSlot(paramtrial.getAddress(),0) - curentry->getGroup();
575  int4 endslot = curentry->getSlot(paramtrial.getAddress(),paramtrial.getSize()-1) - curentry->getGroup();
576  if (endslot < slot) { // With reverse stacks, the ending address may be in an earlier slot
577  int4 tmp = slot;
578  slot = endslot;
579  endslot = tmp;
580  }
581 
582  while(slotlist.size() <= endslot)
583  slotlist.push_back(0);
584  while(slot<=endslot) {
585  slotlist[slot] = 1;
586  slot += 1;
587  }
588  }
589  for(int4 j=0;j<slotlist.size();++j) {
590  if (slotlist[j] == 0) {
591  int4 nextslot = j; // Make copy of j, so that getAddrBySlot can change it
592  Address addr = curentry->getAddrBySlot(nextslot,curentry->getAlign());
593  int4 trialpos = active->getNumTrials();
594  active->registerTrial(addr,curentry->getAlign());
595  ParamTrial &paramtrial(active->getTrial(trialpos));
596  paramtrial.markUnref();
597  paramtrial.setEntry(curentry,0);
598  }
599  }
600  }
601  }
602  active->sortTrials();
603 }

703 void ParamListStandard::forceInactiveChain(ParamActive *active,int4 maxchain,int4 start,int4 stop) const
704 
705 {
706  bool seenchain = false;
707  int4 chainlength = 0;
708  int4 max = -1;
709  for(int4 i=start;i<stop;++i) {
710  ParamTrial &trial(active->getTrial(i));
711  if (trial.getEntry() == (const ParamEntry *)0) continue; // Already know not used
712  if (!trial.isActive()) {
713  if (trial.isUnref()&&active->isRecoverSubcall()) {
714  // If there is no reference to the trial within the function, the only real possibility
715  // is that a register is an input to the calling function and it is being reused (immediately)
716  // to pass the input into the called function. This really can't happen on the stack because
717  // the stack relative caller offset and callee offset are different
718  if (trial.getAddress().getSpace()->getType() == IPTR_SPACEBASE) // So if the parameter is on the stack
719  seenchain = true; // Mark that we have already seen an inactive chain

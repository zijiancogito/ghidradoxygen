209  private void parseRawString(InputStream rawxml) {
210  if (rawxml == null) {
211  return;
212  }
213  XmlPullParser parser = null;
214  try {
215  try {
216  parser =
217  HighFunction.stringTree(
218  rawxml,
219  HighFunction.getErrorHandler(this, "decompiler results for function at " +
220  function.getEntryPoint()));
221  hfunc = null;
222  hparamid = null;
223  docroot = null;
224  parser.start("doc");
225  while(parser.peek().isStart()) {
226  XmlElement el = parser.peek();
227  if (el.getName().equals("function")) {
228  if (hfunc == null) {
229  hfunc = new HighFunction(function, language, compilerSpec, dtmanage, showNamespace);
230  hfunc.readXML(parser);
231  }
232  else { // TODO: This is an ugly kludge to get around duplicate XML tag names
233  docroot = ClangXML.buildClangTree(parser, hfunc);
234  if (docroot == null) {
235  errMsg = "Unable to parse C (xml)";
236  }
237  }
238  }
239  else if (el.getName().equals("parammeasures")) {
240  hparamid =
241  new HighParamID(function, language, compilerSpec, dtmanage,
242  showNamespace);
243  hparamid.readXML(parser);
244  }
245  else {
246  errMsg = "Unknown decompiler tag: "+el.getName();
247  return;
248  }
249  }
250  }
251  catch (PcodeXMLException e) { // Error while walking the DOM
252  errMsg = e.getMessage();
253  hfunc = null;
254  hparamid = null;
255  return;
256  }
257  catch (RuntimeException e) { // Exception from the raw parser
258  errMsg = e.getMessage();
259  hfunc = null;
260  hparamid = null;
261  return;
262  }
263  }
264  finally {
265  if (parser != null) {
266  parser.dispose();
267  }
268  }
269  }

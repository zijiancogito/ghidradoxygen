72 bool Merge::mergeTestRequired(HighVariable *high_out,HighVariable *high_in)
73 
74 {
75  if (high_in == high_out) return true; // Already merged
76 
77  if (high_in->isTypeLock()) // If types are locked
78  if (high_out->isTypeLock()) // dont merge unless
79  if (high_in->getType() != high_out->getType()) return false; // both types are the same
80 
81  if (high_out->isAddrTied()) { // Do not merge address tied input
82  if (high_in->isAddrTied()) {
83  if (high_in->getTiedVarnode()->getAddr() != high_out->getTiedVarnode()->getAddr())
84  return false; // with an address tied output of different address
85  }
86  }
87 
88  if (high_in->isInput()) {
89  // Input and persist must be different vars
90  // as persists inherently have their own input
91  if (high_out->isPersist()) return false;
92  // If we don't prevent inputs and addrtieds from
93  // being merged. Inputs can get merged with the
94  // internal parts of structures on the stack
95  if ((high_out->isAddrTied())&&(!high_in->isAddrTied())) return false;
96  }
97  else if (high_in->isExtraOut())
98  return false;
99  if (high_out->isInput()) {
100  if (high_in->isPersist()) return false;
101  if ((high_in->isAddrTied())&&(!high_out->isAddrTied())) return false;
102  }
103  else if (high_out->isExtraOut())
104  return false;
105 
106  return true;
107 }

116 bool Merge::mergeTestAdjacent(HighVariable *high_out,HighVariable *high_in)
117 
118 {
119  if (!mergeTestRequired(high_out,high_in)) return false;
120 
121  if (high_in->isNameLock() && high_out->isNameLock())
122  return false;

125  if (high_out->getType() != high_in->getType())
126  return false;

128  // as much as possible. See we don't do any speculative
129  // merges with them, UNLESS the illegal input is only
130  // used indirectly

335 void Merge::snipReads(Varnode *vn,list<PcodeOp *> &markedop)
336 
337 {
338  if (markedop.empty()) return;
339 
340  Varnode *uniq;
341  PcodeOp *copyop,*op;
342  BlockBasic *bl;
343  Address pc;
344  int4 slot;
345  PcodeOp *afterop;
346 
347  // Figure out where copy is inserted
348  if (vn->isInput()) {
349  bl = (BlockBasic *) data.getBasicBlocks().getBlock(0);
350  pc = bl->getStart();
351  afterop = (PcodeOp *)0;
352  }
353  else {
354  bl = vn->getDef()->getParent();
355  pc = vn->getDef()->getAddr();
356  if (vn->getDef()->code() == CPUI_INDIRECT) // snip must come after OP CAUSING EFFECT
357  // Not the indirect op itself

359  else
360  afterop = vn->getDef();
361  }
362  copyop = data.newOp(1,pc);
363  data.opSetOpcode(copyop,CPUI_COPY);
364  uniq = data.newUnique(vn->getSize(),vn->getType());
365  data.opSetOutput(copyop,uniq);
366  data.opSetInput(copyop,vn,0);
367  if (afterop == (PcodeOp *)0)
368  data.opInsertBegin(copyop,bl);
369  else
370  data.opInsertAfter(copyop,afterop);
371 
372  list<PcodeOp *>::iterator iter;
373  for(iter=markedop.begin();iter!=markedop.end();++iter) {
374  op = *iter;
375  for(slot=0;slot<op->numInput();++slot)
376  if (op->getIn(slot)==vn) break; // Find the correct slot
377  data.opSetInput(op,uniq,slot);

388 void Merge::eliminateIntersect(Varnode *vn,const vector<BlockVarnode> &blocksort)
389 
390 {
391  list<PcodeOp *> markedop;
392  list<PcodeOp *>::const_iterator oiter;
393  map<int4,CoverBlock>::const_iterator iter,enditer;
394  Varnode *vn2;
395  int4 boundtype;
396  bool insertop;
397 
398  for(oiter=vn->beginDescend();oiter!=vn->endDescend();++oiter) {
399  insertop = false;
400  Cover single;
401  single.addDefPoint(vn);
402  PcodeOp *op = *oiter;
403  single.addRefPoint(op,vn); // Build range for a single read
404  iter = single.begin();
405  enditer = single.end();
406  while(iter != enditer) {
407  int4 blocknum = (*iter).first;
408  ++iter;
409  int4 slot = BlockVarnode::findFront(blocknum,blocksort);
410  if (slot == -1) continue;
411  while(slot < blocksort.size()) {
412  if (blocksort[slot].getIndex() != blocknum)
413  break;
414  vn2 = blocksort[slot].getVarnode();
415  slot += 1;
416  if (vn2 == vn) continue;
417  boundtype = single.containVarnodeDef(vn2);
418  if (boundtype == 0) continue;
419  if (boundtype == 2) { // We have to resolve things defined at same place
420  if (vn2->getDef() == (PcodeOp *)0) {
421  if (vn->getDef() == (PcodeOp *)0) {
422  if (vn < vn2) continue; // Choose an arbitrary order if both are inputs
423  }
424  else
425  continue;
426  }
427  else {
428  if (vn->getDef() != (PcodeOp *)0) {
429  if (vn2->getDef()->getSeqNum().getOrder() < vn->getDef()->getSeqNum().getOrder())
430  continue;
431  }
432  }
433  }
434  else if (boundtype == 3) { // intersection on the tail of the range
435  // For most operations if the READ and WRITE happen on the same op, there is really no cover
436  // intersection because the READ happens before the op and the WRITE happens after, but
437  // if the WRITE is for an INDIRECT that is marking the READING (call) op, and the WRITE is to
438  // an address forced varnode, then because the write varnode must exist just before the op
439  // there really is an intersection.
440  if (!vn2->isAddrForce()) continue;
441  if (!vn2->isWritten()) continue;
442  PcodeOp *indop = vn2->getDef();
443  if (indop->code() != CPUI_INDIRECT) continue;
444  // The vn2 INDIRECT must be linked to the read op
445  if (op != PcodeOp::getOpFromConst(indop->getIn(1)->getAddr())) continue;
446  if (vn->copyShadow(indop->getIn(0))) continue; // If INDIRECT input shadows vn, don't consider as intersection
447  }
448  insertop = true;
449  break; // No need to continue iterating through varnodes in block
450  }
451  if (insertop) break; // No need to continue iterating through blocks
452  }
453  if (insertop)
454  markedop.push_back(op);
455  }
456  snipReads(vn,markedop);
457 }

583 void Merge::mergeOp(PcodeOp *op)
584 
585 {
586  vector<HighVariable *> testlist;
587  HighVariable *high_out;
588  int4 i,nexttrim,max;
589 
590  max = (op->code() == CPUI_INDIRECT) ? 1 : op->numInput();
591  high_out = op->getOut()->getHigh();
592  // First try to deal with non-cover related merge
593  // restrictions
594  for(i=0;i<max;++i) {
595  HighVariable *high_in = op->getIn(i)->getHigh();
596  if (!mergeTestRequired(high_out,high_in)) {
597  trimOpInput(op,i);
598  continue;
599  }
600  for(int4 j=0;j<i;++j)
601  if (!mergeTestRequired(op->getIn(j)->getHigh(),high_in)) {
602  trimOpInput(op,i);
603  break;
604  }
605  }
606  // Now test if a merge violates cover restrictions
607  mergeTest(high_out,testlist);
608  for(i=0;i<max;++i)
609  if (!mergeTest(op->getIn(i)->getHigh(),testlist)) break;
610 
611  if (i != max) { // If there are cover restrictions
612  nexttrim = 0;
613  while(nexttrim < max) {
614  trimOpInput(op,nexttrim); // Trim one of the branches
615  testlist.clear();
616  // Try the merge restriction test again
617  mergeTest(high_out,testlist);
618  for(i=0;i<max;++i)
619  if (!mergeTest(op->getIn(i)->getHigh(),testlist)) break;
620  if (i==max) break; // We successfully test merged everything
621  nexttrim += 1;
622  }
623  if (nexttrim == max) // One last trim we can try
624  trimOpOutput(op);
625  }
626 
627  for(i=0;i<max;++i) { // Try to merge everything for real now
628  if (!mergeTestRequired(op->getOut()->getHigh(),op->getIn(i)->getHigh()))
629  throw LowlevelError("Non-cover related merge restriction violated, despite trims");
630  if (!merge(op->getOut()->getHigh(),op->getIn(i)->getHigh(),false)) {
631  ostringstream errstr;
632  errstr << "Unable to force merge of op at " << op->getSeqNum();
633  throw LowlevelError(errstr.str());
634  }
635  }
636 }

700 void Merge::snipIndirect(PcodeOp *indop)
701 
702 {
703  PcodeOp *op = PcodeOp::getOpFromConst(indop->getIn(1)->getAddr()); // Indirect effect op
704  vector<Varnode *> problemvn;
705  list<PcodeOp *> correctable;
706  vector<int4> correctslot;
707  // Collect instances of output->high that are defined
708  // before (and right up to) op. These need to be snipped.
709  collectCovering(problemvn,indop->getOut()->getHigh(),op);
710  if (problemvn.empty()) return;
711  // Collect vn reads where the snip needs to be.
712  // If cover properly contains op, report an error.

714  // to intersect with indop->output, which it is merged with.
715  if (!collectCorrectable(problemvn,correctable,correctslot,op))
716  throw LowlevelError("Unable to force indirect merge");

721  PcodeOp *snipop,*insertop;
722 
723  // NOTE: the covers for any input to op which is
724  // an instance of the output high must
725  // all intersect so the varnodes must all be
726  // traceable via COPY to the same root
727  snipop = data.newOp(1,op->getAddr());
728  data.opSetOpcode(snipop,CPUI_COPY);

730  data.opSetOutput(snipop,snipvn);
731  data.opSetInput(snipop,refvn,0);
732  data.opInsertBefore(snipop,op);

747 void Merge::mergeIndirect(PcodeOp *indop)
748 
749 {
750  Varnode *outvn = indop->getOut();
751  Varnode *invn0 = indop->getIn(0);
752  if (!outvn->isAddrForce()) { // If the output is NOT address forced
753  mergeOp(indop); // We can merge in the same way as a MULTIEQUAL
754  return;
755  }
756 
757  if (mergeTestRequired(outvn->getHigh(),invn0->getHigh()))
758  if (merge(invn0->getHigh(),outvn->getHigh(),false)) return;
759  snipIndirect(indop); // If we cannot merge, the only thing that can go
760  // wrong with an input trim, is if the output of
761  // indop is involved in the input to the op causing
762  // the indirect effect. So fix this
763 
764  PcodeOp *newop;
765  Varnode *trimvn;

913 bool Merge::merge(HighVariable *high1,HighVariable *high2,bool isspeculative)
914 
915 {
916  if (high1 == high2) return true; // Already merged
917  if (intersection(high1,high2)) return false;
918 
919  // Translate any tests for high2 into tests for high1
920  vector<HighVariable *> yesinter; // Highs that high2 intersects
921  vector<HighVariable *> nointer; // Highs that high2 does not intersect
922  map<HighEdge,bool>::iterator iterfirst = highedgemap.lower_bound( HighEdge(high2,(HighVariable *)0) );
923  map<HighEdge,bool>::iterator iterlast = highedgemap.lower_bound( HighEdge(high2,(HighVariable *)~((uintp)0)) );
924  map<HighEdge,bool>::iterator iter;
925 
926  for(iter=iterfirst;iter!=iterlast;++iter) {
927  HighVariable *b = (*iter).first.b;
928  if (b == high1) continue;
929  if ((*iter).second) // Save all high2's intersections
930  yesinter.push_back(b); // as they are still valid for the merge
931  else {
932  nointer.push_back(b);
933  b->setMark(); // Mark that high2 did not intersect
934  }
935  }
936  // Do a purge of all high2's tests
937  if (iterfirst != iterlast) { // Delete all the high2 tests
938  --iterlast; // Move back 1 to prevent deleting under the iterator
939  for(iter=iterfirst;iter!=iterlast;++iter)
940  highedgemap.erase( HighEdge( (*iter).first.b, (*iter).first.a) );
941  highedgemap.erase( HighEdge( (*iter).first.b, (*iter).first.a) );
942  ++iterlast; // Restore original range (with possibly new open endpoint)
943 
944  highedgemap.erase(iterfirst,iterlast);
945  }
946 
947  iter = highedgemap.lower_bound( HighEdge(high1,(HighVariable *)0) );
948  while((iter!=highedgemap.end())&&((*iter).first.a == high1)) {
949  if (!(*iter).second) { // If test is intersection==false
950  if (!(*iter).first.b->isMark()) // and there was no test with high2
951  highedgemap.erase( iter++ ); // Delete the test
952  else
953  ++iter;
954  }
955  else // Keep any intersection==true tests
956  ++iter;
957  }

959  for(titer=nointer.begin();titer!=nointer.end();++titer)
960  (*titer)->clearMark();
961  // Reinsert high2's intersection==true tests for high1 now
962  for(titer=yesinter.begin();titer!=yesinter.end();++titer) {
963  highedgemap[ HighEdge(high1,*titer) ] = true;
964  highedgemap[ HighEdge(*titer,high1) ] = true;
965  }
966  high1->merge(high2,isspeculative); // Do the actual merge
967  high1->updateCover();
968

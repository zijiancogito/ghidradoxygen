61  class ByteChunk implements Comparable<ByteChunk> {
62  public Address addr;
63  public int min, max;
64  public byte[] val;
65 
66  public ByteChunk(Address ad, int off, byte[] v) {
67  addr = ad.getNewAddress(ad.getOffset() & ~15L);
68  val = new byte[16];
69  min = (int) ad.getOffset() & 15;
70  int len = v.length - off;
71  if (min + len >= 16)
72  len = 16 - min;
73  max = min + len;
74  for (int i = 0; i < 16; ++i)
75  val[i] = 0;
76  for (int i = 0; i < len; ++i)
77  val[min + i] = v[off + i];
78  }
79 
80  public void merge(ByteChunk op2) {
81  for (int i = op2.min; i < op2.max; ++i)
82  val[i] = op2.val[i];
83  if (op2.min < min)
84  min = op2.min;
85  if (op2.max > max)
86  max = op2.max;
87  }
88 
89  @Override
90  public int compareTo(ByteChunk op2) {
91  return addr.compareTo(op2.addr);
92  }
93  }

198  private void dumpBytes(OutputStream debugStream) throws IOException {
199  StringBuilder buf = new StringBuilder();
200  Iterator<ByteChunk> iter = byteset.iterator();
201  AddressSpace lastspace = null;
202  long lastoffset = 0;
203  boolean lastreadonly = false;
204  boolean tagstarted = false;
205  while (iter.hasNext()) {
206  ByteChunk chunk = iter.next();
207  AddressSpace space = chunk.addr.getAddressSpace();
208  boolean readval = isReadOnly(chunk.addr);
209  if (lastreadonly != readval) {
210  lastspace = null; // Force a break in chunk, so we can set new readonly value
211  lastreadonly = readval;
212  }
213 
214  if (tagstarted &&
215  ((chunk.min != 0) || (lastspace != space) || (lastoffset != chunk.addr.getOffset()))) {
216  buf.append("\n</bytechunk>\n");
217  tagstarted = false;
218  }
219  if (!tagstarted) {
220  buf.append("<bytechunk");
221  SpecXmlUtils.encodeStringAttribute(buf, "space", space.getPhysicalSpace().getName());
222  SpecXmlUtils.encodeUnsignedIntegerAttribute(buf, "offset", chunk.addr.getOffset() +
223  chunk.min);
224  if (lastreadonly)
225  SpecXmlUtils.encodeBooleanAttribute(buf, "readonly", lastreadonly);
226  buf.append(">\n");
227  tagstarted = true;
228  }
229  for (int i = 0; i < chunk.min; ++i)
230  buf.append(" "); // pad the hex display to 16 bytes
231  for (int i = chunk.min; i < chunk.max; ++i) {
232  int hi = (chunk.val[i] >> 4) & 0xf;
233  int lo = chunk.val[i] & 0xf;
234  if (hi > 9)
235  hi += 'a' - 10;
236  else
237  hi += '0';
238  if (lo > 9)
239  lo += 'a' - 10;
240  else
241  lo += '0';
242  buf.append((char) hi);
243  buf.append((char) lo);
244  }
245  buf.append('\n');
246  if (chunk.max != 16) {
247  buf.append("</bytechunk>\n");
248  tagstarted = false;
249  }
250  else {
251  lastoffset = chunk.addr.getOffset() + 16;
252  lastspace = space;
253  }
254  }
255  if (tagstarted)
256  buf.append("</bytechunk>\n");
257  debugStream.write(buf.toString().getBytes());
258  }

345  private void dumpPointsetContext(OutputStream debugStream) throws IOException {
346  ArrayList<ContextSymbol> ctxsymbols = getContextSymbols();
347  if (ctxsymbols == null)
348  return;
349  ContextCache ctxcache = new ContextCache();
350  ctxcache.registerVariable(contextRegister);
351  int[] buf = new int[ctxcache.getContextSize()];
352  int[] lastbuf = null;
353 
354  Iterator<Address> iter = contextchange.iterator();
355  while (iter.hasNext()) {
356  Address addr = iter.next();
357  ProgramProcessorContext procctx = new ProgramProcessorContext(progctx, addr);
358  ctxcache.getContext(procctx, buf);
359  StringBuilder stringBuf = new StringBuilder();
360  if (lastbuf != null) { // Check to make sure we don't have identical context data
361  int i;
362  for (i = 0; i < buf.length; ++i)
363  if (buf[i] != lastbuf[i])
364  break;
365  if (i == buf.length)
366  continue; // If all data is identical, then changepoint is not necessary
367  }
368  else
369  lastbuf = new int[buf.length];
370  for (int i = 0; i < buf.length; ++i)
371  lastbuf[i] = buf[i];
372 
373  stringBuf.append("<context_pointset");
374  Varnode.appendSpaceOffset(stringBuf, addr);
375  stringBuf.append(">\n");
376  for (int i = 0; i < ctxsymbols.size(); ++i) {
377  ContextSymbol sym = ctxsymbols.get(i);
378  int sbit = sym.getLow();
379  int ebit = sym.getHigh();
380  int word = sbit / (8 * 4);
381  int startbit = sbit - word * (8 * 4);
382  int endbit = ebit - word * (8 * 4);
383  int shift = (8 * 4) - endbit - 1;
384  int mask = -1 >>> (startbit + shift);
385  int val = (buf[word] >>> shift) & mask;
386  stringBuf.append(" <set");
387  SpecXmlUtils.encodeStringAttribute(stringBuf, "name", sym.getName());
388  SpecXmlUtils.encodeSignedIntegerAttribute(stringBuf, "val", val);
389  stringBuf.append("/>\n");
390  }
391  stringBuf.append("</context_pointset>\n");
392  String end = stringBuf.toString();
393  debugStream.write(end.getBytes());
394  }
395  }

478  public void getPcode(Address addr, Instruction instr) {
479  if (instr != null)
480  try {
481  byte[] bytes;
482  int delaySlotsCnt = instr.getDelaySlotDepth();
483  if (delaySlotsCnt == 0) {
484  bytes = instr.getBytes();
485  }
486  else {
487  // Include delay slot bytes with instruction bytes
488  Listing listing = instr.getProgram().getListing();
489  int byteCnt = instr.getLength();
490  Instruction[] instructions = new Instruction[delaySlotsCnt + 1];
491  instructions[0] = instr;
492  Address nextAddr = instr.getMaxAddress().add(1);
493  for (int i = 1; i <= delaySlotsCnt; i++) {
494  instructions[i] = listing.getInstructionAt(nextAddr);
495  int len = instructions[i].getLength();
496  byteCnt += len;
497  nextAddr.add(len);
498  }
499  bytes = new byte[byteCnt];
500  byteCnt = 0;
501  for (int i = 0; i <= delaySlotsCnt; i++) {
502  byte[] b = instructions[i].getBytes();
503  System.arraycopy(b, 0, bytes, byteCnt, b.length);
504  byteCnt += b.length;
505  }
506  }
507  getBytes(addr, bytes);
508  getContextChangePoints(addr);
509  }
510  catch (MemoryAccessException e) {
511  Msg.error(this, "Unexpected Exception: " + e.getMessage(), e);
512  }
513  }

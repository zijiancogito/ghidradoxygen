112

116  List::const_iterator iter;

118  string p1,p2,p3;

163 string OptionExtraPop::apply(Architecture *glb,const string &p1,const string &p2,const string &p3) const
164 
165 {
166  int4 expop = -300;
167  string res;
168  if (p1 == "unknown")
169  expop = ProtoModel::extrapop_unknown;
170  else {
171  istringstream s1(p1);
172  s1.unsetf(ios::dec | ios::hex | ios::oct); // Let user specify base
173  s1 >> expop;
174  }
175  if (expop == -300)
176  throw ParseError("Bad extrapop adjustment parameter");
177  if (p2.size() != 0) {
178  Funcdata *fd;
179  fd = glb->symboltab->getGlobalScope()->queryFunction( p2 );
180  if (fd == (Funcdata *)0)
181  throw RecovError("Unknown function name: "+p2);
182  fd->getFuncProto().setExtraPop(expop);
183  res = "ExtraPop set for function "+p2;
184  }
185  else {
186  glb->defaultfp->setExtraPop(expop);
187  if (glb->evalfp_current != (ProtoModel *)0)
188  glb->evalfp_current->setExtraPop(expop);
189  if (glb->evalfp_called != (ProtoModel *)0)
190  glb->evalfp_called->setExtraPop(expop);
191  res = "Global extrapop set";
192  }
193  return res;
194 }

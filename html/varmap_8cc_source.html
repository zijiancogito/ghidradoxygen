136 void ScopeLocal::markNotMapped(AddrSpace *spc,uintb first,int4 sz,bool parameter)
137 
138 {
139  if (spaceid != spc) return;
140  uintb last = first + sz - 1;
141  // Do not allow the range to cover the split point between "negative" and "positive" stack offsets
142  if (last < first) // Check for possible wrap around
143  last = spc->getHighest();
144  else if (last > spc->getHighest())
145  last = spc->getHighest();
146  if (parameter) { // Everything above parameter
147  if (stackgrowsnegative) {
148  const Range *rng = localrange.getRange(spc,first);
149  if (rng != (const Range *)0)
150  first = rng->getFirst(); // Everything less is not mapped
151  }
152  else {
153  const Range *rng = localrange.getRange(spc,last);
154  if (rng != (const Range *)0)
155  last = rng->getLast(); // Everything greater is not mapped
156  }
157  sz = (last-first)+1;
158  }
159  Address addr(spaceid,first);
160  // Remove any symbols under range
161  SymbolEntry *overlap = findOverlap(addr,sz);
162  while(overlap != (SymbolEntry *)0) { // For every overlapping entry
163  Symbol *sym = overlap->getSymbol();
164  if ((sym->getFlags()&Varnode::typelock)!=0) {
165  // If the symbol and the use are both as parameters
166  // this is likely the special case of a shared return call sharing the parameter location
167  // of the original function in which case we don't print a warning
168  if ((!parameter) || (sym->getCategory() != 0))
169  fd->warningHeader("Variable defined which should be unmapped: "+sym->getName());
170  return;
171  }
172  removeSymbol(sym);
173  overlap = findOverlap(addr,sz);
174  }
175  glb->symboltab->removeRange(this,spaceid,first,last);
176 }

213 bool ScopeLocal::adjustFit(MapRange &a) const
214 
215 {
216  if (a.size==0) return false; // Nothing to fit
217  if ((a.flags & Varnode::typelock)!=0) return false; // Already entered
218  Address addr(spaceid,a.start);
219  uintb maxsize = getRangeTree().longestFit(addr,a.size);
220  if (maxsize==0) return false;
221  if (maxsize < a.size) { // Suggested range doesn't fit
222  if (maxsize < a.type->getSize()) return false; // Can't shrink that match
223  a.size = (int4)maxsize;
224  }
225  // We want ANY symbol that might be within this range
226  SymbolEntry *entry = findOverlap(addr,a.size);
227  if (entry == (SymbolEntry *)0)
228  return true;
229  if (entry->getAddr() <= addr) {
230  // < generally shouldn't be possible
231  // == we might want to check for anything in -a- after -entry-
232  return false;
233  }
234  maxsize = entry->getAddr().getOffset() - a.start;
235  if (maxsize < a.type->getSize()) return false; // Can't shrink for this type
236  a.size = maxsize;
237  return true;
238 }

294 void AliasChecker::gatherInternal(void) const
295 
296 {
297  calculated = true;
298  aliasboundary = localextreme;
299  Varnode *spacebase = fd->findSpacebaseInput(spaceid);
300  if (spacebase == (Varnode *)0) return; // No possible alias
301 
302  gatherAdditiveBase(spacebase,addbase);
303  for(vector<AddBase>::iterator iter=addbase.begin();iter!=addbase.end();++iter) {
304  uintb offset = gatherOffset((*iter).base);
305  offset = AddrSpace::addressToByte(offset,spaceid->getWordSize()); // Convert to byte offset
306  alias.push_back(offset);
307  if (direction == 1) {
308  if (offset < localboundary) continue; // Parameter ref
309  }
310  else {
311  if (offset > localboundary) continue; // Parameter ref
312  }
313  // Always consider anything AFTER a pointer reference as
314  // aliased, regardless of the stack direction
315  if (offset < aliasboundary)
316  aliasboundary = offset;
317  }
318 }

355 // For every sum that involves \b startvn, collect the final result Varnode of the sum.
356 // A sum is any expression involving only the additive operators
357 // INT_ADD, INT_SUB, PTRADD, PTRSUB, and SEGMENTOP. The routine traverses forward recursively
358 // through all descendants of \b vn that are additive operations and collects all the roots
359 // of the traversed trees.
360 // \param startvn is the Varnode to trace
361 // \param addbase will contain all the collected roots
362 void AliasChecker::gatherAdditiveBase(Varnode *startvn,vector<AddBase> &addbase)
363 
364 {
365  vector<AddBase> vnqueue; // varnodes involved in addition with original vn
366  Varnode *vn,*subvn,*indexvn,*othervn;
367  list<PcodeOp *>::const_iterator iter;
368  PcodeOp *op;
369  bool nonadduse;
370  int4 i=0;
371 
372  vn = startvn;
373  vn->setMark();
374  vnqueue.push_back(AddBase(vn,(Varnode *)0));
375  while(i<vnqueue.size()) {
376  vn = vnqueue[i].base;
377  indexvn = vnqueue[i++].index;
378  nonadduse = false;
379  for(iter=vn->beginDescend();iter!=vn->endDescend();++iter) {
380  op = *iter;
381  switch(op->code()) {
382  case CPUI_COPY:
383  nonadduse = true; // Treat COPY as both non-add use and part of ADD expression
384  subvn = op->getOut();
385  if (!subvn->isMark()) {
386  subvn->setMark();
387  vnqueue.push_back(AddBase(subvn,indexvn));
388  }
389  break;
390  case CPUI_INT_ADD:
391  case CPUI_INT_SUB:
392  case CPUI_PTRADD:
393  othervn = op->getIn(1); // Check if something else is being added in besides a constant
394  if (othervn == vn)
395  othervn = op->getIn(0);
396  if (!othervn->isConstant())
397  indexvn = othervn;
398  case CPUI_PTRSUB:
399  case CPUI_SEGMENTOP:
400  subvn = op->getOut();
401  if (!subvn->isMark()) {
402  subvn->setMark();
403  vnqueue.push_back(AddBase(subvn,indexvn));
404  }
405  break;
406  default:
407  nonadduse = true; // Used in non-additive expression
408  }
409  }
410  if (nonadduse)
411  addbase.push_back(AddBase(vn,indexvn));
412  }
413  for(i=0;i<vnqueue.size();++i)
414  vnqueue[i].base->clearMark();
415 }
416 
417 // Treat \b vn as the result of a series of ADD operations.
418 // Examine all the constant terms of this sum and add them together by traversing
419 // the syntax tree rooted at \b vn, backwards, only through additive operations.
420 // \param vn is the Varnode to gather off of
421 // \return the resulting sub-sum
422 uintb AliasChecker::gatherOffset(Varnode *vn)
423 
424 {
425  uintb retval;
426  Varnode *othervn;
427 
428  if (vn->isConstant()) return vn->getOffset();
429  PcodeOp *def = vn->getDef();
430  if (def == (PcodeOp *)0) return 0;
431  switch(def->code()) {
432  case CPUI_COPY:
433  retval = gatherOffset(def->getIn(0));
434  break;
435  case CPUI_PTRSUB:
436  case CPUI_INT_ADD:
437  retval = gatherOffset(def->getIn(0));
438  retval += gatherOffset(def->getIn(1));
439  break;
440  case CPUI_INT_SUB:
441  retval = gatherOffset(def->getIn(0));
442  retval -= gatherOffset(def->getIn(1));
443  break;
444  case CPUI_PTRADD:
445  othervn = def->getIn(2);
446  retval = gatherOffset(def->getIn(0));
447  // We need to treat PTRADD exactly as if it were encoded as an ADD and MULT
448  // Because a plain MULT truncates the ADD tree
449  // We only follow getIn(1) if the PTRADD multiply is by 1
450  if (othervn->isConstant() && (othervn->getOffset()==1))
451  retval = retval + gatherOffset(def->getIn(1));
452  break;
453  case CPUI_SEGMENTOP:
454  retval = gatherOffset(def->getIn(2));
455  break;
456  default:
457  retval = 0;
458  }
459  return retval & calc_mask(vn->getSize());
460 }

595 
596 { // Gather open-ended ranges. These correspond

621  lo = 0;
622  hi = -1;
623  }

675 static bool range_reconcile(const MapRange *a,const MapRange *b)
676 
677 { // Can the types coexist at the given offsets
678  if (a->type->getSize() < b->type->getSize()) {
679  const MapRange *tmp = b;
680  b = a; // Make sure b is smallest
681  a = tmp;
682  }
683  intb mod = (b->sstart - a->sstart) % a->type->getSize();
684  if (mod < 0)
685  mod += a->type->getSize();
686 
687  Datatype *sub = a->type;
688  uintb umod = mod;
689  while((sub!=(Datatype *)0)&&(sub->getSize() > b->type->getSize()))
690  sub = sub->getSubType(umod,&umod);
691 
692  if (sub == (Datatype *)0) return false;
693  if (umod != 0) return false;
694  if (sub->getSize() < b->type->getSize()) return false;
695  return true;
696 }

698 static bool range_contain(const MapRange *a,const MapRange *b)
699 
700 { // Return true if one contains the other
701  // We assume a starts at least as early as b
702  // and that a and b intersect
703  if (a->sstart == b->sstart) return true;
704  // if (a->sstart==a->send) return true;
705  // if (b->sstart==b->send) return true;
706  if (b->sstart+b->size-1 <= a->sstart+a->size-1) return true;
707  return false;

710 static bool range_preferred(const MapRange *a,const MapRange *b,bool reconcile)
711 
712 { // Return true if a's type is preferred over b's
713  if (a->start != b->start)
714  return true; // Something must occupy a->start to b->start
715  // Prefer the locked type
716  if ((b->flags & Varnode::typelock)!=0) {
717  if ((a->flags & Varnode::typelock)==0)
718  return false;
719  }
720  else if ((a->flags & Varnode::typelock)!=0)
721  return true;
722 
723  if (!reconcile) { // If the ranges don't reconcile
724  if ((a->arrayyes)&&(!b->arrayyes)) // Throw out the open range
725  return false;
726  if ((b->arrayyes)&&(!a->arrayyes))
727  return true;
728  }
729 
730  return (0>a->type->typeOrder(*b->type)); // Prefer the more specific
731 }

733 bool ScopeLocal::rangeAbsorb(MapRange *a,MapRange *b)
734 
735 { // check if -a- is an array and could absorb -b-
736  if (!a->arrayyes) return false;
737  if (a->highind < a->lowind) return false;
738  if ((b->lowind == 0)&&(b->highind==-2)) return false; // Don't merge with bounding range
739  Datatype *settype = a->type;
740  if (settype->getSize() != b->type->getSize()) return false;
741  if (settype->getMetatype() == TYPE_UNKNOWN)
742  settype = b->type;
743  else if (b->type->getMetatype() == TYPE_UNKNOWN) {
744  }
745  else if (settype != b->type) // If they are both not unknown, they must be the same
746  return false;
747  if ((a->flags & Varnode::typelock)!=0) return false;
748  if ((b->flags & Varnode::typelock)!=0) return false;
749  if (a->flags != b->flags) return false;
750  intb diffsz = b->sstart - a->sstart;
751  if ((diffsz % settype->getSize()) != 0) return false;
752  diffsz /= settype->getSize();
753  if (diffsz > a->highind) return false;
754  a->type = settype;
755  if (b->arrayyes && (b->lowind <= b->highind)) { // If b has array indexing
756  int4 triallo = b->lowind + diffsz; // Adjust its indexing to the new base
757  int4 trialhi = b->highind + diffsz;
758  if (triallo < a->lowind) // Check if we can expand the base indexing
759  a->lowind = triallo;
760  if (a->highind < trialhi)
761  a->highind = trialhi;
762  }
763  return true;
764 }

766 void ScopeLocal::rangeUnion(MapRange *a,MapRange *b,bool warning)
767 
768 { // Two ranges intersect, produce the reconciled union (in a)
769  uintb aend,bend;
770  uintb end;
771  Datatype *restype;
772  uint4 flags;
773  bool arrayyes,reconcile;
774  int4 lo,hi;
775 
776  aend = spaceid->wrapOffset(a->start+a->size);
777  bend = spaceid->wrapOffset(b->start+b->size);
778  arrayyes = false;
779  lo = 0;
780  hi = -1;
781  if ((aend==0)||(bend==0))
782  end = 0;
783  else
784  end = (aend > bend) ? aend : bend;
785 
786  if (range_contain(a,b)) { // Check for containment
787  reconcile = range_reconcile(a,b);
788  if (range_preferred(a,b,reconcile)) { // Find bigger type
789  restype = a->type;
790  flags = a->flags;
791  arrayyes = a->arrayyes;
792  lo = a->lowind;
793  hi = a->highind;
794  }
795  else {
796  restype = b->type;
797  flags = b->flags;
798  arrayyes = b->arrayyes;
799  lo = b->lowind;
800  hi = b->highind;
801  }
802  if ((a->start==b->start)&&(a->size==b->size)) {
803  arrayyes = a->arrayyes || b->arrayyes;
804  if (arrayyes) {
805  lo = (a->lowind < b->lowind) ? a->lowind : b->lowind;
806  hi = (a->highind < b->highind) ? b->highind : a->highind;
807  }
808  }
809  if (warning && (!reconcile)) { // See if two types match up
810  if ((!b->arrayyes)&&(!a->arrayyes))
811  overlapproblems = true;
812  }
813  }
814  else {
815  reconcile = false;
816  restype = (Datatype *)0; // Unable to resolve the type
817  flags = 0;
818  }
819  // Check for really problematic cases
820  if (!reconcile) {
821  if ((b->flags & Varnode::typelock)!=0) {
822  if ((a->flags & Varnode::typelock)!=0)
823  throw LowlevelError("Overlapping forced variable types : " + a->type->getName() + " " + b->type->getName());
824  }
825  }
826  if (restype == (Datatype *)0) // If all else fails
827  restype = glb->types->getBase(1,TYPE_UNKNOWN); // Do unknown array (size 1)
828 
829  a->type = restype;
830  a->flags = flags;
831  a->arrayyes = arrayyes;
832  a->lowind = lo;
833  a->highind = hi;
834  if ((!reconcile)&&(a->start != b->start)) { // Truncation is forced
835  if ((a->flags & Varnode::typelock)!=0) { // If a is locked
836  return; // Discard b entirely in favor of a
837  }
838  // Concede confusion about types, set unknown type rather than a or b's type
839  a->size = spaceid->wrapOffset(end-a->start);
840  a->type = glb->types->getBase(a->size,TYPE_UNKNOWN);
841  a->flags = 0;
842  a->arrayyes = false;
843  a->lowind = 0;
844  a->highind = -1;
845  return;
846  }
847  a->size = restype->getSize();
848 }

850 void ScopeLocal::restructure(MapState &state,bool warning)
851 
852 {
853  MapRange cur;
854  MapRange *next;
855  // This implementation does not allow a range
856  // to contain both ~0 and 0
857  overlapproblems = false;
858  if (!state.initialize()) return; // No references to stack at all
859 
860  cur = *state.next();
861  while(state.getNext()) {
862  next = state.next();
863  if (next->sstart < cur.sstart+cur.size) // Do the ranges intersect
864  rangeUnion(&cur,next,warning); // Union them
865  else {

867  if (cur.arrayyes)
868  cur.size = next->sstart-cur.sstart;
869  if (adjustFit(cur))
870  createEntry(cur);
871  cur = *next;
872  }
873  }
874  }
875  // The last range is artificial so we don't
876  // build an entry for it
877 }

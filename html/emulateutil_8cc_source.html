28 uintb EmulatePcodeOp::getLoadImageValue(AddrSpace *spc,uintb off,int4 sz) const
29 
30 {
31  LoadImage *loadimage = glb->loader;
32  uintb res;
33 
34  loadimage->loadFill((uint1 *)&res,sizeof(uintb),Address(spc,off));
35 
36  if ((HOST_ENDIAN==1) != spc->isBigEndian())
37  res = byte_swap(res,sizeof(uintb));
38  if (spc->isBigEndian() && (sz < sizeof(uintb)))
39  res >>= (sizeof(uintb)-sz)*8;
40  else
41  res &= calc_mask(sz);
42  return res;
43 }

325 bool EmulateSnippet::checkForLegalCode(void) const
326 
327 {
328  for(int4 i=0;i<opList.size();++i) {
329  PcodeOpRaw *op = opList[i];
330  VarnodeData *vn;
331  OpCode opc = op->getOpcode();
332  if (opc == CPUI_BRANCHIND || opc == CPUI_CALL || opc == CPUI_CALLIND || opc == CPUI_CALLOTHER ||
333  opc == CPUI_STORE || opc == CPUI_SEGMENTOP || opc == CPUI_CPOOLREF ||
334  opc == CPUI_NEW || opc == CPUI_MULTIEQUAL || opc == CPUI_INDIRECT)
335  return false;
336  if (opc == CPUI_BRANCH) {
337  vn = op->getInput(0);
338  if (vn->space->getType() != IPTR_CONSTANT) // Only relative branching allowed
339  return false;
340  }
341  vn = op->getOutput();
342  if (vn != (VarnodeData *)0) {
343  if (vn->space->getType() != IPTR_INTERNAL)
344  return false; // Can only write to temporaries
345  }
346  for(int4 j=0;j<op->numInput();++j) {
347  vn = op->getInput(j);
348  if (vn->space->getType() == IPTR_PROCESSOR)
349  return false; // Cannot read from normal registers
350  }
351  }
352  return true;
353 }

363 uintb EmulateSnippet::getVarnodeValue(VarnodeData *vn) const
364 
365 {
366  AddrSpace *spc = vn->space;
367  if (spc->getType() == IPTR_CONSTANT)
368  return vn->offset;
369  if (spc->getType() == IPTR_INTERNAL) {
370  map<uintb,uintb>::const_iterator iter;
371  iter = tempValues.find(vn->offset);
372  if (iter != tempValues.end())
373  return (*iter).second; // We have seen this varnode before
374  throw LowlevelError("Read before write in snippet emulation");
375  }
376 
377  return getLoadImageValue(vn->space,vn->offset,vn->size);
378 }

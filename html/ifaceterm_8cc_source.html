60 int4 IfaceTerm::doCompletion(string &line,int4 cursor)
61 
62 { // Try to complete the current command
63  vector<string> fullcommand;
64  istringstream s(line);
65  string tok;
66  vector<IfaceCommand *>::const_iterator first,last;
67  int4 oldsize,match;
68 
69  first = comlist.begin();
70  last = comlist.end();
71  match = expandCom(fullcommand,s,first,last); // Try to expand the command
72  if (match == 0) {
73  *optr << endl << "Invalid command" << endl;
74  return cursor; // No change to command line
75  }
76 
77  // At least one match
78  oldsize = line.size();
79  wordsToString(line,fullcommand);
80  if (match<0)
81  match = -match;
82  else
83  line += ' '; // Provide extra space if command word is complete
84  if (!s.eof()) { // Read any additional parameters back to command line
85  s >> tok >> ws;
86  line += tok; // Assume first space is present before extra params
87  }
88  while(!s.eof()) {
89  line += ' '; // Provide space between extra parameters
90  s >> tok >> ws;
91  line += tok;
92  }
93  if (oldsize < line.size()) // If we have expanded at all
94  return line.size(); // Just display expansion
95 
96  if (match > 1) { // If more than one possible command
97  string complete; // Display all possible completions
98  *optr << endl;
99  for(;first!=last;++first) {
100  (*first)->commandString(complete); // Get possible completion
101  *optr << complete << endl;
102  }
103  }
104  else // Command is unique and expanded
105  *optr << endl << "Command is complete" << endl;
106  return line.size();
107 }

109 void IfaceTerm::readLine(string &line)
110 
111 {
112  char val;
113  int4 escval;
114  int4 cursor,lastlen,i;
115  bool onecharecho;
116  int4 hist;
117  string saveline;
118 
119  line.erase();
120  cursor = 0;
121  hist = 0;
122  do {
123  onecharecho = false;
124  lastlen = line.size();
125  val = sptr->get();
126  if (sptr->eof())
127  val = '\n';
128  switch(val) {
129  case 0x01: // C-a
130  cursor = 0; // Jump to beginning
131  break;
132  case 0x02: // C-b
133  if (cursor>0)
134  cursor -= 1; // back up one char
135  break;
136  case 0x03: // C-c
137  line.erase(); // As if we cleared the line and hit return
138  cursor = 0;
139  val = 0x0a;
140  onecharecho = true;
141  break;
142  case 0x04: // C-d
143  line.erase(cursor,1); // Delete character cursor is on
144  break;
145  case 0x05: // C-e
146  cursor = line.size(); // Jump to end
147  break;
148  case 0x06: // C-f
149  if (cursor < line.size())
150  cursor += 1; // Move forward one character
151  break;
152  case 0x07: // C-g
153  break; // do nothing
154  case 0x09: // TAB
155  cursor = doCompletion(line,cursor);
156  break;
157  case 0x0a: // Newline
158  case 0x0d: // Carriage return
159  cursor = line.size();
160  onecharecho = true;
161  break;
162  case 0x0b: // C-k
163  line.erase(cursor);
164  break;
165  case 0x0c: // C-l
166  break;
167  case 0x0e: // C-n
168  if (hist >0) {
169  hist -= 1; // Get more recent history
170  if (hist>0)
171  getHistory(line,hist-1);
172  else
173  line = saveline;
174  cursor = line.size();
175  }
176  break;
177  case 0x10: // C-p
178  if (hist < getHistorySize()) {
179  hist += 1; // Get more ancient history
180  if (hist==1)
181  saveline = line;
182  getHistory(line,hist-1);
183  cursor = line.size();
184  }
185  break;
186  case 0x12: // C-r
187  break;
188  case 0x15: // C-u
189  line.erase(0,cursor); // Erase up to cursor
190  cursor = 0;
191  break;
192  case 0x1b: // Escape character
193  escval = sptr->get();
194  escval <<= 8;
195  escval += sptr->get();
196  switch(escval) {
197  case 0x4f44: // left arrow
198  if (cursor>0)
199  cursor -= 1;
200  break;
201  case 0x4f43: // right arrow
202  if (cursor<line.size())
203  cursor += 1;
204  break;
205  }
206  break;
207  case 0x08:
208  case 0x7f: // delete
209  if (cursor!=0)
210  line.erase(--cursor,1);
211  break;
212  default:
213  line.insert(cursor++,1,val); // Insert single character
214  if (cursor == line.size())
215  onecharecho = true;
216  break;
217  }
218  if (onecharecho)
219  optr->put(val); // Echo most characters
220  else {
221  optr->put('\r'); // Ontop of old line
222  writePrompt();
223  *optr << line; // print new line
224  for(i=line.size();i<lastlen;++i)
225  optr->put(' '); // Erase any old characters
226  for(i=i-cursor;i>0;--i)
227  optr->put('\b'); // Put cursor in the right place
228  }
229  } while(val != '\n');
230 }

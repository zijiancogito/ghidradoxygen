111 int4 GhidraCommand::doit(void)
112 
113 {
114  status = 0;
115  sout.write("\000\000\001\006",4); // Command response header
116  try {
117  loadParameters();
118  int4 type = ArchitectureGhidra::readToAnyBurst(sin);
119  if (type != 3)
120  throw JavaError("alignment","Missing end of command");
121  rawAction();
122  }
123  catch(XmlError &err) {
124  string errmsg;
125  errmsg = "XML processing error: " + err.explain;
126  ghidra->printMessage( errmsg );
127  }
128  catch(JavaError &err) {
129  ArchitectureGhidra::passJavaException(sout,err.type,err.explain);
130  return status; // Abort sending any results
131  }
132  catch(RecovError &err) {
133  string errmsg;
134  errmsg = "Recoverable Error: " + err.explain;
135  ghidra->printMessage( errmsg );
136  }
137  catch(LowlevelError &err) {
138  string errmsg;
139  errmsg = "Low-level Error: " + err.explain;
140  ghidra->printMessage( errmsg );
141  }
142  sendResult();
143  sout.write("\000\000\001\007",4); // Command response closer
144  sout.flush();
145  return status;
146 }

366 void SetAction::rawAction(void)
367 
368 {
369  res = false;
370 
371  if (actionstring.size() != 0)
372  ghidra->allacts.setCurrent(actionstring);
373  if (printstring.size() != 0) {
374  if (printstring == "tree")
375  ghidra->setSendSyntaxTree(true);
376  else if (printstring == "notree")
377  ghidra->setSendSyntaxTree(false);
378  else if (printstring == "c")
379  ghidra->setSendCCode(true);
380  else if (printstring == "noc")
381  ghidra->setSendCCode(false);
382  else if (printstring == "parammeasures")
383  ghidra->setSendParamMeasures(true);
384  else if (printstring == "noparammeasures")
385  ghidra->setSendParamMeasures(false);
386  else if (printstring == "jumpload")
387  ghidra->flowoptions |= FlowInfo::record_jumploads;
388  else if (printstring == "nojumpload")
389  ghidra->flowoptions &= ~((uint4)FlowInfo::record_jumploads);
390  else
391  throw LowlevelError("Unknown print action: "+printstring);
392  }
393  res = true;
394 }

458 int4 GhidraCapability::readCommand(istream &sin,ostream &out)
459 
460 {
461  string function;
462  int4 type;
463 
464  do {
465  type = ArchitectureGhidra::readToAnyBurst(sin); // Align ourselves
466  } while(type != 2);
467  ArchitectureGhidra::readStringStream(sin,function);
468  map<string,GhidraCommand *>::const_iterator iter;
469  iter = commandmap.find(function);
470  if (iter == commandmap.end()) {
471  out.write("\000\000\001\006",4); // Command response header
472  out.write("\000\000\001\020",4);
473  out << "Bad command: " << function;
474  out.write("\000\000\001\021",4);
475  out.write("\000\000\001\007",4); // Command response closer
476  out.flush();
477  return 0;
478  }
479  return (*iter).second->doit();
480 }

28 static VarnodeTpl *find_size(const ConstTpl &offset,const ConstructTpl *ct)
29 
30 { // Find a defining instance of the local variable
31  // with given -offset-
32  const vector<OpTpl *> &ops(ct->getOpvec());
33  VarnodeTpl *vn;
34  OpTpl *op;
35 
36  for(int4 i=0;i<ops.size();++i) {
37  op = ops[i];
38  vn = op->getOut();
39  if ((vn!=(VarnodeTpl *)0)&&(vn->isLocalTemp())) {
40  if (vn->getOffset() == offset)
41  return vn;
42  }
43  for(int4 j=0;j<op->numInput();++j) {
44  vn = op->getIn(j);
45  if (vn->isLocalTemp()&&(vn->getOffset()==offset))
46  return vn;
47  }
48  }
49  return (VarnodeTpl *)0;
50 }

202  SubtableSymbol *tabsym;

273 bool ConsistencyChecker::sizeRestriction(OpTpl *op,Constructor *ct)
274 
275 { // Make sure op template meets size restrictions
276  // Return false and any info about mismatched sizes
277  int4 vnout,vn0,vn1;
278  AddrSpace *spc;
279 
280  switch(op->getOpcode()) {
281  case CPUI_COPY: // Instructions where all inputs and output are same size
282  case CPUI_INT_2COMP:
283  case CPUI_INT_NEGATE:
284  case CPUI_FLOAT_NEG:
285  case CPUI_FLOAT_ABS:
286  case CPUI_FLOAT_SQRT:
287  case CPUI_FLOAT_CEIL:
288  case CPUI_FLOAT_FLOOR:
289  case CPUI_FLOAT_ROUND:
290  vnout = recoverSize(op->getOut()->getSize(),ct);
291  if (vnout == -1) {
292  printOpError(op,ct,-1,-1,"Using subtable with exports in expression");
293  return false;
294  }
295  vn0 = recoverSize(op->getIn(0)->getSize(),ct);
296  if (vn0 == -1) {
297  printOpError(op,ct,0,0,"Using subtable with exports in expression");
298  return false;
299  }
300  if (vnout == vn0) return true;
301  if ((vnout==0)||(vn0==0)) return true;
302  printOpError(op,ct,-1,0,"Input and output sizes must match");
303  return false;
304  case CPUI_INT_ADD:
305  case CPUI_INT_SUB:
306  case CPUI_INT_XOR:
307  case CPUI_INT_AND:
308  case CPUI_INT_OR:
309  case CPUI_INT_MULT:
310  case CPUI_INT_DIV:
311  case CPUI_INT_SDIV:
312  case CPUI_INT_REM:
313  case CPUI_INT_SREM:
314  case CPUI_FLOAT_ADD:
315  case CPUI_FLOAT_DIV:
316  case CPUI_FLOAT_MULT:
317  case CPUI_FLOAT_SUB:
318  vnout = recoverSize(op->getOut()->getSize(),ct);
319  if (vnout == -1) {
320  printOpError(op,ct,-1,-1,"Using subtable with exports in expression");
321  return false;
322  }
323  vn0 = recoverSize(op->getIn(0)->getSize(),ct);
324  if (vn0 == -1) {
325  printOpError(op,ct,0,0,"Using subtable with exports in expression");
326  return false;
327  }
328  vn1 = recoverSize(op->getIn(1)->getSize(),ct);
329  if (vn1 == -1) {
330  printOpError(op,ct,1,1,"Using subtable with exports in expression");
331  return false;
332  }
333  if ((vnout!=0)&&(vn0!=0)&&(vnout!=vn0)) {
334  printOpError(op,ct,-1,0,"The output and all input sizes must match");
335  return false;
336  }
337  if ((vnout!=0)&&(vn1!=0)&&(vnout!=vn1)) {
338  printOpError(op,ct,-1,1,"The output and all input sizes must match");
339  return false;
340  }
341  if ((vn0!=0)&&(vn1!=0)&&(vn0!=vn1)) {
342  printOpError(op,ct,0,1,"The output and all input sizes must match");
343  return false;
344  }
345  return true;
346  case CPUI_FLOAT_NAN:
347  vnout = recoverSize(op->getOut()->getSize(),ct);
348  if (vnout == -1) {
349  printOpError(op,ct,-1,-1,"Using subtable with exports in expression");
350  return false;
351  }
352  if (vnout != 1) {
353  printOpError(op,ct,-1,-1,"Output must be a boolean (size 1)");
354  return false;
355  }
356  break;
357  case CPUI_INT_EQUAL: // Instructions with bool output, all inputs equal size
358  case CPUI_INT_NOTEQUAL:
359  case CPUI_INT_SLESS:
360  case CPUI_INT_SLESSEQUAL:
361  case CPUI_INT_LESS:
362  case CPUI_INT_LESSEQUAL:
363  case CPUI_INT_CARRY:
364  case CPUI_INT_SCARRY:
365  case CPUI_INT_SBORROW:
366  case CPUI_FLOAT_EQUAL:
367  case CPUI_FLOAT_NOTEQUAL:
368  case CPUI_FLOAT_LESS:
369  case CPUI_FLOAT_LESSEQUAL:
370  vnout = recoverSize(op->getOut()->getSize(),ct);
371  if (vnout == -1) {
372  printOpError(op,ct,-1,-1,"Using subtable with exports in expression");
373  return false;
374  }
375  if (vnout != 1) {
376  printOpError(op,ct,-1,-1,"Output must be a boolean (size 1)");
377  return false;
378  }
379  vn0 = recoverSize(op->getIn(0)->getSize(),ct);
380  if (vn0 == -1) {
381  printOpError(op,ct,0,0,"Using subtable with exports in expression");
382  return false;
383  }
384  vn1 = recoverSize(op->getIn(1)->getSize(),ct);
385  if (vn1 == -1) {
386  printOpError(op,ct,1,1,"Using subtable with exports in expression");
387  return false;
388  }
389  if ((vn0==0)||(vn1==0)) return true;
390  if (vn0 != vn1) {
391  printOpError(op,ct,0,1,"Inputs must be the same size");
392  return false;
393  }
394  return true;
395  case CPUI_BOOL_XOR:
396  case CPUI_BOOL_AND:
397  case CPUI_BOOL_OR:
398  vnout = recoverSize(op->getOut()->getSize(),ct);
399  if (vnout == -1) {
400  printOpError(op,ct,-1,-1,"Using subtable with exports in expression");
401  return false;
402  }
403  if (vnout != 1) {
404  printOpError(op,ct,-1,-1,"Output must be a boolean (size 1)");
405  return false;
406  }
407  vn0 = recoverSize(op->getIn(0)->getSize(),ct);
408  if (vn0 == -1) {
409  printOpError(op,ct,0,0,"Using subtable with exports in expression");
410  return false;
411  }
412  if (vn0 != 1) {
413  printOpError(op,ct,0,0,"Input must be a boolean (size 1)");
414  return false;
415  }
416  return true;
417  case CPUI_BOOL_NEGATE:
418  vnout = recoverSize(op->getOut()->getSize(),ct);
419  if (vnout == -1) {
420  printOpError(op,ct,-1,-1,"Using subtable with exports in expression");
421  return false;
422  }
423  if (vnout != 1) {
424  printOpError(op,ct,-1,-1,"Output must be a boolean (size 1)");
425  return false;
426  }
427  vn0 = recoverSize(op->getIn(0)->getSize(),ct);
428  if (vn0 == -1) {
429  printOpError(op,ct,0,0,"Using subtable with exports in expression");
430  return false;
431  }
432  if (vn0 != 1) {
433  printOpError(op,ct,0,0,"Input must be a boolean (size 1)");
434  return false;
435  }
436  return true;
437  // The shift amount does not necessarily have to be the same size
438  // But the output and first parameter must be same size
439  case CPUI_INT_LEFT:
440  case CPUI_INT_RIGHT:
441  case CPUI_INT_SRIGHT:
442  vnout = recoverSize(op->getOut()->getSize(),ct);
443  if (vnout == -1) {
444  printOpError(op,ct,-1,-1,"Using subtable with exports in expression");
445  return false;
446  }
447  vn0 = recoverSize(op->getIn(0)->getSize(),ct);
448  if (vn0 == -1) {
449  printOpError(op,ct,0,0,"Using subtable with exports in expression");
450  return false;
451  }
452  if ((vnout==0)||(vn0==0)) return true;
453  if (vnout != vn0) {
454  printOpError(op,ct,-1,0,"Output and first input must be the same size");
455  return false;
456  }
457  return true;
458  case CPUI_INT_ZEXT:
459  case CPUI_INT_SEXT:
460  vnout = recoverSize(op->getOut()->getSize(),ct);
461  if (vnout == -1) {
462  printOpError(op,ct,-1,-1,"Using subtable with exports in expression");
463  return false;
464  }
465  vn0 = recoverSize(op->getIn(0)->getSize(),ct);
466  if (vn0 == -1) {
467  printOpError(op,ct,0,0,"Using subtable with exports in expression");
468  return false;
469  }
470  if ((vnout==0)||(vn0==0)) return true;
471  if (vnout == vn0) {
472  dealWithUnnecessaryExt(op,ct);
473  return true;
474  }
475  else if (vnout < vn0) {
476  printOpError(op,ct,-1,0,"Output size must be strictly bigger than input size");
477  return false;
478  }
479  return true;
480  case CPUI_CBRANCH:
481  vn1 = recoverSize(op->getIn(1)->getSize(),ct);
482  if (vn1 == -1) {
483  printOpError(op,ct,1,1,"Using subtable with exports in expression");
484  return false;
485  }
486  if (vn1 != 1) {
487  printOpError(op,ct,1,1,"Input must be a boolean (size 1)");
488  return false;
489  }
490  return true;
491  case CPUI_LOAD:
492  case CPUI_STORE:
493  if (op->getIn(0)->getOffset().getType() != ConstTpl::spaceid)
494  return true;
495  spc = op->getIn(0)->getOffset().getSpace();
496  vn1 = recoverSize(op->getIn(1)->getSize(),ct);
497  if (vn1 == -1) {
498  printOpError(op,ct,1,1,"Using subtable with exports in expression");
499  return false;
500  }
501  if ((vn1!=0)&&(vn1 != spc->getAddrSize())) {
502  printOpError(op,ct,1,1,"Pointer size must match size of space");
503  return false;
504  }
505  return true;
506  case CPUI_SUBPIECE:
507  vnout = recoverSize(op->getOut()->getSize(),ct);
508  if (vnout == -1) {
509  printOpError(op,ct,-1,-1,"Using subtable with exports in expression");
510  return false;
511  }
512  vn0 = recoverSize(op->getIn(0)->getSize(),ct);
513  if (vn0 == -1) {
514  printOpError(op,ct,0,0,"Using subtable with exports in expression");
515  return false;
516  }
517  vn1 = op->getIn(1)->getOffset().getReal();
518  if ((vnout==0)||(vn0==0)) return true;
519  if ((vnout==vn0)&&(vn1==0)) { // No actual truncation is occuring
520  dealWithUnnecessaryTrunc(op,ct);
521  return true;
522  }
523  else if (vnout>=vn0) {
524  printOpError(op,ct,-1,0,"Output must be strictly smaller than input");
525  return false;
526  }
527  if (vnout>vn0-vn1) {
528  printOpError(op,ct,-1,0,"Too much truncation");
529  return false;
530  }
531  return true;
532  default:
533  break;
534  }
535  return true;
536 }

915 void ConsistencyChecker::setPostOrder(SubtableSymbol *root)
916 
917 {
918  postorder.clear();
919  sizemap.clear();
920 
921  // Establish post-order of SubtableSymbols so that we can
922  // recursively fill in sizes of varnodes which are exported
923  // from constructors
924 
925  vector<SubtableSymbol *> path;
926  vector<int4> state;
927  vector<int4> ctstate;
928 
929  sizemap[root] = -1; // Mark root as traversed
930  path.push_back(root);
931  state.push_back(0);
932  ctstate.push_back(0);
933

935  SubtableSymbol *cur = path.back();
936  int4 ctind = state.back();
937  if (ctind >= cur->getNumConstructors()) {
938  path.pop_back(); // Table is fully traversed
939  state.pop_back();
940  ctstate.pop_back();
941  postorder.push_back(cur); // Post the traversed table
942  }
943  else {
944  Constructor *ct = cur->getConstructor(ctind);
945  int4 oper = ctstate.back();
946  if (oper >= ct->getNumOperands()) {
947  state.back() = ctind + 1; // Constructor fully traversed
948  ctstate.back() = 0;

952  OperandSymbol *opsym = ct->getOperand(oper);

955  map<SubtableSymbol *,int4>::const_iterator iter;
956  iter = sizemap.find(subsym);
957  if (iter == sizemap.end()) { // Not traversed yet
958  sizemap[subsym] = -1; // Mark table as traversed
959  path.push_back(subsym); // Recurse
960  state.push_back(0);
961  ctstate.push_back(0);
962  }
963  }
964  }

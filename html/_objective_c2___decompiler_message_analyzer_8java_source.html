118  private void inspectFunction(Program program, DecompileResults results, TaskMonitor monitor) {
119  String currentClassName = null;
120  String currentMethodName = null;
121 
122  HighFunction highFunction = results.getHighFunction();
123  if (highFunction == null) {
124  return;
125  }
126 
127  Function function = results.getFunction();
128  Iterator<PcodeOpAST> pcodeOps = highFunction.getPcodeOps();
129  while (pcodeOps.hasNext()) {
130  if (monitor.isCancelled()) {
131  return;
132  }
133  currentClassName = null;
134  currentMethodName = null;
135  PcodeOpAST op = pcodeOps.next();
136  String mnemonic = op.getMnemonic();
137  if (mnemonic == null || (!mnemonic.equals("CALL") && !mnemonic.equals("CALLIND"))) {
138  continue;
139  }
140  Varnode[] inputs = op.getInputs();
141  if (!isObjcCall(program, inputs[0], monitor)) {
142  continue;
143  }
144  boolean isStret = isStretCall(program, inputs[0], monitor);
145  for (int i = 1; i < inputs.length; i++) {
146  String name;
147  boolean isClass = isClass(i, isStret);
148  boolean isMessage = isMessage(i, isStret);
149  name = getNameForVarnode(program, function, inputs[i], isClass, isMessage, 0, 1,
150  monitor);
151  if (isClass) {
152  currentClassName = name;
153  }
154  else if (isMessage) {
155  currentMethodName = name;
156  }
157  if (currentClassName != null && currentMethodName != null) {
158  break;
159  }
160  }
161 
162  if (currentClassName == null || currentMethodName == null) {
163  continue;
164  }
165 
166  List<String> parameters = new ArrayList<>();
167  int paramStart = isStret ? 4 : 3;
168  for (int i = paramStart; i < inputs.length; i++) {
169  String paramValue =
170  getNameForVarnode(program, function, inputs[i], false, false, 0, 1, monitor);
171  parameters.add(getIvarNameFromQualifiedName(paramValue));
172  }
173  setCommentAndReference(program, currentClassName, currentMethodName, op, parameters);
174  }
175 
176  }

178  private void setCommentAndReference(Program program, String currentClassName,
179  String currentMethodName, PcodeOpAST op, List<String> parameters) {
180  Address objcCallAddress = op.getSeqnum().getTarget();
181  objcCallAddress = getAddressInProgram(program, objcCallAddress.getOffset());
182  Instruction instruction = program.getListing().getInstructionAt(objcCallAddress);
183 
184  String fullyQualifiedName = currentClassName;
185 
186  // If the target is an instance variable, we want to display the
187  // variable name in the comment, but use the class type when
188  // creating the reference.
189  if (currentClassName.contains("::")) {
190  currentClassName = getClassNameFromQualifiedName(fullyQualifiedName);
191  }
192  setReference(objcCallAddress, program, currentClassName, currentMethodName);
193 
194  if (instruction.getComment(CodeUnit.EOL_COMMENT) != null) {
195  return;
196  }
197 
198  currentClassName = getIvarNameFromQualifiedName(fullyQualifiedName);
199 
200  // Formatting based on whether or not the method takes parameters
201  currentMethodName += currentMethodName.contains(":") ? "]" : " ]";
202  String[] split = currentMethodName.split(":");
203  StringBuilder builder = new StringBuilder();
204  builder.append("[" + currentClassName + " " + split[0]);
205  for (int i = 1; i < split.length; i++) {
206  try {
207  builder.append(":" + parameters.get(i - 1) + " ");
208  }
209  catch (Exception e) {
210  // Decompiler found less params than the function should really
211  // have.
212  builder.append(":<<unknown>> ");
213  }
214  builder.append(split[i]);
215  }
216  builder.delete(builder.length() - 2, builder.length() - 1);
217  instruction.setComment(CodeUnit.EOL_COMMENT, builder.toString());
218  }

229  private Address getAddressFromVarnode(Program program, Varnode input, int depth,
230  TaskMonitor monitor) {
231  if (monitor.isCancelled()) {
232  return null;
233  }
234  if (input == null) {
235  return null;
236  }
237  if (depth >= MAX_RECURSION_DEPTH) {
238  return null;
239  }
240  if (!input.isAddress() && !input.isConstant()) {
241  PcodeOp def = input.getDef();
242  if (def == null) {
243  return null;
244  }
245  Varnode[] inputs = def.getInputs();
246  for (Varnode subInput : inputs) {
247  if (monitor.isCancelled()) {
248  return null;
249  }
250  Address address = getAddressFromVarnode(program, subInput, depth + 1, monitor);
251  if (address == null) {
252  continue;
253  }
254  address = getAddressInProgram(program, address.getOffset());
255  if (address != null && program.getMemory().contains(address)) {
256  return address;
257  }
258  }
259  }
260  return input.getAddress();
261  }

273  private String getNameForVarnode(Program program, Function function, Varnode input,
274  boolean isClass, boolean isMethod, int depth, int numInputs, TaskMonitor monitor) {
275  try {
276  if (depth >= MAX_RECURSION_DEPTH) {
277  return "<<unknown>>";
278  }
279  String name = null;
280  if (input == null) {
281  return null;
282  }
283  if (input.isAddress() || input.isConstant()) {
284  long offset = input.getOffset();
285  name = getNameFromOffset(program, offset, input, isClass, isMethod);
286  }
287  PcodeOp def = input.getDef();
288  if (def == null) {
289  if (name == null) {
290  name = getParamNameOrOffset(function, input, isClass, isMethod, numInputs);
291  }
292  return name;
293  }
294  else if (isSuper2Call(program, input) && !isMethod) {
295  name = getSuperClassName(program, input, function);
296  return name;
297  }
298  Varnode[] inputs = def.getInputs();
299 
300  if (isObjcCall(program, inputs[0], monitor)) {
301  Symbol objcSymbol = getSymbolFromVarnode(program, inputs[0], monitor);
302  int classIndex = 1;
303  if (objcSymbol.getName().contains("stret")) {
304  classIndex = 2;
305  }
306  if (inputs.length <= classIndex) {
307  PcodeOp callDefinition = inputs[0].getDef();
308  if (callDefinition == null) {
309  return null;
310  }
311  inputs = new Varnode[] { callDefinition.getInput(classIndex) };
312  }
313  else {
314  inputs = new Varnode[] { inputs[classIndex] };
315  }
316  numInputs = 1;
317  }
318 
319  int index = getIndexOfAddress(inputs);
320  if (index != -1) {
321  name =
322  getNameFromOffset(program, inputs[index].getOffset(), input, isClass, isMethod);
323  if (name != null) {
324  return name;
325  }
326  }
327  for (Varnode subInput : inputs) {
328  // If a name was found, just unwind the recursion. If it is just
329  // a constant (ex. when determining parameters) keep looking
330  // to see if we can find an actual name.
331  if (name != null && !stringIsLong(name)) {
332  break;
333  }
334  name = getNameForVarnode(program, function, subInput, isClass, isMethod, depth + 1,
335  inputs.length, monitor);
336  }
337  return name;
338  }
339  catch (Exception e) {
340  return null;
341  }
342  }

413  private String getNameFromOffset(Program program, long offset, Varnode input, boolean isClass,
414  boolean isMethod) {
415  String name;
416  Address address = getAddressInProgram(program, offset);
417  if (address == null) {
418  return null;
419  }
420  MemoryBlock block = program.getMemory().getBlock(address);
421  if (block == null) {
422  return null;
423  }
424 
425  if (isIvarBlock(block) || isObjcConstBlock(block)) {
426  name = getIvarName(program, address);
427  }
428  else if (isMessageRefsBlock(block)) {
429  name = getFixupMethodName(program, address);
430  }
431  else if (isCFStringBlock(block)) {
432  name = getCFString(program, address);
433  if (name != null) {
434  if (name.startsWith("\\")) {
435  name = "\\" + name;
436  }
437  name = "\"" + name + "\"";
438  }
439  }
440  else if (isDataBlock(block)) {
441  name = getDataName(program, address);
442  if (name != null) {
443  if (name.startsWith("\\")) {
444  name = "\\" + name;
445  }
446  name = "\"" + name + "\"";
447  }
448  }
449  else {
450  Data nameData = program.getListing().getDataAt(address);
451  if (nameData == null) {
452  Function function = program.getListing().getFunctionAt(address);
453  if (function != null && !function.getName().contains("_objc_msgSend")) {
454  DataType returnType = function.getReturnType();
455  name = returnType.getName();
456  return name;
457  }
458  return null;
459  }
460  name = getNameFromData(program, input, isClass, isMethod, address, nameData);
461  }
462  return name;
463  }

486  private String getNameFromData(Program program, Varnode input, boolean isClass,
487  boolean isMethod, Address address, Data nameData) {
488  long offset;
489  String name;
490  if (!nameData.isDefined()) {
491  name = getLabelFromUndefinedData(program, address);
492  }
493  else {
494  Object dataValue = nameData.getValue();
495  if (dataValue instanceof String) {
496  name = (String) dataValue;
497  if (!isClass && !isMethod) {
498  name = "\"" + name + "\"";
499  }
500  }
501  else if (dataValue instanceof Address) {
502  offset = ((Address) dataValue).getOffset();
503  name = getNameFromOffset(program, offset, input, isClass, isMethod);
504  }
505  else {
506  name = getClassName(program, address);
507  if (name == null) {
508  name = getValueAtAddress(program, address);
509  }
510  }
511  }
512  return name;
513  }

566  private String getIvarName(Program program, Address address) {
567  Listing listing = program.getListing();
568  Data ivarOffset = listing.getDataAt(address);
569  ReferenceIterator references = ivarOffset.getReferenceIteratorTo();
570 
571  while (references.hasNext()) {
572  Reference reference = references.next();
573  Address fromAddress = reference.getFromAddress();
574  MemoryBlock block = program.getMemory().getBlock(fromAddress);
575  if (!block.getName().equals(ObjectiveC2_Constants.OBJC2_CONST)) {
576  continue;
577  }
578  Data ivarList = listing.getDataContaining(fromAddress);
579  int numComponents = ivarList.getNumComponents();
580  for (int i = 2; i < numComponents; i++) {
581  Data ivarData = ivarList.getComponent(i);
582  Address ivarAddress = ivarData.getAddress();
583  if (ivarAddress.equals(fromAddress)) {
584  Data typeDataPointer = ivarData.getComponent(2);
585  Object typeAddress = typeDataPointer.getValue();
586  String className = null;
587  if (typeAddress instanceof Address) {
588  Data typeData = listing.getDataAt((Address) typeAddress);
589  className = getClassNameFromIvarData(typeData);
590  }
591  if (className == null) {
592  className = "";
593  }
594 
595  Data nameDataPointer = ivarData.getComponent(1);
596  Object nameAddress = nameDataPointer.getValue();
597  if (nameAddress instanceof Address) {
598  Data nameData = listing.getDataAt((Address) nameAddress);
599  String ivarName = (String) nameData.getValue();
600  return className + "::" + ivarName;
601  }
602  }
603  }
604  }
605  return null;
606  }

687  private String getClassName(Program program, Address toAddress) {
688  try {
689  boolean is32Bit = false;
690 
691  int pointerSize = program.getAddressFactory().getDefaultAddressSpace().getPointerSize();
692  if (pointerSize * 8 == 32) {
693  is32Bit = true;
694  }
695  int nameIndex = is32Bit ? 4 : 3;
696 
697  Data classData = program.getListing().getDefinedDataAt(toAddress);
698 
699  Data classRwPointerData = classData.getComponent(4);
700  Address classRwPointerAddress = (Address) classRwPointerData.getValue();
701 
702  Memory memory = program.getMemory();
703  MemoryBlock block = memory.getBlock(classRwPointerAddress);
704 
705  if (!isObjcConstBlock(block)) {
706  return null;
707  }
708 
709  Data classRwData = program.getListing().getDefinedDataAt(classRwPointerAddress);
710  Data classNamePointerData = classRwData.getComponent(nameIndex);
711 
712  Address classNameAddress = (Address) classNamePointerData.getValue();

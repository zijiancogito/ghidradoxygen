173  private Set<Function> findFunctions(final Program program, ArrayList<Address> locations,
174  final TaskMonitor monitor) throws InterruptedException, Exception, CancelledException {
175 
176  GThreadPool pool = AutoAnalysisManager.getSharedAnalsysThreadPool();
177  FindFunctionCallback callback = new FindFunctionCallback(program);
178 
179  // @formatter:off
180  ConcurrentQ<Address, Function> queue = new ConcurrentQBuilder<Address, Function>()
181  .setCollectResults(true)
182  .setThreadPool(pool)
183  .setMonitor(monitor)
184  .build(callback);
185  // @formatter:on
186 
187  for (Address location : locations) {
188  queue.add(location);
189  }
190 
191  Collection<QResult<Address, Function>> results = queue.waitForResults();
192 
193  Set<Function> functions = new HashSet<>();
194  for (QResult<Address, Function> result : results) {
195  Function function = result.getResult();
196  if (function == null) {
197  continue;
198  }
199  // kids, don't do thunks
200  if (function.isThunk()) {
201  if (function.hasNoReturn()) {
202  hitNonReturningFunction = true;
203  }
204  continue;
205  }
206  functions.add(function);
207  }
208 
209  return functions;
210  }

212  private ArrayList<Address> findLocations(Program program, AddressSetView set,
213  TaskMonitor monitor) throws CancelledException {
214 
215  monitor.setMessage("Finding function locations...");
216  long total = set.getNumAddresses();
217  monitor.initialize(total);
218  Address maxAddress = set.getMaxAddress();
219  ArrayList<Address> locations = new ArrayList<>();
220  Listing list = program.getListing();
221  InstructionIterator iterator = list.getInstructions(set, true);
222  while (iterator.hasNext()) {
223  monitor.checkCanceled();
224 
225  Instruction instruction = iterator.next();
226  FlowType flowType = instruction.getFlowType();
227  if (!flowType.isJump() || !flowType.isComputed()) {
228  if (!isCallFixup(program, instruction, flowType)) {
229  continue;
230  }
231  }
232 
233  Address address = instruction.getMinAddress();
234  locations.add(address);
235  long remaining = (maxAddress.getOffset() - address.getOffset()) + address.getSize();
236  monitor.setProgress(total - remaining);
237  }
238 
239  return locations;
240  }

242  private boolean isCallFixup(Program program, Instruction instr, FlowType flowType) {
243  if (!flowType.isCall()) {
244  return false;
245  }
246 
247  Reference[] referencesFrom = instr.getReferencesFrom();
248  for (Reference reference : referencesFrom) {
249  if (reference.getReferenceType().isCall()) {
250  Function func =
251  program.getFunctionManager().getFunctionAt(reference.getToAddress());
252  if (func != null && func.getCallFixup() != null) {
253  return true;
254  }
255  }
256  }
257  return false;
258  }

279

282  for (Reference element : referencesFrom) {

284  if (referenceType.isComputed()) {

334  private boolean resolveComputableFlow(Address location, TaskMonitor monitor,
335  BasicBlockModel basicBlockModel) throws CancelledException {
336 
337  // get the basic block
338  //
339  // NOTE: Assumption, the decompiler won't get the switch if there is no guard
340 
341  final CodeBlock jumpBlockAt =
342  basicBlockModel.getFirstCodeBlockContaining(location, monitor);
343  // If the jump target can has a computable target with only the instructions in the basic block it is found in
344  // then it isn't a switch statment
345  //
346  // NOTE: Assumption, we have found all flows leading to the switch that might split the basic block
347 
348  final AtomicInteger foundCount = new AtomicInteger(0);
349  SymbolicPropogator prop = new SymbolicPropogator(program);
350  prop.flowConstants(jumpBlockAt.getFirstStartAddress(), jumpBlockAt,
351  new ContextEvaluatorAdapter() {
352  @Override
353  public boolean evaluateReference(VarnodeContext context, Instruction instr,
354  int pcodeop, Address address, int size, RefType refType) {
355  // go ahead and place the reference, since it is a constant.
356  if (refType.isComputed() && refType.isFlow() &&
357  program.getMemory().contains(address)) {
358  propogateCodeMode(context, address);
359  foundCount.incrementAndGet();
360  return true;
361  }
362  return false;
363  }
364 
365  private void propogateCodeMode(VarnodeContext context, Address addr) {
366  // get CodeModeRegister and flow it to destination, if it is set here
367 
368  if (isaModeSwitchRegister == null) {
369  return;
370  }
371  BigInteger value = context.getValue(isaModeSwitchRegister, false);
372  if (value != null && program.getListing().getInstructionAt(addr) == null) {
373  try {
374  program.getProgramContext().setValue(isaModeRegister, addr, addr,
375  value);
376  }
377  catch (ContextChangeException e) {
378  // ignore
379  }
380  }
381  }
382  }, false, monitor);
383 
384  // only found one reference
385  // NOTE: This is overly protective, since we restricted the constant following to the block
386 
387  return foundCount.get() == 1;
388  }
389  }

384  private static ClangToken findClosestAddressedToken(ClangToken token) {
385  if (token == null) {
386  return null;
387  }
388  if (token.getMinAddress() != null) {
389  return token;
390  }
391 
392  List<ClangToken> lineTokens = token.getLineParent().getAllTokens();
393  int tokIndex = -1;
394  int lastIndex = lineTokens.size() - 1;
395  for (int i = 0; i <= lastIndex; i++) {
396  if (lineTokens.get(i) == token) {
397  tokIndex = i;
398  break;
399  }
400  }
401 
402  if (tokIndex != -1) {
403  // look to the right
404  for (int i = tokIndex + 1; i <= lastIndex; i++) {
405  ClangToken tok = lineTokens.get(i);
406  if (tok.getMinAddress() != null) {
407  return tok;
408  }
409  }
410  // look to the left
411  for (int i = tokIndex - 1; i >= 0; i--) {
412  ClangToken tok = lineTokens.get(i);
413  if (tok.getMinAddress() != null) {
414  return tok;
415  }
416  }
417  }
418  return null;

36  @Override
37  public ConditionResult doRun(TaskMonitor monitor) {
38  int threshold = MIN_NUM_FUNCS_DEFAULT;
39  ConditionStatus status = ConditionStatus.Passed;
40  StringBuilder warnings = new StringBuilder();
41  int number;
42  number = checkNumberAnalyzed(program, monitor);
43  if (number < threshold) {
44  status = ConditionStatus.Warning;
45  warnings.append(program.getDomainFile().getName() +
46  " number of functions with signatures from the decompiler parameter id analyzer = " +
47  number + "\n");
48  }
49  return new ConditionResult(status, warnings.toString());
50  }

52  private static int checkNumberAnalyzed(Program prog, TaskMonitor monitor) {
53  FunctionIterator funcIter = prog.getFunctionManager().getFunctions(true);
54  int numFuncsWithParameterID = 0;
55 
56  monitor.setIndeterminate(false);
57  monitor.initialize(prog.getFunctionManager().getFunctionCount());
58  while (funcIter.hasNext() && !monitor.isCancelled()) {
59  monitor.incrementProgress(1);
60  Function func = funcIter.next();
61  Address address = func.getEntryPoint();
62  Instruction inst = prog.getListing().getInstructionAt(address);
63 
64  if (inst != null) {
65  final SourceType signatureSource = func.getSignatureSource();
66  if (signatureSource == SourceType.ANALYSIS) {
67  ++numFuncsWithParameterID;
68  }
69  }
70  }
71 
72  return numFuncsWithParameterID;
73  }

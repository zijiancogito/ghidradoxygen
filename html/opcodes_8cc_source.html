65 OpCode get_opcode(const string &nm)
66 
67 {
68  int4 min = 1; // Don't include BLANK
69  int4 max = CPUI_MAX-1;
70  int4 cur,ind;
71 
72  while(min <= max) { // Binary search
73  cur = (min + max)/2;
74  ind = opcode_indices[cur]; // Get opcode in cur's sort slot
75  if (opcode_name[ind] < nm)
76  min = cur + 1; // Everything equal or below cur is less
77  else if (opcode_name[ind] > nm)
78  max = cur - 1; // Everything equal or above cur is greater
79  else
80  return (OpCode)ind; // Found the match
81  }
82  return (OpCode)0; // Name isn't an op
83 }

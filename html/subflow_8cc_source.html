42 SubvariableFlow::ReplaceVarnode *SubvariableFlow::setReplacement(Varnode *vn,uintb mask,bool &inworklist)
43 
44 { // Mark
45  ReplaceVarnode *res;
46  if (vn->isMark()) { // Already seen before
47  map<Varnode *,ReplaceVarnode>::iterator iter;
48  iter = varmap.find(vn);
49  res = &(*iter).second;
50  inworklist = false;
51  if (res->mask != mask)
52  return (ReplaceVarnode *)0;
53  return res;
54  }
55 
56  if (vn->isConstant()) {
57  inworklist = false;
58  if (sextrestrictions) { // Check that -vn- is a sign extension
59  uintb cval = vn->getOffset();
60  uintb smallval = cval & mask; // From its logical size
61  uintb sextval = sign_extend(smallval,flowsize,vn->getSize());// to its fullsize
62  if (sextval != cval)
63  return (ReplaceVarnode *)0;
64  }
65  return addConstant((ReplaceOp *)0,mask,0,vn->getOffset());
66  }
67 
68  if (vn->isFree())
69  return (ReplaceVarnode *)0; // Abort
70 
71  if (vn->isAddrForce() && (vn->getSize() != flowsize))
72  return (ReplaceVarnode *)0;
73 
74  if (sextrestrictions) {
75  if (vn->getSize() != flowsize) {
76  if ((!aggressive)&& vn->isInput()) return (ReplaceVarnode *)0; // Cannot assume input is sign extended
77  if (vn->isPersist()) return (ReplaceVarnode *)0;
78  }
79  if (vn->isTypeLock()) {
80  if (vn->getType()->getSize() != flowsize)
81  return (ReplaceVarnode *)0;
82  }
83  }
84  else {
85  if (bitsize >= 8) { // Not a flag
86  // If the logical variable is not a flag, don't consider the case where multiple variables
87  // are packed into a single location, i.e. always consider it a single variable
88  if ((!aggressive)&&((vn->getConsume()&~mask)!=0)) // If there is any use of value outside of the logical variable
89  return (ReplaceVarnode *)0; // This probably means the whole thing is a variable, i.e. quit
90  if (vn->isTypeLock()) {
91  int4 sz = vn->getType()->getSize();
92  if (sz != flowsize)
93  return (ReplaceVarnode *)0;
94  }
95  }
96 
97  if (vn->isInput()) { // Must be careful with inputs
98  // Inputs must come in from the right register/memory
99  if (bitsize < 8) return (ReplaceVarnode *)0; // Dont create input flag
100  if ((mask&1)==0) return (ReplaceVarnode *)0; // Dont create unique input
101  // Its extremely important that the code (above) which doesn't allow packed variables be applied
102  // or the mechanisms we use for inputs will give us spurious temporary inputs
103  }
104  }
105 
106  res = & varmap[ vn ];
107  vn->setMark();
108  res->vn = vn;
109  res->replacement = (Varnode *)0;
110  res->mask = mask;
111  res->def = (ReplaceOp *)0;
112  inworklist = true;
113  // Check if vn already represents the logical variable being traced
114  if (vn->getSize() == flowsize) {
115  if (mask == calc_mask(flowsize)) {
116  inworklist = false;
117  res->replacement = vn;
118  }
119  else if (mask == 1) {
120  if ((vn->isWritten())&&(vn->getDef()->isBoolOutput())) {
121  inworklist = false;
122  res->replacement = vn;
123  }
124  }
125  }
126  return res;
127 }

197 bool SubvariableFlow::tryReturnPull(PcodeOp *op,ReplaceVarnode *rvn,int4 slot)
198 
199 { // -rvn- flows to CPUI_RETURN (is probably being returned by function), determine if we can trim varnode to logical size
200  if (slot == 0) return false; // Don't deal with actual return address container
201  if (fd->getFuncProto().isOutputLocked()) return false;
202 
203  if (!returnsTraversed) {
204  // If we plan to truncate the size of a return variable, we need to propagate the logical size to any other
205  // return variables so that there can still be a single return value type for the function

216  if (rep == (ReplaceVarnode *)0)
217  return false;
218  if (inworklist)
219  worklist.push_back(rep);
220  }
221  returnsTraversed = true;
222  }
223  patchlist.push_back(PatchRecord());
224  patchlist.back().type = 2;
225  patchlist.back().pullop = op;
226  patchlist.back().in1 = rvn;
227  patchlist.back().slot = slot;

253 bool SubvariableFlow::traceForward(ReplaceVarnode *rvn)
254 
255 { // Try to trace logical variable through descendant varnodes
256  // updating list/map of replace_ops and replace_varnodes
257  // and the worklist
258  ReplaceOp *rop;
259  PcodeOp *op;
260  Varnode *outvn;
261  int4 slot;
262  int4 sa;
263  uintb newmask;
264  bool booldir;
265  int4 dcount = 0;
266  int4 hcount = 0;
267 
268  list<PcodeOp *>::const_iterator iter,enditer;
269  iter = rvn->vn->beginDescend();
270  enditer = rvn->vn->endDescend();
271  while(iter != enditer) {
272  op = *iter++;
273  outvn = op->getOut();
274  if ((outvn!=(Varnode *)0)&&(outvn->isMark()))
275  continue;
276  dcount += 1; // Count this descendant
277  slot = op->getSlot(rvn->vn);
278  switch(op->code()) {
279  case CPUI_COPY:
280  case CPUI_MULTIEQUAL:
281  case CPUI_INT_NEGATE:
282  case CPUI_INT_XOR:
283  rop = createOpDown(op->code(),op->numInput(),op,rvn,slot);
284  if (!createLink(rop,rvn->mask,-1,outvn)) return false;
285  hcount += 1; // Dealt with this descendant
286  break;
287  case CPUI_INT_OR:
288  if (doesOrSet(op,rvn->mask)!=-1) break; // Subvar set to 1s, truncate flow
289  rop = createOpDown(CPUI_INT_OR,2,op,rvn,slot);
290  if (!createLink(rop,rvn->mask,-1,outvn)) return false;
291  hcount += 1; // Dealt with this descendant
292  break;
293  case CPUI_INT_AND:
294  if ((op->getIn(1)->isConstant())&&(op->getIn(1)->getOffset() == rvn->mask)) {
295  if ((outvn->getSize() == flowsize)&&((rvn->mask & 1)!=0)) {
296  addTerminalPatch(op,rvn);
297  hcount += 1; // Dealt with this descendant
298  break;
299  }
300  // Is the small variable getting zero padded into something that is fully consumed
301  if ((!aggressive)&&(calc_mask(outvn->getSize()) == outvn->getConsume())) {
302  addSuggestedPatch(rvn,op,-1);
303  hcount += 1; // Dealt with this descendant
304  break;
305  }
306  }
307  if (doesAndClear(op,rvn->mask)!=-1) break; // Subvar set to zero, truncate flow
308  rop = createOpDown(CPUI_INT_AND,2,op,rvn,slot);
309  if (!createLink(rop,rvn->mask,-1,outvn)) return false;
310  hcount += 1; // Dealt with this descendant
311  break;
312  case CPUI_INT_ZEXT:
313  case CPUI_INT_SEXT:
314  rop = createOpDown(CPUI_COPY,1,op,rvn,0);
315  if (!createLink(rop,rvn->mask,-1,outvn)) return false;
316  hcount += 1; // Dealt with this descendant
317  break;
318  case CPUI_INT_MULT:
319  if ((rvn->mask & 1)==0)
320  return false; // Cannot account for carry
321  sa = leastsigbit_set(op->getIn(1-slot)->getNZMask());
322  sa &= ~7; // Should be nearest multiple of 8
323  if (bitsize + sa > 8*rvn->vn->getSize()) return false;
324  rop = createOpDown(CPUI_INT_MULT,2,op,rvn,slot);
325  if (!createLink(rop,rvn->mask<<sa,-1,outvn)) return false;
326  hcount += 1;
327  break;
328  case CPUI_INT_ADD:
329  if ((rvn->mask & 1)==0)
330  return false; // Cannot account for carry
331  rop = createOpDown(CPUI_INT_ADD,2,op,rvn,slot);
332  if (!createLink(rop,rvn->mask,-1,outvn)) return false;
333  hcount += 1; // Dealt with this descendant
334  break;
335  case CPUI_INT_LEFT:
336  if (slot == 1) { // Logical flow is into shift amount
337  if ((rvn->mask & 1)==0) return false; // Cannot account for effect of extraneous bits
338  if (bitsize <8) return false;
339  // Its possible that truncating to the logical value could have an effect, if there were non-zero bits
340  // being truncated. Non-zero bits here would mean the shift-amount was very large (>255), indicating the
341  // the result was undefined
342  addTerminalPatchSameOp(op,rvn,slot);
343  hcount += 1;
344  break;
345  }
346  if (!op->getIn(1)->isConstant()) return false; // Dynamic shift
347  sa = (int4)op->getIn(1)->getOffset();
348  newmask = (rvn->mask << sa) & calc_mask( outvn->getSize() );
349  if (newmask == 0) break; // Subvar is cleared, truncate flow
350  if (rvn->mask != (newmask >> sa)) return false; // subvar is clipped
351  // Is the small variable getting zero padded into something that is fully consumed
352  if (((rvn->mask & 1)!=0)&&(sa + bitsize == 8*outvn->getSize())
353  &&(calc_mask(outvn->getSize()) == outvn->getConsume())) {
354  addSuggestedPatch(rvn,op,sa);
355  hcount += 1;
356  break;
357  }
358  rop = createOpDown(CPUI_COPY,1,op,rvn,0);
359  if (!createLink(rop,newmask,-1,outvn)) return false;
360  hcount += 1; // Dealt with this descendant
361  break;
362  case CPUI_INT_RIGHT:
363  case CPUI_INT_SRIGHT:
364  if (slot == 1) { // Logical flow is into shift amount
365  if ((rvn->mask & 1)==0) return false; // Cannot account for effect of extraneous bits
366  if (bitsize <8) return false;
367  addTerminalPatchSameOp(op,rvn,slot);
368  hcount += 1;
369  break;
370  }
371  if (!op->getIn(1)->isConstant()) return false;
372  sa = (int4)op->getIn(1)->getOffset();
373  newmask = rvn->mask >> sa;
374  if (newmask == 0) {
375  if (op->code()==CPUI_INT_RIGHT) break; // subvar is set to zero, truncate flow
376  return false;
377  }
378  if (rvn->mask != (newmask << sa)) return false;
379  if ((outvn->getSize()==flowsize)&&((newmask&1)==1)&&
380  (op->getIn(0)->getNZMask()==rvn->mask)) {
381  addTerminalPatch(op,rvn);
382  hcount += 1; // Dealt with this descendant
383  break;
384  }
385  // Is the small variable getting zero padded into something that is fully consumed
386  if (((newmask&1)==1)&&(sa + bitsize == 8*outvn->getSize())
387  &&(calc_mask(outvn->getSize()) == outvn->getConsume())) {
388  addSuggestedPatch(rvn,op,0);
389  hcount += 1;
390  break;
391  }
392  rop = createOpDown(CPUI_COPY,1,op,rvn,0);
393  if (!createLink(rop,newmask,-1,outvn)) return false;
394  hcount += 1; // Dealt with this descendant
395  break;
396  case CPUI_SUBPIECE:
397  sa = (int4)op->getIn(1)->getOffset() * 8;
398  newmask = (rvn->mask >> sa) & calc_mask(outvn->getSize());
399  if (newmask == 0) break; // subvar is set to zero, truncate flow
400  if (rvn->mask != (newmask << sa)) return false;
401  if (((newmask & 1)!=0)&&(outvn->getSize()==flowsize)) {
402  addTerminalPatch(op,rvn);
403  hcount += 1; // Dealt with this descendant
404  break;
405  }
406  rop = createOpDown(CPUI_COPY,1,op,rvn,0);
407  if (!createLink(rop,newmask,-1,outvn)) return false;
408  hcount += 1; // Dealt with this descendant
409  break;
410  case CPUI_PIECE:
411  if (rvn->vn == op->getIn(0))
412  newmask = rvn->mask << (8*op->getIn(1)->getSize());
413  else
414  newmask = rvn->mask;
415  rop = createOpDown(CPUI_COPY,1,op,rvn,0);
416  if (!createLink(rop,newmask,-1,outvn)) return false;
417  hcount += 1; // Dealt with this descendant
418  break;
419  case CPUI_INT_LESS:
420  case CPUI_INT_LESSEQUAL:
421  outvn = op->getIn(1-slot); // The OTHER side of the comparison
422  if ((!aggressive)&&(((rvn->vn->getNZMask() | rvn->mask) != rvn->mask)))
423  return false; // Everything but logical variable must definitely be zero (unless we are aggressive)
424  if (outvn->isConstant()) {
425  if ((rvn->mask | outvn->getOffset()) != rvn->mask)
426  return false; // Must compare only bits of logical variable
427  }
428  else {
429  if ((!aggressive)&&(((rvn->mask | outvn->getNZMask()) != rvn->mask))) // unused bits of otherside must be zero
430  return false;
431  }
432  if (!createCompareBridge(op,rvn,slot,outvn))
433  return false;
434  hcount += 1; // Dealt with this descendant
435  break;
436  case CPUI_INT_NOTEQUAL:
437  case CPUI_INT_EQUAL:
438  outvn = op->getIn(1-slot); // The OTHER side of the comparison
439  if (bitsize != 1) {
440  if ((!aggressive)&&(((rvn->vn->getNZMask() | rvn->mask) != rvn->mask)))
441  return false; // Everything but logical variable must definitely be zero (unless we are aggressive)
442  if (outvn->isConstant()) {
443  if ((rvn->mask | outvn->getOffset()) != rvn->mask)
444  return false; // Not comparing to just bits of the logical variable
445  }
446  else {
447  if ((!aggressive)&&(((rvn->mask | outvn->getNZMask()) != rvn->mask))) // unused bits must be zero
448  return false;
449  }
450  if (!createCompareBridge(op,rvn,slot,outvn))
451  return false;
452  }
453  else { // Movement of boolean variables
454  if (!outvn->isConstant()) return false;
455  newmask = rvn->vn->getNZMask();
456  if (newmask != rvn->mask) return false;
457  if (op->getIn(1-slot)->getOffset() == (uintb)0)
458  booldir = true;
459  else if (op->getIn(1-slot)->getOffset() == newmask)
460  booldir = false;
461  else
462  return false;
463  if (op->code() == CPUI_INT_EQUAL)
464  booldir = !booldir;
465  if (booldir)
466  addTerminalPatch(op,rvn);
467  else {
468  rop = createOpDown(CPUI_BOOL_NEGATE,1,op,rvn,0);
469  createNewOut(rop,(uintb)1);
470  addTerminalPatch(op,rop->output);
471  }
472  }
473  hcount += 1; // Dealt with this descendant
474  break;
475  case CPUI_CALL:
476  case CPUI_CALLIND:
477  if (!aggressive) return false;
478  if (!tryCallPull(op,rvn,slot)) return false;
479  hcount += 1; // Dealt with this descendant
480  break;
481  case CPUI_RETURN:
482  if (!aggressive) return false;
483  if (!tryReturnPull(op,rvn,slot)) return false;
484  hcount += 1;
485  break;
486  case CPUI_BOOL_NEGATE:
487  case CPUI_BOOL_AND:
488  case CPUI_BOOL_OR:
489  case CPUI_BOOL_XOR:
490  if (bitsize != 1) return false;
491  if (rvn->mask != 1) return false;
492  addBooleanPatch(op,rvn,slot);
493  break;
494  case CPUI_CBRANCH:
495  if ((bitsize != 1)||(slot != 1)) return false;
496  if (rvn->mask != 1) return false;
497  addBooleanPatch(op,rvn,1);
498  hcount += 1;
499  break;
500  default:
501  return false;
502  }
503  }
504  if (dcount != hcount) {
505  // Must account for all descendants of an input
506  if (rvn->vn->isInput()) return false;
507  }
508  return true;
509 }

511 bool SubvariableFlow::traceBackward(ReplaceVarnode *rvn)
512 
513 { // Trace backward through defining op one level
514  // Update worklist, varmap, and oplist
515  // return false if the trace is aborted
516  PcodeOp *op = rvn->vn->getDef();
517  if (op == (PcodeOp *)0) return true; // If vn is input
518  int4 sa;
519  uintb newmask;
520  ReplaceOp *rop;
521 
522  switch(op->code()) {
523  case CPUI_COPY:
524  case CPUI_MULTIEQUAL:
525  case CPUI_INT_NEGATE:
526  case CPUI_INT_XOR:
527  rop = createOp(op->code(),op->numInput(),rvn);
528  for(int4 i=0;i<op->numInput();++i)
529  if (!createLink(rop,rvn->mask,i,op->getIn(i))) // Same inputs and mask
530  return false;
531  return true;
532  case CPUI_INT_AND:
533  sa = doesAndClear(op,rvn->mask);
534  if (sa != -1) {
535  rop = createOp(CPUI_COPY,1,rvn);
536  addConstant(rop,rvn->mask,0,op->getIn(sa)->getOffset());
537  }
538  else {
539  rop = createOp(CPUI_INT_AND,2,rvn);
540  if (!createLink(rop,rvn->mask,0,op->getIn(0))) return false;
541  if (!createLink(rop,rvn->mask,1,op->getIn(1))) return false;
542  }
543  return true;
544  case CPUI_INT_OR:
545  sa = doesOrSet(op,rvn->mask);
546  if (sa != -1) {
547  rop = createOp(CPUI_COPY,1,rvn);
548  addConstant(rop,rvn->mask,0,op->getIn(sa)->getOffset());
549  }
550  else {
551  rop = createOp(CPUI_INT_OR,2,rvn);
552  if (!createLink(rop,rvn->mask,0,op->getIn(0))) return false;
553  if (!createLink(rop,rvn->mask,1,op->getIn(1))) return false;
554  }
555  return true;
556  case CPUI_INT_ZEXT:
557  case CPUI_INT_SEXT:
558  if ((rvn->mask & calc_mask(op->getIn(0)->getSize())) != rvn->mask)
559  break; // Check if subvariable comes through extension
560  rop = createOp(CPUI_COPY,1,rvn);
561  if (!createLink(rop,rvn->mask,0,op->getIn(0))) return false;
562  return true;
563  case CPUI_INT_ADD:
564  if ((rvn->mask & 1)==0)
565  break; // Cannot account for carry
566  if (rvn->mask == (uintb)1)
567  rop = createOp(CPUI_INT_XOR,2,rvn); // Single bit add
568  else
569  rop = createOp(CPUI_INT_ADD,2,rvn);
570  if (!createLink(rop,rvn->mask,0,op->getIn(0))) return false;
571  if (!createLink(rop,rvn->mask,1,op->getIn(1))) return false;
572  return true;
573  case CPUI_INT_LEFT:
574  if (!op->getIn(1)->isConstant()) break; // Dynamic shift
575  sa = (int4)op->getIn(1)->getOffset();
576  newmask = rvn->mask >> sa; // What mask looks like before shift
577  if (newmask == 0) { // Subvariable filled with shifted zero
578  rop = createOp(CPUI_COPY,1,rvn);
579  addConstant(rop,rvn->mask,0,(uintb)0);
580  return true;
581  }
582  if ((newmask<<sa) != rvn->mask)
583  break; // subvariable is truncated by shift
584  rop = createOp(CPUI_COPY,1,rvn);
585  if (!createLink(rop,newmask,0,op->getIn(0))) return false;
586  return true;
587  case CPUI_INT_RIGHT:
588  if (!op->getIn(1)->isConstant()) break; // Dynamic shift
589  sa = (int4)op->getIn(1)->getOffset();
590  newmask = (rvn->mask << sa) & calc_mask(op->getIn(0)->getSize());
591  if (newmask == 0) { // Subvariable filled with shifted zero
592  rop = createOp(CPUI_COPY,1,rvn);
593  addConstant(rop,rvn->mask,0,(uintb)0);
594  return true;
595  }
596  if ((newmask>>sa) != rvn->mask)
597  break; // subvariable is truncated by shift
598  rop = createOp(CPUI_COPY,1,rvn);
599  if (!createLink(rop,newmask,0,op->getIn(0))) return false;
600  return true;
601  case CPUI_INT_SRIGHT:
602  if (!op->getIn(1)->isConstant()) break; // Dynamic shift
603  sa = (int4)op->getIn(1)->getOffset();
604  newmask = (rvn->mask << sa) & calc_mask(op->getIn(0)->getSize());
605  if ((newmask>>sa) != rvn->mask)
606  break; // subvariable is truncated by shift
607  rop = createOp(CPUI_COPY,1,rvn);
608  if (!createLink(rop,newmask,0,op->getIn(0))) return false;
609  return true;
610  case CPUI_INT_MULT:
611  sa = leastsigbit_set(rvn->mask);
612  if (sa!=0) {
613  int4 sa2 = leastsigbit_set(op->getIn(1)->getNZMask());
614  if (sa2 < sa) return false; // Cannot deal with carries into logical multiply
615  newmask = rvn->mask >> sa;
616  rop = createOp(CPUI_INT_MULT,2,rvn);
617  if (!createLink(rop,newmask,0,op->getIn(0))) return false;
618  if (!createLink(rop,rvn->mask,1,op->getIn(1))) return false;
619  }
620  else {
621  if (rvn->mask == (uintb)1)
622  rop = createOp(CPUI_INT_AND,2,rvn); // Single bit multiply
623  else
624  rop = createOp(CPUI_INT_MULT,2,rvn);
625  if (!createLink(rop,rvn->mask,0,op->getIn(0))) return false;
626  if (!createLink(rop,rvn->mask,1,op->getIn(1))) return false;
627  }
628  return true;
629  case CPUI_SUBPIECE:
630  sa = (int4)op->getIn(1)->getOffset() * 8;
631  newmask = rvn->mask << sa;
632  rop = createOp(CPUI_COPY,1,rvn);
633  if (!createLink(rop,newmask,0,op->getIn(0))) return false;
634  return true;
635  case CPUI_PIECE:
636  if ((rvn->mask & calc_mask(op->getIn(1)->getSize()))==rvn->mask) {
637  rop = createOp(CPUI_COPY,1,rvn);
638  if (!createLink(rop,rvn->mask,0,op->getIn(1))) return false;
639  return true;
640  }
641  sa = op->getIn(1)->getSize() * 8;
642  newmask = rvn->mask>>sa;
643  if (newmask<<sa == rvn->mask) {
644  rop = createOp(CPUI_COPY,1,rvn);
645  if (!createLink(rop,newmask,0,op->getIn(0))) return false;
646  return true;
647  }
648  break;
649  case CPUI_INDIRECT:
650  if (aggressive) {
651  if (tryCallReturnPull(op,rvn))
652  return true;
653  }
654  break;
655  case CPUI_INT_EQUAL:
656  case CPUI_INT_NOTEQUAL:
657  case CPUI_INT_SLESS:
658  case CPUI_INT_SLESSEQUAL:
659  case CPUI_INT_LESS:
660  case CPUI_INT_LESSEQUAL:
661  case CPUI_INT_CARRY:
662  case CPUI_INT_SCARRY:
663  case CPUI_INT_SBORROW:
664  case CPUI_BOOL_NEGATE:
665  case CPUI_BOOL_XOR:
666  case CPUI_BOOL_AND:
667  case CPUI_BOOL_OR:
668  case CPUI_FLOAT_EQUAL:
669  case CPUI_FLOAT_NOTEQUAL:
670  case CPUI_FLOAT_LESSEQUAL:
671  case CPUI_FLOAT_NAN:
672  // Mask won't be 1, because setReplacement takes care of it
673  if ((rvn->mask&1)==1) break; // Not normal variable flow
674  // Variable is filled with zero
675  rop = createOp(CPUI_COPY,1,rvn);
676  addConstant(rop,rvn->mask,0,(uintb)0);
677  return true;
678  default:
679  break; // Everything else we abort
680  }
681 
682  return false;
683 }

685 bool SubvariableFlow::traceForwardSext(ReplaceVarnode *rvn)
686 
687 { // Try to trace the logical variable through descendant varnodes, updating map of replacement ops and varnodes
688  // We assume (and check) that the logical variable has always been sign extended (sextstate) into its container
689  ReplaceOp *rop;
690  PcodeOp *op;
691  Varnode *outvn;
692  int4 slot;
693  int4 dcount = 0;
694  int4 hcount = 0;
695 
696  list<PcodeOp *>::const_iterator iter,enditer;
697  iter = rvn->vn->beginDescend();
698  enditer = rvn->vn->endDescend();
699  while(iter != enditer) {
700  op = *iter++;
701  outvn = op->getOut();
702  if ((outvn!=(Varnode *)0)&&(outvn->isMark()))
703  continue;
704  dcount += 1; // Count this descendant
705  slot = op->getSlot(rvn->vn);
706  switch(op->code()) {
707  case CPUI_COPY:
708  case CPUI_MULTIEQUAL:
709  case CPUI_INT_NEGATE:
710  case CPUI_INT_XOR:
711  case CPUI_INT_OR:
712  case CPUI_INT_AND:
713  rop = createOpDown(op->code(),op->numInput(),op,rvn,slot);
714  if (!createLink(rop,rvn->mask,-1,outvn)) return false;
715  hcount += 1;
716  break;
717  case CPUI_INT_SEXT: // extended logical variable into even larger container
718  rop = createOpDown(CPUI_COPY,1,op,rvn,0);
719  if (!createLink(rop,rvn->mask,-1,outvn)) return false;
720  hcount += 1;
721  break;
722  case CPUI_INT_SRIGHT:
723  if (!op->getIn(1)->isConstant()) return false; // Right now we only deal with constant shifts
724  rop = createOpDown(CPUI_INT_SRIGHT,2,op,rvn,0);
725  if (!createLink(rop,rvn->mask,-1,outvn)) return false; // Keep the same mask size
726  addConstant(rop,calc_mask(op->getIn(1)->getSize()),1,op->getIn(1)->getOffset()); // Preserve the shift amount
727  hcount += 1;
728  break;
729  case CPUI_SUBPIECE:
730  if (op->getIn(1)->getOffset() != 0) return false; // Only allow proper truncation
731  if (outvn->getSize() > flowsize) return false;
732  if (outvn->getSize() == flowsize)
733  addTerminalPatch(op,rvn); // Termination of flow, convert SUBPIECE to COPY
734  else
735  addTerminalPatchSameOp(op,rvn,0); // Termination of flow, SUBPIECE truncates even more
736  hcount +=1;
737  break;
738  case CPUI_INT_LESS: // Unsigned comparisons are equivalent at the 2 sizes on sign extended values
739  case CPUI_INT_LESSEQUAL:
740  case CPUI_INT_SLESS:
741  case CPUI_INT_SLESSEQUAL:
742  case CPUI_INT_EQUAL: // Everything works if both sides are sign extended
743  case CPUI_INT_NOTEQUAL:
744  outvn = op->getIn(1-slot); // The OTHER side of the comparison
745  if (!createCompareBridge(op,rvn,slot,outvn)) return false;
746  hcount += 1;
747  break;
748  case CPUI_CALL:
749  case CPUI_CALLIND:
750  if (!aggressive) return false;
751  if (!tryCallPull(op,rvn,slot)) return false;
752  hcount += 1; // Dealt with this descendant
753  break;
754  case CPUI_RETURN:
755  if (!aggressive) return false;
756  if (!tryReturnPull(op,rvn,slot)) return false;
757  hcount += 1;
758  break;
759  default:
760  return false;
761  }
762  }
763  if (dcount != hcount) {
764  // Must account for all descendants of an input
765  if (rvn->vn->isInput()) return false;
766  }
767  return true;
768 }

770 bool SubvariableFlow::traceBackwardSext(ReplaceVarnode *rvn)
771 
772 { // Trace backward through defining op, one level, update worklist and map
773  // We assume (and check) that the logical variable has always been sign extended (sextstate) into its container
774  PcodeOp *op = rvn->vn->getDef();
775  if (op == (PcodeOp *)0) return true; // If vn is input
776  ReplaceOp *rop;
777 
778  switch(op->code()) {
779  case CPUI_COPY:
780  case CPUI_MULTIEQUAL:
781  case CPUI_INT_NEGATE:
782  case CPUI_INT_XOR:
783  case CPUI_INT_AND:
784  case CPUI_INT_OR:
785  rop = createOp(op->code(),op->numInput(),rvn);
786  for(int4 i=0;i<op->numInput();++i)
787  if (!createLink(rop,rvn->mask,i,op->getIn(i))) // Same inputs and mask
788  return false;
789  return true;
790  case CPUI_INT_SEXT:
791  if (flowsize != op->getIn(0)->getSize()) return false;
792  rop = createOp(CPUI_COPY,1,rvn);
793  if (!createLink(rop,rvn->mask,0,op->getIn(0))) return false;
794  return true;
795  case CPUI_INT_SRIGHT:
796  // A sign-extended logical value is arithmetically right-shifted
797  // we can replace with the logical value, keeping the same shift amount
798  if (!op->getIn(1)->isConstant()) return false;
799  rop = createOp(CPUI_INT_SRIGHT,2,rvn);
800  if (!createLink(rop,rvn->mask,0,op->getIn(0))) return false; // Keep the same mask
801  if (rop->input.size()==1)
802  addConstant(rop,calc_mask(op->getIn(1)->getSize()),1,op->getIn(1)->getOffset()); // Preserve the shift amount
803  return true;
804  case CPUI_INDIRECT:
805  if (aggressive) {
806  if (tryCallReturnPull(op,rvn))
807  return true;
808  }
809  break;
810  default:
811  break;
812  }
813  return false;

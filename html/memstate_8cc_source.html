91 void MemoryBank::getPage(uintb addr,uint1 *res,int4 skip,int4 size) const
92 
93 { // Default implementation just iterates using find
94  // but could be optimized
95  uintb ptraddr = addr + skip;
96  uintb endaddr = ptraddr + size;
97  uintb startalign = ptraddr & ~((uintb)(wordsize-1));
98  uintb endalign = endaddr & ~((uintb)(wordsize-1));
99  if ((endaddr & ((uintb)(wordsize-1))) != 0)
100  endalign += wordsize;
101 
102  uintb curval;
103  bool bswap = ((HOST_ENDIAN==1) != space->isBigEndian());
104  uint1 *ptr;
105  do {
106  curval = find(startalign);
107  if (bswap)
108  curval = byte_swap(curval,wordsize);
109  ptr = (uint1 *)&curval;
110  int4 sz = wordsize;
111  if (startalign < addr) {
112  ptr += (addr-startalign);
113  sz = wordsize - (addr-startalign);
114  }
115  if (startalign + wordsize > endaddr)
116  sz -= (startalign + wordsize -endaddr);
117  memcpy(res,ptr,sz);
118  res += sz;
119  startalign += wordsize;
120  } while(startalign != endalign);
121 }

180 void MemoryBank::setValue(uintb offset,int4 size,uintb val)
181 
182 {
183  uintb alignmask = (uintb)(wordsize-1);
184  uintb ind = offset & (~alignmask);
185  int4 skip = offset & alignmask;
186  int4 size1 = wordsize-skip;
187  int4 size2;
188  int4 gap;
189  uintb val1,val2;
190 
191  if (size > size1) { // We have spill over
192  size2 = size - size1;
193  val1 = find(ind);
194  val2 = find(ind+wordsize);
195  gap = wordsize - size2;
196  }
197  else {
198  if (size == wordsize) {
199  insert(ind,val);
200  return;
201  }
202  val1 = find(ind);
203  val2 = 0;
204  gap = size1-size;
205  size1 = size;
206  size2 = 0;
207  }
208 
209  skip = skip * 8; // Convert from byte skip to bit skip
210  gap = gap * 8; // Convert from byte to bits
211  if (space->isBigEndian()) {
212  if (size2 == 0) {
213  val1 &= ~(calc_mask(size1)<<gap);
214  val1 |= val << gap;
215  insert(ind,val1);
216  }
217  else {
218  val1 &= (~((uintb)0)) << 8*size1;
219  val1 |= val >> 8*size2;
220  insert(ind,val1);
221  val2 &= (~((uintb)0)) >> 8*size2;
222  val2 |= val << gap;
223  insert(ind+wordsize,val2);
224  }
225  }
226  else {
227  if (size2 == 0) {
228  val1 &= ~(calc_mask(size1)<<skip);
229  val1 |= val << skip;
230  insert(ind,val1);
231  }
232  else {
233  val1 &= (~((uintb)0)) >> 8*size1;
234  val1 |= val << skip;
235  insert(ind,val1);
236  val2 &= (~((uintb)0)) << 8*size2;
237  val2 |= val >> 8*size1;
238  insert(ind+wordsize,val2);
239  }
240  }
241 }

250 uintb MemoryBank::getValue(uintb offset,int4 size) const
251 
252 {
253  uintb res;
254 
255  uintb alignmask = (uintb) (wordsize-1);
256  uintb ind = offset & (~alignmask);
257  int4 skip = offset & alignmask;
258  int4 size1 = wordsize-skip;
259  int4 size2;
260  int4 gap;
261  uintb val1,val2;
262  if (size > size1) { // We have spill over
263  size2 = size - size1;
264  val1 = find(ind);
265  val2 = find(ind+wordsize);
266  gap = wordsize - size2;
267  }
268  else {
269  val1 = find(ind);
270  val2 = 0;
271  if (size == wordsize)
272  return val1;
273  gap = size1-size;
274  size1 = size;
275  size2 = 0;
276  }
277 
278  if (space->isBigEndian()) {
279  if (size2 == 0)
280  res = val1>>(8*gap);
281  else
282  res = (val1<<(8*size2)) | (val2 >> (8*gap));
283  }
284  else {
285  if (size2 == 0)
286  res = val1 >> (skip*8);
287  else
288  res = (val1>>(skip*8)) | (val2<<(size1*8) );
289  }
290  res &= (uintb)calc_mask(size);
291  return res;
292 }

363 uintb MemoryImage::find(uintb addr) const
364 
365 { // Assume that -addr- is word aligned
366  uintb res = 0; // Make sure all bytes start as 0, as load may not fill all bytes
367  AddrSpace *spc = getSpace();
368  try {
369  uint1 *ptr = (uint1 *)&res;
370  ptr += (HOST_ENDIAN==1) ? (sizeof(uintb) - getWordSize()) : 0;
371  loader->loadFill(ptr,getWordSize(),Address(spc,addr));
372  } catch(DataUnavailError &err) {
373  // Pages not mapped in the load image, are assumed to be zero
374  res = 0;
375  }
376  if ((HOST_ENDIAN==1) != spc->isBigEndian())
377  res = byte_swap(res,getWordSize());
378  return res;
379 }

549 void MemoryHashOverlay::insert(uintb addr,uintb val)
550 
551 {
552  int4 size = address.size();
553  uintb offset = (addr>>alignshift) % size;
554  for(int4 i=0;i<size;++i) {
555  if (address[offset] == addr) { // Address has been seen before
556  value[offset] = val; // Replace old value
557  return;
558  }
559  else if (address[offset] == (uintb)0xBADBEEF) { // Address not seen before
560  address[offset] = addr; // Claim this hash slot
561  value[offset] = val; // Set value
562  return;
563  }
564  offset = (offset + collideskip) % size;
565  }
566  throw LowlevelError("Memory state hash_table is full");
567 }

574 
575 { // Find address in hash-table, or return find from underlying memory

577  uintb offset = (addr>>alignshift) % size;

581  else if (address[offset] == 0xBADBEEF) // Address not seen before

584  }
585 
586  // We didn't find the address in the hashtable
587  if (underlie == (MemoryBank *)0)
588  return (uintb)0;
589  return underlie->find(addr);
590 }

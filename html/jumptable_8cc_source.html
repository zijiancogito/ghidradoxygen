45 void LoadTable::collapseTable(vector<LoadTable> &table)
46 
47 { // Assuming -table- is sorted, collapse sequential LoadTable entries into single LoadTable entries
48  if (table.empty()) return;
49  vector<LoadTable>::iterator iter,lastiter;
50  int4 count = 1;
51  iter = table.begin();
52  lastiter = iter;
53  Address nextaddr = (*iter).addr + (*iter).size * (*iter).num;
54  ++iter;
55  for(;iter!=table.end();++iter) {
56  if (( (*iter).addr == nextaddr ) && ((*iter).size == (*lastiter).size)) {
57  (*lastiter).num += (*iter).num;
58  nextaddr = (*iter).addr + (*iter).size * (*iter).num;
59  }
60  else if (( nextaddr < (*iter).addr )|| ((*iter).size != (*lastiter).size)) {
61  // Starting a new table
62  lastiter++;
63  *lastiter = *iter;
64  nextaddr = (*iter).addr + (*iter).size * (*iter).num;
65  count += 1;
66  }
67  }
68  table.resize(count,LoadTable(nextaddr,0));
69 }

164 uintb EmulateFunction::emulatePath(uintb val,const PathMeld &pathMeld,
165  PcodeOp *startop,Varnode *startvn)
166 {
167  uint4 i;
168  for(i=0;i<pathMeld.numOps();++i)
169  if (pathMeld.getOp(i) == startop) break;
170  if (startop->code() == CPUI_MULTIEQUAL) { // If we start on a MULTIEQUAL
171  int4 j;
172  for(j=0;j<startop->numInput();++j) { // Is our startvn one of the branches
173  if (startop->getIn(j) == startvn)
174  break;
175  }
176  if ((j == startop->numInput())||(i==0)) // If not, we can't continue;
177  throw LowlevelError("Cannot start jumptable emulation with unresolved MULTIEQUAL");
178  // If the startvn was a branch of the MULTIEQUAL, emulate as if we just came from that branch
179  startvn = startop->getOut(); // So the output of the MULTIEQUAL is the new startvn (as if a COPY from old startvn)
180  i -= 1; // Move to the next instruction to be executed
181  startop = pathMeld.getOp(i);
182  }
183  if (i==pathMeld.numOps())
184  throw LowlevelError("Bad jumptable emulation");
185  if (!startvn->isConstant())
186  setVarnodeValue(startvn,val);
187  while(i>0) {
188  PcodeOp *curop = pathMeld.getOp(i);
189  --i;
190  setCurrentOp( curop );
191  try {
192  executeCurrentOp();
193  }
194  catch(DataUnavailError &err) {
195  ostringstream msg;
196  msg << "Could not emulate address calculation at " << curop->getAddr();
197  throw LowlevelError(msg.str());
198  }
199  }
200  Varnode *invn = pathMeld.getOp(0)->getIn(0);
201  return getVarnodeValue(invn);
202 }

421 
422 { // Back up constant normalized value -outvn- to unnormalized
423  Varnode *curvn = outvn;
424  PcodeOp *op;
425  TypeOp *top;
426  int4 slot;
427 
428  while(curvn != invn) {
429  op = curvn->getDef();
430  top = op->getOpcode();
431  for(slot=0;slot<op->numInput();++slot) // Find first non-constant input
432  if (!op->getIn(slot)->isConstant()) break;
433  if (op->getEvalType() == PcodeOp::binary) {
434  const Address &addr(op->getIn(1-slot)->getAddr());
435  uintb otherval;

437  MemoryImage mem(addr.getSpace(),4,1024,fd->getArch()->loader);
438  otherval = mem.getValue(addr.getOffset(),op->getIn(1-slot)->getSize());

442  output = top->recoverInputBinary(slot,op->getOut()->getSize(),output,

444  curvn = op->getIn(slot);

447  output = top->recoverInputUnary(op->getOut()->getSize(),output,op->getIn(slot)->getSize());

456 void JumpBasic::findDeterminingVarnodes(PcodeOp *op,int4 slot)
457 
458 {
459  vector<PcodeOp *> path;
460  vector<int4> slotpath;
461  PcodeOp *curop;
462  Varnode *curvn;
463  bool firstpoint = false; // Have not seen likely switch variable yet
464 
465  path.push_back(op);
466  slotpath.push_back(slot);
467 
468  do { // Traverse through tree of inputs to final address
469  curop = path.back();
470  curvn = curop->getIn(slotpath.back());
471  if (isprune(curvn)) { // Here is a node of the tree
472  if (ispoint(curvn)) { // Is it a possible switch variable
473  if (!firstpoint) { // If it is the first possible
474  pathMeld.set(path,slotpath); // Take the current path as the result
475  firstpoint = true;
476  }
477  else // If we have already seen at least one possible
478  pathMeld.meld(path,slotpath);
479  }
480 
481  slotpath.back() += 1;
482  while(slotpath.back() >= path.back()->numInput()) {
483  path.pop_back();
484  slotpath.pop_back();
485  if (path.empty()) break;
486  slotpath.back() += 1;
487  }
488  }
489  else { // This varnode is not pruned
490  path.push_back(curvn->getDef());
491  slotpath.push_back(0);
492  }
493  } while(path.size() > 1);
494  if (pathMeld.empty()) { // Never found a likely point, which means that
495  // it looks like the address is uniquely determined
496  // but the constants/readonlys haven't been collapsed
497  pathMeld.set(op,op->getIn(slot));
498  }

520 int4 GuardRecord::valueMatch(Varnode *vn2,Varnode *baseVn2,int4 bitsPreserved2) const
521

523  // Return 1, if -vn- and -vn2- are clearly the same value
524  // Return 2, if -vn- and -vn2- are clearly the same value, pending no writes beteen the def of -vn- and -vn2-
525  if (vn == vn2) return 1; // Same varnode, same value
526  PcodeOp *loadOp,*loadOp2;
527  if (bitsPreserved == bitsPreserved2) { // Are the same number of bits being copied
528  if (baseVn == baseVn2) // Are bits being copied from same varnode
529  return 1; // If so, values are the same
530  loadOp = baseVn->getDef(); // Otherwise check if different base varnodes hold same value
531  loadOp2 = baseVn2->getDef();
532  }
533  else {
534  loadOp = vn->getDef(); // Check if different varnodes hold same value
535  loadOp2 = vn2->getDef();
536  }
537  if (loadOp == (PcodeOp *)0) return 0;
538  if (loadOp2 == (PcodeOp *)0) return 0;
539  if (oneOffMatch(loadOp,loadOp2) == 1) // Check for simple duplicate calculations
540  return 1;

542  if (loadOp2->code() != CPUI_LOAD) return 0;

548  if (!ptr2->isWritten()) return 0;

556  if (!constvn2->isConstant()) return 0;

564 { // Return 1 if -op1- and -op2- produce exactly the same value, 0 if otherwise
565  // (one value is allowed to be the zero extension of the other)
566  if (op1->code() != op2->code())

578  if (op2->getIn(0) != op1->getIn(0)) return 0;

588 Varnode *GuardRecord::quasiCopy(Varnode *vn,int4 &bitsPreserved,bool noWholeValue)
589 
590 {
591  Varnode *origVn = vn;
592  bitsPreserved = mostsigbit_set(vn->getNZMask()) + 1;
593  if (bitsPreserved == 0) return vn;
594  uintb mask = 1;
595  mask <<= bitsPreserved;
596  mask -= 1;
597  PcodeOp *op = vn->getDef();
598  Varnode *constVn;
599  while(op != (PcodeOp *)0) {
600  if (noWholeValue && (vn != origVn)) {
601  uintb inputMask = vn->getNZMask() | mask;
602  if (mask == inputMask)
603  return origVn; // vn contains whole value, -noWholeValue- indicates we should abort
604  }
605  switch(op->code()) {
606  case CPUI_COPY:
607  vn = op->getIn(0);
608  op = vn->getDef();
609  break;
610  case CPUI_INT_AND:
611  constVn = op->getIn(1);
612  if (constVn->isConstant() && constVn->getOffset() == mask) {
613  vn = op->getIn(0);
614  op = vn->getDef();
615  }
616  else
617  op = (PcodeOp *)0;
618  break;
619  case CPUI_INT_OR:
620  constVn = op->getIn(1);
621  if (constVn->isConstant() && ((constVn->getOffset() | mask) == (constVn->getOffset() ^ mask))) {
622  vn = op->getIn(0);
623  op = vn->getDef();
624  }
625  else
626  op = (PcodeOp *)0;
627  break;
628  case CPUI_INT_SEXT:
629  case CPUI_INT_ZEXT:
630  if (op->getIn(0)->getSize() * 8 >= bitsPreserved) {
631  vn = op->getIn(0);
632  op = vn->getDef();
633  }
634  else
635  op = (PcodeOp *)0;
636  break;
637  case CPUI_PIECE:
638  if (op->getIn(1)->getSize() * 8 >= bitsPreserved) {
639  vn = op->getIn(1);
640  op = vn->getDef();
641  }
642  else
643  op = (PcodeOp *)0;
644  break;
645  case CPUI_SUBPIECE:
646  constVn = op->getIn(1);
647  if (constVn->isConstant() && constVn->getOffset() == 0) {
648  vn = op->getIn(0);
649  op = vn->getDef();
650  }
651  else
652  op = (PcodeOp *)0;
653  break;
654  default:
655  op = (PcodeOp *)0;
656  break;
657  }
658  }
659  return vn;

691 int4 PathMeld::meldOps(const vector<PcodeOp *> &path,int4 cutOff,const vector<int4> &parentMap)
692 
693 { // Meld old ops (opMeld) with new ops (path), updating rootVn with new commonVn order
694  // Ops should remain in (reverse) execution order
695  // Ops that split (use a vn not in intersection) and do not rejoin (have a predecessor vn in intersection)
696  // get cut
697  // If splitting ops arent can't be ordered with the existing meld, we get a new cut point
698 
699  // First update opMeld.rootVn with new intersection information
700  for(int4 i=0;i<opMeld.size();++i) {
701  int4 pos = parentMap[opMeld[i].rootVn];
702  if (pos == -1) {
703  opMeld[i].op = (PcodeOp *)0; // Op split but did not rejoin
704  }
705  else
706  opMeld[i].rootVn = pos; // New index
707  }
708 
709  // Do a merge sort, keeping ops in execution order
710  vector<RootedOp> newMeld;
711  int4 curRoot = -1;
712  int4 meldPos = 0; // Ops moved from old opMeld into newMeld
713  const BlockBasic *lastBlock = (const BlockBasic *)0;
714  for(int4 i=0;i<cutOff;++i) {
715  PcodeOp *op = path[i]; // Current op in the new path
716  PcodeOp *curOp = (PcodeOp *)0;
717  while(meldPos < opMeld.size()) {
718  PcodeOp *trialOp = opMeld[meldPos].op; // Current op in the old opMeld
719  if (trialOp == (PcodeOp *)0) {
720  meldPos += 1;
721  continue;
722  }
723  if (trialOp->getParent() != op->getParent()) {
724  if (op->getParent() == lastBlock) {
725  curOp = (PcodeOp *)0; // op comes AFTER trialOp
726  break;
727  }
728  else if (trialOp->getParent() != lastBlock) {
729  // Both trialOp and op come from different blocks that are not the lastBlock
730  int4 res = opMeld[meldPos].rootVn; // Force truncatePath at (and above) this op
731 
732  // Found a new cut point
733  opMeld = newMeld; // Take what we've melded so far
734  return res; // return the new cutpoint
735  }
736  }
737  else if (trialOp->getSeqNum().getOrder() <= op->getSeqNum().getOrder()) {
738  curOp = trialOp; // op is equal to or comes later than trialOp
739  break;
740  }
741  lastBlock = trialOp->getParent();
742  newMeld.push_back(opMeld[meldPos]); // Current old op moved into newMeld
743  curRoot = opMeld[meldPos].rootVn;
744  meldPos += 1;
745  }
746  if (curOp == op) {
747  newMeld.push_back(opMeld[meldPos]);
748  curRoot = opMeld[meldPos].rootVn;
749  meldPos += 1;
750  }
751  else {
752  newMeld.push_back(RootedOp(op,curRoot));
753  }
754  lastBlock = op->getParent();
755  }
756  opMeld = newMeld;
757  return -1;
758 }

853 void JumpBasic::analyzeGuards(BlockBasic *bl,int4 pathout)
854 
855 { // Analyze each CBRANCH leading up to -bl- switch.
856  // (if pathout>=0, also analyze the CBRANCH in -bl- that chooses this path)
857  // Analyze the range restrictions on the various variables which allow
858  // control flow to pass through the CBRANCHs to the switch.
859  // Make note of all these restrictions in the guard list
860  // For later determination of the correct switch variable.
861  int4 i,j,indpath;
862  int4 maxbranch = 2; // Maximum number of CBRANCHs to consider
863  int4 maxpullback = 2;
864  bool usenzmask = (jumptable->getStage() == 0);
865 
866  selectguards.clear();
867  BlockBasic *prevbl;
868  Varnode *vn;
869 
870  for(i=0;i<maxbranch;++i) {
871  if ((pathout>=0)&&(bl->sizeOut()==2)) {
872  prevbl = bl;
873  bl = (BlockBasic *)prevbl->getOut(pathout);
874  indpath = pathout;
875  pathout = -1;
876  }
877  else {
878  pathout = -1; // Make sure not to use pathout next time around
879  for(;;) {
880  if (bl->sizeIn() != 1) return; // Assume only 1 path to switch
881  prevbl = (BlockBasic *)bl->getIn(0);
882  if (prevbl->sizeOut() != 1) break; // Is it possible to deviate from switch path in this block
883  bl = prevbl; // If not, back up to next block
884  }
885  indpath = bl->getInRevIndex(0);
886  }
887  PcodeOp *cbranch = prevbl->lastOp();
888  if ((cbranch==(PcodeOp *)0)||(cbranch->code() != CPUI_CBRANCH))
889  break;
890  bool toswitchval = (indpath == 1);
891  if (cbranch->isBooleanFlip())
892  toswitchval = !toswitchval;
893  bl = prevbl;
894  vn = cbranch->getIn(1);
895  CircleRange rng(toswitchval);
896 
897  // The boolean variable could conceivably be the switch variable
898  int4 indpathstore = prevbl->getFlipPath() ? 1-indpath : indpath;
899  selectguards.push_back(GuardRecord(cbranch,indpathstore,rng,vn));
900  for(j=0;j<maxpullback;++j) {
901  Varnode *markup; // Throw away markup information
902  if (!vn->isWritten()) break;
903  vn = rng.pullBack(vn->getDef(),&markup,usenzmask);
904  if (vn == (Varnode *)0) break;
905  if (rng.isEmpty()) break;
906  selectguards.push_back(GuardRecord(cbranch,indpathstore,rng,vn));
907  }
908  }

911 void JumpBasic::calcRange(Varnode *vn,CircleRange &rng) const
912 
913 { // For a putative switch variable, calculate the range of
914  // possible values that variable can have AT the switch
915  // by using the precalculated guard ranges.
916 
917  // Get an initial range, based on the size/type of -vn-
918  if (vn->isConstant())
919  rng = CircleRange(vn->getOffset(),vn->getSize());
920  else if (vn->isWritten() && vn->getDef()->isBoolOutput())
921  rng = CircleRange(0,1,1); // Only 0 or 1 possible
922  else { // Should we go ahead and use nzmask in all cases?
923  uintb mask = calc_mask(vn->getSize());
924  if (vn->isWritten()) {
925  PcodeOp *andop = vn->getDef();
926  if (andop->code() == CPUI_INT_AND) {
927  Varnode *constvn = andop->getIn(1);
928  if (constvn->isConstant()) {
929  mask = coveringmask( constvn->getOffset() );
930  }
931  }
932  }
933  rng = CircleRange(0,mask,mask);
934  setStride(vn,rng);
935  }
936 
937  // Intersect any guard ranges which apply to -vn-
938  int4 bitsPreserved;
939  Varnode *baseVn = GuardRecord::quasiCopy(vn, bitsPreserved, true);
940  vector<GuardRecord>::const_iterator iter;
941  for(iter=selectguards.begin();iter!=selectguards.end();++iter) {
942  const GuardRecord &guard( *iter );
943  int4 matchval = guard.valueMatch(vn,baseVn,bitsPreserved);
944  // if (matchval == 2) TODO: we need to check for aliases
945  if (matchval==0) continue;
946  if (rng.intersect(guard.getRange())!=0) continue;
947  }
948 
949  // It may be an assumption that the switch value is positive
950  // in which case the guard might not check for it. If the
951  // size is too big, we try only positive values
952  if (rng.getSize() > 0x10000) {
953  CircleRange positive(0,rng.getMask()>>1,rng.getMask());
954  positive.intersect(rng);
955  if (!positive.isEmpty())
956  rng = positive;
957  }
958 }

1034 bool JumpBasic::foldInOneGuard(Funcdata *fd,GuardRecord &guard,JumpTable *jump)
1035 
1036 {
1037  PcodeOp *cbranch = guard.getBranch();
1038  int4 indpath = guard.getPath(); // Get stored path to indirect block
1039  BlockBasic *cbranchblock = cbranch->getParent();
1040  if (cbranchblock->getFlipPath()) // Based on whether out branches have been flipped
1041  indpath = 1 - indpath; // get actual path to indirect block
1042  BlockBasic *guardtarget = (BlockBasic *)cbranchblock->getOut(1-indpath);
1043  bool change = false;
1044  uint4 pos;
1045 
1046  // Its possible the guard branch has been converted between the switch recovery and now
1047  if (cbranchblock->sizeOut() != 2) return false; // In which case, we can't fold it in
1048  BlockBasic *switchbl = jump->getIndirectOp()->getParent();
1049  for(pos=0;pos<switchbl->sizeOut();++pos)
1050  if (switchbl->getOut(pos) == guardtarget) break;
1051  if (pos == switchbl->sizeOut()) {
1052  if (BlockBasic::noInterveningStatement(cbranch,indpath,switchbl->lastOp())) {
1053  // Adjust tables and control flow graph
1054  // for new jumptable destination
1055  jump->addBlockToSwitch(guardtarget,0xBAD1ABE1);
1056  jump->setMostCommonIndex(jump->numEntries()-1);
1057  fd->pushBranch(cbranchblock,1-indpath,switchbl);
1058  guard.clear();
1059  change = true;
1060  }
1061  }
1062  else {
1063  // We should probably check that there are no intervening
1064  // statements between the guard and the switch. But the
1065  // fact that the guard target is also a switch target
1066  // is a good indicator that there are none
1067  uintb val = ((indpath==0)!=(cbranch->isBooleanFlip())) ? 0 : 1;
1068  fd->opSetInput(cbranch,fd->newConstant(cbranch->getIn(0)->getSize(),val),1);
1069  jump->setMostCommonBlock(pos); // A guard branch must be most common
1070  guard.clear();
1071  change = true;
1072  }
1073  return change;
1074 }

1325 bool JumpBasic2::recoverModel(Funcdata *fd,PcodeOp *indop,uint4 matchsize,uint4 maxtablesize)
1326 
1327 { // Try to recover a jumptable using the second model
1328  // Basically there is a guard on the main switch variable,
1329  // Along one path, an intermediate value is set to a default constant.
1330  // Along the other path, the intermediate value results in a straight line calculation from the switch var
1331  // The two-pathed intermediate value comes together in a MULTIEQUAL, and there is a straightline
1332  // calculation to the BRANCHIND
1333 
1334  // We piggy back on the partial calculation from the basic model to see if we have the MULTIEQUAL
1335  Varnode *othervn = (Varnode *)0;
1336  PcodeOp *copyop = (PcodeOp *)0;
1337  uintb extravalue = 0;
1338  Varnode *joinvn = extravn; // extravn should be set to as far back as model 1 could trace
1339  if (joinvn == (Varnode *)0) return false;
1340  if (!joinvn->isWritten()) return false;
1341  PcodeOp *multiop = joinvn->getDef();
1342  if (multiop->code() != CPUI_MULTIEQUAL) return false;
1343  if (multiop->numInput() != 2) return false; // Must be exactly 2 paths
1344  // Search for a constant along one of the paths
1345  int4 path;
1346  for(path=0;path<2;++path) {
1347  Varnode *vn = multiop->getIn(path);
1348  if (!vn->isWritten()) continue;

1350  if (copyop->code() != CPUI_COPY) continue;

1352  if (othervn->isConstant()) {
1353  extravalue = othervn->getOffset();

1355  }
1356  }
1357  if (path == 2) return false;

1360  JumpValuesRangeDefault *jdef = new JumpValuesRangeDefault();

1363  jdef->setDefaultVn(joinvn); // Emulate the default calculation from the join point

1373  varnodeIndex += origPathMeld.numCommonVarnode(); // index is pushed up by the append

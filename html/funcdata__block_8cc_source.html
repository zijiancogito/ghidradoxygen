82 void Funcdata::pushMultiequals(BlockBasic *bb)
83 
84 {
85  BlockBasic *outblock;
86  PcodeOp *origop,*replaceop;
87  Varnode *origvn,*replacevn;
88  list<PcodeOp *>::iterator iter;
89  list<PcodeOp *>::const_iterator citer;
90 
91  if (bb->sizeOut()==0) return;
92  if (bb->sizeOut()>1)
93  warningHeader("push_multiequal on block with multiple outputs");
94  outblock = (BlockBasic *) bb->getOut(0); // Take first output block. If this is a
95  // donothing block, it is the only output block
96  for(iter=bb->beginOp();iter!=bb->endOp();++iter) {
97  origop = *iter;
98  if (origop->code() != CPUI_MULTIEQUAL) continue;
99  origvn = origop->getOut();
100  if (origvn->hasNoDescend()) continue;
101  bool needreplace = false;
102  bool neednewunique = false;
103  for(citer=origvn->beginDescend();citer!=origvn->endDescend();++citer) {
104  PcodeOp *op = *citer;
105  if ((op->code()==CPUI_MULTIEQUAL)&&(op->getParent()==outblock)) {
106  if ((origvn->getAddr() == op->getOut()->getAddr())&&origvn->isAddrTied())
107  // If origvn is addrtied and feeds into a MULTIEQUAL at same address in outblock
108  // Then any use of origvn beyond outblock that did not go thru this MULTIEQUAL must have
109  // propagated through some other register. So we make the new MULTIEQUAL write to a unique register
110  neednewunique = true;
111  continue;
112  }
113  needreplace = true;
114  break;
115  }
116  if (!needreplace) continue;
117  // Construct artificial MULTIEQUAL
118  vector<Varnode *> branches;
119  if (neednewunique)
120  replacevn = newUnique(origvn->getSize());
121  else
122  replacevn = newVarnode(origvn->getSize(),origvn->getAddr());
123  for(int4 i=0;i<outblock->sizeIn();++i) {
124  if (outblock->getIn(i) == bb)
125  branches.push_back(origvn);
126  else
127  branches.push_back( replacevn );
128 
129  // In this situation there are other blocks "beyond" outblock which read
130  // origvn defined in bb, but there are other blocks falling into outblock
131  // Assuming the only out of bb is outblock, all heritages of origvn must
132  // come through outblock. Thus any alternate ins to outblock must be
133  // dominated by bb. So the artificial MULTIEQUAL we construct must have
134  // all inputs be origvn
135  }
136  replaceop = newOp(branches.size(),outblock->getStart());
137  opSetOpcode(replaceop,CPUI_MULTIEQUAL);
138  opSetOutput(replaceop,replacevn);
139  opSetAllInput(replaceop,branches);
140  opInsertBegin(replaceop,outblock);
141 
142  // Replace obsolete origvn with replacevn
143  int4 i;
144  int4 outblock_ind = bb->getOutRevIndex(0);
145  list<PcodeOp *>::iterator titer = origvn->descend.begin();
146  while(titer != origvn->descend.end()) {
147  PcodeOp *op = *titer++;
148  i = op->getSlot(origvn);
149  // Do not replace MULTIEQUAL references in the same block
150  // as replaceop. These are patched by block_remove
151  if ((op->code()==CPUI_MULTIEQUAL)&&(op->getParent()==outblock)&&(i==outblock_ind))
152  continue;
153  opSetInput(op,replacevn,i);
154  }
155  }
156 }

239 void Funcdata::blockRemoveInternal(BlockBasic *bb,bool unreachable)
240 
241 {
242  BlockBasic *bbout;
243  Varnode *deadvn;
244  PcodeOp *op,*deadop;
245  list<PcodeOp *>::iterator iter;
246  int4 i,j,blocknum;
247  bool desc_warning;
248 
249  op = bb->lastOp();
250  if ((op != (PcodeOp *)0)&&(op->code() == CPUI_BRANCHIND)) {
251  JumpTable *jt = findJumpTable(op);
252  if (jt != (JumpTable *)0)
253  removeJumpTable(jt);
254  }
255  if (!unreachable) {
256  pushMultiequals(bb); // Make sure data flow is preserved
257 
258  for(i=0;i<bb->sizeOut();++i) {
259  bbout = (BlockBasic *) bb->getOut(i);
260  if (bbout->isDead()) continue;
261  blocknum = bbout->getInIndex(bb); // Get index of bb into bbout
262  for(iter=bbout->beginOp();iter!=bbout->endOp();++iter) {
263  op = *iter;
264  if (op->code() != CPUI_MULTIEQUAL) continue;
265  deadvn = op->getIn(blocknum);
266  opRemoveInput(op,blocknum); // Remove the deleted blocks branch
267  deadop = deadvn->getDef();
268  if ((deadvn->isWritten())&&(deadop->code()==CPUI_MULTIEQUAL)&&(deadop->getParent()==bb)) {
269  // Append new branches
270  for(j=0;j<bb->sizeIn();++j)
271  opInsertInput(op,deadop->getIn(j),op->numInput());
272  }
273  else {
274  for(j=0;j<bb->sizeIn();++j)
275  opInsertInput(op,deadvn,op->numInput()); // Otherwise make copies
276  }
277  opZeroMulti(op);
278  }
279  }
280  }
281  bblocks.removeFromFlow(bb);
282 
283  desc_warning = false;
284  iter = bb->beginOp();
285  while(iter!=bb->endOp()) { // Finally remove all the ops
286  op = *iter;
287  if (op->isAssignment()) { // op still has some descendants
288  deadvn = op->getOut();
289  if (unreachable) {
290  bool undef = descend2Undef(deadvn);
291  if (undef&&(!desc_warning)) { // Mark descendants as undefined
292  warningHeader("Creating undefined varnodes in (possibly) reachable block");
293  desc_warning = true; // Print the warning only once
294  }
295  }
296  if (descendantsOutside(deadvn)) // If any descendants outside of bb
297  throw LowlevelError("Deleting op with descendants\n");
298  }
299  if (op->isCall())
300  deleteCallSpecs(op);
301  iter++; // Increment iterator before unlinking
302  opDestroy(op); // No longer has descendants
303  }
304  bblocks.removeBlock(bb); // Remove the block altogether
305 }

331 bool Funcdata::removeUnreachableBlocks(bool issuewarning,bool checkexistence)
332 
333 {
334  vector<FlowBlock *> list;
335  uint4 i;
336 
337  if (checkexistence) { // Quick check for the existence of unreachable blocks
338  for(i=0;i<bblocks.getSize();++i) {
339  FlowBlock *blk = bblocks.getBlock(i);
340  if (blk->isEntryPoint()) continue; // Don't remove starting component
341  if (blk->getImmedDom() == (FlowBlock *)0) break;
342  }
343  if (i==bblocks.getSize()) return false;
344  }
345  else if (!hasUnreachableBlocks()) // Use cached check
346  return false;
347 
348  // There must be at least one unreachable block if we reach here
349 
350  for(i=0;i<bblocks.getSize();++i) // Find entry point
351  if (bblocks.getBlock(i)->isEntryPoint()) break;
352  bblocks.collectReachable(list,bblocks.getBlock(i),true); // Collect (un)reachable blocks
353 
354  for(int4 i=0;i<list.size();++i) {
355  list[i]->setDead();
356  if (issuewarning) {
357  ostringstream s;
358  BlockBasic *bb = (BlockBasic *)list[i];
359  s << "Removing unreachable block (";
360  s << bb->getStart().getSpace()->getName();
361  s << ',';
362  bb->getStart().printRaw(s);
363  s << ')';
364  warningHeader(s.str());
365  }
366  }
367  for(int4 i=0;i<list.size();++i) {
368  BlockBasic *bb = (BlockBasic *)list[i];
369  while(bb->sizeOut() > 0)
370  branchRemoveInternal(bb,0);
371  }
372  for(int4 i=0;i<list.size();++i) {
373  BlockBasic *bb = (BlockBasic *)list[i];
374  blockRemoveInternal(bb,true);
375  }
376  structureReset();
377  return true;

478 int4 Funcdata::stageJumpTable(JumpTable *jt,PcodeOp *op,FlowInfo *flow)
479 
480 {
481  PcodeOp *partop = (PcodeOp *)0;
482  string oldactname;
483 
484  ostringstream s1;
485  s1 << name << "@@jump@";
486  op->getAddr().printRaw(s1);
487 
488  Funcdata partial(s1.str(),localmap->getParent(),baseaddr);
489  partial.flags |= jumptablerecovery_on; // Mark that this Funcdata object is dedicated to jumptable recovery
490  partial.truncatedFlow(this,flow);
491 
492  partop = partial.findOp(op->getSeqNum());
493 
494  if ((partop==(PcodeOp *)0) ||
495  (partop->code() != CPUI_BRANCHIND)||
496  (partop->getAddr() != op->getAddr()))
497  throw LowlevelError("Error recovering jumptable: Bad partial clone");
498 
499  oldactname = glb->allacts.getCurrentName(); // Save off old action
500  glb->allacts.setCurrent("jumptable");
501  try {
502 #ifdef OPACTION_DEBUG
503  if (jtcallback != (void (*)(Funcdata &orig,Funcdata &fd))0)
504  (*jtcallback)(*this,partial); // Alternative reset/perform
505  else {
506 #endif
507  glb->allacts.getCurrent()->reset( partial );
508  glb->allacts.getCurrent()->perform( partial ); // Simplify the partial function
509 #ifdef OPACTION_DEBUG
510  }
511 #endif
512  glb->allacts.setCurrent(oldactname); // Restore old action
513  if (partop->isDead()) // Indirectop we were trying to recover was eliminated as dead code (unreachable)
514  return 0; // Return jumptable as
515  jt->setLoadCollect(flow->doesJumpRecord());
516  jt->setIndirectOp(partop);
517  if (jt->getStage()>0)
518  jt->recoverMultistage(&partial);
519  else
520  jt->recoverAddresses(&partial); // Analyze partial to recover jumptable addresses
521  }
522  catch(JumptableNotReachableError &err) {
523  glb->allacts.setCurrent(oldactname);
524  return 3;
525  }
526  catch(JumptableThunkError &err) {
527  glb->allacts.setCurrent(oldactname);
528  return 2;
529  }
530  catch(LowlevelError &err) {
531  glb->allacts.setCurrent(oldactname);
532  warning(err.explain,op->getAddr());
533  return 1;
534  }
535  return 0;
536 }

549 JumpTable *Funcdata::recoverJumpTable(PcodeOp *op,FlowInfo *flow,int4 &failuremode)
550 
551 {
552  JumpTable *jt;
553 
554  failuremode = 0;
555  jt = linkJumpTable(op); // Search for pre-existing jumptable
556  if (jt != (JumpTable *)0) {
557  if (!jt->isOverride()) {
558  if (jt->getStage() != 1)
559  return jt; // Previously calculated jumptable (NOT an override and NOT incomplete)
560  }
561  failuremode = stageJumpTable(jt,op,flow); // Recover based on override information
562  if (failuremode != 0)
563  return (JumpTable *)0;
564  jt->setIndirectOp(op); // Relink table back to original op
565  return jt;
566  }
567 
568  if ((flags & jumptablerecovery_dont)!=0)
569  return (JumpTable *)0; // Explicitly told not to recover jumptables
570  JumpTable trialjt(glb);
571  failuremode = stageJumpTable(&trialjt,op,flow);
572  if (failuremode != 0)
573  return (JumpTable *)0;
574  // if (trialjt.is_twostage())
575  // warning("Jumptable maybe incomplete. Second-stage recovery not implemented",trialjt.Opaddress());
576  jt = new JumpTable(&trialjt); // Make the jumptable permanent
577  jumpvec.push_back(jt);
578  jt->setIndirectOp(op); // Relink table back to original op
579  return jt;
580 }

821 void Funcdata::nodeSplitInputPatch(BlockBasic *b,BlockBasic *bprime,int4 inedge)
822 
823 {
824  list<PcodeOp *>::iterator biter,piter;
825  PcodeOp *bop,*pop;
826  Varnode *bvn,*pvn;
827  map<PcodeOp *,PcodeOp *> btop; // Map from b to bprime
828  vector<PcodeOp *> pind; // pop needing b input
829  vector<PcodeOp *> bind; // bop giving input
830  vector<int4> pslot; // slot within pop needing b input
831 
832  biter = b->beginOp();
833  piter = bprime->beginOp();
834 
835  while(piter != bprime->endOp()) {
836  bop = *biter;
837  pop = *piter;
838  btop[bop] = pop; // Establish mapping
839  if (bop->code() == CPUI_MULTIEQUAL) {
840  pop->setNumInputs(1); // One edge now goes into bprime
841  opSetOpcode(pop,CPUI_COPY);
842  opSetInput(pop,bop->getIn(inedge),0);
843  opRemoveInput(bop,inedge); // One edge is removed from b
844  if (bop->numInput() == 1)
845  opSetOpcode(bop,CPUI_COPY);
846  }
847  else if (bop->code() == CPUI_INDIRECT) {
848  throw LowlevelError("Can't handle INDIRECTs in nodesplit");
849  }
850  else if (bop->isCall()) {
851  throw LowlevelError("Can't handle CALLs in nodesplit");
852  }
853  else {
854  for(int4 i=0;i<pop->numInput();++i) {
855  bvn = bop->getIn(i);
856  if (bvn->isConstant())
857  pvn = newConstant(bvn->getSize(),bvn->getOffset());
858  else if (bvn->isAnnotation())
859  pvn = newCodeRef(bvn->getAddr());
860  else if (bvn->isFree())
861  throw LowlevelError("Can't handle free varnode in nodesplit");
862  else {
863  if (bvn->isWritten()) {
864  if (bvn->getDef()->getParent() == b) {
865  pind.push_back(pop); // Need a cross reference
866  bind.push_back(bvn->getDef());
867  pslot.push_back(i);
868  continue;
869  }
870  else
871  pvn = bvn;
872  }
873  else
874  pvn = bvn;
875  }
876  opSetInput(pop,pvn,i);
877  }
878  }
879  ++piter;
880  ++biter;
881  }
882 
883  for(int4 i=0;i<pind.size();++i) {
884  pop = pind[i];
885  PcodeOp *cross = btop[bind[i]];
886  opSetInput(pop,cross->getOut(),pslot[i]);
887  }
888 }

896 void Funcdata::nodeSplit(BlockBasic *b,int4 inedge)
897 
898 { // Split node b along inedge
899  if (b->sizeOut() != 0)
900  throw LowlevelError("Cannot (currently) nodesplit block with out flow");
901  if (b->sizeIn()<=1)
902  throw LowlevelError("Cannot nodesplit block with only 1 in edge");
903  for(int4 i=0;i<b->sizeIn();++i) {
904  if (b->getIn(i)->isMark())
905  throw LowlevelError("Cannot nodesplit block with redundant in edges");
906  b->setMark();
907  }
908  for(int4 i=0;i<b->sizeIn();++i)
909  b->clearMark();

911  // Create duplicate block

918  // We would need to patch outputs here for the more general

920  // any references not in b to varnodes defined in b
921  // need to have MULTIEQUALs defined in b's out blocks
922  // with edges coming from b and bprime
923  structureReset();

961 void Funcdata::spliceBlockBasic(BlockBasic *bl)
962 
963 {
964  BlockBasic *outbl = (BlockBasic *)0;
965  if (bl->sizeOut() == 1) {
966  outbl = (BlockBasic *)bl->getOut(0);
967  if (outbl->sizeIn() != 1)
968  outbl = (BlockBasic *)0;
969  }
970  if (outbl == (BlockBasic *)0)
971  throw LowlevelError("Cannot splice basic blocks");
972  // Remove any jump op at the end of -bl-
973  if (!bl->op.empty()) {
974  PcodeOp *jumpop = bl->op.back();
975  if (jumpop->isBranch())
976  opDestroy(jumpop);
977  }
978  if (!outbl->op.empty()) {
979  // Check for MULTIEQUALs
980  PcodeOp *firstop = outbl->op.front();
981  if (firstop->code() == CPUI_MULTIEQUAL)
982  throw LowlevelError("Splicing block with MULTIEQUAL");
983  firstop->clearFlag(PcodeOp::startbasic);
984  list<PcodeOp *>::iterator iter;
985  // Move ops into -bl-
986  for(iter=outbl->beginOp();iter!=outbl->endOp();++iter) {
987  PcodeOp *op = *iter;
988  op->setParent(bl); // Reset ops parent to -bl-
989  }
990  // Move all ops from -outbl- to end of -bl-
991  bl->op.splice(bl->op.end(),outbl->op,outbl->op.begin(),outbl->op.end());
992  // insertiter should remain valid through splice
993  bl->setOrder(); // Reset the seqnum ordering on all the ops
994  }
995  bl->mergeRange(outbl); // Update the address cover
996  bblocks.spliceBlock(bl);
997  structureReset();
998 }

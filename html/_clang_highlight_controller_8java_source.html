190  public List<ClangToken> addHighlightParen(ClangSyntaxToken tok, Color highlightColor) {
191  ArrayList<ClangToken> tokenList = new ArrayList<ClangToken>();
192  int paren = tok.getOpen();
193  if (paren == -1)
194  paren = tok.getClose();
195  if (paren == -1)
196  return tokenList; // Not a parenthesis
197  ClangNode par = tok.Parent();
198  while (par != null) {
199  boolean outside = true;
200  if (par instanceof ClangTokenGroup) {
201  ArrayList<ClangNode> list = new ArrayList<ClangNode>();
202  ((ClangTokenGroup) par).flatten(list);
203  for (int i = 0; i < list.size(); ++i) {
204  ClangToken tk = (ClangToken) list.get(i);
205  if (tk instanceof ClangSyntaxToken) {
206  ClangSyntaxToken syn = (ClangSyntaxToken) tk;
207  if (syn.getOpen() == paren)
208  outside = false;
209  else if (syn.getClose() == paren) {
210  outside = true;
211  addHighlight(syn, highlightColor);
212  tokenList.add(syn);
213  }
214  }
215  if (!outside) {
216  addHighlight(tk, highlightColor);
217  tokenList.add(tk);
218  }
219  }
220  }
221  par = par.Parent();
222  }
223  return tokenList;
224  }

241  private void highlightBrace(ClangSyntaxToken startToken, ClangNode parent, boolean forward,
242  Color highlightColor) {
243  List<ClangNode> list = new ArrayList<ClangNode>();
244  parent.flatten(list);
245 
246  if (!forward) {
247  Collections.reverse(list);
248  }
249 
250  Stack<ClangSyntaxToken> braceStack = new Stack<ClangSyntaxToken>();
251  for (int i = 0; i < list.size(); ++i) {
252  ClangToken token = (ClangToken) list.get(i);
253  if (token instanceof ClangSyntaxToken) {
254  ClangSyntaxToken syntaxToken = (ClangSyntaxToken) token;
255 
256  if (startToken == syntaxToken) {
257  // found our starting token, take the current value on the stack
258  ClangSyntaxToken matchingBrace = braceStack.pop();
259  matchingBrace.setMatchingToken(true);
260  addHighlight(matchingBrace, highlightColor);
261  return;
262  }
263 
264  if (!isBrace(syntaxToken)) {
265  continue;
266  }
267 
268  if (braceStack.isEmpty()) {
269  braceStack.push(syntaxToken);
270  continue;
271  }
272 
273  ClangSyntaxToken lastToken = braceStack.peek();
274  if (isMatchingBrace(lastToken, syntaxToken)) {
275  braceStack.pop();
276  }
277  else {
278  braceStack.push(syntaxToken);
279  }
280  }
281  }
282  }

299  public void addHighlightFill() {
300  ClangTokenGroup lastgroup = null;
301  ArrayList<ClangNode> newhi = new ArrayList<ClangNode>();
302  ArrayList<Color> newcolor = new ArrayList<Color>();
303  for (ClangToken tok : highlightTokenSet) {
304  if (tok.Parent() instanceof ClangTokenGroup) {
305  ClangTokenGroup par = (ClangTokenGroup) tok.Parent();
306  if (par == lastgroup)
307  continue;
308  lastgroup = par;
309  int beg = -1;
310  int end = par.numChildren();
311  for (int j = 0; j < par.numChildren(); ++j) {
312  if (par.Child(j) instanceof ClangToken) {
313  ClangToken token = (ClangToken) par.Child(j);
314  Color curcolor = token.getHighlight();
315  if (curcolor != null) {
316  if (beg == -1)
317  beg = j;
318  else {
319  end = j;
320  for (int k = beg + 1; k < end; ++k) {
321  if (par.Child(k) instanceof ClangToken) {
322  newhi.add(par.Child(k));
323  newcolor.add(curcolor);
324  }
325  }
326  beg = j;
327  }
328  }
329  else if (token.getMinAddress() != null) {
330  beg = -1;
331  }
332  }
333  else
334  beg = -1;
335  }
336  }
337  }
338  for (int i = 0; i < newhi.size(); ++i) {
339  ClangToken tok = (ClangToken) newhi.get(i);
340  if (tok.getHighlight() != null)
341  continue;
342  addHighlight(tok, newcolor.get(i));
343  }
344  notifyListeners();
345  }

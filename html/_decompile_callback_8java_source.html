336  public String getPcodeInject(String nm, String context, int type) {
337  PcodeInjectLibrary snippetLibrary = pcodecompilerspec.getPcodeInjectLibrary();
338 
339  InjectPayload payload = snippetLibrary.getPayload(type, nm, program, context);
340  if (payload == null) {
341  Msg.warn(this, "Decompiling " + funcEntry + ", no pcode inject with name: " + nm);
342  return null; // No fixup associated with this name
343  }
344  InjectContext con = snippetLibrary.buildInjectContext();
345  PcodeOp[] pcode;
346  try {
347  con.restoreXml(getSAXParser(), context, addrfactory);
348  }
349  catch (PcodeXMLException e) {
350  Msg.error(this, "Decompiling " + funcEntry + ": " + e.getMessage());
351  return null;
352  }
353  try {
354  int fallThruOffset;
355  if (payload.getType() == InjectPayload.EXECUTABLEPCODE_TYPE) {
356  // Executable p-code has no underlying instruction address and
357  // does (should) not use the inst_start, inst_next symbols that need
358  // to know about it.
359  fallThruOffset = 4; // Provide a dummy length for the XML doc
360  }
361  else {
362  Instruction instr = getInstruction(con.baseAddr);
363  if (instr == null) {
364  Msg.warn(this, "Decompiling " + funcEntry + ", pcode inject error at " +
365  con.baseAddr + ": instruction not found");
366  return null;
367  }
368 
369  // get next inst addr for inst_next pcode variable
370  fallThruOffset = instr.getDefaultFallThroughOffset();
371  con.nextAddr = con.baseAddr.add(fallThruOffset);
372 
373  con.refAddr = null;
374  for (Reference ref : program.getReferenceManager().getReferencesFrom(
375  con.baseAddr)) {
376  if (ref.isPrimary() && ref.getReferenceType().isCall()) {
377  con.refAddr = ref.getToAddress();
378  break;
379  }
380  }
381  }
382  pcode = payload.getPcode(program, con);
383  if (pcode == null) {
384  return null; // Just return a null string, which should let the decompiler exit gracefully
385  }
386  String finalPayload =
387  buildInstruction(pcode, fallThruOffset, payload.getParamShift(), addrfactory);
388  if (debug != null) {
389  debug.addInject(con.baseAddr, nm, type, finalPayload);
390  }
391  return finalPayload;
392  }
393  catch (UnknownInstructionException e) {
394  Msg.warn(this, "Decompiling " + funcEntry + ", pcode inject error at " + con.baseAddr +
395  ": " + e.getMessage());
396  }
397  catch (Exception e) {
398  Msg.error(this, "Decompiling " + funcEntry + ", pcode inject error at " + con.baseAddr +
399  ": " + e.getMessage(), e);
400  }
401  return null;
402  }

424  private Instruction pseudoDisassemble(Address addr) throws UnknownInstructionException {
425 
426  Instruction instr;
427 
428  if (lastPseudoInstructionBlock != null) {
429  instr = lastPseudoInstructionBlock.getInstructionAt(addr);
430  if (instr != null) {
431  return instr;
432  }
433  InstructionError error = lastPseudoInstructionBlock.getInstructionConflict();
434  if (error != null && addr.equals(error.getInstructionAddress())) {
435  throw new UnknownInstructionException(error.getConflictMessage());
436  }
437  lastPseudoInstructionBlock = null;
438  }
439 
440  if (pseudoDisassembler == null) {
441  pseudoDisassembler = Disassembler.getDisassembler(program, false, false, false,
442  TaskMonitorAdapter.DUMMY_MONITOR, msg -> {
443  // TODO: Should we log errors?
444  });
445  }
446 
447  RegisterValue entryContext = null;
448  ProgramContext programContext = program.getProgramContext();
449  Register baseContextRegister = programContext.getBaseContextRegister();
450  if (baseContextRegister != null) {
451  entryContext = programContext.getRegisterValue(baseContextRegister, funcEntry);
452  }
453 
454  lastPseudoInstructionBlock =
455  pseudoDisassembler.pseudoDisassembleBlock(addr, entryContext, 64);
456  if (lastPseudoInstructionBlock != null) {
457  InstructionError error = lastPseudoInstructionBlock.getInstructionConflict(); // Look for zero-byte run first
458  if (error != null &&
459  error.getConflictMessage().startsWith("Maximum run of Zero-Byte")) {
460  throw new UnknownInstructionException(error.getConflictMessage()); // Don't return any of the zero-byte instructions
461  }
462  instr = lastPseudoInstructionBlock.getInstructionAt(addr);
463  if (instr != null) {
464  return instr;
465  }
466  if (error != null && addr.equals(error.getInstructionAddress())) {
467  throw new UnknownInstructionException(error.getConflictMessage());
468  }
469  if (MemoryBlock.isExternalBlockAddress(addr, program)) {
470  throw new UnknownInstructionException(
471  "Unable to disassemble EXTERNAL block location: " + addr);
472  }
473  }
474  throw new UnknownInstructionException("Invalid instruction address (improperly aligned)");
475  }

556  private void generateCommentXML(AddressSetView addrset, Address addr, StringBuilder buf,
557  int commenttype) {
558  String typename;
559  switch (commenttype) {
560  case CodeUnit.EOL_COMMENT:
561  typename = "user1";
562  break;
563  case CodeUnit.PRE_COMMENT:
564  typename = "user2";
565  break;
566  case CodeUnit.POST_COMMENT:
567  typename = "user3";
568  break;
569  case CodeUnit.PLATE_COMMENT:
570  typename = "header";
571  break;
572  default:
573  typename = "";
574  break;
575  }
576  AddressIterator iter = listing.getCommentAddressIterator(commenttype, addrset, true);
577  while (iter.hasNext()) {
578  Address commaddr = iter.next();
579  String text = listing.getComment(commenttype, commaddr);
580  if (text != null) {
581  if (commenttype == CodeUnit.PLATE_COMMENT) {
582  // Plate comments on the function entry
583  // address are considered part of the header
584  if (commaddr.equals(addr)) {
585  continue;
586  }
587  }
588 
589  buf.append("<comment");
590  SpecXmlUtils.encodeStringAttribute(buf, "type", typename);
591  buf.append(">\n");
592  buf.append(Varnode.buildXMLAddress(addr));
593  buf.append(Varnode.buildXMLAddress(commaddr));
594  buf.append("\n<text>");
595  SpecXmlUtils.xmlEscape(buf, text);
596  buf.append("</text>\n");
597  buf.append("</comment>\n");
598  }

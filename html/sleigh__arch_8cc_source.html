165 void SleighArchitecture::resolveArchitecture(void)
166 
167 { // Find best architecture
168  if (archid.size() == 0) {
169  if ((target.size()==0)||(target=="default"))
170  archid = loader->getArchType();
171  else
172  archid = target;
173  }
174  if (archid.find("binary-")==0)
175  archid.erase(0,7);
176  else if (archid.find("default-")==0)
177  archid.erase(0,8);
178 
179  archid = normalizeArchitecture(archid);
180  string baseid = archid.substr(0,archid.rfind(':'));
181  int4 i;
182  languageindex = -1;
183  for(i=0;i<description.size();++i) {
184  if (description[i].getId() == baseid) {
185  languageindex = i;
186  if (description[i].isDeprecated())
187  printMessage("WARNING: Language "+baseid+" is deprecated");
188  break;
189  }
190  }
191 
192  if (languageindex == -1)
193  throw LowlevelError("No sleigh specification for "+baseid);
194 }

375 string SleighArchitecture::normalizeArchitecture(const string &nm)
376 
377 {
378  string processor;
379  string endian;
380  string size;
381  string variant;
382  string compile;
383 
384  string::size_type pos[4];
385  int4 i;
386  string::size_type curpos=0;
387  for(i=0;i<4;++i) {
388  curpos = nm.find(':',curpos+1);
389  if (curpos == string::npos) break;
390  pos[i] = curpos;
391  }
392  if ((i!=3)&&(i!=4))
393  throw LowlevelError("Architecture string does not look like sleigh id: "+nm);
394  processor = nm.substr(0,pos[0]);
395  endian = nm.substr(pos[0]+1,pos[1]-pos[0]-1);
396  size = nm.substr(pos[1]+1,pos[2]-pos[1]-1);
397 
398  if (i==4) {
399  variant = nm.substr(pos[2]+1,pos[3]-pos[2]-1);
400  compile = nm.substr(pos[3]+1);
401  }
402  else {
403  variant = nm.substr(pos[2]+1);
404  compile = "default";
405  }
406 
407  processor = normalizeProcessor(processor);
408  endian = normalizeEndian(endian);
409  size = normalizeSize(size);
410  return processor + ':' + endian + ':' + size + ':' + variant + ':' + compile;
411 }

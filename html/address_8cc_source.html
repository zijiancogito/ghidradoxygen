802 int4 power2Divide(int4 n,uint8 divisor,uint8 &q,uint8 &r)
803 
804 {
805  if (divisor == 0) return 2;
806  uint8 power = 1;
807  if (n < 64) {
808  power <<= n;
809  q = power / divisor;
810  r = power % divisor;
811  return 0;
812  }
813  // Divide numerand and divisor by 2^(n-63) to get approximation of result
814  uint8 y = divisor >> (n-64); // Most of the way on divisor
815  if (y == 0) return 1; // Check if result will be too big
816  y >>= 1; // Divide divisor by final bit
817  power <<= 63;
818  uint8 max;
819  if (y == 0) {
820  max = 0;
821  max -= 1; // Could be maximal
822  // Check if divisor is a power of 2
823  if ((((uint8)1) << (n-64)) == divisor)
824  return 1;
825  }
826  else
827  max = power / y + 1;
828  uint8 min = power / (y+1);
829  if (min != 0)
830  min -= 1;
831  uint8 fullpower[2];
832  fullpower[1] = ((uint8)1)<<(n-64);
833  fullpower[0] = 0;
834  uint8 mult[2];
835  mult[0] = 0;
836  mult[1] = 0;
837  uint8 tmpq = 0;
838  while(max > min+1) {
839  tmpq = max + min;
840  if (tmpq < min) {
841  tmpq = (tmpq>>1) + 0x8000000000000000L;
842  }
843  else
844  tmpq >>= 1;
845  mult64to128(mult,divisor,tmpq);
846  if (unsignedCompare128(fullpower,mult) < 0)
847  max = tmpq-1;
848  else
849  min = tmpq;
850  }
851  // min is now our putative quotient
852  if (tmpq != min)
853  mult64to128(mult,divisor,min);
854  unsignedSubtract128(fullpower,mult); // Calculate remainder
855  // min might be 1 too small
856  if (fullpower[1] != 0 || fullpower[0] >= divisor) {
857  q = min + 1;
858  r = fullpower[0] - divisor;
859  }
860  else {
861  q = min;
862  r = fullpower[0];
863  }
864  return 0;
865 }

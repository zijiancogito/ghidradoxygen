35 VarnodeData *PcodeCacher::expandPool(uint4 size)
36 
37 { // Expand the pool so that -size- more elements fit
38  // Return pointer to first available element
39  uint4 curmax = endpool - poolstart;
40  uint4 cursize = curpool - poolstart;
41  if (cursize + size <= curmax)
42  return curpool; // No expansion necessary
43  uint4 increase = (cursize + size) - curmax;
44  if (increase < 100) // Increase by at least 100
45  increase = 100;
46 
47  uint4 newsize = curmax + increase;
48 
49  VarnodeData *newpool = new VarnodeData[newsize];
50  for(uint4 i=0;i<cursize;++i)
51  newpool[i] = poolstart[i]; // Copy old data
52  // Update references to the old pool
53  for(uint4 i=0;i<issued.size();++i) {
54  VarnodeData *outvar = issued[i].outvar;
55  if (outvar != (VarnodeData *)0) {
56  outvar = newpool + (outvar - poolstart);
57  issued[i].outvar = outvar;
58  }
59  VarnodeData *invar = issued[i].invar;
60  if (invar != (VarnodeData *)0) {
61  invar = newpool + (invar - poolstart);
62  issued[i].invar = invar;
63  }
64  }
65  list<RelativeRecord>::iterator iter;
66  for(iter=label_refs.begin();iter!=label_refs.end();++iter) {
67  VarnodeData *ref = (*iter).dataptr;
68  (*iter).dataptr = newpool + (ref - poolstart);
69  }
70 
71  delete [] poolstart; // Free up old pool
72  poolstart = newpool;
73  curpool = newpool + (cursize + size);
74  endpool = newpool + newsize;
75  return newpool + cursize;
76 }

104 void PcodeCacher::resolveRelatives(void)
105 
106 { // Assuming all the PcodeData has been generated for an
107  // instruction, go resolve any relative offsets and back
108  // patch their value(s) into the PcodeData
109  list<RelativeRecord>::const_iterator iter;
110  for(iter=label_refs.begin();iter!=label_refs.end();++iter) {
111  VarnodeData *ptr = (*iter).dataptr;
112  uint4 id = ptr->offset;
113  if ((id >= labels.size())||(labels[id] == 0xbadbeef))
114  throw LowlevelError("Reference to non-existant sleigh label");
115  // Calculate the relative index given the two absolute indices
116  uintb res = labels[id] - (*iter).calling_index;
117  res &= calc_mask( ptr->size );
118  ptr->offset = res;
119  }

161 void SleighBuilder::dump(OpTpl *op)
162 
163 { // Dump on op through low-level dump interface
164  // filling in dynamic loads and stores if necessary
165  PcodeData *thisop;
166  VarnodeData *invars;
167  VarnodeData *loadvars;
168  VarnodeData *storevars;
169  VarnodeTpl *vn,*outvn;
170  int4 isize = op->numInput();
171  // First build all the inputs
172  invars = cache->allocateVarnodes(isize);
173  for(int4 i=0;i<isize;++i) {
174  vn = op->getIn(i);
175  if (vn->isDynamic(*walker)) {
176  generateLocation(vn,invars[i]); // Input of -op- is really temporary storage
177  PcodeData *load_op = cache->allocateInstruction();
178  load_op->opc = CPUI_LOAD;
179  load_op->outvar = invars + i;
180  load_op->isize = 2;
181  loadvars = load_op->invar = cache->allocateVarnodes(2);
182  AddrSpace *spc = generatePointer(vn,loadvars[1]);
183  loadvars[0].space = const_space;
184  loadvars[0].offset = (uintb)(uintp)spc;
185  loadvars[0].size = sizeof(spc);
186  }
187  else
188  generateLocation(vn,invars[i]);
189  }
190  if ((isize>0)&&(op->getIn(0)->isRelative())) {
191  invars->offset += getLabelBase();
192  cache->addLabelRef(invars);
193  }
194  thisop = cache->allocateInstruction();
195  thisop->opc = op->getOpcode();
196  thisop->invar = invars;
197  thisop->isize = isize;
198  outvn = op->getOut();
199  if (outvn != (VarnodeTpl *)0) {
200  if (outvn->isDynamic(*walker)) {
201  storevars = cache->allocateVarnodes(3);
202  generateLocation(outvn,storevars[2]); // Output of -op- is really temporary storage
203  thisop->outvar = storevars+2;
204  PcodeData *store_op = cache->allocateInstruction();
205  store_op->opc = CPUI_STORE;
206  store_op->isize = 3;
207  // store_op->outvar = (VarnodeData *)0;
208  store_op->invar = storevars;
209  AddrSpace *spc = generatePointer(outvn,storevars[1]); // pointer
210  storevars[0].space = const_space;
211  storevars[0].offset = (uintb)(uintp)spc; // space in which to store
212  storevars[0].size = sizeof(spc);
213  }
214  else {
215  thisop->outvar = cache->allocateVarnodes(1);
216  generateLocation(outvn,*thisop->outvar);
217  }
218  }
219 }

284 void SleighBuilder::delaySlot(OpTpl *op)
285 
286 { // Append pcode for an entire instruction (delay slot)
287  // in the middle of the current instruction
288  ParserWalker *tmp = walker;
289  uintb olduniqueoffset = uniqueoffset;
290 
291  Address baseaddr = tmp->getAddr();
292  int4 fallOffset = tmp->getLength();
293  int4 delaySlotByteCnt = tmp->getParserContext()->getDelaySlot();
294  int4 bytecount = 0;
295  do {
296  Address newaddr = baseaddr + fallOffset;
297  setUniqueOffset(newaddr);
298  const ParserContext *pos = discache->getParserContext(newaddr);
299  if (pos->getParserState() != ParserContext::pcode)
300  throw LowlevelError("Could not obtain cached delay slot instruction");
301  int4 len = pos->getLength();
302 
303  ParserWalker newwalker( pos );
304  walker = &newwalker;
305  walker->baseState();
306  build(walker->getConstructor()->getTempl(),-1); // Build the whole delay slot
307  fallOffset += len;
308  bytecount += len;
309  } while(bytecount < delaySlotByteCnt);
310  walker = tmp; // Restore original context
311  uniqueoffset = olduniqueoffset;
312 }

355 void DisassemblyCache::initialize(int4 min,int4 hashsize)
356 
357 {
358  minimumreuse = min;
359  mask = hashsize-1;
360  uintb masktest = coveringmask((uintb)mask);
361  if (masktest != (uintb)mask) // -hashsize- must be a power of 2
362  throw LowlevelError("Bad windowsize for disassembly cache");
363  list = new ParserContext *[minimumreuse];
364  nextfree = 0;
365  hashtable = new ParserContext *[hashsize];
366  for(int4 i=0;i<minimumreuse;++i) {
367  ParserContext *pos = new ParserContext(contextcache);
368  pos->initialize(75,20,constspace);
369  list[i] = pos;
370  }
371  ParserContext *pos = list[0];
372  for(int4 i=0;i<hashsize;++i)
373  hashtable[i] = pos; // Make sure all hashtable positions point to a real ParserContext
374 }

393 ParserContext *DisassemblyCache::getParserContext(const Address &addr)
394 
395 { // Return a (possibly cached) ParserContext that is associated with -addr-
396  // If n different calls to this interface are made with n different Addresses, if
397  // n <= minimumreuse AND
398  // all the addresses are within the windowsize (=mask+1)
399  // then the cacher guarantees that you get all different ParserContext objects
400  int4 hashindex = ((int4) addr.getOffset()) & mask;
401  ParserContext *res = hashtable[ hashindex ];
402  if (res->getAddr() == addr)
403  return res;
404  res = list[ nextfree ];
405  nextfree += 1; // Advance the circular index
406  if (nextfree >= minimumreuse)
407  nextfree = 0;
408  res->setAddr(addr);
409  res->setParserState(ParserContext::uninitialized); // Need to start over with parsing
410  hashtable[ hashindex ] = res; // Stick it into the hashtable
411  return res;
412 }

488 void Sleigh::resolve(ParserContext &pos) const
489 
490 { // Resolve ALL the constructors involved in the
491  // instruction at this address
492  loader->loadFill(pos.getBuffer(),16,pos.getAddr());
493  ParserWalkerChange walker(&pos);
494  pos.deallocateState(walker); // Clear the previous resolve and initialize the walker
495  Constructor *ct,*subct;
496  uint4 off;
497  int4 oper,numoper;
498 
499  pos.setDelaySlot(0);
500  walker.setOffset(0); // Initial offset
501  pos.clearCommits(); // Clear any old context commits
502  pos.loadContext(); // Get context for current address
503  ct = root->resolve(walker); // Base constructor
504  walker.setConstructor(ct);
505  ct->applyContext(walker);
506  while(walker.isState()) {
507  ct = walker.getConstructor();
508  oper = walker.getOperand();
509  numoper = ct->getNumOperands();
510  while(oper < numoper) {
511  OperandSymbol *sym = ct->getOperand(oper);
512  off = walker.getOffset(sym->getOffsetBase()) + sym->getRelativeOffset();
513  pos.allocateOperand(oper,walker); // Descend into new operand and reserve space
514  walker.setOffset(off);
515  TripleSymbol *tsym = sym->getDefiningSymbol();
516  if (tsym != (TripleSymbol *)0) {
517  subct = tsym->resolve(walker);
518  if (subct != (Constructor *)0) {
519  walker.setConstructor(subct);
520  subct->applyContext(walker);
521  break;
522  }
523  }
524  walker.setCurrentLength(sym->getMinimumLength());
525  walker.popOperand();
526  oper += 1;
527  }
528  if (oper >= numoper) { // Finished processing constructor
529  walker.calcCurrentLength(ct->getMinimumLength(),numoper);
530  walker.popOperand();
531  // Check for use of delayslot
532  ConstructTpl *templ = ct->getTempl();
533  if ((templ != (ConstructTpl *)0)&&(templ->delaySlot() > 0))
534  pos.setDelaySlot(templ->delaySlot());
535  }
536  }
537  pos.setNaddr(pos.getAddr()+pos.getLength()); // Update Naddr to pointer after instruction
538  pos.setParserState(ParserContext::disassembly);
539 }

541 void Sleigh::resolveHandles(ParserContext &pos) const
542 
543 { // Resolve handles (assuming Constructors already resolved)
544  TripleSymbol *triple;
545  Constructor *ct;
546  int4 oper,numoper;
547 
548  ParserWalker walker(&pos);
549  walker.baseState();
550  while(walker.isState()) {
551  ct = walker.getConstructor();
552  oper = walker.getOperand();
553  numoper = ct->getNumOperands();
554  while(oper < numoper) {
555  OperandSymbol *sym = ct->getOperand(oper);
556  walker.pushOperand(oper); // Descend into node
557  triple = sym->getDefiningSymbol();
558  if (triple != (TripleSymbol *)0) {
559  if (triple->getType() == SleighSymbol::subtable_symbol)
560  break;
561  else // Some other kind of symbol as an operand
562  triple->getFixedHandle(walker.getParentHandle(),walker);
563  }
564  else { // Must be an expression
565  PatternExpression *patexp = sym->getDefiningExpression();
566  intb res = patexp->getValue(walker);
567  FixedHandle &hand(walker.getParentHandle());
568  hand.space = pos.getConstSpace(); // Result of expression is a constant
569  hand.offset_space = (AddrSpace *)0;
570  hand.offset_offset = (uintb)res;
571  hand.size = 0; // This size should not get used
572  }
573  walker.popOperand();
574  oper += 1;
575  }
576  if (oper >= numoper) { // Finished processing constructor
577  ConstructTpl *templ = ct->getTempl();
578  if (templ != (ConstructTpl *)0) {
579  HandleTpl *res = templ->getResult();
580  if (res != (HandleTpl *)0) // Pop up handle to containing operand
581  res->fix(walker.getParentHandle(),walker);
582  // If we need an indicator that the constructor exports nothing try
583  // else
584  // walker.getParentHandle().setInvalid();
585  }
586  walker.popOperand();
587  }
588  }
589  pos.setParserState(ParserContext::pcode);
590 }

618 int4 Sleigh::oneInstruction(PcodeEmit &emit,const Address &baseaddr) const
619 
620 {
621  int4 fallOffset;
622  if (alignment != 1) {
623  if ((baseaddr.getOffset() % alignment)!=0) {
624  ostringstream s;
625  s << "Instruction address not aligned: " << baseaddr;
626  throw UnimplError(s.str(),0);
627  }
628  }
629 
630  ParserContext *pos = obtainContext(baseaddr,ParserContext::pcode);
631  pos->applyCommits();
632  fallOffset = pos->getLength();
633 
634  if (pos->getDelaySlot()>0) {
635  int4 bytecount = 0;
636  do {
637  // Do not pass pos->getNaddr() to obtainContext, as pos may have been previously cached and had naddr adjusted
638  ParserContext *delaypos = obtainContext(pos->getAddr() + fallOffset,ParserContext::pcode);
639  delaypos->applyCommits();
640  int4 len = delaypos->getLength();
641  fallOffset += len;
642  bytecount += len;
643  } while(bytecount < pos->getDelaySlot());
644  pos->setNaddr(pos->getAddr()+fallOffset);
645  }
646  ParserWalker walker(pos);
647  walker.baseState();
648  pcode_cache.clear();
649  SleighBuilder builder(&walker,discache,&pcode_cache,getConstantSpace(),getUniqueSpace(),unique_allocatemask);
650  try {
651  builder.build(walker.getConstructor()->getTempl(),-1);

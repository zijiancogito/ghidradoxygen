169  void appendText(StringBuffer buffer, FieldRange fieldRange) {
170  int startIndex = fieldRange.getStart().getIndex().intValue();
171  int endIndex = fieldRange.getEnd().getIndex().intValue();
172  if (startIndex == endIndex) { // single line selection (don't include padding)

186  private void appendText(StringBuffer buffer, int lineNumber, FieldSelection singleLineSelection) {
187  if (singleLineSelection.isEmpty()) {
188  return;
189  }
190  FieldRange fieldRange = singleLineSelection.getFieldRange(0);
191  int startColumn = fieldRange.getStart().getCol();
192  int endColumn = Integer.MAX_VALUE;
193  int startRow = fieldRange.getStart().getRow();
194  int endRow = Integer.MAX_VALUE;
195  int startIndex = fieldRange.getStart().getIndex().intValue();
196  int endIndex = fieldRange.getEnd().getIndex().intValue();
197  if (startIndex == endIndex) {
198  endColumn = fieldRange.getEnd().getCol();
199  endRow = fieldRange.getEnd().getRow();
200  }
201 
202  LayoutModel model = provider.getDecompilerPanel().getLayoutModel();
203  Layout layout = model.getLayout(BigInteger.valueOf(lineNumber));
204  ClangTextField field = (ClangTextField) layout.getField(0);
205  int numSpaces = (field.getStartX() - field.getLineNumberWidth()) / spaceCharWidthInPixels;
206  for (int i = 0; i < numSpaces; i++) {
207  buffer.append(' ');
208  }
209 
210  int startPos = field.screenLocationToTextOffset(startRow, startColumn);
211  int endPos = field.screenLocationToTextOffset(endRow, endColumn);
212  for (int i = 0; i < startPos; i++) {
213  buffer.append(' ');
214  }
215  if (startPos >= 0 && endPos >= startPos) {
216  buffer.append(field.getText().substring(startPos, endPos));
217  }
218  }

94 bool CallGraph::findNoEntry(vector<CallGraphNode *> &seeds)
95 
96 { // Find all functions (that are not already marked) that either have no in edges at all,
97  // or have no in edges that haven't been snipped as part of cycles
98 
99  map<Address,CallGraphNode>::iterator iter;
100  CallGraphNode *lownode = (CallGraphNode *)0;
101  bool allcovered = true;
102  bool newseeds = false;
103 
104  for(iter=graph.begin();iter!=graph.end();++iter) {
105  CallGraphNode &node( (*iter).second );
106  if (node.isMark()) continue;
107  if ((node.inedge.size() == 0)||((node.flags&CallGraphNode::onlycyclein)!=0)) {
108  seeds.push_back(&node);
109  node.flags |= CallGraphNode::mark | CallGraphNode::entrynode;
110  newseeds = true;
111  }
112  else {
113  allcovered = false;
114  // We need to worry about the case where everything is in a cycle, so we don't find a natural root
115  // We use the node with the lowest number of in edges as a pseudo root
116  if (lownode == (CallGraphNode *)0)
117  lownode = &node;
118  else {
119  if (node.numInEdge() < lownode->numInEdge())
120  lownode = &node;
121  }
122  }
123  }
124  if ((!newseeds)&&(!allcovered)) {
125  seeds.push_back(lownode);
126  lownode->flags |= CallGraphNode::mark | CallGraphNode::entrynode;
127  }
128  return allcovered;
129 }

131 void CallGraph::snipCycles(CallGraphNode *node)
132 
133 { // Snip any cycles starting from root -node-
134  CallGraphNode *next;
135  vector<LeafIterator> stack;
136 
137  node->flags |= CallGraphNode::currentcycle;
138  stack.push_back(LeafIterator(node));
139 
140  while(!stack.empty()) {
141  CallGraphNode *cur = stack.back().node; // Current node
142  int4 st = stack.back().outslot; // which out edge we will follow
143  if (st >= cur->outedge.size()) {
144  cur->flags &= ~((uint4)CallGraphNode::currentcycle);
145  stack.pop_back();
146  }
147  else {
148  stack.back().outslot += 1;
149  if ((cur->outedge[st].flags&CallGraphEdge::cycle)!=0) continue;
150  next = cur->outedge[st].to;
151  if ((next->flags&CallGraphNode::currentcycle)!=0) { // Found a cycle
152  snipEdge(cur,st);
153  continue;
154  }
155  else if ((next->flags&CallGraphNode::mark)!=0) { // Already traced before
156  cur->outedge[st].flags |= CallGraphEdge::dontfollow;
157  continue;
158  }
159  next->parentedge = cur->outedge[st].complement;
160  next->flags |= (CallGraphNode::currentcycle| CallGraphNode::mark);
161  stack.push_back(LeafIterator(next));
162  }
163  }
164 }

166 void CallGraph::snipEdge(CallGraphNode *node,int4 i)
167 
168 {
169  node->outedge[i].flags |= CallGraphEdge::cycle | CallGraphEdge::dontfollow;
170  int4 toi = node->outedge[i].complement;
171  CallGraphNode *to = node->outedge[i].to;
172  to->inedge[toi].flags |= CallGraphEdge::cycle;
173  bool onlycycle = true;
174  for(uint4 j=0;j<to->inedge.size();++j) {
175  if ((to->inedge[j].flags & CallGraphEdge::cycle)==0) {
176  onlycycle = false;
177  break;
178  }
179  }
180  if (onlycycle)
181  to->flags |= CallGraphNode::onlycyclein;
182 }

246 
247 {
248  int4 i;
249  for(i=0;i<from->outedge.size();++i) {
250  CallGraphNode *outnode = from->outedge[i].to;
251  if (outnode == to) return; // Already have an out edge
252  if (to->entryaddr < outnode->entryaddr) break;

254 
255  CallGraphEdge &fromedge( insertBlankEdge(from,i) );

260

354 void CallGraph::cycleStructure(void)
355 
356 { // Generate list of seeds nodes (from which we can get to everything)
357  if (!seeds.empty())
358  return;
359  uint4 walked = 0;
360  bool allcovered;
361 
362  do {
363  allcovered = findNoEntry(seeds);
364  while(walked < seeds.size()) {
365  CallGraphNode *rootnode = seeds[walked];
366  rootnode->parentedge = walked;
367  snipCycles(rootnode);
368  walked += 1;
369  }
370  } while(!allcovered);
371  clearMarks();

409 
410 { // Build edges from a disassembled (decompiled) function
411  CallGraphNode *fdnode = findNode(fd->getAddress());

107  /*
108  * The method indicates whether the function is in a block of code that is considered external or "glue"--meaning
109  * that we don't want to analyze the code that might be there, yet we might have signatures for the function or
110  * what it provides linkage to that we do not want to wipe. We want to keep what is already there.
111  * TODO: This implementation below, with hard-coded specific block names (other than EXTERNAL, which is created
112  * by analysis) will need to be revisited. Perhaps a flag will be set by the importer on the blocks that we should
113  * ignore.
114  */
115  private boolean funcIsExternalGlue(Function func) {
116  String blockName = program.getMemory().getBlock(func.getEntryPoint()).getName();
117  return (blockName.equals("EXTERNAL") || blockName.equals(".plt") || blockName.equals("__stub_helper"));
118  }

120  private void resetFunctionSourceTypes(Set<Address> set) {
121  FunctionManager functionManager = program.getFunctionManager();
122  //For those functions that we will process, appropriately clear SourceType, meaning
123  // move ANALYSIS (or other, depending on ClearLevel) SourceType back to DEFAULT SourceType,
124  // but keep higher SoureTypes fixed. Should we do this for individual parameters and returns too? --> yes/no TODO
125  for (Address entryPoint : set) {
126  Function func = functionManager.getFunctionAt(entryPoint);
127  try {
128  //Do not clear prototypes of "pseudo-analyzed" external functions.
129  if (func.isExternal()) {
130  continue;
131  }
132  if (funcIsExternalGlue(func)) {
133  continue;
134  }
135 
136  // TODO: should refactor to avoid changing source type to default
137  // since decompile could fail and leave the source types changed.
138  Parameter retParam = func.getReturn();
139  if (retParam != null) {
140  if (!retParam.getSource().isHigherPriorityThan(sourceTypeClearLevel)) {
141  func.setReturn(retParam.getDataType(), retParam.getVariableStorage(),
142  SourceType.DEFAULT);
143  }
144  }
145  if (!func.getSignatureSource().isHigherPriorityThan(sourceTypeClearLevel)) {
146  func.setSignatureSource(SourceType.DEFAULT);
147  }
148  }
149  catch (InvalidInputException e) {
150  Msg.warn(
151  this,
152  "Error changing signature SourceType (should not since Input is the same) on--" +
153  func.getName(), e);
154  }
155  }

158  private void analyzeFunction(DecompInterface decomplib, Function f, TaskMonitor monitor) {
159 
160  // TODO: in addition to signature-source-type we should also verify that a named calling
161  // has been specified since auto-storage assignments may be based on the incorrect default
162  // (or unknown) calling convention.
163 
164  // TODO: How should we handle cases with unassigned storage on one or more parameters
165  // but with good type info and a non-default source type
166 
167  if (f == null || f.isThunk() || f.getSignatureSource() != SourceType.DEFAULT) {
168  return;
169  }
170 
171  //We didn't "wipe" previous results of external functions, but we also do not want
172  // to set new results.
173  if (f.isExternal()) {
174  return;
175  }
176  if (funcIsExternalGlue(f)) {
177  return;
178  }
179 
180  try {
181  DecompileResults decompRes = null;
182  if (monitor.isCancelled()) {
183  return;
184  }
185 
186  decompRes = decomplib.decompileFunction(f, decompilerTimeoutSecs, monitor);
187  setStatusMsg(decompRes.getErrorMessage());
188 
189  if (monitor.isCancelled()) {
190  return;
191  }
192 
193  boolean goodInfo = false;
194  if (decompRes.decompileCompleted()) {
195  // if results lack sanity, don't store them
196  if (hasInconsistentResults(decompRes)) {
197  setStatusMsg("Error function " + decompRes.getFunction() +
198  " has inconsistent parameters");
199  return;
200  }
201 
202  if (commitDataTypes) {
203  HighFunction hfunc = decompRes.getHighFunction();
204  if (hfunc == null) {
205  return;
206  }
207  HighFunctionDBUtil.commitParamsToDatabase(hfunc, true, SourceType.ANALYSIS);
208  boolean commitReturn = true;
209  if (!commitVoidReturn) {
210  DataType returnType = hfunc.getFunctionPrototype().getReturnType();
211  if (returnType instanceof VoidDataType)
212  commitReturn = false;
213  }
214  if (commitReturn) {
215  HighFunctionDBUtil.commitReturnToDatabase(hfunc, SourceType.ANALYSIS);
216  }
217  goodInfo = true;
218  }
219  else {
220  HighParamID hparamid = decompRes.getHighParamID();
221 
222  hparamid.storeParametersToDatabase(commitDataTypes, SourceType.ANALYSIS);
223 
224  // Not doing anything with ExtraPop or setStackPurgeSize(int change)
225  hparamid.storeReturnToDatabase(commitDataTypes, SourceType.ANALYSIS);
226  goodInfo = true;
227  }
228 
229  checkModelNameConsistency(decompRes.getFunction());
230  }
231 
232  if (!monitor.isCancelled() && !goodInfo) {
233  String msg = getStatusMsg();
234  msg = msg == null ? "" : ": " + msg;
235  Msg.debug(this, " Failed to decompile function: " + f.getName() + msg);
236  }
237  }
238  catch (Exception e) {
239  if (!monitor.isCancelled()) {
240  String errMsg = e.getMessage();
241  if (errMsg == null) {
242  errMsg = "Error decompiling function: " + e;
243  }
244  setStatusMsg(errMsg);
245  }
246  }
247  }

255  private boolean hasInconsistentResults(DecompileResults decompRes) {
256  HighFunction hfunc = decompRes.getHighFunction();
257 
258  if (hfunc == null) {
259  return false;
260  }
261 
262  Iterator<HighSymbol> symIter = hfunc.getLocalSymbolMap().getSymbols();
263  while (symIter.hasNext()) {
264  HighSymbol sym = symIter.next();
265  HighVariable highVar = sym.getHighVariable();
266  if (!(highVar instanceof HighLocal)) {
267  continue;
268  }
269 
270  if (!sym.getName().startsWith("in_")) {
271  continue;
272  }
273  // TODO: THIS IS A HACK!
274  if (sym.getName().equals("in_FS_OFFSET")) {
275  continue;
276  }
277  if (!sym.getHighVariable().getStorage().isRegisterStorage()) {
278  continue;
279  }
280 
281  Function func = hfunc.getFunction();
282  if (func != null) {
283  Address entryPoint = func.getEntryPoint();
284  BookmarkManager bookmarkManager =
285  hfunc.getFunction().getProgram().getBookmarkManager();
286  bookmarkManager.setBookmark(
287  entryPoint,
288  BookmarkType.WARNING,
289  "DecompilerParamID",
290  "Problem recovering parameters in function " + func.getName() + " at " +
291  func.getEntryPoint() + " unknown input variable " + sym.getName());
292  }
293  return true;
294 
295  // TODO: should unaff_ be checked?
296 
297  // TODO: should wierd stack references be checked?
298  }
299 
300  return false;
301  }

303  private void checkModelNameConsistency(Function func) {
304  int paramCount = func.getParameterCount();
305 
306  String modelName = func.getCallingConventionName();
307 
308  if (func.getStackPurgeSize() == 0 && paramCount > 0 &&
309  modelName.equals(CompilerSpec.CALLING_CONVENTION_stdcall)) {
310  try {
311  func.setCallingConvention(CompilerSpec.CALLING_CONVENTION_cdecl);
312  }
313  catch (InvalidInputException e) {
314  setStatusMsg("Invalid Calling Convention " + CompilerSpec.CALLING_CONVENTION_cdecl +

369

372  Function function = program.getFunctionManager().getFunctionAt(address);

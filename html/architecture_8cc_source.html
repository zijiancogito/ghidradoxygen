798 {

800  bool stackGrowth = true; // Default stack growth is in negative direction
801  if (basespace == (AddrSpace *)0)
802  throw LowlevelError("Unknown space name: "+el->getAttributeValue("space"));
803 
804  bool isreversejustify = false;
805  int4 numattr = el->getNumAttributes();
806  for(int4 i=0;i<numattr;++i) {
807  const string &attr( el->getAttributeName(i) );
808  if (attr == "reversejustify")
809  isreversejustify = xml_readbool(el->getAttributeValue(i));
810  else if (attr == "growth")
811  stackGrowth = el->getAttributeValue(i) == "negative";
812  }
813

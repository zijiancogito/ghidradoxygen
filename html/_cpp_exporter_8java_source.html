121  parallelDecompiler.dispose();

133  private void decompileAndExport(AddressSetView addrSet, Program program,
134  PrintWriter headerWriter, PrintWriter cFileWriter,
135  ChunkingParallelDecompiler<CPPResult> parallelDecompiler,
136  ChunkingTaskMonitor chunkingMonitor)
137  throws InterruptedException, Exception, CancelledException {
138 
139  int functionCount = program.getFunctionManager().getFunctionCount();
140  chunkingMonitor.doInitialize(functionCount);
141 
142  Listing listing = program.getListing();
143  FunctionIterator iterator = listing.getFunctions(addrSet, true);
144  List<Function> functions = new ArrayList<>();
145  for (int i = 0; iterator.hasNext(); i++) {
146  //
147  // Write results every so many items so that we don't blow out memory
148  //
149  if (i % 10000 == 0) {
150  List<CPPResult> results = parallelDecompiler.decompileFunctions(functions);
151  writeResults(results, headerWriter, cFileWriter, chunkingMonitor);
152  functions.clear();
153  }
154 
155  functions.add(iterator.next());
156  }
157 
158  // handle any remaining functions
159  List<CPPResult> results = parallelDecompiler.decompileFunctions(functions);
160  writeResults(results, headerWriter, cFileWriter, chunkingMonitor);
161  }

163  private void writeResults(List<CPPResult> results, PrintWriter headerWriter,
164  PrintWriter cFileWriter, TaskMonitor monitor) throws CancelledException {
165  monitor.checkCanceled();
166 
167  Collections.sort(results);
168 
169  StringBuilder headers = new StringBuilder();
170  StringBuilder bodies = new StringBuilder();
171  for (CPPResult entry : results) {
172  monitor.checkCanceled();
173  if (entry != null) {
174  String headerCode = entry.getHeaderCode();
175  if (headerCode != null) {
176  headers.append(headerCode);
177  headers.append(EOL);
178  }
179 
180  String bodyCode = entry.getBodyCode();
181  if (bodyCode != null) {
182  bodies.append(bodyCode);
183  bodies.append(EOL);
184  }
185  }
186  }
187 
188  monitor.checkCanceled();
189 
190  if (headerWriter != null) {
191  headerWriter.println(headers.toString());
192  }
193  if (cFileWriter != null) {
194  cFileWriter.print(bodies.toString());
195  }
196  }

424  @Override
425  public CPPResult process(Function function, TaskMonitor monitor) throws Exception {
426  if (monitor.isCancelled()) {
427  return null;
428  }
429 
430  DecompInterface decompiler = pool.get();
431  try {
432  CPPResult result = doWork(function, decompiler, monitor);
433  return result;
434  }
435  finally {

440  private CPPResult doWork(Function function, DecompInterface decompiler,
441  TaskMonitor monitor) {
442  Address entryPoint = function.getEntryPoint();
443  CodeUnit codeUnitAt = function.getProgram().getListing().getCodeUnitAt(entryPoint);
444  if (codeUnitAt == null || !(codeUnitAt instanceof Instruction)) {
445  return new CPPResult(entryPoint, function.getPrototypeString(false, false), null);
446  }
447 
448  monitor.setMessage("Decompiling " + function.getName());
449 
450  DecompileResults dr =
451  decompiler.decompileFunction(function, options.getDefaultTimeout(), monitor);
452  String errorMessage = dr.getErrorMessage();
453  if (!"".equals(errorMessage)) {
454  Msg.warn(CppExporter.this, "Error decompiling: " + errorMessage);
455  if (options.isWARNCommentIncluded()) {
456  monitor.incrementProgress(1);
457  return new CPPResult(entryPoint, null,
458  "/*" + EOL + "Unable to decompile '" + function.getName() + "'" + EOL +
459  "Cause: " + errorMessage + EOL + "*/" + EOL);
460  }
461  return null;
462  }
463 
464  DecompiledFunction decompiledFunction = dr.getDecompiledFunction();
465  return new CPPResult(entryPoint, decompiledFunction.getSignature(),
466  decompiledFunction.getC());
467  }
468  }

78 // This is the only important method for the LoadImage. It returns bytes from the static array
79 // depending on the address range requested
80 void MyLoadImage::loadFill(uint1 *ptr,int4 size,const Address &addr)
81 
82 {
83  uintb start = addr.getOffset();
84  uintb max = baseaddr + (length-1);
85  for(int4 i=0;i<size;++i) { // For every byte requestes
86  uintb curoff = start + i; // Calculate offset of byte
87  if ((curoff < baseaddr)||(curoff>max)) { // If byte does not fall in window
88  ptr[i] = 0; // return 0
89  continue;
90  }
91  uintb diff = curoff - baseaddr;
92  ptr[i] = data[(int4)diff]; // Otherwise return data from our window
93  }
94 }

160 static void dumpPcode(Translate &trans)
161 
162 { // Dump pcode translation of machine instructions
163  PcodeRawOut emit; // Set up the pcode dumper
164  AssemblyRaw assememit; // Set up the disassembly dumper
165  int4 length; // Number of bytes of each machine instruction
166 
167  Address addr(trans.getDefaultSpace(),0x80483b4); // First address to translate
168  Address lastaddr(trans.getDefaultSpace(),0x80483bf); // Last address
169 
170  while(addr < lastaddr) {
171  cout << "--- ";
172  trans.printAssembly(assememit,addr);
173  length = trans.oneInstruction(emit,addr); // Translate instruction
174  addr = addr + length; // Advance to next instruction
175  }

250 static void doEmulation(Translate &trans,LoadImage &loader)
251 
252 {
253  // Set up memory state object
254  MemoryImage loadmemory(trans.getDefaultSpace(),8,4096,&loader);
255  MemoryPageOverlay ramstate(trans.getDefaultSpace(),8,4096,&loadmemory);
256  MemoryHashOverlay registerstate(trans.getSpaceByName("register"),8,4096,4096,(MemoryBank *)0);
257  MemoryHashOverlay tmpstate(trans.getUniqueSpace(),8,4096,4096,(MemoryBank *)0);
258 
259  MemoryState memstate(&trans); // Instantiate the memory state object
260  memstate.setMemoryBank(&ramstate);
261  memstate.setMemoryBank(&registerstate);
262  memstate.setMemoryBank(&tmpstate);
263 
264  BreakTableCallBack breaktable(&trans); // Set up the callback object
265  EmulatePcodeCache emulater(&trans,&memstate,&breaktable); // Set up the emulator
266 
267  // Set up the initial register state for execution
268  memstate.setValue("ESP",0xbffffffc);
269  emulater.setExecuteAddress(Address(trans.getDefaultSpace(),0x80483b4));
270 
271  // Register callbacks
272  PutsCallBack putscallback;
273  PrintfCallBack printfcallback;
274  TerminateCallBack terminatecallback;
275  breaktable.registerAddressCallback(Address(trans.getDefaultSpace(),0x80482c8),&putscallback);
276  breaktable.registerAddressCallback(Address(trans.getDefaultSpace(),0x80482b8),&printfcallback);
277  breaktable.registerAddressCallback(Address(trans.getDefaultSpace(),0x804846b),&terminatecallback);
278 
279  emulater.setHalt(false);
280 
281  do {
282  emulater.executeInstruction();
283  } while(!emulater.getHalt());
284 }

286 int main(int argc,char **argv)
287 
288 {
289  if (argc != 2) {
290  cerr << "USAGE: " << argv[0] << " disassemble" << endl;
291  cerr << " " << argv[0] << " pcode" << endl;
292  cerr << " " << argv[0] << " emulate" << endl;
293  return 2;
294  }
295  string action(argv[1]);
296 
297  // Set up the loadimage
298  MyLoadImage loader(0x80483b4,myprog,408);
299  // loader->open();
300  // loader->adjustVma(adjustvma);
301 
302  // Set up the context object
303  ContextInternal context;
304 
305  // Set up the assembler/pcode-translator
306  string sleighfilename = "specfiles/x86.sla";
307  Sleigh trans(&loader,&context);
308 
309  // Read sleigh file into DOM
310  DocumentStorage docstorage;
311  Element *sleighroot = docstorage.openDocument(sleighfilename)->getRoot();
312  docstorage.registerTag(sleighroot);
313  trans.initialize(docstorage); // Initialize the translator
314 
315  // Now that context symbol names are loaded by the translator
316  // we can set the default context
317 
318  context.setVariableDefault("addrsize",1); // Address size is 32-bit
319  context.setVariableDefault("opsize",1); // Operand size is 32-bit
320 
321  if (action == "disassemble")
322  dumpAssembly(trans);
323  else if (action == "pcode")
324  dumpPcode(trans);
325  else if (action == "emulate")
326  doEmulation(trans,loader);
327  else
328  cerr << "Unknown action: "+action << endl;
329 }

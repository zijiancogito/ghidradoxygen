32  int4 exttype1 = intPromotionType(vn);

47 bool CastStrategyC::checkIntPromotionForExtension(const PcodeOp *op) const
48 
49 {
50  const Varnode *vn = op->getIn(0);
51  int4 exttype = intPromotionType(vn);
52  if (exttype == NO_PROMOTION) return false;
53  if (exttype == UNKNOWN_PROMOTION) return true; // If there is an extension and we don't know type, we need a cast
54 
55  // Test if the promotion extension matches the explicit extension
56  if (((exttype & UNSIGNED_EXTENSION) != 0) && (op->code() == CPUI_INT_ZEXT)) return false;
57  if (((exttype & SIGNED_EXTENSION) != 0) && (op->code() == CPUI_INT_SEXT)) return false;
58  return true; // Otherwise we need a cast before we extend

61 int4 CastStrategyC::localExtensionType(const Varnode *vn) const
62 
63 {
64  type_metatype meta = vn->getHigh()->getType()->getMetatype();
65  int4 natural; // 1= natural zero extension, 2= natural sign extension
66  if ((meta == TYPE_UINT)||(meta == TYPE_BOOL)||(meta == TYPE_UNKNOWN))
67  natural = UNSIGNED_EXTENSION;
68  else if (meta == TYPE_INT)
69  natural = SIGNED_EXTENSION;
70  else
71  return UNKNOWN_PROMOTION;
72  if (vn->isConstant()) {
73  if (!signbit_negative(vn->getOffset(),vn->getSize())) // If the high-bit is zero
74  return EITHER_EXTENSION; // Can be viewed as either extension
75  return natural;
76  }
77  if (vn->isExplicit())
78  return natural;
79  if (!vn->isWritten())
80  return UNKNOWN_PROMOTION;
81  const PcodeOp *op = vn->getDef();
82  if (op->isBoolOutput())
83  return EITHER_EXTENSION;
84  OpCode opc = op->code();
85  if ((opc == CPUI_CAST)||(opc == CPUI_LOAD)||op->isCall())
86  return natural;
87  if (opc == CPUI_INT_AND) { // This is kind of recursing
88  const Varnode *tmpvn = op->getIn(1);
89  if (tmpvn->isConstant()) {
90  if (!signbit_negative(tmpvn->getOffset(),tmpvn->getSize()))
91  return EITHER_EXTENSION;
92  return natural;
93  }
94  }
95  return UNKNOWN_PROMOTION;
96 }

98 int4 CastStrategyC::intPromotionType(const Varnode *vn) const
99 
100 {
101  int4 val;
102  if (vn->getSize() >= promoteSize)
103  return NO_PROMOTION;
104  if (vn->isConstant())
105  return localExtensionType(vn);
106  if (vn->isExplicit())
107  return NO_PROMOTION;
108  if (!vn->isWritten()) return UNKNOWN_PROMOTION;
109  const PcodeOp *op = vn->getDef();
110  const Varnode *othervn;
111  switch(op->code()) {
112  case CPUI_INT_AND:
113  othervn = op->getIn(1);
114  if ((localExtensionType(othervn) & UNSIGNED_EXTENSION) != 0)
115  return UNSIGNED_EXTENSION;
116  othervn = op->getIn(0);
117  if ((localExtensionType(othervn) & UNSIGNED_EXTENSION) != 0)
118  return UNSIGNED_EXTENSION; // If either side has zero extension, result has zero extension
119  break;
120  case CPUI_INT_RIGHT:
121  othervn = op->getIn(0);
122  val = localExtensionType(othervn);
123  if ((val & UNSIGNED_EXTENSION) != 0) // If the input provably zero extends
124  return val; // then the result is a zero extension (plus possibly a sign extension)
125  break;
126  case CPUI_INT_SRIGHT:
127  othervn = op->getIn(0);
128  val = localExtensionType(othervn);
129  if ((val & SIGNED_EXTENSION) != 0) // If input can be construed as a sign-extension
130  return val; // then the result is a sign extension (plus possibly a zero extension)
131  break;
132  case CPUI_INT_XOR:
133  case CPUI_INT_OR:
134  case CPUI_INT_DIV:
135  case CPUI_INT_REM:
136  othervn = op->getIn(0);
137  if ((localExtensionType(othervn) & UNSIGNED_EXTENSION) == 0)
138  return UNKNOWN_PROMOTION;
139  othervn = op->getIn(1);
140  if ((localExtensionType(othervn) & UNSIGNED_EXTENSION) == 0)
141  return UNKNOWN_PROMOTION;
142  return UNSIGNED_EXTENSION; // If both sides have zero extension, result has zero extension
143  case CPUI_INT_SDIV:
144  case CPUI_INT_SREM:
145  othervn = op->getIn(0);
146  if ((localExtensionType(othervn) & SIGNED_EXTENSION) == 0)
147  return UNKNOWN_PROMOTION;
148  othervn = op->getIn(1);
149  if ((localExtensionType(othervn) & SIGNED_EXTENSION) == 0)
150  return UNKNOWN_PROMOTION;
151  return SIGNED_EXTENSION; // If both sides have sign extension, result has sign extension
152  case CPUI_INT_NEGATE:
153  case CPUI_INT_2COMP:
154  othervn = op->getIn(0);
155  if ((localExtensionType(othervn) & SIGNED_EXTENSION) != 0)
156  return SIGNED_EXTENSION;
157  break;
158  case CPUI_INT_ADD:
159  case CPUI_INT_SUB:
160  case CPUI_INT_LEFT:
161  case CPUI_INT_MULT:
162  break;
163  default:
164  return NO_PROMOTION; // No integer promotion at all
165  }
166  return UNKNOWN_PROMOTION;
167 }

169 Datatype *CastStrategyC::castStandard(Datatype *reqtype,Datatype *curtype,
170  bool care_uint_int,bool care_ptr_uint) const
171 
172 { // Generic casting rules that apply for most ops
173  if (curtype == reqtype) return (Datatype *)0; // Types are equal, no cast required
174  Datatype *reqbase = reqtype;
175  Datatype *curbase = curtype;
176  bool isptr = false;
177  while((reqbase->getMetatype()==TYPE_PTR)&&(curbase->getMetatype()==TYPE_PTR)) {
178  reqbase = ((const TypePointer *)reqbase)->getPtrTo();
179  curbase = ((const TypePointer *)curbase)->getPtrTo();
180  care_uint_int = true;
181  isptr = true;
182  }
183  if (curtype == reqtype) return (Datatype *)0; // Different typedefs could point to the same type
184  if ((reqbase->getMetatype()==TYPE_VOID)||(reqbase->getMetatype()==TYPE_VOID))
185  return (Datatype *)0; // Don't cast from or to VOID
186  if (reqbase->getSize() != curbase->getSize()) return reqtype; // Always cast change in size
187  switch(reqbase->getMetatype()) {
188  case TYPE_UNKNOWN:
189  return (Datatype *)0;
190  case TYPE_UINT:
191  if (!care_uint_int) {
192  type_metatype meta = curbase->getMetatype();
193  // Note: meta can be TYPE_UINT if curbase is typedef/enumerated
194  if ((meta==TYPE_UNKNOWN)||(meta==TYPE_INT)||(meta==TYPE_UINT)||(meta==TYPE_BOOL))
195  return (Datatype *)0;
196  }
197  else {
198  type_metatype meta = curbase->getMetatype();
199  if ((meta == TYPE_UINT)||(meta==TYPE_BOOL)) // Can be TYPE_UINT for typedef/enumerated
200  return (Datatype *)0;
201  if (isptr && (meta==TYPE_UNKNOWN)) // Don't cast pointers to unknown
202  return (Datatype *)0;
203  }
204  if ((!care_ptr_uint)&&(curbase->getMetatype()==TYPE_PTR))
205  return (Datatype *)0;
206  break;
207  case TYPE_INT:
208  if (!care_uint_int) {
209  type_metatype meta = curbase->getMetatype();
210  // Note: meta can be TYPE_INT if curbase is an enumerated type
211  if ((meta==TYPE_UNKNOWN)||(meta==TYPE_INT)||(meta==TYPE_UINT)||(meta==TYPE_BOOL))
212  return (Datatype *)0;
213  }
214  else {
215  type_metatype meta = curbase->getMetatype();
216  if ((meta == TYPE_INT)||(meta == TYPE_BOOL))
217  return (Datatype *)0; // Can be TYPE_INT for typedef/enumerated/char
218  if (isptr && (meta==TYPE_UNKNOWN)) // Don't cast pointers to unknown
219  return (Datatype *)0;
220  }
221  break;
222  case TYPE_CODE:
223  if (curbase->getMetatype() == TYPE_CODE) {
224  // Don't cast between function pointer and generic code pointer
225  if (((TypeCode *)reqbase)->getPrototype() == (const FuncProto *)0)
226  return (Datatype *)0;
227  if (((TypeCode *)curbase)->getPrototype() == (const FuncProto *)0)
228  return (Datatype *)0;
229  }
230  break;
231  default:
232  break;
233  }
234 
235  return reqtype;
236 }

78  private List<Function> filterFunctions(Program p, FunctionIterator iter, TaskMonitor monitor) {
79 
80  List<Function> results = new ArrayList<>();
81  Listing listing = p.getListing();
82  while (iter.hasNext()) {
83  Function f = iter.next();
84 
85  if (monitor.isCancelled()) {
86  return Collections.emptyList();
87  }
88 
89  Address entryPoint = f.getEntryPoint();
90  CodeUnit codeUnitAt = listing.getCodeUnitAt(entryPoint);
91  if (codeUnitAt == null) {
92  continue;
93  }
94 
95  if (codeUnitAt instanceof Instruction) {
96  results.add(f);
97  }
98  }
99  return results;
100  }

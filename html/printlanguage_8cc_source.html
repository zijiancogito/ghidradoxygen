62 bool OpToken::parentheses(const OpToken &op2,int4 stage) const
63 
64 {
65  switch(type) {
66  case space:
67  case binary:
68  if (precedence > op2.precedence) return true;
69  if (precedence < op2.precedence) return false;
70  if (associative && (this == &op2)) return false;
71  // If operators are adjacent to each other, the
72  // operator printed first must be evaluated first
73  // In this case op2 must be evaluated first, so we
74  // check if it is printed first (in first stage of binary)
75  if ((op2.type==postsurround)&&(stage==0)) return false;
76  return true;
77  case unary_prefix:
78  if (precedence > op2.precedence) return true;
79  if (precedence < op2.precedence) return false;
80  // if (associative && (this == &op2)) return false;
81  if ((op2.type==unary_prefix)||(op2.type==presurround)) return false;
82  return true;
83  case postsurround:
84  if (stage==1) return false; // Inside the surround
85  if (precedence > op2.precedence) return true;
86  if (precedence < op2.precedence) return false;
87  // If the precedences are equal, we know this postsurround
88  // comes after, so op2 being first doesn't need parens
89  if ((op2.type==postsurround)||(op2.type==binary)) return false;
90  // if (associative && (this == &op2)) return false;
91  return true;
92  case presurround:
93  if (stage==0) return false; // Inside the surround
94  if (precedence > op2.precedence) return true;
95  if (precedence < op2.precedence) return false;
96  // if (associative && (this == &op2)) return false;
97  if ((op2.type==unary_prefix)||(op2.type==presurround)) return false;
98  return true;
99  }
100 
101  return true;
102 }

166 void PrintLanguage::pushOp(const OpToken *tok,const PcodeOp *op)
167 
168 {
169  if (pending < nodepend.size()) // Pending varnode pushes before op
170  recurse(); // So we must recurse
171 
172  bool paren;
173  int4 id;
174 
175  if (revpol.empty()) {
176  paren = false;
177  id = emit->openGroup();

179  else {

181  paren = revpol.back().tok->parentheses(*tok,revpol.back().visited);
182  if (paren)

184  else
185  id = emit->openGroup();
186  }
187  revpol.push_back(ReversePolish());
188  revpol.back().tok = tok;
189  revpol.back().visited = 0;
190  revpol.back().paren = paren;

199 void PrintLanguage::pushAtom(const Atom &atom)
200 
201 {
202  if (pending < nodepend.size()) // pending varnodes before atom
203  recurse(); // So we must recurse
204 
205  if (revpol.empty())
206  emitAtom(atom);
207  else {
208  emitOp(revpol.back());
209  emitAtom(atom);
210  do {
211  revpol.back().visited += 1;
212  if (revpol.back().visited == revpol.back().tok->stage) {
213  emitOp(revpol.back());
214  if (revpol.back().paren)
215  emit->closeParen(')',revpol.back().id);
216  else
217  emit->closeGroup(revpol.back().id);
218  revpol.pop_back();
219  }
220  else
221  break;
222  } while(!revpol.empty());
223  }

257 {

314 void PrintLanguage::emitOp(const ReversePolish &entry)
315 
316 {
317  switch(entry.tok->type) {
318  case OpToken::binary:
319  if (entry.visited!=1) return;
320  emit->spaces(entry.tok->spacing,entry.tok->bump); // Spacing around operator
321  emit->tagOp(entry.tok->print,EmitXml::no_color,entry.op);
322  emit->spaces(entry.tok->spacing,entry.tok->bump);
323  break;
324  case OpToken::unary_prefix:
325  if (entry.visited!=0) return;
326  emit->tagOp(entry.tok->print,EmitXml::no_color,entry.op);
327  emit->spaces(entry.tok->spacing,entry.tok->bump);
328  break;
329  case OpToken::postsurround:
330  if (entry.visited==0) return;
331  if (entry.visited==1) { // Front surround token
332  emit->spaces(entry.tok->spacing,entry.tok->bump);
333  entry.id2 = emit->openParen(entry.tok->print[0]);
334  emit->spaces(0,entry.tok->bump);
335  }
336  else { // Back surround token
337  emit->closeParen(entry.tok->print[1],entry.id2);
338  }
339  break;
340  case OpToken::presurround:
341  if (entry.visited==2) return;
342  if (entry.visited==0) { // Front surround token
343  entry.id2 = emit->openParen(entry.tok->print[0]);
344  }
345  else { // Back surround token
346  emit->closeParen(entry.tok->print[1],entry.id2);
347  emit->spaces(entry.tok->spacing,entry.tok->bump);
348  }
349  break;
350  case OpToken::space: // Like binary but just a space between
351  if (entry.visited != 1) return;
352  emit->spaces(entry.tok->spacing,entry.tok->bump);
353  break;
354  }
355 }

394 bool PrintLanguage::unicodeNeedsEscape(int4 codepoint)
395 
396 {
397  if (codepoint < 0x20) { // C0 Control characters
398  return true;
399  }
400  if (codepoint < 0x7F) { // Printable ASCII
401  switch(codepoint) {
402  case 92: // back-slash
403  case '"':
404  case '\'':
405  return true;
406  }
407  return false;
408  }
409  if (codepoint < 0x100) {
410  if (codepoint > 0xa0) { // Printable codepoints A1-FF
411  return false;
412  }
413  return true;
414  }
415  if (codepoint >= 0x2fa20) { // Up to last currently defined language
416  return true;
417  }
418  if (codepoint < 0x2000) {
419  if (codepoint >= 0x180b && codepoint <= 0x180e) {
420  return true; // Mongolian separators
421  }
422  if (codepoint == 0x61c) {
423  return true; // arabic letter mark
424  }
425  if (codepoint == 0x1680) {
426  return true; // ogham space mark
427  }
428  return false;
429  }
430  if (codepoint < 0x3000) {
431  if (codepoint < 0x2010) {
432  return true; // white space and separators
433  }
434  if (codepoint >= 0x2028 && codepoint <= 0x202f) {
435  return true; // white space and separators
436  }
437  if (codepoint == 0x205f || codepoint == 0x2060) {
438  return true; // white space and word joiner
439  }
440  if (codepoint >= 0x2066 && codepoint <= 0x206f) {
441  return true; // bidirectional markers
442  }
443  return false;
444  }
445  if (codepoint < 0xe000) {
446  if (codepoint == 0x3000) {
447  return true; // ideographic space
448  }
449  return false;
450  }
451  if (codepoint < 0xf900) {
452  return true; // private use
453  }
454  if (codepoint >= 0xfe00 && codepoint <= 0xfe0f) {
455  return true; // variation selectors
456  }
457  if (codepoint == 0xfeff) {
458  return true; // zero width non-breaking space
459  }
460  if (codepoint >= 0xfff0 && codepoint <= 0xffff) {
461  return true; // interlinear specials
462  }
463  return false;

470 void PrintLanguage::writeUtf8(ostream &s,int4 codepoint)
471 
472 {
473  uint1 bytes[4];
474  int4 size;
475 
476  if (codepoint < 0)
477  throw LowlevelError("Negative unicode codepoint");
478  if (codepoint < 128) {
479  s.put((uint1)codepoint);
480  return;
481  }
482  int4 bits = mostsigbit_set(codepoint) + 1;
483  if (bits > 21)
484  throw LowlevelError("Bad unicode codepoint");
485  if (bits < 12) { // Encode with two bytes
486  bytes[0] = 0xc0 ^ ((codepoint >> 6)&0x1f);
487  bytes[1] = 0x80 ^ (codepoint & 0x3f);
488  size = 2;
489  }
490  else if (bits < 17) {
491  bytes[0] = 0xe0 ^ ((codepoint >> 12)&0xf);
492  bytes[1] = 0x80 ^ ((codepoint >> 6)&0x3f);
493  bytes[2] = 0x80 ^ (codepoint & 0x3f);
494  size = 3;
495  }
496  else {
497  bytes[0] = 0xf0 ^ ((codepoint >> 18) & 7);
498  bytes[1] = 0x80 ^ ((codepoint >> 12) & 0x3f);
499  bytes[2] = 0x80 ^ ((codepoint >> 6) & 0x3f);
500  bytes[3] = 0x80 ^ (codepoint & 0x3f);
501  size = 4;
502  }
503  s.write((char *)bytes, size);
504 }

535 int4 PrintLanguage::getCodepoint(const uint1 *buf,int4 charsize,bool bigend,int4 &skip)
536 
537 {
538  int4 codepoint;
539  int4 sk = 0;
540  if (charsize==2) { // UTF-16
541  codepoint = readUtf16(buf,bigend);
542  sk += 2;
543  if ((codepoint>=0xD800)&&(codepoint<=0xDBFF)) { // high surrogate
544  int4 trail=readUtf16(buf+2,bigend);
545  sk += 2;
546  if ((trail<0xDC00)||(trail>0xDFFF)) return -1; // Bad trail
547  codepoint = (codepoint<<10) + trail + (0x10000 - (0xD800 << 10) - 0xDC00);
548  }
549  else if ((codepoint>=0xDC00)&&(codepoint<=0xDFFF)) return -1; // trail before high
550  }
551  else if (charsize==1) { // UTF-8
552  int4 val = buf[0];
553  if ((val&0x80)==0) {
554  codepoint = val;
555  sk = 1;
556  }
557  else if ((val&0xe0)==0xc0) {
558  int4 val2 = buf[1];
559  sk = 2;
560  if ((val2&0xc0)!=0x80) return -1; // Not a valid UTF8-encoding
561  codepoint = ((val&0x1f)<<6) | (val2 & 0x3f);
562  }
563  else if ((val&0xf0)==0xe0) {
564  int4 val2 = buf[1];
565  int4 val3 = buf[2];
566  sk = 3;
567  if (((val2&0xc0)!=0x80)||((val3&0xc0)!=0x80)) return -1; // invalid encoding
568  codepoint = ((val&0xf)<<12) | ((val2&0x3f)<<6) | (val3 & 0x3f);
569  }
570  else if ((val&0xf8)==0xf0) {
571  int4 val2 = buf[1];
572  int4 val3 = buf[2];
573  int4 val4 = buf[3];
574  sk = 4;
575  if (((val2&0xc0)!=0x80)||((val3&0xc0)!=0x80)||((val4&0xc0)!=0x80)) return -1; // invalid encoding
576  codepoint = ((val&7)<<18) | ((val2&0x3f)<<12) | ((val3&0x3f)<<6) | (val4 & 0x3f);
577  }
578  else
579  return -1;
580  }
581  else if (charsize == 4) { // UTF-32
582  sk = 4;
583  if (bigend)
584  codepoint = (buf[0]<<24) + (buf[1]<<16) + (buf[2]<<8) + buf[3];
585  else
586  codepoint = (buf[3]<<24) + (buf[2]<<16) + (buf[1]<<8) + buf[0];
587  }
588  else
589  return -1;
590  if (codepoint >= 0xd800 && codepoint <= 0xdfff)
591  return -1; // Reserved for surrogates, invalid codepoints
592  skip = sk;
593  return codepoint;
594 }

622 
623 {

626  pending = nodepend.size(); // Lay claim to the rest
627  while(final < pending) {
628  const Varnode *vn = nodepend.back().vn;
629  const PcodeOp *op = nodepend.back().op;

635  else

638  }
639  mods = modsave;

679 void PrintLanguage::emitLineComment(int4 indent,const Comment *comm)
680 
681 {
682  const string &text( comm->getText() );
683  const AddrSpace *spc = comm->getAddr().getSpace();
684  uintb off = comm->getAddr().getOffset();
685  if (indent <0)
686  indent = line_commentindent; // User specified default indent
687  emit->tagLine(indent);
688  int4 id = emit->startComment();
689  // The comment delimeters should not be printed as
690  // comment tags, so that they won't get filled
691  emit->tagComment(commentstart.c_str(),EmitXml::comment_color,
692  spc,off);
693  int4 pos = 0;
694  while(pos < text.size()) {
695  char tok = text[pos++];
696  if ((tok==' ')||(tok=='\t')) {
697  int4 count = 1;
698  while(pos<text.size()) {
699  tok = text[pos];
700  if ((tok!=' ')&&(tok!='\t')) break;
701  count += 1;
702  pos += 1;
703  }
704  emit->spaces(count);
705  }
706  else if (tok=='\n')
707  emit->tagLine();
708  else if (tok=='\r') {
709  }
710  else {
711  int4 count = 1;
712  while(pos < text.size()) {
713  tok = text[pos];
714  if (isspace(tok)) break;
715  count += 1;
716  pos += 1;
717  }
718  string sub = text.substr(pos-count,count);
719  emit->tagComment(sub.c_str(),EmitXml::comment_color,
720  spc,off);
721  }
722  }
723  if (commentend.size() != 0)
724  emit->tagComment(commentend.c_str(),EmitXml::comment_color,
725  spc,off);
726  emit->stopComment(id);

793 
794 {

796 
797  uintb tmp = val;
798  int4 dig,setdig;
799  if (tmp==0) return 10;

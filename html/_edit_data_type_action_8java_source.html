112  DataType baseDataType = DataTypeUtils.getBaseDataType(dataType);
113  final DataTypeManagerService service = tool.getService(DataTypeManagerService.class);

117  @Override
118  public void actionPerformed(ActionContext context) {
119  // Note: we intentionally do this check here and not in isEnabledForContext() so
120  // that global events do not get triggered.
121  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
122  if (decompilerActionContext.isDecompiling()) {
123  Msg.showInfo(getClass(),
124  context.getComponentProvider().getComponent(),
125  "Decompiler Action Blocked", "You cannot perform Decompiler actions while the Decompiler is busy");
126  return;
127  }
128 
129  DecompilerPanel decompilerPanel = controller.getDecompilerPanel();
130  ClangToken tokenAtCursor = decompilerPanel.getTokenAtCursor();
131  HighVariable variable = tokenAtCursor.getHighVariable();
132 
133  DataType dataType = variable.getDataType();
134  DataType baseDataType = DataTypeUtils.getBaseDataType(dataType);
135  DataTypeManager dataTypeManager = decompilerActionContext.getProgram().getDataTypeManager();
136  DataTypeManager baseDtDTM = baseDataType.getDataTypeManager();
137  if (baseDtDTM != dataTypeManager) {
138  baseDataType = baseDataType.clone(dataTypeManager);
139  }
140  final DataTypeManagerService service = tool.getService(DataTypeManagerService.class);
141  service.edit(baseDataType);

30 void PreferSplitManager::fillinInstance(SplitInstance *inst,bool bigendian,bool sethi,bool setlo)
31 
32 { // Define the varnode pieces of -inst-
33  Varnode *vn = inst->vn;
34  int4 losize;
35  if (bigendian)
36  losize = vn->getSize() - inst->splitoffset;
37  else
38  losize = inst->splitoffset;
39  int4 hisize = vn->getSize() - losize;
40  if (vn->isConstant()) {
41  uintb origval = vn->getOffset();
42 
43  uintb loval = origval & calc_mask( losize );// Split the constant into two pieces
44  uintb hival = (origval >> 8*losize) & calc_mask( hisize );
45  if (setlo && (inst->lo == (Varnode *)0))
46  inst->lo = data->newConstant(losize,loval);
47  if (sethi && (inst->hi == (Varnode *)0))
48  inst->hi = data->newConstant(hisize,hival);
49  }
50  else {
51  if (bigendian) {
52  if (setlo && (inst->lo == (Varnode *)0))
53  inst->lo = data->newVarnode(losize,vn->getAddr() + inst->splitoffset);
54  if (sethi && (inst->hi == (Varnode *)0))
55  inst->hi = data->newVarnode(hisize,vn->getAddr());
56  }
57  else {
58  if (setlo && (inst->lo == (Varnode *)0))
59  inst->lo = data->newVarnode(losize,vn->getAddr());
60  if (sethi && (inst->hi == (Varnode *)0))
61  inst->hi = data->newVarnode(hisize,vn->getAddr() + inst->splitoffset);
62  }
63  }

66 void PreferSplitManager::createCopyOps(SplitInstance *ininst,SplitInstance *outinst,PcodeOp *op,bool istemp)
67 
68 { // Create COPY ops based on input -ininst- and output -outinst- to replace -op-
69  PcodeOp *hiop = data->newOp(1,op->getAddr()); // Create two new COPYs
70  PcodeOp *loop = data->newOp(1,op->getAddr());
71  data->opSetOpcode(hiop,CPUI_COPY);
72  data->opSetOpcode(loop,CPUI_COPY);
73 
74  data->opInsertAfter(loop,op); // Insert new COPYs at same position as original operation
75  data->opInsertAfter(hiop,op);
76  data->opUnsetInput(op,0); // Unset input so we can reassign free inputs to new ops
77 
78  data->opSetOutput(hiop,outinst->hi); // Outputs
79  data->opSetOutput(loop,outinst->lo);
80  data->opSetInput(hiop,ininst->hi,0);
81  data->opSetInput(loop,ininst->lo,0);
82  tempsplits.push_back(hiop);
83  tempsplits.push_back(loop);

246 
247 { // Knowing -op- is a CPUI_SUBPIECE that extracts a logical piece from -inst-, rewrite it to a copy
248  Varnode *vn = inst->vn;

268 void PreferSplitManager::splitLoad(SplitInstance *inst,PcodeOp *op)
269 
270 { // Knowing -op- is a CPUI_LOAD that defines the -inst- varnode, split it into two pieces
271  bool bigendian = inst->vn->getSpace()->isBigEndian();
272  fillinInstance(inst,bigendian,true,true);
273  PcodeOp *hiop = data->newOp(2,op->getAddr()); // Create two new LOAD ops
274  PcodeOp *loop = data->newOp(2,op->getAddr());
275  PcodeOp *addop = data->newOp(2,op->getAddr());
276  Varnode *ptrvn = op->getIn(1);
277 
278  data->opSetOpcode(hiop,CPUI_LOAD);
279  data->opSetOpcode(loop,CPUI_LOAD);
280 
281  data->opSetOpcode(addop,CPUI_INT_ADD); // Create a new ADD op to calculate and hold the second pointer
282 
283  data->opInsertAfter(loop,op);
284  data->opInsertAfter(hiop,op);
285  data->opInsertAfter(addop,op);
286  data->opUnsetInput(op,1); // Free up ptrvn
287 
288  Varnode *addvn = data->newUniqueOut(ptrvn->getSize(),addop);
289  data->opSetInput(addop,ptrvn,0);
290  data->opSetInput(addop,data->newConstant(ptrvn->getSize(),inst->splitoffset),1);
291 
292  data->opSetOutput(hiop,inst->hi); // Outputs are the pieces of the original
293  data->opSetOutput(loop,inst->lo);
294  Varnode *spaceid = op->getIn(0);
295  AddrSpace *spc = Address::getSpaceFromConst(spaceid->getAddr());
296  spaceid = data->newConstant(spaceid->getSize(),spaceid->getOffset()); // Duplicate original spaceid into new LOADs
297  data->opSetInput(hiop,spaceid,0);
298  spaceid = data->newConstant(spaceid->getSize(),spaceid->getOffset());
299  data->opSetInput(loop,spaceid,0);
300  if (ptrvn->isFree()) // Don't read a free varnode twice
301  ptrvn = data->newVarnode(ptrvn->getSize(),ptrvn->getSpace(),ptrvn->getOffset());
302 
303  if (spc->isBigEndian()) {
304  data->opSetInput(hiop,ptrvn,1);
305  data->opSetInput(loop,addvn,1);
306  }
307  else {
308  data->opSetInput(hiop,addvn,1);
309  data->opSetInput(loop,ptrvn,1);
310  }
311 }

364 bool PreferSplitManager::splitVarnode(SplitInstance *inst)
365 
366 { // Test if -vn- can be readily split, if so, do the split
367  Varnode *vn = inst->vn;
368  bool istemp;
369  if (vn->isWritten()) {
370  if (!vn->hasNoDescend()) return false; // Already linked in
371  PcodeOp *op = vn->getDef();
372  switch (op->code()) {
373  case CPUI_COPY:
374  if (!testDefiningCopy(inst,op,istemp))
375  return false;
376  splitDefiningCopy(inst,op,istemp);
377  break;
378  case CPUI_PIECE:
379  if (!testPiece(inst,op))
380  return false;
381  splitPiece(inst,op);
382  break;
383  case CPUI_LOAD:
384  if (!testLoad(inst,op))
385  return false;
386  splitLoad(inst,op);
387  break;
388  case CPUI_INT_ZEXT:
389  if (!testZext(inst,op))
390  return false;
391  splitZext(inst,op);
392  break;
393  default:
394  return false;
395  }
396  data->opDestroy(op);
397  }
398  else {
399  if (!vn->isFree()) return false; // Make sure vn is not already a marked input
400  PcodeOp *op = vn->loneDescend();
401  if (op == (PcodeOp *)0) // vn must be read exactly once
402  return false;
403  switch(op->code()) {
404  case CPUI_COPY:
405  if (!testReadingCopy(inst,op,istemp))
406  return false;
407  splitReadingCopy(inst,op,istemp);
408  break;
409  case CPUI_SUBPIECE:
410  if (!testSubpiece(inst,op))
411  return false;
412  splitSubpiece(inst,op);
413  return true; // Do not destroy op, it has been transformed
414  case CPUI_STORE:
415  if (!testStore(inst,op))
416  return false;
417  splitStore(inst,op);
418  break;
419  default:
420  return false;
421  }
422  data->opDestroy(op); // Original op is now dead
423  }
424  return true;
425 }

427 void PreferSplitManager::splitRecord(const PreferSplitRecord &rec)
428 
429 {
430  Address addr = rec.storage.getAddr();
431  VarnodeLocSet::const_iterator iter,enditer;
432 
433  SplitInstance inst((Varnode *)0,rec.splitoffset);
434  iter = data->beginLoc(rec.storage.size,addr);

442  iter = data->beginLoc(rec.storage.size,addr);

562 void PreferSplitManager::splitAdditional(void)
563 
564 {
565  vector<PcodeOp *> defops;
566  for(int4 i=0;i<tempsplits.size();++i) {
567  PcodeOp *op = tempsplits[i]; // Look at everything connected to COPYs in -tempsplits-
568  if (op->isDead()) continue;
569  Varnode *vn = op->getIn(0);
570  if (vn->isWritten()) {
571  PcodeOp *defop = vn->getDef();
572  if (defop->code() == CPUI_SUBPIECE) { // SUBPIECEs flowing into the COPY
573  Varnode *invn = defop->getIn(0);
574  if (invn->getSpace()->getType() == IPTR_INTERNAL) // Might be from a temporary that needs further splitting
575  defops.push_back(defop);
576  }
577  }
578  list<PcodeOp *>::const_iterator iter,enditer;
579  iter = op->getOut()->beginDescend();
580  enditer = op->getOut()->endDescend();
581  while(iter != enditer) {
582  PcodeOp *defop = *iter;
583  ++iter;
584  if (defop->code() == CPUI_PIECE) { // COPY flowing into PIECEs
585  Varnode *outvn = defop->getOut();
586  if (outvn->getSpace()->getType() == IPTR_INTERNAL) // Might be to a temporary that needs further splitting
587  defops.push_back(defop);
588  }
589  }
590  }
591  for(int4 i=0;i<defops.size();++i) {
592  PcodeOp *op = defops[i];
593  if (op->isDead()) continue;
594  if (op->code() == CPUI_PIECE) {
595  int4 splitoff;
596  Varnode *vn = op->getOut();
597  if (vn->getSpace()->isBigEndian())
598  splitoff = op->getIn(0)->getSize();
599  else
600  splitoff = op->getIn(1)->getSize();
601  SplitInstance inst(vn,splitoff);
602  if (testTemporary(&inst))
603  splitTemporary(&inst);
604  }
605  else if (op->code() == CPUI_SUBPIECE) {
606  int4 splitoff;
607  Varnode *vn = op->getIn(0);
608  uintb suboff = op->getIn(1)->getOffset();

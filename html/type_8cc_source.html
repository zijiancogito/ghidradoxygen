110 int4 Datatype::compareDependency(const Datatype &op) const
111 
112 {
113  if (size != op.size) return (op.size-size);
114  if (metatype != op.metatype) return (metatype < op.metatype) ? -1 : 1;
115  uint4 fl = flags & (~coretype);
116  uint4 opfl = op.flags & (~coretype);
117  // We need to be careful here, we compare flags so that enum types are more specific than base int or uint,

119  // we don't want char to be more specific than int1 because char is the default size 1 integer type.

542 void TypeEnum::setNameMap(const map<uintb,string> &nmap)
543 
544 {
545  map<uintb,string>::const_iterator iter;
546  uintb curmask,lastmask;
547  int4 maxbit;
548  int4 curmaxbit;
549  bool fieldisempty;
550 
551  namemap = nmap;
552  masklist.clear();
553 
554  flags &= ~((uint4)poweroftwo);
555 
556  maxbit = 8 * size - 1;
557 
558  curmaxbit = 0;
559  while(curmaxbit <= maxbit) {
560  curmask = 1;
561  curmask <<= curmaxbit;
562  lastmask = 0;
563  fieldisempty = true;
564  while(curmask != lastmask) { // Repeat until there is no change in the current mask
565  lastmask = curmask; // Note changes from last time through
566 
567  for(iter=namemap.begin();iter!=namemap.end();++iter) { // For every named enumeration value
568  uintb val = (*iter).first;
569  if ((val & curmask) != 0) { // If the value shares ANY bits in common with the current mask
570  curmask |= val; // Absorb ALL defined bits of the value into the current mask
571  fieldisempty = false;
572  }
573  }
574 
575  // Fill in any holes in the mask (bit field must consist of contiguous bits
576  int4 lsb = leastsigbit_set(curmask);
577  int4 msb = mostsigbit_set(curmask);
578  if (msb > curmaxbit)
579  curmaxbit = msb;
580 
581  uintb mask1 = 1;
582  mask1 = (mask1 << lsb) - 1; // every bit below lsb is set to 1
583  uintb mask2 = 1;
584  mask2 <<= msb;
585  mask2 <<= 1;
586  mask2 -= 1; // every bit below or equal to msb is set to 1
587  curmask = mask1 ^ mask2;
588  }
589  if (fieldisempty) { // If no value hits this bit
590  if (!masklist.empty())
591  masklist.back() |= curmask; // Include the bit with the previous mask
592  else
593  masklist.push_back(curmask);
594  }
595  else
596  masklist.push_back(curmask);
597  curmaxbit += 1;
598  }
599  if (masklist.size() > 1)
600  flags |= poweroftwo;
601 }

614

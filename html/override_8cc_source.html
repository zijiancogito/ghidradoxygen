284 
285 {

287  multistagejump.empty() && flowoverride.empty())

310  s << "<indirectoverride>";

317 
318  for(fiter=protoover.begin();fiter!=protoover.end();++fiter) {

322  s << "</protooverride>\n";

324

348 void Override::restoreXml(const Element *el,Architecture *glb)
349 
350 {
351  const List &list(el->getChildren());
352  List::const_iterator iter;
353 
354  for(iter=list.begin();iter!=list.end();++iter) {
355  const Element *subel = *iter;
356 
357  if (subel->getName() == "indirectoverride") {
358  const List &list2(subel->getChildren());
359  List::const_iterator iter2 = list2.begin();
360  Address callpoint = Address::restoreXml(*iter2,glb);
361  ++iter2;
362  Address directcall = Address::restoreXml(*iter2,glb);
363  insertIndirectOverride(callpoint,directcall);
364  }
365  else if (subel->getName() == "protooverride") {
366  const List &list2(subel->getChildren());
367  List::const_iterator iter2 = list2.begin();
368  Address callpoint = Address::restoreXml(*iter2,glb);
369  ++iter2;
370  FuncProto *fp = new FuncProto();
371  fp->setInternal(glb->defaultfp,glb->types->getTypeVoid());
372  fp->restoreXml(*iter2,glb);
373  insertProtoOverride(callpoint,fp);
374  }
375  else if (subel->getName() == "forcegoto") {
376  const List &list2(subel->getChildren());
377  List::const_iterator iter2 = list2.begin();
378  Address targetpc = Address::restoreXml(*iter2,glb);
379  ++iter2;
380  Address destpc = Address::restoreXml(*iter2,glb);
381  insertForceGoto(targetpc,destpc);
382  }
383  else if (subel->getName() == "deadcodedelay") {
384  int4 delay = -1;
385  istringstream s(subel->getAttributeValue("delay"));
386  s.unsetf(ios::dec | ios::hex | ios::oct);
387  s >> delay;
388  AddrSpace *spc = glb->getSpaceByName( subel->getAttributeValue("space"));
389  if ((spc == (AddrSpace *)0)||(delay < 0))
390  throw LowlevelError("Bad deadcodedelay tag");
391  insertDeadcodeDelay(spc,delay);
392  }
393  else if (subel->getName() == "multistagejump") {
394  const Element *tmpel = subel->getChildren().front();
395  Address callpoint = Address::restoreXml(tmpel,glb);
396  insertMultistageJump(callpoint);
397  }
398  else if (subel->getName() == "multistagejump") {
399  const Element *tmpel = subel->getChildren().front();
400  Address callpoint = Address::restoreXml(tmpel,glb);
401  insertMultistageJump(callpoint);
402  }
403  else if (subel->getName() == "flow") {
404  uint4 type = stringToType(subel->getAttributeValue("type"));
405  const Element *tmpel = subel->getChildren().front();
406  Address addr = Address::restoreXml(tmpel,glb);
407  if ((type == Override::NONE)||(addr.isInvalid()))
408  throw LowlevelError("Bad flowoverride tag");
409  insertFlowOverride(addr,type);
410  }
411  }
412 
413 }

42 const VarnodeData &GhidraTranslate::getRegister(const string &nm) const
43 
44 {
45  map<string,VarnodeData>::const_iterator iter = nm2addr.find(nm);
46  if (iter != nm2addr.end())
47  return (*iter).second;
48  Document *doc;
49  try {
50  doc = glb->getRegister(nm); // Ask Ghidra client about the register
51  }
52  catch(XmlError &err) {
53  ostringstream errmsg;
54  errmsg << "Error parsing XML response for query of register: " << nm;
55  errmsg << " -- " << err.explain;

57  }

59  throw LowlevelError("No register named "+nm);

64  vndata.space = regaddr.getSpace();

66  vndata.size = regsize;
67  delete doc;

100 int4 GhidraTranslate::oneInstruction(PcodeEmit &emit,const Address &baseaddr) const
101 
102 {
103  int4 offset;
104  uint1 *doc;
105  try {
106  doc = glb->getPcodePacked(baseaddr); // Request p-code for one instruction
107  }
108  catch(JavaError &err) {
109  ostringstream s;
110  s << "Error generating pcode at address: " << baseaddr.getShortcut();
111  baseaddr.printRaw(s);
112  throw LowlevelError(s.str());
113  }
114  if (doc == (uint1 *)0) {
115  ostringstream s;
116  s << "No pcode could be generated at address: " << baseaddr.getShortcut();
117  baseaddr.printRaw(s);
118  throw BadDataError(s.str());
119  }
120 
121  uintb val;
122  const uint1 *ptr = PcodeEmit::unpackOffset(doc+1,val);
123  offset = (int4)val;
124 
125  if (*doc == PcodeEmit::unimpl_tag) {
126  ostringstream s;
127  s << "Instruction not implemented in pcode:\n ";
128  baseaddr.printRaw(s);
129  delete [] doc;
130  throw UnimplError(s.str(),offset);
131  }
132 
133  int4 spcindex = (int4)(*ptr++ - 0x20);
134  AddrSpace *spc = getSpace(spcindex);
135  uintb instoffset;
136  ptr = PcodeEmit::unpackOffset(ptr,instoffset);
137  Address pc(spc,instoffset);
138 
139  while(*ptr == PcodeEmit::op_tag)
140  ptr = emit.restorePackedOp(pc,ptr,this);
141  delete [] doc;
142  return offset;
143 }

148 void GhidraTranslate::restoreXml(const Element *el)
149 
150 {
151  setBigEndian(xml_readbool(el->getAttributeValue("bigendian")));
152  {
153  istringstream s(el->getAttributeValue("uniqbase"));
154  s.unsetf(ios::dec | ios::hex | ios::oct);
155  uintm ubase;
156  s >> ubase;
157  setUniqueBase(ubase);
158  }
159  const List &list(el->getChildren());
160  List::const_iterator iter;
161  iter = list.begin();
162  restoreXmlSpaces(*iter,this);
163  ++iter;
164  while(iter != list.end()) {
165  const Element *subel = *iter;
166  if (subel->getName() == "truncate_space") {
167  TruncationTag tag;
168  tag.restoreXml(subel);
169  truncateSpace(tag);
170  }
171  ++iter;
172  }
173 }

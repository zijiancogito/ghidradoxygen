24 void InjectPayload::readParameter(const Element *el,string &name,uint4 &size)
25 
26 {
27  name = "";
28  size = 0;
29  int4 num = el->getNumAttributes();
30  for(int4 i=0;i<num;++i) {
31  if (el->getAttributeName(i) == "name")
32  name = el->getAttributeValue(i);
33  else if (el->getAttributeName(i) == "size") {
34  istringstream s(el->getAttributeValue(i));
35  s.unsetf(ios::dec | ios::hex | ios::oct);
36  s >> size;
37  }
38  }
39  if (name.size()==0)
40  throw LowlevelError("Missing inject parameter name");
41 }

62 void InjectPayload::restoreXml(const Element *el)
63 
64 {
65  paramshift = 0;
66  dynamic = false;
67  int4 num = el->getNumAttributes();
68  for(int4 i=0;i<num;++i) {
69  const string &elname(el->getAttributeName(i));
70  if (elname == "paramshift") {
71  istringstream s(el->getAttributeValue(i));
72  s.unsetf(ios::dec | ios::hex | ios::oct);
73  s >> paramshift;
74  }
75  else if (elname == "dynamic")
76  dynamic = xml_readbool(el->getAttributeValue(i));
77  }
78  const List &list(el->getChildren());
79  List::const_iterator iter;
80  for(iter=list.begin();iter!=list.end();++iter) {
81  const Element *subel = *iter;
82  if (subel->getName() == "input") {
83  string paramName;
84  uint4 size;
85  readParameter(subel,paramName,size);
86  inputlist.push_back(InjectParameter(paramName,size));
87  }
88  else if (subel->getName() == "output") {
89  string paramName;
90  uint4 size;
91  readParameter(subel,paramName,size);
92  output.push_back(InjectParameter(paramName,size));
93  }
94  }
95  orderParameters();
96 }

110 void ExecutablePcode::build(void)
111 
112 {
113  if (built) return;
114  InjectContext &icontext(glb->pcodeinjectlib->getCachedContext());
115  icontext.clear();
116  uintb uniqReserve = 0x10; // Temporary register space reserved for inputs and output
117  Architecture *glb = emulator.getArch();
118  AddrSpace *codeSpace = glb->getDefaultSpace();
119  AddrSpace *uniqSpace = glb->getUniqueSpace();
120  icontext.baseaddr = Address(codeSpace,0x1000); // Fake address
121  icontext.nextaddr = icontext.baseaddr;
122  for(int4 i=0;i<sizeInput();++i) { // Skip the first operand containing the injectid
123  InjectParameter &param( getInput(i) );
124  icontext.inputlist.push_back(VarnodeData());
125  icontext.inputlist.back().space = uniqSpace;
126  icontext.inputlist.back().offset = uniqReserve;
127  icontext.inputlist.back().size = param.getSize();
128  inputList.push_back(uniqReserve);
129  uniqReserve += 0x20;
130  }
131  for(int4 i=0;i<sizeOutput();++i) {
132  InjectParameter &param( getOutput(i) );
133  icontext.output.push_back(VarnodeData());
134  icontext.output.back().space = uniqSpace;
135  icontext.output.back().offset = uniqReserve;
136  icontext.output.back().size = param.getSize();
137  outputList.push_back(uniqReserve);
138  uniqReserve += 0x20;
139  }
140  emitter = emulator.buildEmitter(glb->pcodeinjectlib->getBehaviors(),uniqReserve);
141  inject(icontext,*emitter);
142  delete emitter;
143  emitter = (PcodeEmit *)0;
144  if (!emulator.checkForLegalCode())
145  throw LowlevelError("Illegal p-code in executable snippet");
146  built = true;
147 }

157 uintb ExecutablePcode::evaluate(const vector<uintb> &input)
158 
159 {
160  build(); // Build the PcodeOpRaws (if we haven't before)
161  emulator.resetMemory();
162  if (input.size() != inputList.size())
163  throw LowlevelError("Wrong number of input parameters to executable snippet");
164  if (outputList.size() == 0)
165  throw LowlevelError("No registered outputs to executable snippet");
166  for(int4 i=0;i<input.size();++i)
167  emulator.setVarnodeValue(inputList[i], input[i]);
168  while(!emulator.getHalt())
169  emulator.executeCurrentOp();
170  return emulator.getTempValue(outputList[0]);
171 }

250 int4 PcodeInjectLibrary::getPayloadId(int4 type,const string &nm) const
251 
252 {
253  map<string,int4>::const_iterator iter;
254  if (type == InjectPayload::CALLFIXUP_TYPE) {
255  iter = callFixupMap.find(nm);
256  if (iter == callFixupMap.end())
257  return -1;
258  }
259  else if (type == InjectPayload::CALLOTHERFIXUP_TYPE) {
260  iter = callOtherFixupMap.find(nm);
261  if (iter == callOtherFixupMap.end())
262  return -1;
263  }
264  else if (type == InjectPayload::CALLMECHANISM_TYPE) {
265  iter = callMechFixupMap.find(nm);
266  if (iter == callMechFixupMap.end())
267  return -1;
268  }
269  else {
270  iter = scriptMap.find(nm);
271  if (iter == scriptMap.end())
272  return -1;
273  }
274  return (*iter).second;
275 }

290 int4 Action::perform(Funcdata &data)
291 
292 {
293  int4 res;
294 
295  do {
296  switch(status) {
297  case status_start:
298  count = 0; // No changes made yet by this action
299  if (checkStartBreak()) {
300  status = status_breakstarthit;
301  return -1; // Indicate partial completion
302  }
303  count_tests += 1;
304  case status_breakstarthit:
305  case status_repeat:
306  lcount = count;
307  case status_mid:
308 #ifdef OPACTION_DEBUG
309  data.debugActivate();
310 #endif
311  res = apply(data); // Start or continue action
312 #ifdef OPACTION_DEBUG
313  data.debugModPrint(getName());
314 #endif
315  if (res < 0) { // negative indicates partial completion
316  status = status_mid;
317  return res;
318  }
319  else if (lcount < count) { // Action has been applied
320  issueWarning(data.getArch());
321  count_apply += 1;
322  if (checkActionBreak()) {
323  status = status_actionbreak;
324  return -1; // Indicate action breakpoint
325  }
326 #ifdef OPACTION_DEBUG
327  else if (data.debugBreak()) {
328  status = status_actionbreak;
329  data.debugHandleBreak();
330  return -1;
331  }
332 #endif
333  }
334  break;
335  case status_end:
336  return 0; // Rule applied, do not repeat until reset
337  break;
338  case status_actionbreak: // Returned -1 last time, but we do not reapply
339  break; // we either repeat, or return our count
340  }
341  status = status_repeat;
342  } while((lcount<count)&&((flags&rule_repeatapply)!=0));
343 
344  if ((flags&(rule_onceperfunc|rule_oneactperfunc))!=0) {
345  if ((count>0)||((flags&rule_onceperfunc)!=0))
346  status = status_end;
347  else
348  status = status_start;
349  }
350  else
351  status = status_start;
352 
353  return count;
354 }

489 int4 ActionGroup::apply(Funcdata &data)
490 
491 {
492  int4 res;
493 
494  if (status != status_mid)
495  state = list.begin(); // Initialize the derived action
496  for(;state!=list.end();++state) {
497  res = (*state)->perform(data);
498  if (res>0) { // A change was made
499  count += res;
500  if (checkActionBreak()) { // Check if this is an action breakpoint
501  ++state;
502  return -1;
503  }
504  }
505  else if (res<0) // Partial completion of member
506  return -1; // equates to partial completion of group action
507  }
508 
509  return 0; // Indicate successful completion
510 }

536 int4 ActionRestartGroup::apply(Funcdata &data)
537 
538 {
539  int4 res;
540 
541  if (curstart == -1) return 0; // Already completed
542  for(;;) {
543  res = ActionGroup::apply(data);
544  if (res != 0) return res;
545  if (!data.hasRestartPending()) {
546  curstart = -1;
547  return 0;
548  }
549  if (data.isJumptableRecoveryOn()) // Don't restart within jumptable recovery
550  return 0;
551  curstart += 1;
552  if (curstart > maxrestarts) {
553  data.warningHeader("Exceeded maximum restarts with more pending");
554  curstart = -1;
555  return 0;
556  }
557  data.getArch()->clearAnalysis(&data);
558 
559  // Reset everything but ourselves
560  vector<Action *>::iterator iter;
561  for(iter=list.begin();iter!=list.end();++iter)
562  (*iter)->reset(data); // Reset each subrule
563  status = status_start;
564  }

805 int4 ActionPool::processOp(PcodeOp *op,Funcdata &data)
806 
807 {
808  Rule *rl;
809  int4 res;
810  uint4 opc;
811 
812  if (op->isDead()) {
813  op_state++;
814  data.opDeadAndGone(op);
815  rule_index = 0;
816  return 0;
817  }
818  opc = op->code();
819  while(rule_index < perop[opc].size()) {
820  rl = perop[opc][rule_index++];
821  if (rl->isDisabled()) continue;
822 #ifdef OPACTION_DEBUG
823  data.debugActivate();
824 #endif
825  rl->count_tests += 1;
826  res = rl->applyOp(op,data);
827 #ifdef OPACTION_DEBUG
828  data.debugModPrint(rl->getName());
829 #endif
830  if (res>0) {
831  rl->count_apply += 1;
832  count += res;
833  rl->issueWarning(data.getArch()); // Check if we need to issue a warning
834  if (rl->checkActionBreak())
835  return -1;
836 #ifdef OPACTION_DEBUG
837  if (data.debugBreak()) {
838  data.debugHandleBreak();
839  return -1;
840  }
841 #endif
842  if (op->isDead()) break;
843  if (opc != op->code()) { // Set of rules to apply to this op has changed
844  opc = op->code();
845  rule_index = 0;
846  }
847  }
848  else if (opc != op->code()) {
849  data.getArch()->printMessage("ERROR: Rule " + rl->getName() + " changed op without returning result of 1!");
850  opc = op->code();
851  rule_index = 0;
852  }
853  }
854  op_state++;
855  rule_index = 0;
856 
857  return 0;
858 }

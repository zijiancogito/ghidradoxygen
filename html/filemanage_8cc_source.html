275 void FileManage::splitPath(const string &full,string &path,string &base)
276 
277 { // Split path string -full- into its -base-name and -path- (relative or absolute)
278  // If there is no path, i.e. only a basename in full, then -path- will return as an empty string
279  // otherwise -path- will be non-empty and end in a separator character
280  string::size_type end = full.size()-1;
281  if (full[full.size()-1] == separator) // Take into account terminating separator
282  end = full.size()-2;
283  string::size_type pos = full.rfind(separator,end);
284  if (pos == string::npos) { // Didn't find any separator
285  base = full;
286  path.clear();
287  }
288  else {
289  string::size_type sz = (end - pos);
290  base = full.substr(pos+1,sz);
291  path = full.substr(0,pos+1);
292  }
293 }

343  vector<string> pathels;

372  }

375  if (pathels[i] != "Ghidra") continue;
376  string root;
377  if (testDevelopmentPath(pathels,i,root))
378  return root;
379  if (testInstallPath(pathels,i,root))

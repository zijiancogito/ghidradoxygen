19 static void print_varnode_vertex(Varnode *vn,ostream &s)
20 
21 {
22  PcodeOp *op;
23 
24  if (vn == (Varnode *)0) return;
25  if (vn->isMark()) return;
26  AddrSpace *spc = vn->getSpace();
27  if (spc->getType() == IPTR_FSPEC) return;
28  if (spc->getType() == IPTR_IOP) return;
29  s << dec << 'v' << vn->getCreateIndex() << ' ' << spc->getName();
30  s << " var ";
31 
32  vn->printRawNoMarkup(s);
33 
34  op = vn->getDef();
35  if (op != (PcodeOp *)0)
36  s << ' ' << hex << op->getAddr().getOffset();
37  else if (vn->isInput())
38  s << " i";
39  else
40  s << " <na>";
41  s << endl;
42  vn->setMark();
43 }

66 static void dump_varnode_vertex(Funcdata &data,ostream &s)
67 
68 {
69  list<PcodeOp *>::const_iterator oiter;
70  PcodeOp *op;
71  int4 i,start,stop;
72 
73  s << "\n\n// Add Vertices\n";
74  s << "*CMD=*COLUMNAR_INPUT,\n";
75  s << " Command=AddVertices,\n";
76  s << " Parsing=WhiteSpace,\n";
77  s << " Fields=({Name=Internal, Location=1},\n";
78  s << " {Name=SubClass, Location=2},\n";
79  s << " {Name=Type, Location=3},\n";
80  s << " {Name=Name, Location=4},\n";
81  s << " {Name=Address, Location=5});\n\n";
82  s << "//START:varnodes\n";
83 
84  for(oiter=data.beginOpAlive();oiter!=data.endOpAlive();++oiter) {
85  op = *oiter;
86  print_varnode_vertex(op->getOut(),s);
87  start = 0;
88  stop = op->numInput();
89  switch(op->code()) {
90  case CPUI_LOAD:
91  case CPUI_STORE:
92  case CPUI_BRANCH:
93  case CPUI_CALL:
94  start = 1;
95  break;
96  case CPUI_INDIRECT:
97  stop = 1;
98  break;
99  default:
100  break;
101  }
102  for(i=start;i<stop;++i)
103  print_varnode_vertex(op->getIn(i),s);
104  }
105  s << "*END_COLUMNS\n";
106  for(oiter=data.beginOpAlive();oiter!=data.endOpAlive();++oiter) {
107  op = *oiter;
108  if (op->getOut()!=(Varnode *)0)
109  op->getOut()->clearMark();
110  for(i=0;i<op->numInput();++i)
111  op->getIn(i)->clearMark();
112  }
113 }

139 static void print_edges(PcodeOp *op,ostream &s)
140 
141 {
142  Varnode *vn;
143  int4 i,start,stop;
144 
145  vn = op->getOut();
146  if (vn != (Varnode *)0)
147  s << dec << 'o' << op->getTime() << " v" << vn->getCreateIndex() << " output\n";
148  start = 0;
149  stop = op->numInput();
150  switch(op->code()) {
151  case CPUI_LOAD:
152  case CPUI_STORE:
153  case CPUI_BRANCH:
154  case CPUI_CALL:
155  start = 1;
156  break;
157  case CPUI_INDIRECT:
158  stop = 1;
159  break;
160  default:
161  break;
162  }
163  for(i=start;i<stop;++i) {
164  vn = op->getIn(i);
165  spacetype tp = vn->getSpace()->getType();
166  if ((tp != IPTR_FSPEC)&&(tp != IPTR_IOP))
167  s << dec << 'v' << vn->getCreateIndex() << " o" << op->getTime() << " input\n";
168  }
169 }

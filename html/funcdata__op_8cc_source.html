429 Varnode *Funcdata::createStackRef(AddrSpace *spc,uintb off,PcodeOp *op,Varnode *stackptr,bool insertafter)
430 
431 {
432  PcodeOp *addop;
433  Varnode *addout;
434  int4 addrsize;
435 
436  // Calculate CURRENT stackpointer as base for relative offset
437  if (stackptr == (Varnode *)0) // If we are not reusing an old reference to the stack pointer
438  stackptr = newSpacebasePtr(spc); // create a new reference
439  addrsize = stackptr->getSize();
440  addop = newOp(2,op->getAddr());
441  opSetOpcode(addop,CPUI_INT_ADD);
442  addout = newUniqueOut(addrsize,addop);
443  opSetInput(addop,stackptr,0);
444  off = AddrSpace::byteToAddress(off,spc->getWordSize());
445  opSetInput(addop,newConstant(addrsize,off),1);
446  if (insertafter)
447  opInsertAfter(addop,op);
448  else
449  opInsertBefore(addop,op);
450 
451  AddrSpace *containerid = spc->getContain();
452  SegmentOp *segdef = glb->userops.getSegmentOp(containerid->getIndex());

457  Varnode *segout = newUniqueOut(containerid->getAddrSize(),segop);
458  opSetInput(segop,newVarnodeSpace(containerid),0);
459  opSetInput(segop,newConstant(segdef->getBaseSize(),0),1);
460  opSetInput(segop,addout,2);

462  addout = segout;

547 PcodeOp *Funcdata::canonicalReturnOp(void) const
548 
549 {
550  bool hasnohigh = !isHighOn();
551  PcodeOp *res = (PcodeOp *)0;
552  Datatype *bestdt = (Datatype *)0;
553  list<PcodeOp *>::const_iterator iter,iterend;
554  iterend = endOp(CPUI_RETURN);
555  for(iter=beginOp(CPUI_RETURN);iter!=iterend;++iter) {
556  PcodeOp *retop = *iter;
557  if (retop->isDead()) continue;
558  if (retop->getHaltType()!=0) continue;
559  if (retop->numInput() > 1) {
560  if (hasnohigh) return retop;
561  Varnode *vn = retop->getIn(1);
562  Datatype *ct = vn->getHigh()->getType();
563  if (bestdt == (Datatype *)0) {
564  res = retop;
565  bestdt = ct;
566  }
567  else if (ct->typeOrder(*bestdt) < 0) {
568  res = retop;
569  bestdt = ct;
570  }
571  }
572  }
573  return res;
574 }

746 void Funcdata::truncatedFlow(const Funcdata *fd,const FlowInfo *flow)
747 
748 {
749  if (!obank.empty())
750  throw LowlevelError("Trying to do truncated flow on pre-existing pcode");
751 
752  list<PcodeOp *>::const_iterator oiter; // Clone the raw pcode
753  for(oiter=fd->obank.beginDead();oiter!=fd->obank.endDead();++oiter)
754  cloneOp(*oiter,(*oiter)->getSeqNum());
755  obank.setUniqId(fd->obank.getUniqId());
756 
757  // Clone callspecs
758  for(int4 i=0;i<fd->qlst.size();++i) {
759  FuncCallSpecs *oldspec = fd->qlst[i];
760  PcodeOp *newop = findOp(oldspec->getOp()->getSeqNum());
761  FuncCallSpecs *newspec = oldspec->clone(newop);
762  Varnode *invn0 = newop->getIn(0);
763  if (invn0->getSpace()->getType() == IPTR_FSPEC) { // Replace embedded pointer to callspec
764  Varnode *newvn0 = newVarnodeCallSpecs(newspec);
765  opSetInput(newop,newvn0,0);
766  deleteVarnode(invn0);
767  }
768  qlst.push_back(newspec);
769  }
770 
771  vector<JumpTable *>::const_iterator jiter; // Clone the jumptables
772  for(jiter=fd->jumpvec.begin();jiter!=fd->jumpvec.end();++jiter) {
773  PcodeOp *indop = (*jiter)->getIndirectOp();
774  if (indop == (PcodeOp *)0) // If indirect op has not been linked, this is probably a jumptable override
775  continue; // that has not been reached by the flow yet, so we ignore/truncate it
776  PcodeOp *newop = findOp(indop->getSeqNum());
777  if (newop == (PcodeOp *)0)
778  throw LowlevelError("Could not trace jumptable across partial clone");
779  JumpTable *jtclone = new JumpTable(*jiter);
780  jtclone->setIndirectOp(newop);
781  jumpvec.push_back(jtclone);
782  }
783 
784  FlowInfo partialflow(*this,obank,bblocks,qlst,flow); // Clone the flow
785  if (partialflow.hasInject())
786  partialflow.injectPcode();
787  // Clear error reporting flags
788  // Keep possible unreachable flag
789  partialflow.clearFlags(~((uint4)FlowInfo::possible_unreachable));
790 
791  partialflow.generateBlocks(); // Generate basic blocks for partial flow
792  flags |= blocks_generated;
793 }

807 bool Funcdata::inlineFlow(Funcdata *inlinefd,FlowInfo &flow,PcodeOp *callop)
808 
809 {
810  inlinefd->getArch()->clearAnalysis(inlinefd);
811  FlowInfo inlineflow(*inlinefd,inlinefd->obank,inlinefd->bblocks,inlinefd->qlst);
812  inlinefd->obank.setUniqId( obank.getUniqId() );
813 
814  // Generate the pcode ops to be inlined
815  Address baddr(baseaddr.getSpace(),0);
816  Address eaddr(baseaddr.getSpace(),~((uintb)0));
817  inlineflow.setRange(baddr,eaddr);
818  inlineflow.setFlags(FlowInfo::error_outofbounds|FlowInfo::error_unimplemented|
819  FlowInfo::error_reinterpreted|FlowInfo::flow_forinline);
820  inlineflow.forwardRecursion(flow);
821  inlineflow.generateOps();
822 
823  if (inlineflow.checkEZModel()) {
824  // With an EZ clone there are no jumptables to clone
825  list<PcodeOp *>::const_iterator oiter = obank.endDead();
826  --oiter; // There is at least one op
827  flow.inlineEZClone(inlineflow,callop->getAddr());
828  ++oiter;
829  if (oiter != obank.endDead()) { // If there was at least one PcodeOp cloned
830  PcodeOp *firstop = *oiter;
831  oiter = obank.endDead();
832  --oiter;
833  PcodeOp *lastop = *oiter;
834  obank.moveSequenceDead(firstop,lastop,callop); // Move cloned sequence to right after callop
835  if (callop->isBlockStart())
836  firstop->setFlag(PcodeOp::startbasic); // First op of inline inherits callop's startbasic flag
837  else
838  firstop->clearFlag(PcodeOp::startbasic);
839  }
840  opDestroyRaw(callop);
841  }
842  else {
843  Address retaddr;
844  if (!flow.testHardInlineRestrictions(inlinefd,callop,retaddr))
845  return false;
846  vector<JumpTable *>::const_iterator jiter; // Clone any jumptables from inline piece

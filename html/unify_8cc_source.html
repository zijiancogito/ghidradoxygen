263 uintb ConstantExpression::getConstant(UnifyState &state) const
264 
265 {
266  OpBehavior *behavior = state.getBehavior(opc);
267  if (behavior->isSpecial())
268  throw LowlevelError("Cannot evaluate special operator in constant expression");
269  uintb res;
270  if (behavior->isUnary()) {
271  uintb ourconst1 = expr1->getConstant(state);
272  res = behavior->evaluateUnary(sizeof(uintb),sizeof(uintb),ourconst1);
273  }
274  else {
275  uintb ourconst1 = expr1->getConstant(state);
276  uintb ourconst2 = expr2->getConstant(state);
277  res = behavior->evaluateBinary(sizeof(uintb),sizeof(uintb),ourconst1,ourconst2);
278  }
279  return res;
280 }

282 void ConstantExpression::writeExpression(ostream &s,UnifyCPrinter &printstate) const
283 
284 {
285  int4 type; // 0=binary 1=unarypre 2=unarypost 3=func
286  string name; // name of operator
287  switch(opc) {
288  case CPUI_INT_ADD:

292  case CPUI_INT_SUB:
293  type=0;

296  case CPUI_INT_AND:
297  type=0;
298  name = " & ";
299  break;
300  case CPUI_INT_OR:
301  type=0;
302  name = " | ";
303  break;
304  case CPUI_INT_XOR:
305  type=0;
306  name = " ^ ";
307  break;
308  case CPUI_INT_MULT:
309  type = 0;
310  name = " * ";
311  break;
312  case CPUI_INT_DIV:
313  type = 0;
314  name = " / ";
315  break;
316  case CPUI_INT_EQUAL:
317  type = 0;
318  name = " == ";
319  break;
320  case CPUI_INT_NOTEQUAL:
321  type = 0;
322  name = " != ";
323  break;
324  // case CPUI_INT_SLESS:
325  // case CPUI_INT_SLESSEQUAL:
326  case CPUI_INT_LESS:
327  type = 0;
328  name = " < ";
329  break;
330  case CPUI_INT_LESSEQUAL:
331  type = 0;
332  name = " <= ";
333  break;
334  // case CPUI_INT_ZEXT:
335  // case CPUI_INT_SEXT:
336  // case CPUI_INT_CARRY:
337  // case CPUI_INT_SCARRY:
338  // case CPUI_INT_SBORROW:
339  case CPUI_INT_LEFT:
340  type = 0;
341  name = " << ";
342  break;
343  case CPUI_INT_RIGHT:
344  type = 0;
345  name = " >> ";
346  break;
347  // case CPUI_INT_SRIGHT:
348  default:
349  throw LowlevelError("Unable to generate C for this expression element");
350  }
351  if (type==0) {
352  s << '(';
353  expr1->writeExpression(s,printstate);
354  s << name;
355  expr2->writeExpression(s,printstate);
356  s << ')';
357  }
358  else if (type==1) {
359  s << '(' << name;
360  expr1->writeExpression(s,printstate);
361  s << ')';
362  }
363  else if (type==2) {
364  s << '(';
365  expr1->writeExpression(s,printstate);
366  s << name << ')';
367  }

370  expr1->writeExpression(s,printstate);
371  s << ')';
372  }

1033 bool ConstraintGroup::step(UnifyState &state)
1034 
1035 {
1036  TraverseGroupState *traverse = (TraverseGroupState *)state.getTraverse(uniqid);
1037 
1038  UnifyConstraint *subconstraint;
1039  TraverseConstraint *subtraverse;
1040  int4 subindex;
1041  int4 stateint;
1042  int4 max = constraintlist.size();
1043  do {
1044  stateint = traverse->getState();
1045  if (stateint == 0) { // Attempt a step at current constraint
1046  subindex = traverse->getCurrentIndex();
1047  subtraverse = traverse->getSubTraverse(subindex);
1048  subconstraint = constraintlist[subindex];
1049  if (subconstraint->step(state)) {
1050  traverse->setState(1); // Now try a push
1051  subindex += 1;
1052  traverse->setCurrentIndex(subindex);
1053  }
1054  else {
1055  subindex -= 1;
1056  if (subindex < 0) return false; // Popped off the top
1057  traverse->setCurrentIndex(subindex);
1058  traverse->setState(0); // Try a step next
1059  }
1060  }
1061  else if (stateint == 1) { // Push
1062  subindex = traverse->getCurrentIndex();
1063  subtraverse = traverse->getSubTraverse(subindex);
1064  subconstraint = constraintlist[subindex];
1065  subconstraint->initialize(state);
1066  traverse->setState(0); // Try a step next
1067  }
1068  else { // Very first time through
1069  traverse->setCurrentIndex(0);
1070  subindex = 0;
1071  subtraverse = traverse->getSubTraverse(subindex);
1072  subconstraint = constraintlist[subindex];
1073  subconstraint->initialize(state); // Initialize the very first subcontraint
1074  traverse->setState(0); // Now try a step
1075  }
1076  } while(subindex < max);
1077  subindex -= 1;
1078  traverse->setCurrentIndex(subindex);
1079  traverse->setState(0); // Have full solution, do step next, to get to next solution
1080  return true;
1081 }

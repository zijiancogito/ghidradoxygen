116  @Override
117  public void actionPerformed(ActionContext context) {

119  // that global events do not get triggered.
120  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
121  if (decompilerActionContext.isDecompiling()) {

129  ClangToken tokenAtCursor = decompilerPanel.getTokenAtCursor();
130  HighVariable variable = null;
131  Structure struct = null;
132  DataTypeComponent comp = null;
133

144  Msg.showError(this, tool.getToolFrame(), "Retype Failed",
145  "Failed to re-type structure: " + struct.getName());
146  return;
147  }
148  comp = struct.getComponentAt(offset);
149  if (comp == null)
150  dataType = chooseDataType(DataType.DEFAULT);
151  else
152  dataType = chooseDataType(comp.getDataType());
153  }
154  else if (tokenAtCursor.Parent() instanceof ClangReturnType) {
155  ClangReturnType parent = (ClangReturnType) tokenAtCursor.Parent();
156  dataType = chooseDataType(parent.getDataType());
157  if (dataType == null) {
158  return;
159  }
160  retypeReturnType(dataType, parent);
161  return;
162  }
163  else {
164  variable = tokenAtCursor.getHighVariable();
165  if (variable == null) {
166  Address addr = RenameVariableAction.getStorageAddress(tokenAtCursor, controller);
167  if (addr == null)
168  return;
169  variable = RenameVariableAction.forgeHighVariable(addr, controller);
170  if (variable == null)
171  return;
172  }
173  dataType = chooseDataType(variable.getDataType());
174  }
175  if (dataType == null)
176  return;
177  if (struct != null)
178  retypeStructVariable(struct, comp, dataType);
179  else
180  retypeVariable(variable, tokenAtCursor.getVarnode(), dataType);

183  private void retypeReturnType(DataType dataType, ClangReturnType parent) {
184  Program program = controller.getProgram();
185  HighFunction hfunction = getHighFunctionFromReturnTypeToken(parent);
186  if (hfunction == null) {
187  return;
188  }
189  boolean commitRequired = checkFullCommit(null, hfunction);
190  if (commitRequired) {
191  int resp =
192  OptionDialog.showOptionDialog(
193  tool.getToolFrame(),
194  "Parameter Commit Required",
195  "Retyping the return value requires all other parameters to be committed!\nContinue with retype?",
196  "Continue");
197  if (resp != OptionDialog.OPTION_ONE) {
198  return;
199  }
200  }
201  Function function = hfunction.getFunction();
202  int transactionID = program.startTransaction("Retype return type");
203  try {
204  if (commitRequired) {
205  try {
206  HighFunctionDBUtil.commitParamsToDatabase(hfunction, true,
207  SourceType.USER_DEFINED);
208  }
209  catch (DuplicateNameException e) {
210  throw new AssertException("Unexpected exception", e);
211  }
212  catch (InvalidInputException e) {
213  Msg.showError(this, null, "Parameter Commit Failed", e.getMessage());
214  }
215  }
216  function.setReturnType(dataType, SourceType.USER_DEFINED);
217  }
218  catch (InvalidInputException e) {
219  Msg.showError(this, tool.getToolFrame(), "Retype Failed",
220  "Failed to re-type return type '" + getName() + "': " + e.getMessage());
221  }
222  program.endTransaction(transactionID, true);
223  }

254  private void retypeVariable(HighVariable var, Varnode exactSpot, DataType dt) {
255  HighFunction hfunction = var.getHighFunction();
256 
257  boolean commitRequired = checkFullCommit(var, hfunction);
258  if (commitRequired) {
259  int resp =
260  OptionDialog.showOptionDialog(
261  tool.getToolFrame(),
262  "Parameter Commit Required",
263  "Retyping a parameter requires all other parameters to be committed!\nContinue with retype?",
264  "Continue");
265  if (resp != OptionDialog.OPTION_ONE) {
266  return;
267  }
268  exactSpot = null; // Don't try to split out if commit is required
269  }
270 
271  if (exactSpot != null) { // The user pointed at a particular usage, not just the vardecl
272  try {
273  var = hfunction.splitOutMergeGroup(var, exactSpot);
274  }
275  catch (PcodeException e) {
276  Msg.showError(this, tool.getToolFrame(), "Retype Failed", e.getMessage());
277  return;
278  }
279  }
280  Program program = controller.getProgram();
281  int transaction = program.startTransaction("Retype Variable");
282  try {
283  if (commitRequired) {
284  try {
285  HighFunctionDBUtil.commitParamsToDatabase(hfunction, true,
286  SourceType.USER_DEFINED);
287  HighFunctionDBUtil.commitReturnToDatabase(hfunction, SourceType.USER_DEFINED);
288  }
289  catch (DuplicateNameException e) {
290  throw new AssertException("Unexpected exception", e);
291  }
292  catch (InvalidInputException e) {
293  Msg.showError(this, null, "Parameter Commit Failed", e.getMessage());
294  }
295  }
296  HighFunctionDBUtil.updateDBVariable(var, null, dt, SourceType.USER_DEFINED);
297  }
298  catch (DuplicateNameException e) {
299  throw new AssertException("Unexpected exception", e);
300  }
301  catch (InvalidInputException e) {
302  Msg.showError(this, tool.getToolFrame(), "Retype Failed",
303  "Failed to re-type variable '" + var.getName() + "': " + e.getMessage());
304  }
305  finally {
306  program.endTransaction(transaction, true);
307  }
308  }

318  private int getEndComponentIndex(Structure struct, DataTypeComponent comp, DataType newtype) {
319  int newlen = newtype.getLength();
320  if (newlen <= 0)
321  return -1; // Don't support variable length types
322  DataType curtype = comp.getDataType();
323  newlen -= curtype.getLength();
324  if (newlen < 0)
325  return -1; // new size is smaller than original size
326  int index = comp.getOrdinal();
327  while (newlen > 0) {
328  index += 1;
329  if (index >= struct.getNumComponents())
330  return -1; // Not enough space in the structure
331  comp = struct.getComponent(index);
332 // String nm = comp.getFieldName();
333 // if ((nm !=null)&&(nm.length()!=0))
334 // return -1;
335  curtype = comp.getDataType();
336 // if (!Undefined.isUndefined(curtype))
337 // return -1; // Overlaps non-undefined datatype
338  if (curtype != DataType.DEFAULT)
339  return -1; // Only allow overwrite of placeholder components
340  newlen -= curtype.getLength();
341  }
342  if (newlen < 0)
343  return -1; // Partial field
344  return index;
345  }

347  private void retypeStructVariable(Structure dt, DataTypeComponent comp, DataType newtype) {
348  Program program = controller.getProgram();
349  if (comp == null) {
350  if (!dt.isNotYetDefined()) {
351  Msg.showError(this, tool.getToolFrame(), "Retype Failed",
352  "Could not find component of '" + dt.getName() + "' to retype");
353  return;
354  }
355  // note if we reach here the offset must be zero, so assume we are inserting newtype
356  int transaction = program.startTransaction("Retype Structure Field");
357  try {
358  dt.insert(0, newtype);
359  }
360  finally {
361  program.endTransaction(transaction, true);
362  }
363  return;
364  }
365  int startind = comp.getOrdinal();
366  int endind = getEndComponentIndex(dt, comp, newtype);
367  if (endind < 0) {
368  Msg.showError(this, tool.getToolFrame(), "Retype Failed",
369  "Failed to re-type structure '" + dt.getName() + "': Datatype did not fit");
370  return;
371  }
372  int transaction = program.startTransaction("Retype Structure Field");
373  try {
374  for (int i = endind; i > startind; --i) { // Clear all but first field
375  dt.clearComponent(i);
376  }
377  dt.replaceAtOffset(comp.getOffset(), newtype, newtype.getLength(), comp.getFieldName(),
378  comp.getComment());
379  }
380  finally {
381  program.endTransaction(transaction, true);
382  }
383  }

392  public static boolean checkFullCommit(HighVariable var, HighFunction hfunction) {
393  if ((var != null) && (!(var instanceof HighParam)))
394  return false;
395  Function function = hfunction.getFunction();
396  Parameter[] parameters = function.getParameters();
397  LocalSymbolMap localSymbolMap = hfunction.getLocalSymbolMap();
398  int numParams = localSymbolMap.getNumParams();
399  if (numParams != parameters.length)
400  return true;
401 
402  int skipslot = -1;
403  if (var != null)
404  skipslot = ((HighParam) var).getSlot();
405  for (int i = 0; i < numParams; i++) {
406  HighParam param = localSymbolMap.getParam(i);
407  if (param.getSlot() != i) // Slot must match
408  return true;
409  VariableStorage storage = param.getStorage();
410  if (!storage.equals(parameters[i].getVariableStorage())) // Storage must match
411  return true;
412  if (skipslot != i) { // Compare datatypes unless it is the specific -var- we are skipping
413  if (!param.getDataType().isEquivalent(parameters[i].getDataType()))
414  return true;
415  }
416  }
417 
418  if (var != null) { // A null var indicates we are changing the returntype anyway, so we don't need to check it
419  DataType funcReturnType = function.getReturnType();
420  if (funcReturnType != DataType.DEFAULT) {
421  DataType hfuncReturnType = hfunction.getFunctionPrototype().getReturnType();
422  if (!funcReturnType.equals(hfuncReturnType))
423  return true;
424  }
425  }
426 
427  return false;
428  }

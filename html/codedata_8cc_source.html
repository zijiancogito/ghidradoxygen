82 void DisassemblyEngine::disassemble(const Address &addr,DisassemblyResult &res)
83 
84 {
85  jumpaddr.clear();
86  lastop = CPUI_COPY;
87  hascall = false;
88  hitsaddress = false;
89  res.flags = 0;
90  try {
91  res.length = trans->oneInstruction(*this,addr);
92  } catch(BadDataError &err) {
93  res.success = false;
94  return;
95  } catch(DataUnavailError &err) {
96  res.success = false;
97  return;
98  } catch(UnimplError &err) {
99  res.length = err.instruction_length;
100  }
101  res.success = true;
102  if (hascall)
103  res.flags |= CodeUnit::call;
104  if (hitsaddress) {
105  res.flags |= CodeUnit::targethit;
106  res.targethit = targethit;
107  }
108  Address lastaddr = addr + res.length;
109  switch(lastop) {
110  case CPUI_BRANCH:
111  case CPUI_BRANCHIND:
112  if (hitsaddress)
113  res.flags |= CodeUnit::thunkhit; // Hits target via indirect jump
114  break;
115  case CPUI_RETURN:
116  break;
117  default:
118  res.flags |= CodeUnit::fallthru;
119  break;
120  }
121  for(int4 i=0;i<jumpaddr.size();++i) {
122  if (jumpaddr[i] == lastaddr)
123  res.flags |= CodeUnit::fallthru;
124  else if (jumpaddr[i] != addr) {
125  res.flags |= CodeUnit::jump;
126  res.jumpaddress = jumpaddr[i];
127  }
128  }
129 }

161 void CodeDataAnalysis::processTaint(void)
162 
163 {
164  map<Address,CodeUnit>::iterator iter = taintlist.back();
165  taintlist.pop_back();
166 
167  CodeUnit &cu((*iter).second);
168  cu.flags |= CodeUnit::notcode;
169  Address startaddr = (*iter).first;
170  Address endaddr = startaddr + cu.size;
171  if (iter != codeunit.begin()) {
172  --iter;
173  CodeUnit &cu2( (*iter).second);
174  if ((cu2.flags & (CodeUnit::fallthru&CodeUnit::notcode))==CodeUnit::fallthru) { // not "notcode" and fallthru
175  Address addr2 = (*iter).first + cu.size;
176  if (addr2 == startaddr)
177  taintlist.push_back(iter);
178  }
179  }
180  map<AddrLink,uint4>::iterator ftiter,diter,enditer;
181  ftiter = fromto_crossref.lower_bound(AddrLink(startaddr));
182  enditer = fromto_crossref.lower_bound(AddrLink(endaddr));
183  fromto_crossref.erase(ftiter,enditer); // Erase all cross-references coming out of this block
184 
185  ftiter = tofrom_crossref.lower_bound(AddrLink(startaddr));
186  enditer = tofrom_crossref.lower_bound(AddrLink(endaddr));
187  while(ftiter != enditer) {
188  pushTaintAddress((*ftiter).first.b);
189  diter = ftiter;
190  ++ftiter;
191  tofrom_crossref.erase(diter);
192  }

260 Address CodeDataAnalysis::disassembleBlock(const Address &addr,const Address &endaddr)
261 
262 {
263  DisassemblyResult disresult;
264  vector<CodeUnit> codevec;
265  map<AddrLink,uint4> fromto_vec;
266  bool flowin = false;
267  bool hardend = false;
268 
269  Address curaddr = addr;
270  map<Address,CodeUnit>::iterator iter;
271  iter = codeunit.lower_bound(addr);
272  Address lastaddr;
273  if (iter != codeunit.end()) {
274  lastaddr = (*iter).first;
275  if (endaddr < lastaddr) {
276  lastaddr = endaddr;
277  hardend = true;
278  }
279  }
280  else {
281  lastaddr = endaddr;
282  hardend = true;
283  }
284  for(;;) {
285  disengine.disassemble(curaddr,disresult);
286  codevec.push_back(CodeUnit());
287  if (!disresult.success) {
288  codevec.back().flags = CodeUnit::notcode;
289  codevec.back().size = 1;
290  curaddr = curaddr + 1;
291  break;
292  }
293  if ((disresult.flags & CodeUnit::jump)!=0) {
294  fromto_vec[ AddrLink(curaddr,disresult.jumpaddress)] = disresult.flags;
295  }
296  codevec.back().flags = disresult.flags;
297  codevec.back().size = disresult.length;
298  curaddr = curaddr + disresult.length;
299  while(lastaddr < curaddr) {
300  if ((!hardend)&&((*iter).second.flags & CodeUnit::notcode)!=0) {
301  if ((*iter).second.size == 1) {
302  map<Address,CodeUnit>::iterator iter2 = iter;
303  ++iter; // We delete the bad disassembly, as it looks like it is unaligned
304  codeunit.erase(iter2);
305  if (iter != codeunit.end()) {
306  lastaddr = (*iter).first;
307  if (endaddr < lastaddr) {
308  lastaddr = endaddr;
309  hardend = true;
310  }
311  }
312  else {
313  lastaddr = endaddr;
314  hardend = true;
315  }
316  }
317  else {
318  disresult.success = false;
319  flowin = true;
320  break;
321  }
322  }
323  else {
324  disresult.success = false;
325  break;
326  }
327  }
328  if (!disresult.success)
329  break;
330  if (curaddr == lastaddr) {
331  if (((*iter).second.flags & CodeUnit::notcode)!=0) {
332  flowin = true;
333  break;
334  }
335  }
336  if (((disresult.flags & CodeUnit::fallthru)==0)||(curaddr==lastaddr)) { // found the end of a block
337  return commitCodeVec(addr,codevec,fromto_vec);
338  }
339  }
340  // If we reach here, we have bad disassembly of some sort
341  CodeUnit &cu( codeunit[ addr ] );
342  cu.flags = CodeUnit::notcode;
343  if (hardend && (lastaddr < curaddr))
344  curaddr = lastaddr;
345  int4 wholesize = curaddr.getOffset() - addr.getOffset();
346  if ((!flowin) && (wholesize < 10)) {
347  wholesize = 1;
348  }
349  cu.size = wholesize;
350  curaddr = addr + cu.size;
351  return curaddr;
352 }

504 bool CodeDataAnalysis::repairJump(const Address &addr,int4 max)
505 
506 { // Assume -addr- is a correct instruction start. Try to repair
507  // disassembly for up to -max- instructions following it,
508  // trying to get back on cut
509  DisassemblyResult disresult;
510  vector<CodeUnit> codevec;
511  map<AddrLink,uint4> fromto_vec;
512  Address curaddr = addr;
513  map<Address,CodeUnit>::iterator iter;
514  int4 count = 0;
515 
516  iter = codeunit.lower_bound(addr);
517  if (iter == codeunit.end()) return false;
518  for(;;) {
519  count += 1;
520  if (count >=max) return false;
521  while ((*iter).first < curaddr) {
522  ++iter;
523  if (iter == codeunit.end()) return false;
524  }
525  if (curaddr == (*iter).first) break; // Back on cut
526  disengine.disassemble(curaddr,disresult);
527  if (!disresult.success) return false;
528  codevec.push_back(CodeUnit());
529  if ((disresult.flags & CodeUnit::jump)!=0) {
530  fromto_vec[ AddrLink(curaddr,disresult.jumpaddress) ] = disresult.flags;
531  }
532  codevec.back().flags = disresult.flags;
533  codevec.back().size = disresult.length;
534  curaddr = curaddr + disresult.length;
535  }
536  clearCodeUnits(addr,curaddr);
537  commitCodeVec(addr,codevec,fromto_vec);
538  return true;
539 }

541 void CodeDataAnalysis::findOffCut(void)
542 
543 {
544  map<AddrLink,uint4>::iterator iter;
545  map<Address,CodeUnit>::iterator citer;
546 
547  iter = tofrom_crossref.begin();
548  while(iter!=tofrom_crossref.end()) {
549  Address addr = (*iter).first.a; // Destination of a jump
550  citer = codeunit.lower_bound(addr);
551  if (citer != codeunit.end()) {
552  if ((*citer).first == addr) { // Not off cut
553  CodeUnit &cu( (*citer).second );
554  if ((cu.flags & (CodeUnit::hit_by_fallthru|CodeUnit::hit_by_call))==
555  (CodeUnit::hit_by_fallthru|CodeUnit::hit_by_call)) {
556  // Somebody falls through into the call
557  --citer;
558  checkErrantStart(citer);
559  }
560  ++iter;
561  continue;
562  }
563  }
564  if (citer == codeunit.begin()) {
565  ++iter;
566  continue;
567  }
568  --citer; // Last lessthan or equal
569  if ((*citer).first == addr) {
570  ++iter;
571  continue; // on cut
572  }
573  Address endaddr = (*citer).first + (*citer).second.size;
574  if (endaddr <= addr) {
575  ++iter;
576  continue;
577  }
578  if (!checkErrantStart(citer)) {
579  ++iter;
580  continue;
581  }
582  AddrLink addrlink = (*iter).first;
583  repairJump(addr,10); // This may delete tofrom_crossref nodes
584  iter = tofrom_crossref.upper_bound(addrlink);
585  }

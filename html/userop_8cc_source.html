84 void OpFollow::restoreXml(const Element *el)
85 
86 {
87  const string &name(el->getAttributeValue("code"));
88  if (name=="INT_ZEXT")
89  opc = CPUI_INT_ZEXT;
90  else if (name=="INT_LEFT")
91  opc = CPUI_INT_LEFT;
92  else if (name=="INT_AND")
93  opc = CPUI_INT_AND;
94  else

99

134 uintb SegmentOp::executeSide(const vector<OpFollow> &follow,uintb input)
135 
136 {
137  for(int4 i=0;i<follow.size();++i) {
138  switch(follow[i].opc) {
139  case CPUI_INT_AND:
140  input &= follow[i].val;
141  break;
142  case CPUI_INT_LEFT:
143  input <<= follow[i].val;
144  break;
145  default:
146  break;
147  }
148  }
149  return input;
150 }

152 bool SegmentOp::unify(Funcdata &data,PcodeOp *op,
153  vector<Varnode *> &bindlist) const
154 {
155  Varnode *basevn,*innervn;
156 
157  // Segmenting is done by a user defined p-code op, so this is what we look for
158  // The op must have innervn and basevn (if base is present) as inputs
159  // so there isn't much to follow. The OpFollow arrays are no
160  // longer needed for unification but are needed to provide
161  // a definition for the userop
162  if (op->code() != CPUI_CALLOTHER) return false;
163  if (op->getIn(0)->getOffset() != useropindex) return false;
164  if (op->numInput() != 3) return false;
165  innervn = op->getIn(1);
166  if (basepresent) {
167  basevn = op->getIn(1);
168  innervn = op->getIn(2);
169  if (basevn->isConstant())
170  basevn = data.newConstant(baseinsize,basevn->getOffset());
171  bindlist[1] = basevn;
172  }
173  else
174  bindlist[1] = (Varnode *)0;
175  if (innervn->isConstant())
176  innervn = data.newConstant(innerinsize,innervn->getOffset());
177  bindlist[0] = innervn;
178  return true;
179 }

195 void SegmentOp::restoreXml(const Element *el)
196 
197 {
198  spc = glb->getSpaceByName(el->getAttributeValue("space"));
199  baseinsize = 0;
200  innerinsize = 0;
201  bool userdefined = false;
202  forcesegment = true;
203  supportsfarpointer = false;
204  name = "segment"; // Default name, might be overridden by userop attribute
205  for(int4 i=0;i<el->getNumAttributes();++i) {
206  const string &nm(el->getAttributeName(i));
207  if (nm == "space") continue;
208  else if (nm == "baseinsize") {
209  istringstream s(el->getAttributeValue(i));
210  s.unsetf(ios::dec | ios::hex | ios::oct);
211  s >> baseinsize;
212  }
213  else if (nm == "innerinsize") {
214  istringstream s1(el->getAttributeValue(i));
215  s1.unsetf(ios::dec | ios::hex | ios::oct);
216  s1 >> innerinsize;
217  }
218  else if (nm == "farpointer")
219  supportsfarpointer = true;
220  else if (nm == "userop") { // Based on existing sleigh op
221  name = el->getAttributeValue(i);
222  UserPcodeOp *otherop = glb->userops.getOp(name);
223  if (otherop != (UserPcodeOp *)0) {
224  userdefined = true;
225  useropindex = otherop->getIndex();
226  if (dynamic_cast<UnspecializedPcodeOp *>(otherop) == (UnspecializedPcodeOp *)0)
227  throw LowlevelError("Redefining userop "+name);
228  }
229  }
230  else if (nm == "force")
231  forcesegment = xml_readbool(el->getAttributeValue(i));
232  else
233  throw LowlevelError("Bad segmentop tag attribute: "+nm);
234  }
235  if (!userdefined)
236  throw LowlevelError("Missing userop attribute in segmentop tag");
237  basepresent = (baseinsize != 0);
238 
239  const List &list(el->getChildren());
240  List::const_iterator iter;
241  for(iter=list.begin();iter!=list.end();++iter) {
242  const Element *subel = *iter;
243  if (subel->getName()=="baseop") {
244  basefollow.push_back(OpFollow());
245  basefollow.back().restoreXml(subel);
246  }
247  else if (subel->getName()=="innerop") {
248  innerfollow.push_back(OpFollow());
249  innerfollow.back().restoreXml(subel);
250  }
251  else if (subel->getName()=="constresolve") {
252  int4 sz;
253  const List &sublist(subel->getChildren());
254  if (!sublist.empty()) {
255  List::const_iterator subiter = sublist.begin();
256  const Element *subsubel = *subiter;
257  Address addr = Address::restoreXml(subsubel,glb,sz);
258  constresolve.space = addr.getSpace();
259  constresolve.offset = addr.getOffset();
260  constresolve.size = sz;
261  }
262  }
263  else
264  throw LowlevelError("Bad segment pattern tag: "+subel->getName());
265  }
266 }

395 void UserOpManage::registerOp(UserPcodeOp *op)
396 
397 {
398  int4 ind = op->getIndex();
399  if (ind < 0) throw LowlevelError("UserOp not assigned an index");
400 
401  map<string,UserPcodeOp *>::iterator iter;
402  iter = useropmap.find(op->getName());
403  if (iter != useropmap.end()) {
404  UserPcodeOp *other = (*iter).second;
405  if (other->getIndex() != ind)
406  throw LowlevelError("Conflicting indices for userop name "+op->getName());
407  }
408 
409  while(useroplist.size() <= ind)
410  useroplist.push_back((UserPcodeOp *)0);
411  if (useroplist[ind] != (UserPcodeOp *)0) {
412  if (useroplist[ind]->getName() != op->getName())
413  throw LowlevelError("User op "+op->getName()+" has same index as "+useroplist[ind]->getName());
414  // We assume this registration customizes an existing userop
415  delete useroplist[ind]; // Delete the old spec
416  }
417  useroplist[ind] = op; // Index crossref
418  useropmap[op->getName()] = op; // Name crossref
419 
420  SegmentOp *s_op = dynamic_cast<SegmentOp *>(op);
421  if (s_op != (SegmentOp *)0) {
422  int4 index = s_op->getSpace()->getIndex();
423 
424  while(segmentop.size() <= index)
425  segmentop.push_back((SegmentOp *)0);
426 
427  if (segmentop[index] != (SegmentOp *)0)
428  throw LowlevelError("Multiple segmentops defined for same space");
429  segmentop[index] = s_op;
430  return;
431  }
432  VolatileReadOp *tmpVolRead = dynamic_cast<VolatileReadOp *>(op);
433  if (tmpVolRead != (VolatileReadOp *)0) {
434  if (vol_read != (VolatileReadOp *)0)
435  throw LowlevelError("Multiple volatile reads registered");
436  vol_read = tmpVolRead;
437  return;
438  }
439  VolatileWriteOp *tmpVolWrite = dynamic_cast<VolatileWriteOp *>(op);
440  if (tmpVolWrite != (VolatileWriteOp *)0) {
441  if (vol_write != (VolatileWriteOp *)0)
442  throw LowlevelError("Multiple volatile writes registered");
443  vol_write = tmpVolWrite;
444  }
445 }

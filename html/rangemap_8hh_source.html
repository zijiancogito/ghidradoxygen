17 // A container for records occupying (possibly overlapping)
18 // intervals. I.e. a map from a linear ordered domain to
19 // (multiple) records.
20 // recordtype is the type of a record
21 // must support
22 // constructor(first,last)
23 // getFirst() beginning of range
24 // getLast() end of range (inclusive)
25 // getSubsort()
26 // initialize() initialization with inittype object
27 // must define types
28 // linetype
29 // subsorttype
30 // inittype
31 // linetype is the type of elements in the linear domain
32 // must support <,<=,==,!=, +(integer) -(integer)
33 // subsorttype - overlapping intervals can be subsorted
34 // must suport <
35 // null or false initialization produces minimal value
36 // true initialization produces maximal value
37 // copy constructor

137 template<typename _recordtype>
138 void rangemap<_recordtype>::zip(linetype i,typename std::multiset<AddrRange>::iterator iter)
139 
140 { // Remove the partition boundary occurring right after i
141  // This should run in O(k)
142  linetype f = (*iter).first;
143  while((*iter).last == i)
144  tree.erase(iter++);
145  i = i+1;
146  while((iter!=tree.end())&&((*iter).first==i)) {
147  (*iter).first = f;
148  ++iter;
149  }

152 template<typename _recordtype>
153 void rangemap<_recordtype>::unzip(linetype i,typename std::multiset<AddrRange>::iterator iter)
154 
155 { // Create a new partition boundary right after i
156  // This should run in O(k), where k is the number
157  // of intervals intersecting the point i
158  // iter should be the first interval containing i
159  typename std::multiset<AddrRange>::iterator hint = iter;
160  if ((*iter).last == i) return; // Can't split size 1 (i.e. split already present)
161  linetype f;
162  linetype plus1 = i+1;
163  while((iter!=tree.end())&&((*iter).first<=i)) {
164  f = (*iter).first;
165  (*iter).first = plus1;
166  typename std::multiset<AddrRange>::iterator newiter;
167  newiter = tree.insert(hint,AddrRange(i,(*iter).subsort));
168  const AddrRange &newrange( *newiter );
169  newrange.first = f;
170  newrange.a = (*iter).a;
171  newrange.b = (*iter).b;
172  newrange.value = (*iter).value;
173  ++iter;
174  }
175 }

177 template<typename _recordtype>
178 typename std::list<_recordtype>::iterator
179 rangemap<_recordtype>::insert(const inittype &data,linetype a,linetype b)
180 
181 { // Insert a new record into the container at inclusive range [a,b]
182  linetype f=a;
183  typename std::list<_recordtype>::iterator liter;
184  typename std::multiset<AddrRange>::iterator low = tree.lower_bound(AddrRange(f));
185 
186  if (low != tree.end()) {
187  if ((*low).first < f) // Check if left boundary refines existing partition
188  unzip(f-1,low); // If so do the refinement
189  }
190 
191  record.push_front( _recordtype(a,b) );
192  record.front().initialize( data );
193  liter = record.begin();
194 
195  AddrRange addrrange(b,(*liter).getSubsort());
196  addrrange.a = a;
197  addrrange.b = b;
198  addrrange.value = liter;
199  typename std::multiset<AddrRange>::iterator spot = tree.lower_bound(addrrange);
200  // Where does the new record go in full list, insert it
201  record.splice( (spot==tree.end()) ? record.end():(*spot).value,
202  record,liter);
203 
204  while((low != tree.end())&&((*low).first<=b)) {
205  if (f <= (*low).last) { // Do we overlap at all
206  if (f < (*low).first) {
207  // Assume the hint makes this insert an O(1) op
208  addrrange.first = f;
209  addrrange.last = (*low).first-1;
210  tree.insert(low,addrrange);
211  f = (*low).first;
212  }
213  if ((*low).last <= b) { // Insert as much of interval as we can
214  addrrange.first = f;
215  addrrange.last = (*low).last;
216  tree.insert(low,addrrange);
217  if ((*low).last==b) break; // Did we manage to insert it all
218  f = (*low).last + 1;
219  }
220  else if (b < (*low).last) { // We can insert everything left, but must refine
221  unzip(b,low);
222  break;
223  }
224  }
225  ++low;
226  }
227  if (f <= b) {
228  addrrange.first = f;
229  addrrange.last = b;
230  tree.insert(addrrange);
231  }
232 
233  return liter;
234 }

236 template<typename _recordtype>
237 void rangemap<_recordtype>::erase(typename std::list<_recordtype>::iterator v)
238 
239 {
240  linetype a = (*v).getFirst();
241  linetype b = (*v).getLast();
242  bool leftsew = true;
243  bool rightsew = true;
244  bool rightoverlap = false;
245  bool leftoverlap = false;
246  typename std::multiset<AddrRange>::iterator low = tree.lower_bound(AddrRange(a));
247  typename std::multiset<AddrRange>::iterator uplow = low;
248 
249  linetype aminus1 = a-1;
250  while (uplow != tree.begin()) {
251  --uplow;
252  if ((*uplow).last != aminus1) break;
253  if ((*uplow).b == aminus1) {
254  leftsew = false; // Still a split between a-1 and a
255  break;
256  }
257  }
258  do {
259  if ((*low).value == v)
260  tree.erase(low++);
261  else {
262  if ((*low).a < a)
263  leftoverlap = true; // a splits somebody else
264  else if ((*low).a == a)
265  leftsew = false; // Somebody else splits at a (in addition to v)
266  if (b < (*low).b)
267  rightoverlap = true; // b splits somebody else
268  else if ((*low).b == b)
269  rightsew = false; // Somebody else splits at b (in addition to v)
270  low++;
271  }
272  } while ((low != tree.end())&&((*low).first<=b));
273  if (low != tree.end()) {
274  if ((*low).a-1 == b)
275  rightsew = false;
276  }
277  if (leftsew&&leftoverlap)
278  zip(a-1,tree.lower_bound(AddrRange(a-1)));
279  if (rightsew&&rightoverlap)
280  zip(b,tree.lower_bound(AddrRange(b)));
281  record.erase(v);
282 }

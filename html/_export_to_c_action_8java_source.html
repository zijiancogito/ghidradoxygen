48  @Override

51  return false;

54  if (decompilerActionContext.isDecompiling()) {

78  @Override
79  public void actionPerformed(ActionContext context) {
80  // Note: we intentionally do this check here and not in isEnabledForContext() so
81  // that global events do not get triggered.
82  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
83  if (decompilerActionContext.isDecompiling()) {
84  Msg.showInfo(getClass(), context.getComponentProvider().getComponent(),
85  "Decompiler Action Blocked",
86  "You cannot perform Decompiler actions while the Decompiler is busy");
87  return;
88  }
89 
90  File lastUsedFile = readLastUsedFile();
91 
92  String[] extensions = new String[] { "h", "c", "cpp" };
93  GhidraFileChooser fileChooser = new GhidraFileChooser(controller.getDecompilerPanel());
94  fileChooser.setFileFilter(new ExtensionFileFilter(extensions, "C/C++ Files"));
95  if (lastUsedFile != null) {
96  fileChooser.setSelectedFile(lastUsedFile);
97  }
98  File file = fileChooser.getSelectedFile();
99  if (file == null) {
100  return;
101  }
102 
103  saveLastUsedFileFile(file);
104 
105  boolean hasExtension = false;
106  String path = file.getAbsolutePath();
107  for (String element : extensions) {
108  if (path.toLowerCase().endsWith("." + element)) {
109  hasExtension = true;
110  }
111  }
112 
113  if (!hasExtension) {
114  file = new File(path + ".c");
115  }
116 
117  if (file.exists()) {
118  if (OptionDialog.showYesNoDialog(controller.getDecompilerPanel(),
119  "Overwrite Existing File?", "Do you want to overwrite the existing file?") == OptionDialog.OPTION_TWO) {
120  return;
121  }
122  }
123 
124  try {
125  PrintWriter writer = new PrintWriter(new FileOutputStream(file));
126  ClangTokenGroup grp = controller.getCCodeModel();
127  PrettyPrinter printer = new PrettyPrinter(controller.getFunction(), grp);
128  DecompiledFunction decompFunc = printer.print(true);
129  writer.write(decompFunc.getC());
130  writer.close();
131  controller.setStatusMessage("Successfully exported function(s) to " +
132  file.getAbsolutePath());
133  }
134  catch (IOException e) {
135  Msg.showError(getClass(), controller.getDecompilerPanel(), "Export to C Failed",
136  "Error exporting to C: " + e);
137  }
138  }

53  if (decompilerActionContext.isDecompiling()) {
54  // Let this through here and handle it in actionPerformed(). This lets us alert
55  // the user that they have to wait until the decompile is finished. If we are not
56  // enabled at this point, then the keybinding will be propagated to the global

64  @Override
65  public void actionPerformed(ActionContext context) {
66  // Note: we intentionally do this check here and not in isEnabledForContext() so
67  // that global events do not get triggered.
68  DecompilerActionContext decompilerActionContext = (DecompilerActionContext) context;
69  if (decompilerActionContext.isDecompiling()) {
70  Msg.showInfo(getClass(),

72  "Decompiler Action Blocked", "You cannot perform Decompiler actions while the Decompiler is busy");

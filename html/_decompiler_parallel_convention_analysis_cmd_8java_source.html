104  /*
105  * The method indicates whether the function is in a block of code that is considered external or "glue"--meaning
106  * that we don't want to analyze the code that might be there, yet we might have signatures for the function or
107  * what it provides linkage to that we do not want to wipe. We want to keep what is already there.
108  * TODO: This implementation below, with hard-coded specific block names (other than EXTERNAL, which is created
109  * by analysis) will need to be revisited. Perhaps a flag will be set by the importer on the blocks that we should
110  * ignore.
111  */
112  private boolean funcIsExternalGlue(Function func) {
113  String blockName = program.getMemory().getBlock(func.getEntryPoint()).getName();
114  return (blockName.equals("EXTERNAL") || blockName.equals(".plt") ||
115  blockName.equals("__stub_helper"));
116  }

124  private void analyzeFunction(Function f, TaskMonitor monitor) {
125 
126  // do some simple checks just in case
127  if (f == null || f.isThunk() || isInStdNamespace(f)) {
128  return;
129  }
130 
131  // if custom storage already enabled or calling convention known, return
132  if (f.hasCustomVariableStorage() ||
133  !f.getCallingConventionName().equals(Function.UNKNOWN_CALLING_CONVENTION_STRING)) {
134  return;
135  }
136 
137  //We didn't "wipe" previous results of external functions, but we also do not want
138  // to set new results.
139  if (f.isExternal()) {
140  return;
141  }
142 
143  if (funcIsExternalGlue(f)) {
144  return;
145  }
146 
147  SourceType signatureSource = f.getSignatureSource();
148 
149  try {
150  DecompileResults decompRes = null;
151  if (monitor.isCancelled()) {
152  return;
153  }
154 
155  // reset the sourcetype so that no signature information goes to the decompiler
156  // we will set if back later.
157  f.setSignatureSource(SourceType.DEFAULT);
158 
159  decompRes = this.decompiler.decompileFunction(f, decompilerTimeoutSecs, monitor);
160  setStatusMsg(decompRes.getErrorMessage());
161 
162  if (monitor.isCancelled()) {
163  return;
164  }
165 
166  if (!decompRes.decompileCompleted()) {
167  return;
168  }
169 
170  HighFunction highFunction = decompRes.getHighFunction();
171  String modelName = highFunction.getFunctionPrototype().getModelName();
172 
173  // TODO: Need to check the calling convention name
174  // what does decompiler return if it doesn't know convention, or guessed?
175  if (!modelName.equals(Function.DEFAULT_CALLING_CONVENTION_STRING)) {
176  signatureSource =
177  updateCallingConvention(f, signatureSource, highFunction, modelName);
178  }
179 
180  String errMsg = getStatusMsg();
181  if (!monitor.isCancelled() && (errMsg != null && errMsg.length() != 0)) {
182  Msg.debug(this, " Failed to decompile function: " + f.getName() + " - " + errMsg);
183  }
184  }
185  catch (Exception e) {
186  if (!monitor.isCancelled()) {
187  String errMsg = e.getMessage();
188  if (errMsg == null) {
189  errMsg = "Error decompiling function: " + e;
190  }
191  setStatusMsg(errMsg);
192  }
193  }
194  finally {
195  f.setSignatureSource(signatureSource);
196  }
197  }

199  private SourceType updateCallingConvention(Function f, SourceType signatureSource,
200  HighFunction highFunction, String modelName) throws InvalidInputException {
201  // do the number of parameters disagree and decompiler says there is one more
202  Namespace parentNamespace = f.getParentNamespace();
203  if (f.getParameterCount() + 1 == highFunction.getFunctionPrototype().getNumParams()) {
204  // does it have a namespace
205  if (parentNamespace.getID() != Namespace.GLOBAL_NAMESPACE_ID &&
206  // prevent accidental treatment of std namespace as class
207  !parentNamespace.getName().equals(STD_NAMESPACE)) {
208  // does it have a this call convention that is the equivalent of the stdcall
209  PrototypeModel callingConvention = program.getCompilerSpec().getCallingConvention(
210  CompilerSpec.CALLING_CONVENTION_thiscall);
211  if (callingConvention != null) {
212  modelName = CompilerSpec.CALLING_CONVENTION_thiscall;
213  }
214  }
215  }
216  // Then is __thiscall, create an object and new parameter if it doesn't have one yet.
217  if (modelName.equals(CompilerSpec.CALLING_CONVENTION_stdcall) &&
218  f.getStackPurgeSize() == 0 && f.getParameterCount() > 0) {
219  // if has parameters, and there is no purge, it can't be a stdcall, change it to cdecl
220  if (program.getLanguageID().getIdAsString().startsWith("x86:LE:32")) {
221  modelName = CompilerSpec.CALLING_CONVENTION_cdecl;
222  // it could be a this call...
223  }
224  }
225  if (parentNamespace.getSymbol().getSymbolType() == SymbolType.NAMESPACE &&
226  modelName.equals(CompilerSpec.CALLING_CONVENTION_thiscall)) {
227  NamespaceUtils.convertNamespaceToClass(f.getParentNamespace());
228  }
229  f.setCallingConvention(modelName);
230  if (signatureSource == SourceType.DEFAULT) {
231  signatureSource = SourceType.ANALYSIS;
232  }
233  return signatureSource;
234  }

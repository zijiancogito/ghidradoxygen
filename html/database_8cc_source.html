409 void FunctionSymbol::buildType(int4 size)
410 
411 {
412  TypeFactory *types = scope->getArch()->types;
413  type = types->getTypeCode();
414  // Entries for functions have small size starting at the entry address
415  // of the function in order to deal with non-contiguous functions
416  // The size used to always be 1, but now we need sizes (slightly) larger than 1
417  // to accomodate pointer constants that encode extra information in the lower bit(s)
418  // of an otherwise aligned pointer. If the encoding is not aprior detected, it is interpreted
419  // initially as a straight address that comes up 1 (or more) bytes off of the start of the function
420  // In order to detect this, we need to lay down a slightly larger size than 1
421  if (size > 1)
422  type = types->getTypeArray(size,type);
423 
424  flags |= Varnode::namelock | Varnode::typelock;
425 }

497 bool EquateSymbol::isValueClose(uintb op2Value,int4 size) const
498 
499 {
500  if (value == op2Value) return true;
501  uintb mask = calc_mask(size);
502  uintb maskValue = value & mask;
503  if (maskValue != value) { // If '1' bits are getting masked off
504  // Make sure only sign-extension is getting masked off
505  if (value != sign_extend(maskValue,size,sizeof(uintb)))
506  return false;
507  }
508  if (maskValue == (op2Value & mask)) return true;
509  if (maskValue == (~op2Value & mask)) return true;
510  if (maskValue == (-op2Value & mask)) return true;
511  if (maskValue == ((op2Value + 1) & mask)) return true;
512  if (maskValue == ((op2Value -1) & mask)) return true;
513  return false;
514 }

19 int4 TokenPattern::resolveTokens(const TokenPattern &tok1,const TokenPattern &tok2)
20 
21 { // Use the token lists to decide how the two patterns
22  // should be aligned relative to each other
23  // return how much -tok2- needs to be shifted
24  // and set the resulting tokenlist and ellipses
25  bool reversedirection = false;
26  leftellipsis = false;
27  rightellipsis = false;
28  int4 ressa = 0;
29  int4 minsize = tok1.toklist.size() < tok2.toklist.size() ? tok1.toklist.size() : tok2.toklist.size();
30  if (minsize == 0) {
31  // Check if pattern doesn't care about tokens
32  if ((tok1.toklist.size()==0)&&(tok1.leftellipsis==false)&&(tok1.rightellipsis==false)) {
33  toklist = tok2.toklist;
34  leftellipsis = tok2.leftellipsis;
35  rightellipsis = tok2.rightellipsis;
36  return 0;
37  }
38  else if ((tok2.toklist.size()==0)&&(tok2.leftellipsis==false)&&(tok2.rightellipsis==false)) {
39  toklist = tok1.toklist;
40  leftellipsis = tok1.leftellipsis;
41  rightellipsis = tok1.rightellipsis;
42  return 0;
43  }
44  // If one of the ellipses is true then the pattern
45  // still cares about tokens even though none are
46  // specified
47  }
48 
49  if (tok1.leftellipsis) {
50  reversedirection = true;
51  if (tok2.rightellipsis)
52  throw SleighError("Right/left ellipsis");
53  else if (tok2.leftellipsis)
54  leftellipsis = true;
55  else if (tok1.toklist.size() != minsize)
56  throw SleighError("Mismatched pattern sizes");
57  else if (tok1.toklist.size()==tok2.toklist.size())
58  throw SleighError("Pattern size cannot vary (missing ... ?)");
59  }
60  else if (tok1.rightellipsis) {
61  if (tok2.leftellipsis)
62  throw SleighError("Left/right ellipsis");
63  else if (tok2.rightellipsis)
64  rightellipsis = true;
65  else if (tok1.toklist.size() != minsize)
66  throw SleighError("Mismatched pattern sizes");
67  else if (tok1.toklist.size()==tok2.toklist.size())
68  throw SleighError("Pattern size cannot vary (missing ... ?)");
69  }
70  else {
71  if (tok2.leftellipsis) {
72  reversedirection = true;
73  if (tok2.toklist.size() != minsize)
74  throw SleighError("Mismatched pattern sizes");
75  else if (tok1.toklist.size()==tok2.toklist.size())
76  throw SleighError("Pattern size cannot vary (missing ... ?)");
77  }
78  else if (tok2.rightellipsis) {
79  if (tok2.toklist.size() != minsize)
80  throw SleighError("Mismatched pattern sizes");
81  else if (tok1.toklist.size()==tok2.toklist.size())
82  throw SleighError("Pattern size cannot vary (missing ... ?)");
83  }
84  else {
85  if (tok2.toklist.size() != tok1.toklist.size())
86  throw SleighError("Mismatched pattern sizes");
87  }
88  }
89  if (reversedirection) {
90  for(int4 i=0;i<minsize;++i)
91  if (tok1.toklist[tok1.toklist.size()-1-i] != tok2.toklist[tok2.toklist.size()-1-i])
92  throw SleighError("Mismatched tokens when combining patterns");
93  if (tok1.toklist.size() <= tok2.toklist.size())
94  for(int4 i=minsize;i<tok2.toklist.size();++i)
95  ressa += tok2.toklist[tok2.toklist.size()-1-i]->getSize();
96  else
97  for(int4 i=minsize;i<tok1.toklist.size();++i)
98  ressa += tok1.toklist[tok1.toklist.size()-1-i]->getSize();
99  if (tok1.toklist.size() < tok2.toklist.size())
100  ressa = -ressa;
101  }
102  else {
103  for(int4 i=0;i<minsize;++i)
104  if (tok1.toklist[i] != tok2.toklist[i])
105  throw SleighError("Mismatched tokens when combining patterns");
106  }
107  // Save the results into -this-
108  if (tok1.toklist.size() <= tok2.toklist.size())
109  toklist = tok2.toklist;
110  else
111  toklist = tok1.toklist;
112  return ressa;
113 }

115 PatternBlock *TokenPattern::buildSingle(int4 startbit,int4 endbit,uintm byteval)
116 
117 { // Create a mask/value pattern within a single word
118  // The field is given by the bitrange [startbit,endbit]
119  // bit 0 is the MOST sig bit of the word
120  // use the least sig bits of byteval to fill in
121  // the field's value
122  uintm mask;
123  int4 offset = 0;
124  int4 size = endbit-startbit+1;
125  while(startbit >= 8) {
126  offset += 1;
127  startbit -= 8;
128  endbit -= 8;
129  }
130  mask = (~((uintm)0)) << (sizeof(uintm)*8-size);
131  byteval = (byteval << (sizeof(uintm)*8-size))& mask;
132  mask >>= startbit;
133  byteval >>= startbit;
134  return new PatternBlock(offset,mask,byteval);
135 }

137 PatternBlock *TokenPattern::buildBigBlock(int4 size,int4 bitstart,int4 bitend,intb value)
138 
139 { // Build pattern block given a bigendian contiguous
140  // range of bits and a value for those bits
141  int4 tmpstart,startbit,endbit;
142  PatternBlock *tmpblock,*block;
143 
144  startbit = 8*size - 1 - bitend;
145  endbit = 8*size - 1 - bitstart;
146 
147  block = (PatternBlock *)0;
148  while(endbit >= startbit) {
149  tmpstart = endbit - (endbit & 7);
150  if (tmpstart < startbit)
151  tmpstart = startbit;
152  tmpblock = buildSingle(tmpstart,endbit,(uintm)value);
153  if (block == (PatternBlock *)0)
154  block = tmpblock;
155  else {
156  PatternBlock *newblock = block->intersect(tmpblock);
157  delete block;
158  delete tmpblock;
159  block = newblock;
160  }
161  value >>= (endbit-tmpstart+1);
162  endbit = tmpstart - 1;
163  }
164  return block;
165 }

167 PatternBlock *TokenPattern::buildLittleBlock(int4 size,int4 bitstart,int4 bitend,intb value)
168 
169 { // Build pattern block given a littleendian contiguous
170  // range of bits and a value for those bits
171  PatternBlock *tmpblock,*block;
172  int4 startbit,endbit;
173 
174  block = (PatternBlock *)0;
175 
176  // we need to convert a bit range specified on a little endian token where the
177  // bit indices label the least sig bit as 0 into a bit range on big endian bytes
178  // where the indices label the most sig bit as 0. The reversal due to
179  // little->big endian cancels part of the reversal due to least->most sig bit
180  // labelling, but not on the lower 3 bits. So the transform becomes
181  // leave the upper bits the same, but transform the lower 3-bit value x into 7-x.
182 
183  startbit = (bitstart/8) * 8; // Get the high-order portion of little/LSB labelling
184  endbit = (bitend/8) * 8;
185  bitend = bitend % 8; // Get the low-order portion of little/LSB labelling
186  bitstart = bitstart % 8;
187 
188  if (startbit == endbit) {
189  startbit += 7 - bitend;
190  endbit += 7 - bitstart;
191  block = buildSingle(startbit,endbit,(uintm)value);
192  }
193  else {
194  block = buildSingle(startbit,startbit+(7-bitstart),(uintm)value);
195  value >>= (8-bitstart); // Cut off bits we just encoded
196  startbit += 8;
197  while(startbit != endbit) {
198  tmpblock = buildSingle(startbit,startbit+7,(uintm)value);
199  if (block == (PatternBlock *)0)
200  block = tmpblock;
201  else {
202  PatternBlock *newblock = block->intersect(tmpblock);
203  delete block;
204  delete tmpblock;
205  block = newblock;
206  }
207  value >>= 8;
208  startbit += 8;
209  }
210  tmpblock = buildSingle(endbit+(7-bitend),endbit+7,(uintm)value);
211  if (block == (PatternBlock *)0)
212  block = tmpblock;
213  else {
214  PatternBlock *newblock = block->intersect(tmpblock);
215  delete block;
216  delete tmpblock;
217  block = newblock;
218  }
219  }
220  return block;
221 }

317 TokenPattern TokenPattern::doCat(const TokenPattern &tokpat) const
318 
319 { // Return Concatenation of -this- and -tokpat-
320  TokenPattern res((Pattern *)0);
321  int4 sa;
322 
323  res.leftellipsis = leftellipsis;
324  res.rightellipsis = rightellipsis;
325  res.toklist = toklist;
326  if (rightellipsis||tokpat.leftellipsis) { // Check for interior ellipsis
327  if (rightellipsis) {
328  if (!tokpat.alwaysInstructionTrue())
329  throw SleighError("Interior ellipsis in pattern");
330  }
331  if (tokpat.leftellipsis) {
332  if (!alwaysInstructionTrue())
333  throw SleighError("Interior ellipsis in pattern");
334  res.leftellipsis = true;
335  }
336  sa = -1;
337  }
338  else {
339  sa = 0;
340  vector<Token *>::const_iterator iter;
341 
342  for(iter=toklist.begin();iter!=toklist.end();++iter)
343  sa += (*iter)->getSize();
344  for(iter=tokpat.toklist.begin();iter!=tokpat.toklist.end();++iter)
345  res.toklist.push_back(*iter);
346  res.rightellipsis = tokpat.rightellipsis;
347  }
348  if (res.rightellipsis && res.leftellipsis)
349  throw SleighError("Double ellipsis in pattern");
350  if (sa < 0)
351  res.pattern = pattern->doAnd(tokpat.pattern,0);
352  else
353  res.pattern = pattern->doAnd(tokpat.pattern,sa);
354  return res;
355 }

357 TokenPattern TokenPattern::commonSubPattern(const TokenPattern &tokpat) const
358 
359 { // Construct pattern that matches anything
360  // that matches either -this- or -tokpat-
361  TokenPattern patres((Pattern *)0); // Empty shell
362  int4 i;
363  bool reversedirection = false;
364 
365  if (leftellipsis||tokpat.leftellipsis) {
366  if (rightellipsis||tokpat.rightellipsis)
367  throw SleighError("Right/left ellipsis in commonSubPattern");
368  reversedirection = true;
369  }
370 
371  // Find common subset of tokens and ellipses
372  patres.leftellipsis = leftellipsis || tokpat.leftellipsis;
373  patres.rightellipsis = rightellipsis || tokpat.rightellipsis;
374  int4 minnum = toklist.size();
375  int4 maxnum = tokpat.toklist.size();
376  if (maxnum < minnum) {
377  int4 tmp = minnum;
378  minnum = maxnum;
379  maxnum = tmp;
380  }
381  if (reversedirection) {
382  for(i=0;i<minnum;++i) {
383  Token *tok = toklist[toklist.size()-1-i];
384  if (tok == tokpat.toklist[tokpat.toklist.size()-1-i])
385  patres.toklist.insert(patres.toklist.begin(),tok);
386  else
387  break;
388  }
389  if (i<maxnum)
390  patres.leftellipsis = true;
391  }
392  else {
393  for(i=0;i<minnum;++i) {
394  Token *tok = toklist[i];
395  if (tok == tokpat.toklist[i])
396  patres.toklist.push_back(tok);
397  else
398  break;
399  }
400  if (i<maxnum)
401  patres.rightellipsis = true;
402  }
403 
404  patres.pattern = pattern->commonSubPattern(tokpat.pattern,0);
405  return patres;
406 }

472 static intb getInstructionBytes(ParserWalker &walker,int4 bytestart,int4 byteend,bool bigendian)
473 
474 { // Build a intb from the instruction bytes
475  intb res = 0;
476  uintm tmp;
477  int4 size,tmpsize;
478 
479  size = byteend-bytestart+1;
480  tmpsize = size;
481  while(tmpsize >= sizeof(uintm)) {
482  tmp = walker.getInstructionBytes(bytestart,sizeof(uintm));
483  res <<= 8*sizeof(uintm);
484  res |= tmp;
485  bytestart += sizeof(uintm);
486  tmpsize -= sizeof(uintm);
487  }
488  if (tmpsize > 0) {
489  tmp = walker.getInstructionBytes(bytestart,tmpsize);
490  res <<= 8*tmpsize;
491  res |= tmp;
492  }
493  if (!bigendian)
494  byte_swap(res,size);
495  return res;
496 }

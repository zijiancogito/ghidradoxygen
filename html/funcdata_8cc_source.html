201 void Funcdata::spacebase(void)
202 
203 {
204  VarnodeLocSet::const_iterator iter,enditer;
205  int4 i,j,numspace;
206  Varnode *vn;
207  AddrSpace *spc;
208 
209  for(j=0;j<glb->numSpaces();++j) {
210  spc = glb->getSpace(j);
211  numspace = spc->numSpacebase();
212  for(i=0;i<numspace;++i) {
213  const VarnodeData &point(spc->getSpacebase(i));
214  // Find input varnode at this size and location
215  Datatype *ct = glb->types->getTypeSpacebase(spc,getAddress());
216  Datatype *ptr = glb->types->getTypePointer(point.size,ct,spc->getWordSize());
217 
218  iter = vbank.beginLoc(point.size,Address(point.space,point.offset));
219  enditer = vbank.endLoc(point.size,Address(point.space,point.offset));
220  while(iter != enditer) {
221  vn = *iter++;
222  if (vn->isFree()) continue;
223  if (vn->isSpacebase()) { // This has already been marked spacebase
224  // We have given it a chance for descendants to
225  // be eliminated naturally, now force a split if
226  // it still has multiple descendants
227  PcodeOp *op = vn->getDef();
228  if ((op != (PcodeOp *)0)&&(op->code() == CPUI_INT_ADD))
229  splitUses(vn);
230  }
231  else {
232  vn->setFlags(Varnode::spacebase); // Mark all base registers (not just input)
233  if (vn->isInput()) // Only set type on the input spacebase register
234  vn->updateType(ptr,true,true);
235  }
236  }
237  }
238  }

289 void Funcdata::spacebaseConstant(PcodeOp *op,int4 slot,SymbolEntry *entry,const Address &rampoint,uintb origval,int4 origsize)
290 
291 {
292  PcodeOp *addop;
293  int4 sz = rampoint.getAddrSize();
294  AddrSpace *spaceid = rampoint.getSpace();
295  Datatype *sb_type = glb->types->getTypeSpacebase(spaceid,Address());
296  sb_type = glb->types->getTypePointer(sz,sb_type,spaceid->getWordSize());
297  Varnode *spacebase_vn,*outvn,*newconst;
298 
299  uintb extra = rampoint.getOffset() - entry->getAddr().getOffset(); // Offset from beginning of entry
300  extra = AddrSpace::byteToAddress(extra,rampoint.getSpace()->getWordSize()); // Convert to address units
301 
302  spacebase_vn = newConstant(sz,0);
303  spacebase_vn->updateType(sb_type,true,true);
304  spacebase_vn->setFlags(Varnode::spacebase);
305  addop = newOp(2,op->getAddr());
306  opSetOpcode(addop,CPUI_PTRSUB);
307  outvn = newUniqueOut(sz,addop);
308  // Make sure newconstant and extra preserve origval in address units
309  uintb newconstoff = origval - extra; // everything is already in address units
310  newconst = newConstant(sz,newconstoff);
311  newconst->setPtrCheck(); // No longer need to check this constant as a pointer
312  if (spaceid->isTruncated())
313  addop->setPtrFlow();
314  opSetInput(addop,spacebase_vn,0);
315  opSetInput(addop,newconst,1);
316  opInsertBefore(addop,op);
317  Symbol *sym = entry->getSymbol();
318  Datatype *entrytype = sym->getType();
319  Datatype *ptrentrytype = glb->types->getTypePointer(sz,entrytype,spaceid->getWordSize());
320  bool typelock = sym->isTypeLocked();
321  if (typelock && (entrytype->getMetatype() == TYPE_UNKNOWN))
322  typelock = false;
323  outvn->updateType(ptrentrytype,typelock,true);
324  if (extra != 0) {
325  PcodeOp *extraop = newOp(2,op->getAddr());
326  opSetOpcode(extraop,CPUI_INT_ADD);
327  Varnode *outvn2 = newUniqueOut(sz,extraop);
328  Varnode *extconst = newConstant(sz,extra);
329  extconst->setPtrCheck();
330  opSetInput(extraop,outvn,0);
331  opSetInput(extraop,extconst,1);
332  outvn = outvn2;
333  opInsertBefore(extraop,op);
334  }
335  if (sz != origsize) { // There is a change in size from address -> varnode
336  PcodeOp *zextop = newOp(1,op->getAddr());
337  Varnode *outvn2 = newUniqueOut(origsize,zextop);
338  opSetOpcode(zextop,CPUI_INT_ZEXT); // Create an extension to get back to original varnode size
339  opSetInput(zextop,outvn,0);
340  opInsertBefore(zextop,op);
341  outvn = outvn2;
342  }
343  opSetInput(op,outvn,slot);
344 }

441 int4 Funcdata::fillinExtrapop(void)
442 
443 {
444  if (hasNoCode()) // If no code to make a decision on
445  return funcp.getExtraPop(); // either we already know it or we don't
446 
447  if (funcp.getExtraPop() != ProtoModel::extrapop_unknown)
448  return funcp.getExtraPop(); // If we already know it, just return it
449 
450  list<PcodeOp *>::const_iterator iter = beginOp(CPUI_RETURN);
451  if (iter == endOp(CPUI_RETURN)) return 0; // If no return statements, answer is irrelevant
452 
453  PcodeOp *retop = *iter;
454  uint1 buffer[4];
455 
456  glb->loader->loadFill(buffer,4,retop->getAddr());
457 
458  // We are assuming x86 code here
459  int4 extrapop = 4; // The default case
460  if (buffer[0] == 0xc2) {
461  extrapop = buffer[2]; // Pull out immediate 16-bits
462  extrapop <<= 8;
463  extrapop += buffer[1];
464  extrapop += 4; // extra 4 for the return address
465  }
466  funcp.setExtraPop( extrapop ); // Save what we have learned on the prototype
467  return extrapop;

692 void Funcdata::restoreXml(const Element *el)
693 
694 {
695  // clear(); // Shouldn't be needed
696  name.clear();
697  size = -1;
698  AddrSpace *stackid = glb->getStackSpace();
699  for(int4 i=0;i<el->getNumAttributes();++i) {
700  if (el->getAttributeName(i) == "name")
701  name = el->getAttributeValue(i);
702  else if (el->getAttributeName(i) == "size") {
703  istringstream s( el->getAttributeValue(i));
704  s.unsetf(ios::dec | ios::hex | ios::oct);
705  s >> size;
706  }
707  else if (el->getAttributeName(i) == "nocode") {
708  if (xml_readbool(el->getAttributeValue(i)))
709  flags |= no_code;
710  }
711  }
712  if (name.size() == 0)
713  throw LowlevelError("Missing function name");
714  if (size == -1)
715  throw LowlevelError("Missing function size");
716  const List &list( el->getChildren() );
717  List::const_iterator iter = list.begin();
718  baseaddr = Address::restoreXml( *iter, glb );
719  ++iter;
720  for(;iter!=list.end();++iter) {
721  if ((*iter)->getName() == "localdb") {
722  if (localmap != (ScopeLocal *)0)
723  throw LowlevelError("Pre-existing local scope when restoring: "+name);
724  ScopeLocal *newMap = new ScopeLocal(stackid,this,glb);
725  glb->symboltab->restoreXmlScope(*iter,newMap); // May delete newMap and throw
726  localmap = newMap;
727  }
728  // else if ((*iter)->getName() == "scope") {
729  // const Element *scopeel = *iter;
730  // ScopeInternal *subscope = new ScopeInternal("",glb);
731  // subscope->restrictScope(this);
732  // glb->symboltab->restore_nonglobal_xml(scopeel,subscope);
733  // }
734  else if ((*iter)->getName() == "override")
735  localoverride.restoreXml(*iter,glb);
736  else if ((*iter)->getName() == "prototype") {
737  if (localmap == (ScopeLocal *)0) {
738  // If we haven't seen a <localdb> tag yet, assume we have a default local scope
739  ScopeLocal *newMap = new ScopeLocal(stackid,this,glb);
740  Scope *scope = glb->symboltab->getGlobalScope();
741  glb->symboltab->attachScope(newMap,scope); // May delete newMap and throw
742  localmap = newMap;
743  }
744  funcp.setScope(localmap,baseaddr+ -1); // localmap built earlier
745  funcp.restoreXml(*iter,glb);
746  }
747  else if ((*iter)->getName() == "jumptablelist")
748  restoreXmlJumpTable(*iter);
749  }
750  if (localmap == (ScopeLocal *)0) { // Seen neither <localdb> or <prototype>
751  // This is a function shell, so we provide default locals
752  ScopeLocal *newMap = new ScopeLocal(stackid,this,glb);
753  Scope *scope = glb->symboltab->getGlobalScope();
754  glb->symboltab->attachScope(newMap,scope); // May delete newMap and throw
755  localmap = newMap;
756  funcp.setScope(localmap,baseaddr+ -1);
757  }
758  localmap->resetLocalWindow();
759 }

113 uintb ConstTpl::fix(const ParserWalker &walker) const
114 
115 { // Get the value of the ConstTpl in context
116  // NOTE: if the property is dynamic this returns the property
117  // of the temporary storage
118  switch(type) {
119  case j_start:
120  return walker.getAddr().getOffset(); // Fill in starting address placeholder with real address
121  case j_next:
122  return walker.getNaddr().getOffset(); // Fill in next address placeholder with real address
123  case j_flowref:
124  return walker.getRefAddr().getOffset();
125  case j_flowref_size:
126  return walker.getRefAddr().getAddrSize();
127  case j_flowdest:
128  return walker.getDestAddr().getOffset();
129  case j_flowdest_size:
130  return walker.getDestAddr().getAddrSize();
131  case j_curspace_size:
132  return walker.getCurSpace()->getAddrSize();
133  case j_curspace:
134  return (uintb)(uintp)walker.getCurSpace();
135  case handle:
136  {
137  const FixedHandle &hand(walker.getFixedHandle(value.handle_index));
138  switch(select) {
139  case v_space:
140  if (hand.offset_space == (AddrSpace *)0)
141  return (uintb)(uintp)hand.space;
142  return (uintb)(uintp)hand.temp_space;
143  case v_offset:
144  if (hand.offset_space==(AddrSpace *)0)
145  return hand.offset_offset;
146  return hand.temp_offset;
147  case v_size:
148  return hand.size;
149  case v_offset_plus:
150  if (hand.space != walker.getConstSpace()) { // If we are not a constant
151  if (hand.offset_space==(AddrSpace *)0)
152  return hand.offset_offset + (value_real&0xffff); // Adjust offset by truncation amount
153  return hand.temp_offset + (value_real&0xffff);
154  }
155  else { // If we are a constant, we want to return a shifted value
156  uintb val;
157  if (hand.offset_space==(AddrSpace *)0)
158  val = hand.offset_offset;
159  else
160  val = hand.temp_offset;
161  val >>= 8 * (value_real>>16);
162  return val;
163  }
164  }
165  break;
166  }
167  case j_relative:
168  case real:
169  return value_real;
170  case spaceid:
171  return (uintb)(uintp)value.spaceid;
172  }
173  return 0; // Should never reach here
174 }

254 void ConstTpl::transfer(const vector<HandleTpl *> &params)
255 
256 { // Replace old handles with new handles
257  if (type != handle) return;
258  HandleTpl *newhandle = params[value.handle_index];
259 
260  switch(select) {
261  case v_space:
262  *this = newhandle->getSpace();
263  break;
264  case v_offset:
265  *this = newhandle->getPtrOffset();
266  break;
267  case v_offset_plus:
268  {
269  uintb tmp = value_real;
270  *this = newhandle->getPtrOffset();
271  if (type == real) {
272  value_real += (tmp&0xffff);
273  }
274  else if ((type == handle)&&(select == v_offset)) {
275  select = v_offset_plus;
276  value_real = tmp;
277  }
278  else
279  throw LowlevelError("Cannot truncate macro input in this way");
280  break;
281  }
282  case v_size:
283  *this = newhandle->getSize();
284  break;
285  }

510 bool VarnodeTpl::adjustTruncation(int4 sz,bool isbigendian)
511 
512 { // We know this->offset is an offset_plus, check that the truncation is in bounds (given -sz-)
513  // adjust plus for endianness if necessary
514  // return true if truncation is in bounds
515  if (size.getType() != ConstTpl::real)
516  return false;
517  int4 numbytes = (int4) size.getReal();
518  int4 byteoffset = (int4) offset.getReal();
519  if (numbytes + byteoffset > sz) return false;
520 
521  // Encode the original truncation amount with the plus value
522  uintb val = byteoffset;
523  val <<= 16;
524  if (isbigendian) {
525  val |= (uintb)(sz - (numbytes+byteoffset));
526  }
527  else {
528  val |= (uintb) byteoffset;
529  }
530 
531 
532  offset = ConstTpl(ConstTpl::handle,offset.getHandleIndex(),ConstTpl::v_offset_plus,val);
533  return true;
534 }

584 void HandleTpl::fix(FixedHandle &hand,const ParserWalker &walker) const
585 
586 {
587  if (ptrspace.getType() == ConstTpl::real) {
588  // The export is unstarred, but this doesn't mean the varnode
589  // being exported isn't dynamic
590  space.fillinSpace(hand,walker);
591  hand.size = size.fix(walker);
592  ptroffset.fillinOffset(hand,walker);
593  }
594  else {
595  hand.space = space.fixSpace(walker);
596  hand.size = size.fix(walker);
597  hand.offset_offset = ptroffset.fix(walker);
598  hand.offset_space = ptrspace.fixSpace(walker);
599  if (hand.offset_space->getType()==IPTR_CONSTANT) {
600  // Handle could have been dynamic but wasn't
601  hand.offset_space = (AddrSpace *)0;
602  hand.offset_offset = AddrSpace::addressToByte(hand.offset_offset,hand.space->getWordSize());
603  hand.offset_offset = hand.space->wrapOffset(hand.offset_offset);
604  }
605  else {
606  hand.offset_size = ptrsize.fix(walker);
607  hand.temp_space = temp_space.fixSpace(walker);
608  hand.temp_offset = temp_offset.fix(walker);
609  }
610  }
611 }

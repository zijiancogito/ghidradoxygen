144  int i = 0;

174  Element xmlElement = saveState.getXmlElement("Provider" + i);

262  @Override
263  public void processEvent(PluginEvent event) {
264  if (event instanceof ProgramClosedPluginEvent) {
265  Program program = ((ProgramClosedPluginEvent) event).getProgram();
266  programClosed(program);
267  return;
268  }
269  if (connectedProvider == null) {
270  return;
271  }
272 
273  if (event instanceof ProgramActivatedPluginEvent) {
274  currentProgram = ((ProgramActivatedPluginEvent) event).getActiveProgram();
275  connectedProvider.doSetProgram(currentProgram);
276  }
277  else if (event instanceof ProgramLocationPluginEvent) {
278  ProgramLocation location = ((ProgramLocationPluginEvent) event).getLocation();
279  Address address = location.getAddress();
280  if (address.isExternalAddress()) {
281  return;
282  }
283  if (currentProgram != null) {
284  Listing listing = currentProgram.getListing();
285  CodeUnit codeUnit = listing.getCodeUnitContaining(address);
286  if (codeUnit instanceof Data) {
287  return;
288  }
289  }
290  currentLocation = location;
291  // delay location change to allow immediate location changes to
292  // settle down. This happens when switching program tabs in
293  // code browser which produces multiple location changes
294  delayedLocationUpdateMgr.updateLater();
295  }
296  else if (event instanceof ProgramSelectionPluginEvent) {
297  currentSelection = ((ProgramSelectionPluginEvent) event).getSelection();
298  connectedProvider.setSelection(currentSelection);
299  }
300 
301  }

303  private void programClosed(Program closedProgram) {
304  Iterator<DecompilerProvider> iterator = disconnectedProviders.iterator();
305  while (iterator.hasNext()) {
306  DecompilerProvider provider = iterator.next();
307  if (provider.getProgram() == closedProgram) {
308  iterator.remove();
309  removeProvider(provider);
310  }
311  }
312  if (connectedProvider != null) {
313  connectedProvider.programClosed(closedProgram);
314  }

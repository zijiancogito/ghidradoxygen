86 void InjectPayloadSleigh::checkParameterRestrictions(InjectContextSleigh &con,
87  const vector<InjectParameter> &inputlist,
88  const vector<InjectParameter> &output,
89  const string &source)
90 { // Verify that the storage locations passed in -con- match the restrictions set for this payload
91  if (inputlist.size() != con.inputlist.size())
92  throw LowlevelError("Injection parameter list has different number of parameters than p-code operation: "+source);
93  for(int4 i=0;i<inputlist.size();++i) {
94  uint4 sz = inputlist[i].getSize();
95  if ((sz != 0) && (sz != con.inputlist[i].size))
96  throw LowlevelError("P-code input parameter size does not match injection specification: "+source);
97  }
98  if (output.size() != con.output.size())
99  throw LowlevelError("Injection output does not match output of p-code operation: "+source);
100  for(int4 i=0;i<output.size();++i) {
101  uint4 sz = output[i].getSize();
102  if ((sz != 0) && (sz != con.output[i].size))
103  throw LowlevelError("P-code output size does not match injection specification: "+source);
104  }
105 }

107 void InjectPayloadSleigh::setupParameters(InjectContextSleigh &con,ParserWalkerChange &walker,
108  const vector<InjectParameter> &inputlist,
109  const vector<InjectParameter> &output,
110  const string &source)
111 { // Set-up operands in the parser state so that they pick up storage locations in InjectContext
112  checkParameterRestrictions(con,inputlist,output,source);
113  ParserContext *pos = walker.getParserContext();
114  for(int4 i=0;i<inputlist.size();++i) {
115  pos->allocateOperand(inputlist[i].getIndex(),walker);
116  VarnodeData &data( con.inputlist[i] );
117  FixedHandle &hand(walker.getParentHandle());
118  hand.space = data.space;
119  hand.offset_offset = data.offset;
120  hand.size = data.size;
121  hand.offset_space = (AddrSpace *)0;
122  walker.popOperand();
123  }
124  for(int4 i=0;i<output.size();++i) {
125  pos->allocateOperand(output[i].getIndex(),walker);
126  VarnodeData &data( con.output[i] );
127  FixedHandle &hand(walker.getParentHandle());
128  hand.space = data.space;
129  hand.offset_offset = data.offset;
130  hand.size = data.size;
131  hand.offset_space = (AddrSpace *)0;
132  walker.popOperand();
133  }
134 }

297 void PcodeInjectLibrarySleigh::parseInject(InjectPayload *payload)
298 
299 {
300  if (payload->isDynamic())
301  return;
302  if (slgh == (const SleighBase *)0) { // Make sure we have the sleigh AddrSpaceManager
303  slgh = (const SleighBase *)glb->translate;
304  if (slgh == (const SleighBase *)0)
305  throw LowlevelError("Registering pcode snippet before language is instantiated");
306  }
307  if (contextCache.pos == (ParserContext *)0) { // Make sure we have a context
308  contextCache.pos = new ParserContext((ContextCache *)0);
309  contextCache.pos->initialize(8,8,slgh->getConstantSpace());
310  }
311  PcodeSnippet compiler(slgh);
312 // compiler.clear(); // Not necessary unless we reuse
313  for(int4 i=0;i<payload->sizeInput();++i) {
314  InjectParameter &param( payload->getInput(i) );
315  compiler.addOperand(param.getName(),param.getIndex());
316  }
317  for(int4 i=0;i<payload->sizeOutput();++i) {
318  InjectParameter &param( payload->getOutput(i) );
319  compiler.addOperand(param.getName(),param.getIndex());
320  }
321  if (payload->getType() == InjectPayload::EXECUTABLEPCODE_TYPE) {
322  compiler.setUniqueBase(0x2000); // Don't need to deconflict with anything other injects
323  ExecutablePcodeSleigh *sleighpayload = (ExecutablePcodeSleigh *)payload;
324  istringstream s(sleighpayload->parsestring);
325  if (!compiler.parseStream(s))
326  throw LowlevelError(payload->getSource() + ": Unable to compile pcode: "+compiler.getErrorMessage());
327  sleighpayload->tpl = compiler.releaseResult();
328  sleighpayload->parsestring = ""; // No longer need the memory
329  }
330  else {
331  compiler.setUniqueBase(tempbase);
332  InjectPayloadSleigh *sleighpayload = (InjectPayloadSleigh *)payload;
333  istringstream s(sleighpayload->parsestring);
334  if (!compiler.parseStream(s))
335  throw LowlevelError(payload->getSource() + ": Unable to compile pcode: "+compiler.getErrorMessage());
336  tempbase = compiler.getUniqueBase();
337  sleighpayload->tpl = compiler.releaseResult();
338  sleighpayload->parsestring = ""; // No longer need the memory
339  }
340 }

342 int4 PcodeInjectLibrarySleigh::allocateInject(const string &sourceName,const string &name,int4 type)
343 
344 {
345  int4 injectid = injection.size();
346  if (type == InjectPayload::CALLFIXUP_TYPE)

357 void PcodeInjectLibrarySleigh::registerInject(int4 injectid)
358 
359 {
360  InjectPayload *payload = injection[injectid];
361  if (payload->isDynamic()) {
362  InjectPayload *sub = new InjectPayloadDynamic(glb,payload->getName(),payload->getType());
363  delete payload;
364  payload = sub;
365  injection[injectid] = payload;
366  }
367  switch(payload->getType()) {
368  case InjectPayload::CALLFIXUP_TYPE:
369  registerCallFixup(payload->getName(), injectid);
370  parseInject(payload);
371  break;
372  case InjectPayload::CALLOTHERFIXUP_TYPE:
373  registerCallOtherFixup(payload->getName(), injectid);
374  parseInject(payload);
375  break;
376  case InjectPayload::CALLMECHANISM_TYPE:
377  registerCallMechanism(payload->getName(), injectid);
378  parseInject(payload);
379  break;
380  case InjectPayload::EXECUTABLEPCODE_TYPE:
381  registerExeScript(payload->getName(), injectid);
382  parseInject(payload);
383  break;
384  default:
385  throw LowlevelError("Unknown p-code inject type");
386  }
387 }

89 uint4 ToOpEdge::hash(uint4 reg) const
90 
91 {
92  reg = crc_update(reg,(uint4)slot);
93  reg = crc_update(reg,DynamicHash::transtable[op->code()]);
94  uintb val = op->getSeqNum().getAddr().getOffset();
95  int4 sz = op->getSeqNum().getAddr().getAddrSize();
96  for(int4 i=0;i<sz;++i) {
97  reg = crc_update(reg,(uint4)val); // Hash in the address
98  val >>= 8;
99  }
100  return reg;

122 void DynamicHash::buildVnDown(const Varnode *vn)
123 
124 {
125  list<PcodeOp *>::const_iterator iter;
126  uint4 insize = opedge.size();
127 
128  for(iter=vn->beginDescend();iter!=vn->endDescend();++iter) {
129  const PcodeOp *op = *iter;
130  const Varnode *tmpvn = vn;
131  while(transtable[op->code()]==0) {
132  tmpvn = op->getOut();
133  if (tmpvn == (const Varnode *)0) {
134  op = (const PcodeOp *)0;
135  break;
136  }
137  op = tmpvn->loneDescend();
138  if (op == (const PcodeOp *)0) break;
139  }
140  if (op == (const PcodeOp *)0) continue;
141  int4 slot = op->getSlot(tmpvn);
142  opedge.push_back(ToOpEdge(op,slot));
143  }
144  if ((uint4)opedge.size()-insize > 1)
145  sort(opedge.begin()+insize,opedge.end());
146 }

210 void DynamicHash::calcHash(const Varnode *root,uint4 method)
211 
212 {
213  vnproc = 0;
214  opproc = 0;
215  opedgeproc = 0;
216 
217  vnedge.push_back(root);
218  gatherUnmarkedVn();
219  for(uint4 i=vnproc;i<markvn.size();++i)
220  buildVnUp(markvn[i]);
221  for(;vnproc<markvn.size();++vnproc)
222  buildVnDown(markvn[vnproc]);
223 
224  switch(method) {
225  case 0:
226  break;
227  case 1:
228  gatherUnmarkedOp();
229  for(;opproc<markop.size();++opproc)
230  buildOpUp(markop[opproc]);
231 
232  gatherUnmarkedVn();
233  for(;vnproc<markvn.size();++vnproc)
234  buildVnUp(markvn[vnproc]);
235  break;
236  case 2:
237  gatherUnmarkedOp();
238  for(;opproc<markop.size();++opproc)
239  buildOpDown(markop[opproc]);
240 
241  gatherUnmarkedVn();
242  for(;vnproc<markvn.size();++vnproc)
243  buildVnDown(markvn[vnproc]);
244  break;
245  case 3:
246  gatherUnmarkedOp();
247  for(;opproc<markop.size();++opproc)
248  buildOpUp(markop[opproc]);
249 
250  gatherUnmarkedVn();
251  for(;vnproc<markvn.size();++vnproc)
252  buildVnDown(markvn[vnproc]);
253  break;
254  default:
255  break;
256  }
257 
258  for(uint4 i=0;i<markvn.size();++i) // Clear our marks
259  markvn[i]->clearMark();
260  for(uint4 i=0;i<markop.size();++i)
261  markop[i]->clearMark();
262 
263  if (opedge.size() == 0) {
264  hash = (uint8)0;
265  addrresult = Address();
266  return;
267  }
268 
269  uint4 reg = 0x3ba0fe06; // Calculate the 32-bit hash
270 
271  // Hash in information about the root
272  reg = crc_update(reg,(uint4)root->getSize());
273  if (root->isConstant()) {
274  uintb val = root->getOffset();
275  for(int4 i=0;i<root->getSize();++i) {
276  reg = crc_update(reg,(uint4)val);
277  val >>= 8;
278  }
279  }
280 
281  for(uint4 i=0;i<opedge.size();++i)
282  reg = opedge[i].hash(reg);
283 
284  // Build the final 64-bit hash
285  const PcodeOp *op = (const PcodeOp *)0;
286  int4 slot = 0;
287  uint4 ct;
288  bool attachedop = true;
289  for(ct=0;ct<opedge.size();++ct) { // Find op that is directly attached to -root- i.e. not a skip op
290  op = opedge[ct].getOp();
291  slot = opedge[ct].getSlot();
292  if ((slot < 0) && (op->getOut() == root)) break;
293  if ((slot >=0) && (op->getIn(slot)==root)) break;
294  }
295  if (ct == opedge.size()) { // If everything attached to the root was a skip op
296  op = opedge[0].getOp(); // Return op that is not attached directly
297  slot = opedge[0].getSlot();
298  attachedop = false;
299  }
300 
301  // 15 bits unused
302  hash = attachedop ? 0 : 1;
303  hash <<= 4;
304  hash |= method; // 4-bits
305  hash <<= 7;
306  hash |= (uint8)op->code(); // 7-bits
307  hash <<= 5;
308  hash |= (uint8)(slot & 0x1f); // 5-bits
309 
310  hash <<= 32;
311  hash |= (uint8)reg; // 32-bits for the neighborhood hash
312  addrresult = op->getSeqNum().getAddr();
313 }

330 void DynamicHash::uniqueHash(const Varnode *root,Funcdata *fd)
331 
332 {
333  vector<Varnode *> vnlist;
334  vector<Varnode *> vnlist2;
335  vector<Varnode *> champion;
336  uint4 method;
337  uint8 tmphash;
338  Address tmpaddr;
339  uint4 maxduplicates = 8;
340 
341  for(method=0;method<4;++method) {
342  clear();
343  calcHash(root,method);
344  if (hash == 0) return; // Can't get a good hash
345  tmphash = hash;
346  tmpaddr = addrresult;
347  vnlist.clear();
348  vnlist2.clear();
349  gatherFirstLevelVars(vnlist,fd,tmpaddr,tmphash);
350  for(uint4 i=0;i<vnlist.size();++i) {
351  Varnode *tmpvn = vnlist[i];
352  clear();
353  calcHash(tmpvn,method);
354  if (hash == tmphash) { // Hash collision
355  vnlist2.push_back(tmpvn);
356  if (vnlist2.size()>maxduplicates) break;
357  }
358  }
359  if (vnlist2.size() <= maxduplicates) {
360  if ((champion.size()==0)||(vnlist2.size() < champion.size())) {
361  champion = vnlist2;
362  if (champion.size()==1) break; // Current hash is unique
363  }
364  }
365  }
366  if (champion.empty()) {
367  hash = (uint8)0;
368  addrresult = Address(); // Couldn't find a unique hash
369  return;
370  }
371  uint4 total = (uint4)champion.size() - 1; // total is in range [0,maxduplicates-1]
372  uint4 pos;
373  for(pos=0;pos<=total;++pos)
374  if (champion[pos] == root) break;
375  if (pos > total) {
376  hash = (uint8)0;
377  addrresult = Address();
378  return;
379  }
380  hash = tmphash | ((uint8)pos << 49); // Store three bits for position with list of duplicate hashes
381  hash |= ((uint8)total << 52); // Store three bits for total number of duplicate hashes
382  addrresult = tmpaddr;
383 }

426 void DynamicHash::gatherFirstLevelVars(vector<Varnode *> &varlist,Funcdata *fd,const Address &addr,uint8 h)
427 
428 {
429  OpCode opc = getOpCodeFromHash(h);
430  int4 slot = getSlotFromHash(h);
431  bool isnotattached = getIsNotAttached(h);
432  PcodeOpTree::const_iterator iter = fd->beginOp(addr);
433  PcodeOpTree::const_iterator enditer = fd->endOp(addr);
434 
435  while(iter!=enditer) {
436  PcodeOp *op = (*iter).second;
437  ++iter;
438  if (op->code() != opc) continue;
439  if (slot <0) {
440  Varnode *vn = op->getOut();
441  if (vn != (Varnode *)0) {
442  if (isnotattached) { // If original varnode was not attached to (this) op
443  op = vn->loneDescend();
444  if (op != (PcodeOp *)0) {
445  if (transtable[op->code()] == 0) { // Check for skipped op
446  vn = op->getOut();
447  if (vn == (Varnode *)0) continue;
448  }
449  }
450  }
451  varlist.push_back(vn);
452  }
453  }
454  else if (slot < op->numInput()) {
455  Varnode *vn = op->getIn(slot);
456  if (isnotattached) {
457  op = vn->getDef();
458  if ((op != (PcodeOp *)0)&&(transtable[op->code()]==0))
459  vn = op->getIn(0);
460  }
461  varlist.push_back(vn);
462  }
463  }
464 }

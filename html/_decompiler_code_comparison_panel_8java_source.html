552  private void refreshPanel(int leftOrRight) {
553  // Hold onto functions for reloading them after the indicated side is cleared,
554  // because that will have cleared it in the functions array.
555  Function leftFunction = functions[LEFT];
556  Function rightFunction = functions[RIGHT];
557 
558  // Save the location so it can be restored after getting new decompiler results.
559  FieldLocation leftCursorLocation =
560  getLeftDecompilerPanel().getFieldPanel().getCursorLocation();
561  FieldLocation rightCursorLocation =
562  getRightDecompilerPanel().getFieldPanel().getCursorLocation();
563 
564  MyDecompileResultsListener listener =
565  new MyDecompileResultsListener(leftCursorLocation, rightCursorLocation);
566 
567  // Clear any previous listener that is for a decompiler load that hasn't finished.
568  // This can simply clear since it is the only one that is adding them to the list.
569  dualDecompileResultsListenerList.clear();
570 
571  // Clear the left or right function by passing null to the load method
572  // and then reload it below to get it to update.
573  loadFunctions(((leftOrRight == LEFT) ? null : leftFunction),
574  ((leftOrRight == RIGHT) ? null : rightFunction));
575 
576  // Setup to restore location to left or right decompiler panel.
577  addDualDecompileResultsListener(listener);
578 
579  // Reload the left or right function to get it to update.
580  loadFunctions(leftFunction, rightFunction);
581  }

633  public void decompileResultsSet(final DecompileData myLeftDecompileData,
634  final DecompileData myRightDecompileData) {
635  SwingUtilities.invokeLater(() -> {
636  if (myLeftDecompileData != null) {
637  // The left side may have reloaded with decompiler results,
638  // so restore the cursor location.
639  restoreCursor(getLeftDecompilerPanel(), leftCursorLocation);
640  }
641  if (myRightDecompileData != null) {
642  // The right side may have reloaded with decompiler results,
643  // so restore the cursor location.
644  restoreCursor(getRightDecompilerPanel(), rightCursorLocation);
645  }
646 
647  // The listener did its job so now remove it.
648  removeDualDecompileResultsListener(MyDecompileResultsListener.this);
649  });

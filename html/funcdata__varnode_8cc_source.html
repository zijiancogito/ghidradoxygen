327 
328 {
329  Varnode *invn;
330 
331  if (vn->isInput()) return vn; // Already an input
332  // First we check if it overlaps any other varnode
333  VarnodeDefSet::const_iterator iter;
334  iter = vbank.beginDef(Varnode::input,vn->getAddr()+vn->getSize());
335

369 void Funcdata::adjustInputVarnodes(const Address &addr,int4 size)
370 
371 {
372  Address endaddr = addr + (size-1);
373  vector<Varnode *> inlist;
374  VarnodeDefSet::const_iterator iter,enditer;
375  iter = vbank.beginDef(Varnode::input,addr);
376  enditer = vbank.endDef(Varnode::input,endaddr);
377  while(iter != enditer) {
378  Varnode *vn = *iter;
379  ++iter;
380  if (vn->getOffset() + (vn->getSize()-1) > endaddr.getOffset())
381  throw LowlevelError("Cannot properly adjust input varnodes");
382  inlist.push_back(vn);
383  }
384 
385  for(uint4 i=0;i<inlist.size();++i) {
386  Varnode *vn = inlist[i];
387  int4 sa = addr.justifiedContain(size,vn->getAddr(),vn->getSize(),false);
388  if ((!vn->isInput())||(sa < 0)||(size<=vn->getSize()))
389  throw LowlevelError("Bad adjustment to input varnode");
390  PcodeOp *subop = newOp(2,getAddress());
391  opSetOpcode(subop,CPUI_SUBPIECE);
392  opSetInput(subop,newConstant(4,sa),1);
393  Varnode *newvn = newVarnodeOut(vn->getSize(),vn->getAddr(),subop);
394  // newvn must not be free in order to give all vn's descendants
395  opInsertBegin(subop,(BlockBasic *)bblocks.getBlock(0));
396  totalReplace(vn,newvn);
397  deleteVarnode(vn); // Get rid of old input before creating new input
398  inlist[i] = newvn;
399  }
400  // Now that all the intersecting inputs have been pulled out, we can create the new input
401  Varnode *invn = newVarnode(size,addr);
402  invn = setInputVarnode(invn);
403  // The new input may cause new heritage and "Heritage AFTER dead removal" errors
404  // So tell heritage to ignore it
405  // FIXME: It would probably be better to insert this directly into heritage's globaldisjoint
406  invn->setWriteMask();
407  // Now change all old inputs to be created as SUBPIECE from the new input
408  for(uint4 i=0;i<inlist.size();++i) {
409  PcodeOp *op = inlist[i]->getDef();
410  opSetInput(op,invn,0);
411  }
412 }

420 {

423  Varnode *badconst;

516 bool Funcdata::fillinReadOnly(Varnode *vn)
517 
518 {
519  if (vn->isWritten()) { // Can't replace output with constant
520  PcodeOp *defop = vn->getDef();
521  if (defop->isMarker())
522  defop->setFlag(PcodeOp::warning); // Not a true write, ignore it
523  else if (!defop->isWarning()) { // No warning generated before
524  defop->setFlag(PcodeOp::warning);
525  ostringstream s;
526  if ((!vn->isAddrForce())||(!vn->hasNoDescend())) {
527  s << "Read-only address (";
528  s << vn->getSpace()->getName();
529  s << ',';
530  vn->getAddr().printRaw(s);
531  s << ") is written";
532  warning(s.str(),defop->getAddr());
533  }
534  }
535  return false; // No change was made
536  }
537 
538  uintb res;
539  uint1 bytes[32];
540  try {
541  glb->loader->loadFill(bytes,vn->getSize(),vn->getAddr());
542  } catch(DataUnavailError &err) { // Could not get value from LoadImage
543  vn->clearFlags(Varnode::readonly); // Treat as writeable
544  return true;
545  }
546 
547  if (vn->getSpace()->isBigEndian()) { // Big endian
548  res = 0;
549  for(int4 i=0;i<vn->getSize();++i) {
550  res <<= 8;
551  res |= bytes[i];
552  }
553  }
554  else {
555  res = 0;
556  for(int4 i=vn->getSize()-1;i>=0;--i) {
557  res <<= 8;
558  res |= bytes[i];
559  }
560  }
561  // Replace all references to vn
562  bool changemade = false;
563  list<PcodeOp *>::const_iterator iter;
564  PcodeOp *op;
565  int4 i;
566  Datatype *locktype = vn->isTypeLock() ? vn->getType() : (Datatype *)0;
567 
568  iter = vn->beginDescend();
569  while(iter != vn->endDescend()) {
570  op = *iter++;
571  i = op->getSlot(vn);
572  if (op->isMarker()) { // Must be careful putting constants in here
573  if ((op->code()!=CPUI_INDIRECT)||(i!=0)) continue;
574  Varnode *outvn = op->getOut();
575  if (outvn->getAddr() == vn->getAddr()) continue; // Ignore indirect to itself
576  // Change the indirect to a COPY
577  opRemoveInput(op,1);
578  opSetOpcode(op,CPUI_COPY);
579  }
580  Varnode *cvn = newConstant(vn->getSize(),res);
581  if (locktype != (Datatype *)0)
582  cvn->updateType(locktype,true,true); // Try to pass on the locked datatype
583  opSetInput(op,cvn,i);
584  changemade = true;
585  }
586  return changemade;
587 }

595 bool Funcdata::replaceVolatile(Varnode *vn)
596 
597 {
598  PcodeOp *newop;
599  if (vn->isWritten()) { // A written value
600  VolatileWriteOp *vw_op = glb->userops.getVolatileWrite();
601  if (!vn->hasNoDescend()) throw LowlevelError("Volatile memory was propagated");
602  PcodeOp *defop = vn->getDef();
603  newop = newOp(3,defop->getAddr());
604  opSetOpcode(newop,CPUI_CALLOTHER);
605  // Create a userop of type specified by vw_op
606  opSetInput(newop,newConstant(4,vw_op->getIndex()),0);
607  // The first parameter is the offset of volatile memory location
608  opSetInput(newop,newCodeRef(vn->getAddr()),1);
609  // Replace the volatile variable with a temp
610  Varnode *tmp = newUnique(vn->getSize());
611  opSetOutput(defop,tmp);
612  // The temp is the second parameter to the userop
613  opSetInput(newop,tmp,2);
614  opInsertAfter(newop,defop); // Insert after defining op
615  }
616  else { // A read value
617  VolatileReadOp *vr_op = glb->userops.getVolatileRead();
618  if (vn->hasNoDescend()) return false; // Dead
619  PcodeOp *readop = vn->loneDescend();
620  if (readop == (PcodeOp *)0)
621  throw LowlevelError("Volatile memory value used more than once");
622  newop = newOp(2,readop->getAddr());
623  opSetOpcode(newop,CPUI_CALLOTHER);
624  // Create a temp to replace the volatile variable
625  Varnode *tmp = newUniqueOut(vn->getSize(),newop);
626  // Create a userop of type specified by vr_op
627  opSetInput(newop,newConstant(4,vr_op->getIndex()),0);
628  // The first parameter is the offset of the volatile memory loc
629  opSetInput(newop,newCodeRef(vn->getAddr()),1);
630  opSetInput(readop,tmp,readop->getSlot(vn));
631  opInsertBefore(newop,readop); // Insert before read
632  }
633  if (vn->isTypeLock()) // If the original varnode had a type locked on it
634  newop->setAdditionalFlag(PcodeOp::special_prop); // Mark this op as doing special propagation
635  return true;

722 void Funcdata::calcNZMask(void)
723 
724 {
725  vector<PcodeOp *> opstack;
726  vector<int4> slotstack;
727  list<PcodeOp *>::const_iterator oiter;
728 
729  for(oiter=beginOpAlive();oiter!=endOpAlive();++oiter) {
730  PcodeOp *op = *oiter;
731  if (op->isMark()) continue;
732  opstack.push_back(op);
733  slotstack.push_back(0);
734  op->setMark();
735 
736  do {
737  // Get next edge
738  op = opstack.back();
739  int4 slot = slotstack.back();
740  if (slot >= op->numInput()) { // If no edge left
741  Varnode *outvn = op->getOut();
742  if (outvn != (Varnode *)0) {
743  outvn->nzm = op->getNZMaskLocal(true);
744  }
745  opstack.pop_back(); // Pop a level
746  slotstack.pop_back();
747  continue;
748  }
749  slotstack.back() = slot + 1; // Advance to next input
750  // Determine if we want to traverse this edge
751  if (op->code() == CPUI_MULTIEQUAL) {
752  if (op->getParent()->isLoopIn(slot)) // Clip looping edges
753  continue;
754  }
755  // Traverse edge indicated by slot
756  Varnode *vn = op->getIn(slot);
757  if (!vn->isWritten()) {
758  if (vn->isConstant())
759  vn->nzm = vn->getOffset();
760  else {
761  vn->nzm = calc_mask(vn->getSize());
762  if (vn->isSpacebase())
763  vn->nzm &= ~((uintb)0xff); // Treat spacebase input as aligned
764  }
765  }
766  else if (!vn->getDef()->isMark()) { // If haven't traversed before
767  opstack.push_back(vn->getDef());

1467 int4 AncestorRealistic::uponPop(State &state,int4 pop_command)
1468 
1469 {
1470  if (state.op->code() == CPUI_MULTIEQUAL) { // All the interesting action happens for MULTIEQUAL branch points
1471  State &prevstate( stateStack[ stateStack.size()-2 ]); // State previous the one being popped
1472  if (pop_command == pop_fail) { // For a pop_fail, we always pop and pass along the fail
1473  multiDepth -= 1;
1474  stateStack.pop_back();
1475  return pop_command;
1476  }
1477  else if ((pop_command == pop_solid)&&(multiDepth == 1)&&(state.op->numInput()==2))
1478  prevstate.markSolid(state.slot); // Indicate we have seen a "solid" that could override a "failkill"
1479  else if (pop_command == pop_failkill)
1480  prevstate.markKill(); // Indicate we have seen a "failkill" along at least one path of MULTIEQUAL
1481  state.slot += 1; // Move to the next sibling
1482  if (state.slot == state.op->numInput()) { // If we have traversed all siblings
1483  if (prevstate.seenSolid()) { // If we have seen an overriding "solid" along at least one path
1484  pop_command = pop_success; // this is always a success
1485  if (prevstate.seenKill()) { // UNLESS we have seen a failkill
1486  if (allowFailingPath) {
1487  if (!checkConditionalExe(state)) // that can NOT be attributed to conditional execution
1488  pop_command = pop_fail; // in which case we fail despite having solid movement
1489  else
1490  trial->setCondExeEffect(); // Slate this trial for additional testing
1491  }
1492  else
1493  pop_command = pop_fail;
1494  }
1495  }
1496  else if (prevstate.seenKill()) // If we have seen a failkill without solid movement
1497  pop_command = pop_failkill; // this is always a failure
1498  else
1499  pop_command = pop_success; // seeing neither solid nor failkill is still a success
1500  multiDepth -= 1;
1501  stateStack.pop_back();
1502  return pop_command;
1503  }
1504  state.vn = state.op->getIn(state.slot); // Advance to next sibling
1505  return enter_node;
1506  }
1507  else {
1508  stateStack.pop_back();
1509  return pop_command;
1510  }

1521

1524  allowFailingPath = allowFail;
1525  markedVn.clear(); // Make sure to clear out any old data
1526  stateStack.clear();

1528  // If the parameter itself is an input, we don't consider this realistic, we expect to see active

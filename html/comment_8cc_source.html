268 bool CommentSorter::findPosition(Subsort &subsort,Comment *comm,const Funcdata *fd)
269 
270 {
271  if (comm->getType() == 0) return false;
272  const Address &fad( fd->getAddress() );
273  if (((comm->getType() & (Comment::header | Comment::warningheader))!=0)&&(comm->getAddr() == fad)) {
274  // If it is a header comment at the address associated with the beginning of the function
275  subsort.setHeader(header_basic);
276  return true;
277  }
278 
279  // Try to find block containing comment
280  // Find op at lowest address greater or equal to comment's address
281  PcodeOpTree::const_iterator opiter = fd->beginOp(comm->getAddr());
282  PcodeOp *backupOp = (PcodeOp *)0;
283  if (opiter != fd->endOpAll()) { // If there is an op at or after the comment
284  PcodeOp *op = (*opiter).second;
285  BlockBasic *block = op->getParent();
286  if (block->contains(comm->getAddr())) { // If the op's block contains the address
287  // Associate comment with this op
288  subsort.setBlock(block->getIndex(), (uint4)op->getSeqNum().getOrder());
289  return true;
290  }
291  if (comm->getAddr() == op->getAddr())
292  backupOp = op;
293  }
294  if (opiter != fd->beginOpAll()) { // If there is a previous op
295  --opiter;
296  PcodeOp *op = (*opiter).second;
297  BlockBasic *block = op->getParent();
298  if (block->contains(comm->getAddr())) { // If the op's block contains the address
299  // Treat the comment as being in this block at the very end
300  subsort.setBlock(block->getIndex(),0xffffffff);
301  return true;
302  }
303  }
304  if (backupOp != (PcodeOp *)0) {
305  // Its possible the op migrated from its original basic block.
306  // Since the address matches exactly, hang the comment on it
307  subsort.setBlock(backupOp->getParent()->getIndex(),(uint4)backupOp->getSeqNum().getOrder());
308  return true;
309  }
310  if (fd->beginOpAll() == fd->endOpAll()) { // If there are no ops at all
311  subsort.setBlock(0,0); // Put comment at the beginning of the first block
312  return true;
313  }
314  if (displayUnplacedComments) {
315  subsort.setHeader(header_unplaced);
316  return true;
317  }
318  return false; // Basic block containing comment has been excised
319 }

328 void CommentSorter::setupFunctionList(uint4 tp,const Funcdata *fd,const CommentDatabase &db,bool displayUnplaced)
329 {
330  commmap.clear();
331  displayUnplacedComments = displayUnplaced;
332  if (tp == 0) return;
333  const Address &fad( fd->getAddress() );
334  CommentSet::const_iterator iter = db.beginComment(fad);
335  CommentSet::const_iterator lastiter = db.endComment(fad);
336  Subsort subsort;
337 
338  subsort.pos = 0;
339 
340  while(iter != lastiter) {
341  Comment *comm = *iter;
342  if (findPosition(subsort, comm, fd)) {
343  commmap[ subsort ] = comm;
344  subsort.pos += 1; // Advance the uniqueness counter
345  }
346  ++iter;
347  }

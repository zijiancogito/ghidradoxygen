62 uintm ParserContext::getInstructionBits(int4 startbit,int4 size,uint4 off) const
63 
64 {
65  off += (startbit/8);
66  if (off >= 16)
67  throw BadDataError("Instruction is using more than 16 bytes");
68  const uint1 *ptr = buf + off;
69  startbit = startbit % 8;
70  int4 bytesize = (startbit+size-1)/8 + 1;
71  uintm res = 0;
72  for(int4 i=0;i<bytesize;++i) {
73  res <<= 8;
74  res |= ptr[i];
75  }
76  res <<= 8*(sizeof(uintm)-bytesize)+startbit; // Move starting bit to highest position
77  res >>= 8*sizeof(uintm)-size; // Shift to bottom of intm
78  return res;

81 uintm ParserContext::getContextBytes(int4 bytestart,int4 size) const
82 
83 { // Get bytes from context into a uintm
84  int4 intstart = bytestart / sizeof(uintm);
85  uintm res = context[ intstart ];
86  int4 byteOffset = bytestart % sizeof(uintm);
87  int4 unusedBytes = sizeof(uintm) - size;
88  res <<= byteOffset*8;
89  res >>= unusedBytes*8;
90  int4 remaining = size - sizeof(uintm) + byteOffset;
91  if ((remaining > 0)&&(++intstart < contextsize)) { // If we extend beyond boundary of a single uintm
92  uintm res2 = context[ intstart ];
93  unusedBytes = sizeof(uintm) - remaining;
94  res2 >>= unusedBytes * 8;
95  res |= res2;
96  }
97  return res;
98 }

134 
135 {
136  if (contextcommit.empty()) return;
137  ParserWalker walker(this);
138  walker.baseState();
139 
140  vector<ContextSet>::iterator iter;
141 
142  for(iter=contextcommit.begin();iter!=contextcommit.end();++iter) {
143  TripleSymbol *sym = (*iter).sym;
144  Address commitaddr;
145  if (sym->getType() == SleighSymbol::operand_symbol) {
146  // The value for an OperandSymbol is probabably already
147  // calculated, we just need to find the right
148  // tree node of the state
149  int4 i = ((OperandSymbol *)sym)->getIndex();
150  FixedHandle &h((*iter).point->resolve[i]->hand);
151  commitaddr = Address(h.space,h.offset_offset);
152  }
153  else {
154  FixedHandle hand;
155  sym->getFixedHandle(hand,walker);
156  commitaddr = Address(hand.space,hand.offset_offset);
157  }
158  if (commitaddr.isConstant()) {
159  // If the symbol handed to globalset was a computed value, the getFixedHandle calculation
160  // will return a value in the constant space. If this is a case, we explicitly convert the
161  // offset into the current address space
162  uintb newoff = AddrSpace::addressToByte(commitaddr.getOffset(),addr.getSpace()->getWordSize());
163  commitaddr = Address(addr.getSpace(),newoff);
164  }
165 
166  // Commit context change
167  if ((*iter).flow) // The context flows
168  contcache->setContext(commitaddr,(*iter).num,(*iter).mask,(*iter).value);
169  else { // Set the context so that is doesn't flow
170  Address nextaddr = commitaddr + 1;
171  if (nextaddr.getOffset() < commitaddr.getOffset())
172  contcache->setContext(commitaddr,(*iter).num,(*iter).mask,(*iter).value);
173  else
174  contcache->setContext(commitaddr,nextaddr,(*iter).num,(*iter).mask,(*iter).value);
175  }
176  }

179 void ParserWalker::setOutOfBandState(Constructor *ct,int4 index,ConstructState *tempstate,const ParserWalker &otherwalker)
180 
181 { // Initialize walker for future calls into getInstructionBytes assuming -ct- is the current position in the walk
182  const ConstructState *pt = otherwalker.point;
183  int4 curdepth = otherwalker.depth;
184  while(pt->ct != ct) {
185  if (curdepth <= 0) return;
186  curdepth -= 1;
187  pt = pt->parent;
188  }
189  OperandSymbol *sym = ct->getOperand(index);
190  int4 i = sym->getOffsetBase();
191  // if i<0, i.e. the offset of the operand is constructor relative
192  // its possible that the branch corresponding to the operand
193  // has not been constructed yet. Context expressions are
194  // evaluated BEFORE the constructors branches are created.
195  // So we have to construct the offset explicitly.
196  if (i<0)
197  tempstate->offset = pt->offset + sym->getRelativeOffset();
198  else
199  tempstate->offset = pt->resolve[index]->offset;

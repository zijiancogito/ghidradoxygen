25 uintm CoverBlock::getUIndex(const PcodeOp *op)
26 
27 {
28  uintp switchval = (uintp)op;
29  switch(switchval) {
30  case 0: // Special marker for very beginning of block
31  return (uintm)0;
32  case 1: // Special marker for very end of block
33  return ~((uintm)0);
34  case 2: // Special marker for input
35  return (uintm)0;
36  }
37  if (op->isMarker()) {
38  if (op->code() == CPUI_MULTIEQUAL) // MULTIEQUALs are considered very beginning

43  }

55 int4 CoverBlock::intersect(const CoverBlock &op2) const
56 
57 {
58  uintm ustart,ustop;
59  uintm u2start,u2stop;
60 
61  if (empty()) return 0;
62  if (op2.empty()) return 0;
63 
64  ustart = getUIndex(start);
65  ustop = getUIndex(stop);
66  u2start = getUIndex(op2.start);
67  u2stop = getUIndex(op2.stop);
68  if (ustart <= ustop) {
69  if (u2start <= u2stop) { // We are both one piece
70  if ((ustop<=u2start)||(u2stop<=ustart)) {
71  if ((ustart==u2stop)||(ustop==u2start))
72  return 1; // Boundary intersection
73  else
74  return 0; // No intersection
75  }
76  }
77  else { // They are two-piece, we are one-piece
78  if ((ustart>=u2stop)&&(ustop<=u2start)) {
79  if ((ustart==u2stop)||(ustop==u2start))
80  return 1;
81  else
82  return 0;
83  }
84  }
85  }
86  else {
87  if (u2start <= u2stop) { // They are one piece, we are two-piece
88  if ((u2start>=ustop)&&(u2stop<=ustart)) {
89  if ((u2start==ustop)||(u2stop==ustart))
90  return 1;
91  else
92  return 0;
93  }
94  }
95  // If both are two-pieces, then the intersection must be an interval
96  }
97  return 2; // Interval intersection
98 }

143 void CoverBlock::merge(const CoverBlock &op2)
144 
145 {
146  bool internal1,internal2,internal3,internal4;
147  uintm ustart,u2start;
148 
149  if (op2.empty()) return; // Nothing to merge in
150  if (empty()) {
151  start = op2.start;
152  stop = op2.stop;
153  return;
154  }
155  ustart = getUIndex(start);
156  u2start = getUIndex(op2.start);
157  // Is start contained in op2
158  internal4 = ((ustart==(uintm)0)&&(op2.stop==(const PcodeOp *)1));
159  internal1 = internal4 || op2.contain(start);
160  // Is op2.start contained in this
161  internal3 = ((u2start==0)&&(stop==(const PcodeOp *)1));
162  internal2 = internal3 || contain(op2.start);
163 
164  if (internal1&&internal2)
165  if ((ustart!=u2start)|| internal3 || internal4) { // Covered entire block
166  setAll();
167  return;
168  }
169  if (internal1)
170  start = op2.start; // Pick non-internal start
171  else if ((!internal1)&&(!internal2)) { // Disjoint intervals
172  if (ustart < u2start) // Pick earliest start
173  stop = op2.stop; // then take other stop
174  else
175  start = op2.start;
176  return;
177  }
178  if (internal3 || op2.contain(stop)) // Pick non-internal stop
179  stop = op2.stop;
180 }

266 
267 {

269  int4 res,newres;
270 
271  res = 0;
272  iter = cover.begin();
273  iter2 = op2.cover.begin();
274 
275  for(;;) {
276  if (iter == cover.end()) return res;
277  if (iter2 == op2.cover.end()) return res;
278 
279  if ((*iter).first < (*iter2).first)
280  ++iter;
281  else if ((*iter).first > (*iter2).first)
282  ++iter2;

284  newres = (*iter).second.intersect((*iter2).second);
285  if (newres == 2) return 2;
286  if (newres == 1)
287  res = 1; // At least a point intersection
288  ++iter;

291  }

507  uintm ustop;
508 
509  bl = ref->getParent();
510  CoverBlock &block(cover[bl->getIndex()]);
511  if (block.empty()) {
512  block.setEnd(ref);
513  }
514  else {
515  if (block.contain(ref)) {

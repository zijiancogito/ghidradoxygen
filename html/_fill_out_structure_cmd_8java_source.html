194  private void pushIntoCalls(Structure structDT) {
195  AddressSet doneSet = new AddressSet();
196  DataType pointerDT = new PointerDataType(structDT);
197 
198  while (addressToCallInputMap.size() > 0) {
199  currentCallDepth += 1;
200  if (currentCallDepth > maxCallDepth) {
201  return;
202  }
203  HashMap<Address, Integer> savedList = addressToCallInputMap;
204  addressToCallInputMap = new HashMap<>();
205  Set<Address> keys = savedList.keySet();
206  Iterator<Address> keyIter = keys.iterator();
207  while (keyIter.hasNext()) {
208  Address addr = keyIter.next();
209 
210  if (doneSet.contains(addr)) {
211  continue;
212  }
213  doneSet.addRange(addr, addr);
214  Function func = currentProgram.getFunctionManager().getFunctionAt(addr);
215  int paramIndex = savedList.get(addr);
216  Address storageAddr = computeParamAddress(func, paramIndex, pointerDT);
217  HighVariable paramHighVar = computeHighVariable(storageAddr, func);
218  if (paramHighVar != null) {
219  fillOutStructureDef(paramHighVar);
220  populateStructure(structDT);
221  }
222  }
223  }

282  HighVariable highVar = null;

291  HighFunction highFunc = results.getHighFunction();
292 
293  // no decompile...
294  if (highFunc == null) {
295  return null;
296  }
297 
298  // try to map the variable
299  HighSymbol sym =
300  highFunc.getMappedSymbol(storageAddress, function.getEntryPoint().subtractWrap(1L));
301  if (sym == null) {
302  sym = highFunc.getMappedSymbol(storageAddress, null);
303  }
304  if (sym == null) {
305  sym = highFunc.getMappedSymbol(storageAddress, function.getEntryPoint());
306  }
307  if (sym == null) {
308  sym = highFunc.getLocalSymbolMap().findLocal(storageAddress,
309  function.getEntryPoint().subtractWrap(1L));
310  }
311  if (sym == null) {
312  sym = highFunc.getLocalSymbolMap().findLocal(storageAddress, null);
313  }
314  if (sym == null) {
315  sym = highFunc.getLocalSymbolMap().findLocal(storageAddress,
316  function.getEntryPoint());
317  }
318  if (sym == null) {
319  return null;
320  }
321 
322  highVar = sym.getHighVariable();
323  }
324  finally {

530  private void fillOutStructureDef(HighVariable var) {
531  Varnode startVN = var.getRepresentative();
532  ArrayList<PointerRef> todoList = new ArrayList<PointerRef>();
533  HashSet<Varnode> doneList = new HashSet<>();
534 
535  todoList.add(new PointerRef(startVN, 0)); // Base Varnode on the todo list
536 
537  // while Todo list not empty
538  while (!todoList.isEmpty()) {
539  PointerRef currentRef = todoList.remove(0);
540  if (currentRef.varnode == null) {
541  continue;
542  }
543 
544  Varnode[] instances = currentRef.varnode.getHigh().getInstances();
545  // println("");
546  for (Varnode iVn : instances) {
547  Iterator<PcodeOp> descendants = iVn.getDescendants();
548  while (descendants.hasNext()) {
549  PcodeOp pcodeOp = descendants.next();
550  Varnode output = pcodeOp.getOutput();
551  Varnode[] inputs = pcodeOp.getInputs();
552  // println("off=" + offset + " " + pcodeOp.getSeqnum().getTarget().toString() + " : "
553  // + pcodeOp.toString());
554 
555  DataType outDt;
556  long newOff;
557  switch (pcodeOp.getOpcode()) {
558  case PcodeOp.INT_SUB:
559  case PcodeOp.INT_ADD:
560  if (!inputs[1].isConstant()) {
561  break;
562  }
563  long value = getSigned(inputs[1]);
564  newOff = currentRef.offset +
565  ((pcodeOp.getOpcode() == PcodeOp.INT_ADD) ? value : (-value));
566  if (sanityCheck(newOff)) { // should this offset create a location in the structure?
567  putOnList(output, newOff, todoList, doneList);
568  maxOffset = computeMax(maxOffset, newOff, 0);
569  }
570  break;
571  case PcodeOp.PTRADD:
572  if (!inputs[1].isConstant() || !inputs[2].isConstant()) {
573  break;
574  }
575  newOff =
576  currentRef.offset + getSigned(inputs[1]) * inputs[2].getOffset();
577  if (sanityCheck(newOff)) { // should this offset create a location in the structure?
578  putOnList(output, newOff, todoList, doneList);
579  maxOffset = computeMax(maxOffset, newOff, 0);
580  }
581  break;
582  case PcodeOp.PTRSUB:
583  if (!inputs[1].isConstant()) {
584  break;
585  }
586  long subOff = currentRef.offset + getSigned(inputs[1]);
587  if (sanityCheck(subOff)) { // should this offset create a location in the structure?
588  putOnList(output, subOff, todoList, doneList);
589  maxOffset = computeMax(maxOffset, subOff, 0);
590  }
591  break;
592  case PcodeOp.SEGMENTOP:
593  // treat segment op as if it were a cast to complete the value
594  // The segment adds in some unknown base value.
595  // get output and add to the Varnode Todo list
596  putOnList(output, currentRef.offset, todoList, doneList);
597  break;
598 
599  case PcodeOp.LOAD:
600  outDt = output.getHigh().getDataType();
601  if (outDt != null) {
602  offsetToDataTypeMap.put(Long.valueOf(currentRef.offset), outDt);
603  }
604  maxOffset = computeMax(maxOffset, currentRef.offset, output.getSize());
605  break;
606  case PcodeOp.STORE:
607  // create a location in the struct
608  //use the type of the varnode being put in to the structure
609  if (pcodeOp.getSlot(iVn) != 1) {
610  break; // store must be into the target structure
611  }
612  outDt = inputs[2].getHigh().getDataType();
613  int outLen = 1; // Storing at least one byte
614  if (outDt != null) {
615  offsetToDataTypeMap.put(Long.valueOf(currentRef.offset), outDt);
616  outLen = outDt.getLength();
617  }
618 
619  maxOffset = computeMax(maxOffset, currentRef.offset, outLen);
620  break;
621  case PcodeOp.CAST:
622  putOnList(output, currentRef.offset, todoList, doneList);
623  break;
624  case PcodeOp.MULTIEQUAL:
625  putOnList(output, currentRef.offset, todoList, doneList);
626  break;
627  case PcodeOp.COPY:
628  putOnList(output, currentRef.offset, todoList, doneList);
629  break;
630  case PcodeOp.CALL:
631  if (currentRef.offset == 0) { // If pointer is passed directly (no offset)
632  // find it as an input
633  int slot = pcodeOp.getSlot(iVn);
634  if (slot > 0 && slot < pcodeOp.getNumInputs()) {
635  putOnCallParamList(pcodeOp.getInput(0).getAddress(), slot - 1);
636  }
637  }
638  break;
639  }
640 
641  }
642  }
643  }

663  private long getSigned(Varnode varnode) {
664  long mask = 0x80L << ((varnode.getSize() - 1) * 8);
665  long value = varnode.getOffset();
666  if ((value & mask) != 0) {
667  value |= (0xffffffffffffffffL << ((varnode.getSize() - 1) * 8));

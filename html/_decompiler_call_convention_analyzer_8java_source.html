126  private void runDecompilerAnalysis(Program program, AddressSetView functionEntries,
127  TaskMonitor monitor) throws InterruptedException, Exception {
128 
129  CachingPool<DecompInterface> decompilerPool =
130  new CachingPool<>(new DecompilerFactory(program));
131  QRunnable<Address> callback = new ParallelDecompilerCallback(decompilerPool);
132 
133  ConcurrentGraphQ<Address> queue = null;
134 
135  monitor.initialize(functionEntries.getNumAddresses());
136 
137  try {
138  monitor.setMessage("Analyzing Call Hierarchy...");
139  AcyclicCallGraphBuilder builder =
140  new AcyclicCallGraphBuilder(program, functionEntries, true);
141  DependencyGraph<Address> graph = builder.getDependencyGraph(monitor);
142  if (graph.isEmpty()) {
143  return;
144  }
145 
146  GThreadPool pool = AutoAnalysisManager.getSharedAnalsysThreadPool();
147  queue = new ConcurrentGraphQ<>(callback, graph, pool, monitor);
148 
149  monitor.setMessage("Analyzing Call Conventions...");
150 
151  queue.execute();
152  }
153  finally {
154  if (queue != null) {
155  queue.dispose();
156  }
157  decompilerPool.dispose();
158  }

161  private void performConventionAnalysis(Function function, DecompInterface decompiler,
162  TaskMonitor monitor) {
163  DecompilerParallelConventionAnalysisCmd cmd = new DecompilerParallelConventionAnalysisCmd(
164  function, decompiler, decompilerTimeoutSecondsOption);
165  boolean applyTo = cmd.applyTo(function.getProgram(), monitor);
166 
167  // there was an error recovering, mark it so we don't do it again.
168  if (!applyTo) {
169  BookmarkManager bkMgr = function.getProgram().getBookmarkManager();
170  bkMgr.setBookmark(function.getEntryPoint(), BookmarkType.WARNING,
171  COULD_NOT_RECOVER_CALLING_CONVENTION, cmd.getStatusMsg());
172  }
173  }

175  private AddressSetView findLocations(Program program, AddressSetView set, TaskMonitor monitor)
176  throws CancelledException {
177 
178  AddressSet functionEntries = new AddressSet();
179 
180  FunctionIterator functions = program.getFunctionManager().getFunctions(set, true);
181 
182  // must be a function defined
183  BookmarkManager bkMgr = program.getBookmarkManager();
184 
185  for (Function function : functions) {
186  monitor.checkCanceled();
187 
188  // must not be a bookmark here, and paying attention to bookmarks
189  if (!ignoreBookmarks) {
190  Bookmark bookmark = bkMgr.getBookmark(function.getEntryPoint(),
191  BookmarkType.WARNING, COULD_NOT_RECOVER_CALLING_CONVENTION);
192  if (bookmark != null) {
193  continue;
194  }
195  }
196 
197  // must not be thunk or inline
198  if (function.isThunk() || function.isInline()) {
199  continue;
200  }
201 
202  if (function.isExternal()) {
203  continue;
204  }
205 
206  // don't do call fixup
207  if (function.getCallFixup() != null) {
208  continue;
209  }
210 
211  // must be an unknown signature
212  String callingConventionName = function.getCallingConventionName();
213 
214  if (!callingConventionName.equals(Function.UNKNOWN_CALLING_CONVENTION_STRING)) {
215  continue;
216  }
217 
218  // don't touch custom storage
219  if (function.hasCustomVariableStorage()) {
220  continue;
221  }
222 
223  if (hasImportedSignatureWithinNamespace(function) ||
224  hasDefinedParameterTypes(function)) {
225  functionEntries.add(function.getEntryPoint());
226  }
227  }
228 
229  return functionEntries;
230  }
231 
232  private boolean hasImportedSignatureWithinNamespace(Function function) {
233  return function.getSignatureSource() == SourceType.IMPORTED &&
234  function.getParentNamespace().getID() != Namespace.GLOBAL_NAMESPACE_ID;
235  }
236 
237  private boolean hasDefinedParameterTypes(Function function) {
238  ParameterDefinition[] arguments = function.getSignature().getArguments();
239  for (ParameterDefinition parameterDefinition : arguments) {
240  DataType dataType = parameterDefinition.getDataType();
241  if (dataType == DefaultDataType.dataType || Undefined.isUndefined(dataType)) {
242  continue;
243  }
244  return true;
245  }
246  return false;
247  }

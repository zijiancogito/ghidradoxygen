205  protected void initializeProcess() throws IOException, DecompileException {
206  if (decompCallback == null) {
207  throw new IOException("Program not opened in decompiler");
208  }
209  if (decompProcess == null) {
210  decompProcess = DecompileProcessFactory.get();
211  }
212  else if (!decompProcess.isReady()) {
213  DecompileProcessFactory.release(decompProcess);
214  decompProcess = DecompileProcessFactory.get();
215  }
216  // use static uniqueBase since we don't know how many dynamically generated
217  // variables Ghidra may add to the language/compile-spec uniqueBase
218  long uniqueBase = 0x10000000;
219  String tspec = pcodelanguage.buildTranslatorTag(program.getAddressFactory(), uniqueBase, null);
220  String coretypes = dtmanage.buildCoreTypes();
221  SleighLanguageDescription sleighdescription =
222  (SleighLanguageDescription) pcodelanguage.getLanguageDescription();
223  ResourceFile pspecfile = sleighdescription.getSpecFile();
224  String pspecxml = fileToString(pspecfile);
225  String cspecxml = compilerSpec.getCompilerSpecString();
226 
227  decompCallback.setNativeMessage(null);
228  decompProcess.registerProgram(decompCallback, pspecxml, cspecxml, tspec, coretypes);
229  String nativeMessage = decompCallback.getNativeMessage();
230  if ((nativeMessage != null) && (nativeMessage.length() != 0)) {
231  throw new IOException("Could not register program: " + nativeMessage);
232  }
233  if (xmlOptions != null) {
234  decompProcess.setMaxResultSize(xmlOptions.getMaxPayloadMBytes());
235  decompProcess.setShowNamespace(xmlOptions.isDisplayNamespaces());
236  if (!decompProcess.sendCommand1Param("setOptions", xmlOptions.getXML(this)).toString().equals("t")) {
237  throw new IOException("Did not accept decompiler options");
238  }
239  }
240  if (actionname == null) {
241  throw new IOException("Decompile action not specified");
242  }
243  if (!actionname.equals("decompile")) {
244  if (!decompProcess.sendCommand2Params("setAction", actionname, "").toString().equals("t")) {
245  throw new IOException("Could not set decompile action");
246  }
247  }
248  if (!printSyntaxTree) {
249  if (!decompProcess.sendCommand2Params("setAction", "", "notree").toString().equals("t")) {
250  throw new IOException("Could not turn off syntax tree");
251  }
252  }
253  if (!printCCode) {
254  if (!decompProcess.sendCommand2Params("setAction", "", "noc").toString().equals("t")) {
255  throw new IOException("Could not turn off C printing");
256  }
257  }
258  if (sendParamMeasures) {
259  if (!decompProcess.sendCommand2Params("setAction", "", "parammeasures").toString().equals("t")) {
260  throw new IOException("Could not turn on sending of parameter measures");
261  }
262  }
263  if (jumpLoad) {
264  if (!decompProcess.sendCommand2Params("setAction", "", "jumpload").toString().equals("t")) {
265  throw new IOException("Could not turn on jumptable loads");
266  }
267  }
268  }

293  public synchronized boolean openProgram(Program prog) {
294  decompileMessage = "";
295  program = prog;
296  Language lang = prog.getLanguage();
297  if (!lang.supportsPcode()) {
298  decompileMessage = "Language does not support PCode.";
299  return false;
300  }
301  pcodelanguage = (SleighLanguage)lang;
302  CompilerSpec spec = prog.getCompilerSpec();
303  if (!(spec instanceof BasicCompilerSpec)) {
304  decompileMessage =
305  "Language has unsupported compiler spec: " + spec.getClass().getName();
306  return false;
307  }
308  compilerSpec = (BasicCompilerSpec) spec;
309 
310  dtmanage = new PcodeDataTypeManager(prog);
311  try {
312  decompCallback =
313  new DecompileCallback(prog, pcodelanguage, program.getCompilerSpec(), dtmanage);
314  initializeProcess();
315  if (!decompProcess.isReady()) {
316  throw new IOException("Unable to start decompiler process");
317  }
318  decompileMessage = decompCallback.getNativeMessage();
319  if (!isErrorMessage()) {
320  return true;
321  }
322  }
323  catch (Exception ex) {
324  decompileMessage = ex.getMessage();
325  if (decompProcess == null) {
326  return false;
327  }
328  stopProcess();
329  }
330  program = null;
331  decompCallback = null;
332 
333  return false;
334  }

633  public synchronized BlockGraph structureGraph(BlockGraph ingraph,AddressFactory factory,int timeoutSecs,TaskMonitor monitor) {
634  decompileMessage = "";
635  if (monitor != null && monitor.isCancelled())
636  return null;

638  monitor.addCancelledListener(monitorListener);
639  LimitedByteBuffer res = null;
640  BlockGraph resgraph = null;
641  try {
642  StringWriter writer = new StringWriter();
643  ingraph.saveXml(writer);
644  verifyProcess();
645  res = decompProcess.sendCommand1ParamTimeout("structureGraph", writer.toString(), timeoutSecs);
646  decompileMessage = decompCallback.getNativeMessage();
647  if (res != null) {
648  XmlPullParser parser = HighFunction.stringTree(res.getInputStream(), HighFunction.getErrorHandler(this, "Results for structureGraph command"));
649  resgraph = new BlockGraph();
650  resgraph.restoreXml(parser, factory);
651  resgraph.transferObjectRef(ingraph);
652  }
653  }
654  catch (Exception ex) {
655  decompileMessage = "Exception while graph structuring: " + ex.getMessage() + '\n';
656  }
657  finally {
658  if (monitor != null) {

661  }

673  public synchronized DecompileResults decompileFunction(Function func, int timeoutSecs,
674  TaskMonitor monitor) {
675 
676  decompileMessage = "";
677  if (monitor != null && monitor.isCancelled()) {
678  return null;
679  }
680 
681  LimitedByteBuffer res = null;
682  if (monitor != null) {
683  monitor.addCancelledListener(monitorListener);
684  }
685 
686  if (program == null) {
687  return new DecompileResults(func, pcodelanguage, null, dtmanage, decompileMessage, null,
688  DecompileProcess.DisposeState.DISPOSED_ON_CANCEL,
689  false /* cancelled--doesn't matter */);
690  }
691 
692  try {
693  Address funcEntry = func.getEntryPoint();
694  if (debug != null) {
695  debug.setFunction(func);
696  }
697  decompCallback.setFunction(func, funcEntry, debug);
698  String addrstring = Varnode.buildXMLAddress(funcEntry);
699  verifyProcess();
700  res =
701  decompProcess.sendCommand1ParamTimeout("decompileAt", addrstring.toString(),
702  timeoutSecs);
703  decompileMessage = decompCallback.getNativeMessage();
704  }
705  catch (Exception ex) {
706  decompileMessage = "Exception while decompiling " +func.getEntryPoint() + ": "+ ex.getMessage() + '\n';
707  }
708  finally {
709  if (monitor != null) {
710  monitor.removeCancelledListener(monitorListener);
711  }
712  }
713  if (debug != null) {
714  debug.shutdown(pcodelanguage, xmlOptions.getXML(this));
715  debug = null;
716  }
717 
718  DecompileProcess.DisposeState processState;
719  if (decompProcess != null) {
720  processState = decompProcess.getDisposeState();
721  if (decompProcess.getDisposeState() == DecompileProcess.DisposeState.NOT_DISPOSED) {
722  flushCache();
723  }
724  }
725  else {
726  processState = DecompileProcess.DisposeState.DISPOSED_ON_CANCEL;
727  }
728 
729  InputStream stream = null;
730  if (res != null)
731  stream = res.getInputStream();
732  return new DecompileResults(func, pcodelanguage, compilerSpec, dtmanage, decompileMessage,
733  stream, processState, isDisplayNamespace());
734  }

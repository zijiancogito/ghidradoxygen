422  public void grabFromProgram(Program program) {
423  // Default values, even if there is no program
424  displayLanguage = BasicCompilerSpec.DECOMPILER_OUTPUT_DEF;
425  protoEvalModel = "default";
426  if (program == null) {
427  return;
428  }
429 
430  CompilerSpec cspec = program.getCompilerSpec();
431  PrototypeModel model = (PrototypeModel) cspec.getPrototypeEvaluationModel(program);
432  if (model != null) {
433  String modelname = model.getName();
434  if (modelname != null) {
435  protoEvalModel = modelname;
436  }
437  }
438  displayLanguage = cspec.getDecompilerOutputLanguage(program);
439  }

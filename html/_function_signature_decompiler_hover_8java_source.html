75 
76  if (!enabled || programLocation == null) {
77  return null;

79 
80  if (!(field instanceof ClangTextField)) {
81  return null;
82  }
83 
84  ClangToken token = ((ClangTextField) field).getToken(fieldLocation);
85  if (token instanceof ClangFuncNameToken) {
86 
87  // Obvious function reference
88  String name = token.getText();
89  Symbol symbol = getSymbolForLocation(program, name);
90  if (symbol == null) {
91  return null;
92  }
93 
94  if (symbol.getSymbolType() == SymbolType.FUNCTION) {
95  Function function = program.getFunctionManager().getFunctionAt(symbol.getAddress());
96  String content = ToolTipUtils.getToolTipText(function, false);
97  return createTooltipComponent(content);
98  }
99  }
100  else if (token instanceof ClangVariableToken) {
101 
102  // Reference to function-address: "x = &foo;" where 'foo' is a function
103  Varnode vn = ((ClangVariableToken) token).getVarnode();
104  if (vn == null) {
105  return null;
106  }
107 
108  if (!(vn.getHigh() instanceof HighConstant)) {
109  return null;
110  }
111 
112  HighConstant hv = (HighConstant) vn.getHigh();
113  long offset = vn.getOffset();
114  int sz = vn.getSize();
115  boolean isSigned = true;
116  if (hv.getDataType() instanceof AbstractIntegerDataType) {
117  isSigned = ((AbstractIntegerDataType) hv.getDataType()).isSigned();
118  }
119 
120  if (sz > 8) {
121  // our Scalar can currently only handle long values
122  return null;
123  }
124 
125  Scalar scalar = new Scalar(sz * 8, offset, isSigned);
126  long scalarLong = scalar.getValue();
127  AddressFactory factory = program.getAddressFactory();
128  AddressSpace space = factory.getDefaultAddressSpace();
129  try {
130  Address asAddress = factory.getAddress(space.getBaseSpaceID(), scalarLong);
131  Function function = program.getListing().getFunctionAt(asAddress);
132  if (function != null) {
133  String content = ToolTipUtils.getToolTipText(function, false);
134  content = content.replaceFirst(HTML,
135  HTML + italic(bold("Reference to Function")) + "<br/><br/>");
136 
137  return createTooltipComponent(content);
138  }
139  }
140  catch (AddressOutOfBoundsException ex) {
141  return null; // Constant does not make sense as an address
142  }
143  }
144 
145  return null;
146  }

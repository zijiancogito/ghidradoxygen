648 template<typename _type>
649 void circularqueue<_type>::setMax(int4 sz)
650

655  left = 1; // This operation empties queue
656  right = 0;

664 template<typename _type>
665 void circularqueue<_type>::expand(int4 amount)
666 
667 {
668  _type *newcache = new _type [ max + amount ];
669 
670  int4 i=left;
671  int4 j=0;
672 
673  // Assume there is at least one element in queue
674  while(i != right) {
675  newcache[j++] = cache[i];
676  i = (i+1)%max;
677  }
678  newcache[j] = cache[i]; // Copy rightmost
679  left=0;
680  right = j;
681 
682  delete [] cache;
683  cache = newcache;
684  max += amount;
685 }

102 uintm PcodeOp::getCseHash(void) const
103 
104 {
105  uintm hash;
106  if ((getEvalType()&(PcodeOp::unary|PcodeOp::binary))==0) return ((uintm)0);
107  if (code()==CPUI_COPY) return ((uintm)0); // Let copy propagation deal with this
108 
109  hash = (output->getSize()<<8) | (uintm)code();
110  for(int4 i=0;i<inrefs.size();++i) {
111  const Varnode *vn = getIn(i);
112  hash = (hash<<8) | (hash>>(sizeof(uintm)*8-8));
113  if (vn->isConstant())
114  hash ^= (uintm)vn->getOffset();
115  else
116  hash ^= (uintm)vn->getCreateIndex(); // Hash in pointer itself as unique id
117  }
118  return hash;
119 }

379 uintb PcodeOp::getNZMaskLocal(bool cliploop) const
380 
381 {
382  int4 sa,sz1,sz2,size;
383  uintb resmask,val;
384 
385  size = output->getSize();
386  uintb fullmask = calc_mask( size );
387 
388  switch(opcode->getOpcode()) {
389  case CPUI_INT_EQUAL:
390  case CPUI_INT_NOTEQUAL:
391  case CPUI_INT_SLESS:
392  case CPUI_INT_SLESSEQUAL:
393  case CPUI_INT_LESS:
394  case CPUI_INT_LESSEQUAL:
395  case CPUI_INT_CARRY:
396  case CPUI_INT_SCARRY:
397  case CPUI_INT_SBORROW:
398  case CPUI_BOOL_NEGATE:
399  case CPUI_BOOL_XOR:
400  case CPUI_BOOL_AND:
401  case CPUI_BOOL_OR:
402  case CPUI_FLOAT_EQUAL:
403  case CPUI_FLOAT_NOTEQUAL:
404  case CPUI_FLOAT_LESS:
405  case CPUI_FLOAT_LESSEQUAL:
406  case CPUI_FLOAT_NAN:
407  resmask=1; // Only 1 bit not guaranteed to be 0
408  break;
409  case CPUI_COPY:
410  case CPUI_INT_ZEXT:
411  resmask = getIn(0)->getNZMask();
412  break;
413  case CPUI_INT_SEXT:
414  resmask = sign_extend( getIn(0)->getNZMask(), getIn(0)->getSize(), size);
415  break;
416  case CPUI_INT_XOR:
417  case CPUI_INT_OR:
418  resmask = getIn(0)->getNZMask();
419  if (resmask != fullmask)
420  resmask |= getIn(1)->getNZMask();
421  break;
422  case CPUI_INT_AND:
423  resmask = getIn(0)->getNZMask();
424  if (resmask != 0)
425  resmask &= getIn(1)->getNZMask();
426  break;
427  case CPUI_INT_LEFT:
428  if (!getIn(1)->isConstant())
429  resmask = fullmask;
430  else {
431  sa = getIn(1)->getOffset(); // Get shift amount
432  resmask = getIn(0)->getNZMask();
433  resmask = pcode_left(resmask,sa) & fullmask;
434  }
435  break;
436  case CPUI_INT_RIGHT:
437  if (!getIn(1)->isConstant())
438  resmask = fullmask;
439  else {
440  sz1 = getIn(0)->getSize();
441  sa = getIn(1)->getOffset(); // Get shift amount
442  resmask = getIn(0)->getNZMask();
443  resmask = pcode_right(resmask,sa);
444  if (sz1 > sizeof(uintb)) {
445  // resmask did not hold most sig bits of mask
446  if (sa >= 8*sz1)
447  resmask = 0;
448  else if (sa >= 8*sizeof(uintb)) {
449  // Full mask shifted over 8*sizeof(uintb)
450  resmask = calc_mask( sz1-sizeof(uintb) );
451  // Shift over remaining portion of sa
452  resmask >>= (sa-8*sizeof(uintb));
453  }
454  else {
455  // Fill in one bits from part of mask not originally
456  // calculated
457  uintb tmp = 0;
458  tmp -= 1;
459  tmp <<= (8*sizeof(uintb)-sa);
460  resmask |= tmp;
461  }
462  }
463  }
464  break;
465  case CPUI_INT_SRIGHT:
466  if ((!getIn(1)->isConstant())||(size > sizeof(uintb)))
467  resmask = fullmask;
468  else {
469  sa = getIn(1)->getOffset(); // Get shift amount
470  resmask = getIn(0)->getNZMask();
471  if ((resmask & (fullmask ^ (fullmask>>1))) == 0) { // If we know sign bit is zero
472  resmask = pcode_right(resmask,sa); // Same as CPUI_INT_RIGHT
473  }
474  else {
475  resmask = pcode_right(resmask,sa);
476  resmask |= (fullmask >> sa) ^ fullmask; // Don't know what the new high bits are
477  }
478  }
479  break;
480  case CPUI_INT_DIV:
481  val = getIn(0)->getNZMask();
482  resmask = coveringmask(val);
483  if (getIn(1)->isConstant()) {
484  // Dividing by power of 2 is equiv to right shift
485  // if the denom is bigger than a power of 2, then
486  // the result still has at least that many highsig zerobits
487  sa = mostsigbit_set(getIn(1)->getNZMask());
488  if (sa != -1)
489  resmask >>= sa; // Add sa additional zerobits
490  }
491  break;
492  case CPUI_INT_REM:
493  val = (getIn(1)->getNZMask()-1); // Result is less than modulus
494  resmask = coveringmask(val);
495  break;
496  case CPUI_SUBPIECE:
497  resmask = getIn(0)->getNZMask();
498  resmask >>= 8*getIn(1)->getOffset();
499  resmask &= fullmask;
500  break;
501  case CPUI_PIECE:
502  resmask = getIn(0)->getNZMask();
503  resmask <<= 8*getIn(1)->getSize();
504  resmask |= getIn(1)->getNZMask();
505  break;
506  case CPUI_INT_MULT:
507  val = getIn(0)->getNZMask();
508  resmask = getIn(1)->getNZMask();
509  sz1 = mostsigbit_set(val);
510  if (sz1 == -1)
511  resmask = 0;
512  else {
513  sz2 = mostsigbit_set(resmask);
514  if (sz2 == -1)
515  resmask = 0;
516  else {
517  if (sz1 + sz2 < 8*size-2)
518  fullmask >>= (8*size-2-sz1-sz2);
519  sz1 = leastsigbit_set(val);
520  sz2 = leastsigbit_set(resmask);
521  resmask = (~((uintb)0))<<(sz1+sz2);
522  resmask &= fullmask;
523  }
524  }
525  break;
526  case CPUI_INT_ADD:
527  resmask = getIn(0)->getNZMask();
528  if (resmask!=fullmask) {
529  resmask |= getIn(1)->getNZMask();
530  resmask |= (resmask<<1); // Account for possible carries
531  resmask &= fullmask;
532  }
533  break;
534  case CPUI_MULTIEQUAL:
535  if (inrefs.size()==0)
536  resmask = fullmask;
537  else {
538  int4 i=0;
539  resmask = 0;
540  if (cliploop) {
541  for(;i<inrefs.size();++i) {
542  if (parent->isLoopIn(i)) continue;
543  resmask |= getIn(i)->getNZMask();
544  }
545  }
546  else {
547  for(;i<inrefs.size();++i)
548  resmask |= getIn(i)->getNZMask();
549  }
550  }
551  break;
552  case CPUI_CALL:
553  case CPUI_CALLIND:
554  case CPUI_CPOOLREF:
555  if (isCalculatedBool())
556  resmask = 1; // In certain cases we know the output is strictly boolean
557  else
558  resmask = fullmask;
559  break;
560  default:
561  resmask = fullmask;
562  break;
563  }
564  return resmask;
565 }

798  // In this case we know an instruction is contiguous
799  // in the dead list
800  list<PcodeOp *>::const_iterator iter = op->insertiter;
801  ++iter;
802  if (iter != deadlist.end()) {
803  retop = *iter;
804  if (!retop->isInstructionStart()) // If the next in dead list is not marked
805  return retop; // It is in the same instruction, and is the fallthru
806  }
807  --iter;
808  SeqNum max = op->getSeqNum();
809  while(!(*iter)->isInstructionStart()) // Find start of instruction
810  --iter;
811  // Find biggest sequence number in this instruction
812  // This is probably -op- itself because it is the
813  // last op in the instruction, but it might not be
814  // because of delay slot reordering
815  while((iter!=deadlist.end())&&(*iter != op)) {
816  if (max < (*iter)->getSeqNum())
817  max = (*iter)->getSeqNum();
818  ++iter;
819  }
820  PcodeOpTree::const_iterator nextiter = optree.upper_bound(max);
821  if (nextiter == optree.end()) return (PcodeOp *)0;
822  retop = (*nextiter).second;
823  return retop;
824  }
825  else
826  return op->nextOp();
827 }

892 static int4 functionalEqualityLevel0(Varnode *vn1,Varnode *vn2)
893 
894 { // Return 0 if -vn1- and -vn2- must hold same value
895  // Return -1 if they definitely don't hold same value
896  // Return 1 if the same value depends on ops writing to -vn1- and -vn2-
897  if (vn1==vn2) return 0;

923 int4 functionalEqualityLevel(Varnode *vn1,Varnode *vn2,Varnode **res1,Varnode **res2)
924 
925 {
926  int4 testval = functionalEqualityLevel0(vn1,vn2);
927  if (testval != 1) return testval;
928  PcodeOp *op1 = vn1->getDef();
929  PcodeOp *op2 = vn2->getDef();
930  OpCode opc = op1->code();
931 
932  if (opc != op2->code()) return -1;
933 
934  int4 num = op1->numInput();
935  if (num != op2->numInput()) return -1;
936  if (op1->isMarker()) return -1;
937  if (op2->isCall()) return -1;
938  if (opc == CPUI_LOAD) {
939  // FIXME: We assume two loads produce the same
940  // result if the address is the same and the loads
941  // occur in the same instruction
942  if (op1->getAddr() != op2->getAddr()) return -1;
943  }
944  if (num >= 3) {
945  if (opc != CPUI_PTRADD) return -1; // If this is a PTRADD
946  if (op1->getIn(2)->getOffset() != op2->getIn(2)->getOffset()) return -1; // Make sure the elsize constant is equal
947  num = 2; // Otherwise treat as having 2 inputs
948  }
949  for(int4 i=0;i<num;++i) {
950  res1[i] = op1->getIn(i);
951  res2[i] = op2->getIn(i);
952  }
953 
954  testval = functionalEqualityLevel0(res1[0],res2[0]);
955  if (testval == 0) { // A match locks in this comparison ordering
956  if (num==1) return 0;
957  testval = functionalEqualityLevel0(res1[1],res2[1]);
958  if (testval==0) return 0;
959  if (testval < 0) return -1;
960  res1[0] = res1[1]; // Match is contingent on second pair
961  res2[0] = res2[1];
962  return 1;
963  }
964  if (num == 1) return testval;
965  int4 testval2 = functionalEqualityLevel0(res1[1],res2[1]);
966  if (testval2 == 0) { // A match locks in this comparison ordering
967  return testval;
968  }
969  int4 unmatchsize;
970  if ((testval==1)&&(testval2==1))
971  unmatchsize = 2;
972  else
973  unmatchsize = -1;
974 
975  if (!op1->isCommutative()) return unmatchsize;
976  // unmatchsize must be 2 or -1 here on a commutative operator,
977  // try flipping
978  int4 comm1 = functionalEqualityLevel0(res1[0],res2[1]);
979  int4 comm2 = functionalEqualityLevel0(res1[1],res2[0]);
980  if ((comm1==0) && (comm2==0))
981  return 0;
982  if ((comm1<0)||(comm2<0))
983  return unmatchsize;
984  if (comm1==0) { // AND (comm2==1)
985  res1[0] = res1[1]; // Left over unmatch is res1[1] and res2[0]
986  return 1;
987  }
988  if (comm2==0) { // AND (comm1==1)
989  res2[0] = res2[1]; // Left over unmatch is res1[0] and res2[1]
990  return 1;
991  }
992  // If we reach here (comm1==1) AND (comm2==1)
993  if (unmatchsize == 2) // If the original ordering wasn't impossible
994  return 2; // Prefer the original ordering
995  Varnode *tmpvn = res2[0]; // Otherwise swap the ordering
996  res2[0] = res2[1];
997  res2[1] = tmpvn;
998  return 2;
999 }

1022 bool functionalDifference(Varnode *vn1,Varnode *vn2,int4 depth)
1023 
1024 {
1025  PcodeOp *op1,*op2;
1026  int4 i,num;
1027 
1028  if (vn1 == vn2) return false;
1029  if ((!vn1->isWritten())||(!vn2->isWritten())) {
1030  if (vn1->isConstant() && vn2->isConstant())
1031  return !(vn1->getAddr()==vn2->getAddr());
1032  if (vn1->isInput()&&vn2->isInput()) return false; // Might be the same
1033  if (vn1->isFree()||vn2->isFree()) return false; // Might be the same
1034  return true;
1035  }
1036  op1 = vn1->getDef();
1037  op2 = vn2->getDef();
1038  if (op1->code() != op2->code()) return true;
1039  num = op1->numInput();
1040  if (num != op2->numInput()) return true;
1041  if (depth==0) return true; // Different as far as we can tell
1042  depth -= 1;
1043  for(i=0;i<num;++i)
1044  if (functionalDifference(op1->getIn(i),op2->getIn(i),depth))
1045  return true;
1046  return false;
1047 }

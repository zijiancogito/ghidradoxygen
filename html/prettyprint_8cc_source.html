562 void EmitPrettyPrint::expand(void)
563 
564 {
565  int4 max = tokqueue.getMax();
566  int4 left = tokqueue.bottomref();
567  tokqueue.expand(200);
568  // Expanding puts the leftmost element at reference 0
569  // So we need to adjust references
570  for(int4 i=0;i<max;++i)
571  scanqueue.ref(i) = (scanqueue.ref(i) + max - left) % max;
572  // The number of elements in scanqueue is always less than
573  // or equal to the number of elements in tokqueue, so
574  // if we keep scanqueue and tokqueue with the same max
575  // we don't need to check for scanqueue overflow
576  scanqueue.expand(200);
577 }

582 void EmitPrettyPrint::overflow(void)
583 
584 {
585  int4 half = maxlinesize / 2;
586  for(int4 i=indentstack.size()-1;i>=0;--i) {
587  if (indentstack[i] < half)
588  indentstack[i] = half;

602 void EmitPrettyPrint::print(const TokenSplit &tok)
603 
604 {
605  int4 val = 0;
606 
607  switch(tok.getClass()) {
608  case TokenSplit::ignore:
609  tok.print(lowlevel); // Markup or other that doesn't use space
610  break;
611  case TokenSplit::begin_indent:
612  val = indentstack.back() - tok.getIndentBump();
613  indentstack.push_back(val);
614 #ifdef PRETTY_DEBUG
615  checkid.push_back(tok.getCount());
616 #endif
617  break;
618  case TokenSplit::begin_comment:
619  commentmode = true;
620  // fallthru, treat as a group begin
621  case TokenSplit::begin:
622  tok.print(lowlevel);
623  indentstack.push_back(spaceremain);
624 #ifdef PRETTY_DEBUG
625  checkid.push_back(tok.getCount());
626 #endif
627  break;
628  case TokenSplit::end_indent:
629  if (indentstack.empty())
630  throw LowlevelError("indent error");
631 #ifdef PRETTY_DEBUG
632  if (checkid.empty() || (checkid.back() != tok.getCount()))
633  throw LowlevelError("mismatch1");
634  checkid.pop_back();
635  if (indentstack.empty())
636  throw LowlevelError("Empty indent stack");
637 #endif
638  indentstack.pop_back();
639  break;
640  case TokenSplit::end_comment:
641  commentmode = false;
642  // fallthru, treat as a group end
643  case TokenSplit::end:
644  tok.print(lowlevel);
645 #ifdef PRETTY_DEBUG
646  if (checkid.empty() || (checkid.back() != tok.getCount()))
647  throw LowlevelError("mismatch2");
648  checkid.pop_back();
649  if (indentstack.empty())
650  throw LowlevelError("indent error");
651 #endif
652  indentstack.pop_back();
653  break;
654  case TokenSplit::tokenstring:
655  if (tok.getSize() > spaceremain)
656  overflow();
657  tok.print(lowlevel);
658  spaceremain -= tok.getSize();
659  break;
660  case TokenSplit::tokenbreak:
661  if (tok.getSize() > spaceremain) {
662  if (tok.getTag() == TokenSplit::line_t) // Absolute indent
663  spaceremain = maxlinesize - tok.getIndentBump();
664  else { // relative indent
665  val = indentstack.back() - tok.getIndentBump();
666  // If creating a line break doesn't save that much
667  // don't do the line break
668  if ((tok.getNumSpaces() <= spaceremain)&&
669  (val-spaceremain < 10)) {
670  lowlevel->spaces(tok.getNumSpaces());
671  spaceremain -= tok.getNumSpaces();
672  return;
673  }
674  indentstack.back() = val;
675  spaceremain = val;
676  }
677  lowlevel->tagLine(maxlinesize-spaceremain);
678  if (commentmode &&(commentfill.size() != 0)) {
679  lowlevel->print(commentfill.c_str(),EmitXml::comment_color);
680  spaceremain -= commentfill.size();
681  }
682  }
683  else {
684  lowlevel->spaces(tok.getNumSpaces());
685  spaceremain -= tok.getNumSpaces();
686  }
687  break;
688  }
689 }

698 void EmitPrettyPrint::advanceleft(void)
699 
700 {
701  int4 l = tokqueue.bottom().getSize();
702  while(l >= 0) {
703  const TokenSplit &tok( tokqueue.bottom() );
704  print(tok);
705  switch(tok.getClass()) {
706  case TokenSplit::tokenbreak:
707  leftotal += tok.getNumSpaces();
708  break;
709  case TokenSplit::tokenstring:
710  leftotal += l;

714  }
715  tokqueue.popbottom();
716  if (tokqueue.empty()) break;
717  l = tokqueue.bottom().getSize();
718  }

729 void EmitPrettyPrint::scan(void)
730 
731 {
732  if (tokqueue.empty()) // If we managed to overflow queue
733  expand(); // Expand it
734  // Delay creating reference until after the possible expansion
735  TokenSplit &tok( tokqueue.top() );
736  switch(tok.getClass()) {
737  case TokenSplit::begin_comment:
738  case TokenSplit::begin:
739  if (scanqueue.empty()) {
740  leftotal = rightotal = 1;
741  }
742  tok.setSize( -rightotal );
743  scanqueue.push() = tokqueue.topref();
744  break;
745  case TokenSplit::end_comment:
746  case TokenSplit::end:
747  tok.setSize(0);
748  if (!scanqueue.empty()) {
749  TokenSplit &ref( tokqueue.ref( scanqueue.pop() ) );
750  ref.setSize( ref.getSize() + rightotal );
751  if ((ref.getClass() == TokenSplit::tokenbreak)&&(!scanqueue.empty())) {
752  TokenSplit &ref2( tokqueue.ref( scanqueue.pop() ) );
753  ref2.setSize( ref2.getSize() + rightotal );
754  }
755  if (scanqueue.empty())
756  advanceleft();
757  }
758  break;
759  case TokenSplit::tokenbreak:
760  if (scanqueue.empty()) {
761  leftotal = rightotal = 1;
762  }
763  else {
764  TokenSplit &ref( tokqueue.ref( scanqueue.top() ) );
765  if (ref.getClass() == TokenSplit::tokenbreak) {
766  scanqueue.pop();
767  ref.setSize( ref.getSize() + rightotal );
768  }
769  }
770  tok.setSize( -rightotal );
771  scanqueue.push() = tokqueue.topref();
772  rightotal += tok.getNumSpaces();
773  break;
774  case TokenSplit::begin_indent:
775  case TokenSplit::end_indent:
776  case TokenSplit::ignore:
777  tok.setSize(0);
778  break;
779  case TokenSplit::tokenstring:
780  if (!scanqueue.empty()) {
781  rightotal += tok.getSize();
782  while(rightotal-leftotal > spaceremain) {
783  TokenSplit &ref( tokqueue.ref( scanqueue.popbottom() ) );
784  ref.setSize(999999);
785  advanceleft();
786  if (scanqueue.empty()) break;
787  }
788  }
789  }
790 }

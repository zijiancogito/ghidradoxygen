329 inline uintb AddrSpace::wrapOffset(uintb off) const {
330  if (off <= highest) // Comparison is unsigned
331  return off;
332  intb mod = (intb)(highest+1);
333  intb res = (intb)off % mod; // remainder is signed
334  if (res<0) // Remainder may be negative
335  res += mod; // Adding mod guarantees res is in (0,mod)
336  return (uintb)res;
337 }

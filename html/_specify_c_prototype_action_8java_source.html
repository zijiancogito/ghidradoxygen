126  private void verifyDynamicEditorModel(HighFunction hf, FunctionEditorModel model) {
127 
128  FunctionPrototype functionPrototype = hf.getFunctionPrototype();
129  int decompParamCnt = functionPrototype.getNumParams();
130 
131  List<ParamInfo> parameters = model.getParameters();
132  int modelParamCnt = parameters.size();
133 
134  // growth accounts for auto param injection
135  int autoParamCnt = modelParamCnt - decompParamCnt;
136 
137  // make sure decomp params account for injected auto params
138  boolean useCustom = (decompParamCnt < autoParamCnt);
139 
140  for (int i = 0; i < autoParamCnt && !useCustom; i++) {
141  if (i >= decompParamCnt) {
142  useCustom = true;
143  }
144  else {
145  VariableStorage modelParamStorage = parameters.get(i).getStorage();
146  VariableStorage decompParamStorage = functionPrototype.getParam(i).getStorage();
147  if (!modelParamStorage.equals(decompParamStorage)) {
148  useCustom = true;
149  }
150  }
151  }
152 
153  if (!useCustom) {
154  // remove original params which replicate auto params
155  for (int i = 0; i < autoParamCnt; i++) {
156  model.setSelectedParameterRow(new int[] { autoParamCnt });
157  model.removeParameters();
158  }
159 
160  // verify remaining parameter storage
161  for (int i = autoParamCnt; i < decompParamCnt; i++) {
162  VariableStorage modelParamStorage = parameters.get(i).getStorage();
163  VariableStorage decompParamStorage = functionPrototype.getParam(i).getStorage();
164  if (!modelParamStorage.equals(decompParamStorage)) {
165  useCustom = true;
166  break;
167  }
168  }
169  }
170 
171  // TODO: return storage not currently returned from Decompiler
172 
173  if (useCustom) {
174  // Force custom storage
175  model.setUseCustomizeStorage(true);
176  model.setFunctionData(buildSignature(hf));
177  model.setReturnStorage(functionPrototype.getReturnStorage());
178  parameters = model.getParameters();
179  for (int i = 0; i < decompParamCnt; i++) {
180  model.setParameterStorage(parameters.get(i),
181  functionPrototype.getParam(i).getStorage());
182  }
183  }
184  }

186  private FunctionDefinitionDataType buildSignature(HighFunction hf) {
187  // try to look up the function that is at the current cursor location
188  // If there isn't one, just use the function we are in.
189 
190  Function func = hf.getFunction();
191  FunctionDefinitionDataType fsig = new FunctionDefinitionDataType(CategoryPath.ROOT,
192  func.getName(), func.getProgram().getDataTypeManager());
193  FunctionPrototype functionPrototype = hf.getFunctionPrototype();
194 
195  int np = hf.getLocalSymbolMap().getNumParams();
196  fsig.setReturnType(functionPrototype.getReturnType());
197 
198  ParameterDefinition[] args = new ParameterDefinitionImpl[np];
199  for (int i = 0; i < np; i++) {
200  HighParam parm = hf.getLocalSymbolMap().getParam(i);
201  args[i] = new ParameterDefinitionImpl(parm.getName(), parm.getDataType(), null);
202  }
203  fsig.setArguments(args);
204  fsig.setVarArgs(functionPrototype.isVarArg());
205  return fsig;
206  }

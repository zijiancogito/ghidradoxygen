166 
167 {
168  {
169  uint4 size;
170  istringstream s(el->getAttributeValue("scopesize"));
171  s.unsetf(ios::dec | ios::hex | ios::oct);
172  s >> size;

175  {

178  s.unsetf(ios::dec | ios::hex | ios::oct);
179  s >> size;
180  symbollist.resize(size,(SleighSymbol *)0);
181  }
182  const List &list(el->getChildren());
183  List::const_iterator iter;
184  iter = list.begin();
185  for(int4 i=0;i<table.size();++i) { // Restore the scopes
186  Element *subel = *iter;
187  if (subel->getName() != "scope")
188  throw SleighError("Misnumbered symbol scopes");
189  uintm id;
190  uintm parent;
191  {
192  istringstream s(subel->getAttributeValue("id"));
193  s.unsetf(ios::dec | ios::hex | ios::oct);
194  s >> id;
195  }
196  {
197  istringstream s(subel->getAttributeValue("parent"));
198  s.unsetf(ios::dec | ios::hex | ios::oct);
199  s >> parent;
200  }
201  SymbolScope *parscope = (parent==id) ? (SymbolScope *)0 : table[parent];
202  table[id] = new SymbolScope( parscope, id );
203  ++iter;
204  }

206 
207  // Now restore the symbol shells
208  for(int4 i=0;i<symbollist.size();++i) {
209  restoreSymbolHeader(*iter);
210  ++iter;
211  }
212  // Now restore the symbol content
213  while(iter != list.end()) {
214  Element *subel = *iter;
215  uintm id;
216  SleighSymbol *sym;
217  {
218  istringstream s(subel->getAttributeValue("id"));
219  s.unsetf(ios::dec | ios::hex | ios::oct);
220  s >> id;

224  ++iter;
225  }

268 void SymbolTable::purge(void)
269 
270 { // Get rid of unsavable symbols and scopes
271  SleighSymbol *sym;
272  for(int4 i=0;i<symbollist.size();++i) {
273  sym = symbollist[i];
274  if (sym == (SleighSymbol *)0) continue;
275  if (sym->scopeid != 0) { // Not in global scope
276  if (sym->getType() == SleighSymbol::operand_symbol) continue;
277  }
278  else {
279  switch(sym->getType()) {
280  case SleighSymbol::space_symbol:
281  case SleighSymbol::token_symbol:
282  case SleighSymbol::epsilon_symbol:
283  case SleighSymbol::section_symbol:
284  break;
285  case SleighSymbol::macro_symbol:
286  { // Delete macro's local symbols
287  MacroSymbol *macro = (MacroSymbol *)sym;
288  for(int4 i=0;i<macro->getNumOperands();++i) {
289  SleighSymbol *opersym = macro->getOperand(i);
290  table[opersym->scopeid]->removeSymbol(opersym);
291  symbollist[opersym->id] = (SleighSymbol *)0;
292  delete opersym;
293  }
294  break;
295  }
296  case SleighSymbol::subtable_symbol:
297  { // Delete unused subtables
298  SubtableSymbol *subsym = (SubtableSymbol *)sym;
299  if (subsym->getPattern() != (TokenPattern *)0) continue;
300  for(int4 i=0;i<subsym->getNumConstructors();++i) { // Go thru each constructor
301  Constructor *con = subsym->getConstructor(i);
302  for(int4 j=0;j<con->getNumOperands();++j) { // Go thru each operand
303  OperandSymbol *oper = con->getOperand(j);
304  table[oper->scopeid]->removeSymbol(oper);
305  symbollist[oper->id] = (SleighSymbol *)0;
306  delete oper;
307  }
308  }
309  break; // Remove the subtable symbol itself
310  }
311  default:
312  continue;
313  }
314  }
315  table[sym->scopeid]->removeSymbol(sym); // Remove the symbol
316  symbollist[i] = (SleighSymbol *)0;
317  delete sym;
318  }
319  for(int4 i=1;i<table.size();++i) { // Remove any empty scopes
320  if (table[i]->tree.empty()) {
321  delete table[i];
322  table[i] = (SymbolScope *)0;
323  }
324  }
325  renumber();
326 }

329 
330 { // Renumber all the scopes and symbols
331  // so that there are no gaps
332  vector<SymbolScope *> newtable;
333  vector<SleighSymbol *> newsymbol;

343  // Now renumber the symbols

346  sym = symbollist[i];

349  sym->id = newsymbol.size();
350  newsymbol.push_back(sym);
351  }
352  }

1749 {
1750  if (pattern != (TokenPattern *)0) return pattern; // Already built
1751 
1752  pattern = new TokenPattern();

1754  bool recursion = false;
1755  // Generate pattern for each operand, store in oppattern
1756  for(int4 i=0;i<operands.size();++i) {
1757  OperandSymbol *sym = operands[i];

1764  if (recursion) {
1765  throw SleighError("Illegal recursion");
1766  }
1767  // We should also check that recursion is rightmost extreme
1768  recursion = true;
1769  oppattern.push_back(TokenPattern());
1770  }
1771  else
1772  oppattern.push_back(*subsym->buildPattern(s));
1773  }
1774  else
1775  oppattern.push_back(triple->getPatternExpression()->genMinPattern(oppattern));
1776  }
1777  else if (defexp != (PatternExpression *)0)
1778  oppattern.push_back(defexp->genMinPattern(oppattern));

1781  }

1785  sym->setVariableLength();
1786  }
1787

1789  throw SleighError("Missing equation");
1790

1792  pateq->genPattern(oppattern);
1793  *pattern = pateq->getTokenPattern();
1794  if (pattern->alwaysFalse())
1795  throw SleighError("Impossible pattern");
1796  if (recursion)
1797  pattern->setRightEllipsis(true);
1798  minimumlength = pattern->getMinimumLength(); // Get length of the pattern in bytes
1799

1802  if (!pateq->resolveOperandLeft(resolve))
1803  throw SleighError("Unable to resolve operand offsets");
1804

1806  int4 base,offset;
1807  OperandSymbol *sym = operands[i];
1808  if (sym->isOffsetIrrelevant()) {
1809  sym->offsetbase = -1;
1810  sym->reloffset = 0;
1811  continue;
1812  }
1813  base = sym->offsetbase;
1814  offset = sym->reloffset;
1815  while(base >= 0) {
1816  sym = operands[base];
1817  if (sym->isVariableLength()) break; // Cannot resolve to absolute
1818  base = sym->offsetbase;
1819  offset += sym->getMinimumLength();
1820  offset += sym->reloffset;
1821  if (base < 0) {
1822  operands[i]->offsetbase = base;
1823  operands[i]->reloffset = offset;
1824  }
1825  }
1826  }
1827 
1828  // Make sure context expressions are valid
1829  for(int4 i=0;i<context.size();++i)
1830  context[i]->validate();
1831 
1832  orderOperands(); // Order the operands based on offset dependency
1833  return pattern;
1834 }

2211 void DecisionNode::orderPatterns(DecisionProperties &props)
2212 
2213 {
2214  // This is a tricky routine. When this routine is called, the patterns remaining in the
2215  // the decision node can no longer be distinguished by examining additional bits. The basic
2216  // idea here is that the patterns should be ordered so that the most specialized should come
2217  // first in the list. Pattern 1 is a specialization of pattern 2, if the set of instructions
2218  // matching 1 is contained in the set matching 2. So in the simplest case, the pattern order
2219  // should represent a strict nesting. Unfortunately, there are many potential situations where
2220  // patterns don't necessarily nest.
2221  // 1) An "or" of two patterns. This can be an explicit '|' operator in the Constructor, in
2222  // which case this can be detected because the two patterns point to the same constructor
2223  // But the "or" can be implied across two constructors that do the same thing. This should
2224  // probably be flagged as an error except in the following case.
2225  // 2) Two patterns aren't properly nested, but they are "resolved" by a third pattern which
2226  // covers the intersection of the first two patterns. Sometimes its easier to specify
2227  // three cases that need to be distinguished in this way.
2228  // 3) Recursive constructors that use a "guard" context bit. The guard bit is used to prevent
2229  // the recursive constructor from matching repeatedly, but it's too much work to put a
2230  // constraint an the bit for every other pattern.
2231  // 4) Other situations where the ability to distinguish between constructors is hidden in
2232  // the subconstructors.
2233  // This routine can determine if an intersection results from case 1) or case 2)
2234  int4 i,j,k;
2235  vector<pair<DisjointPattern *,Constructor *> > newlist;
2236  vector<pair<DisjointPattern *,Constructor *> > conflictlist;
2237 
2238  // Check for identical patterns
2239  for(i=0;i<list.size();++i) {

2241  DisjointPattern *ipat = list[i].first;
2242  DisjointPattern *jpat = list[j].first;
2243  if (ipat->identical(jpat))
2244  props.identicalPattern(list[i].second,list[j].second);
2245  }
2246  }
2247 
2248  newlist = list;
2249  for(i=0;i<list.size();++i) {
2250  for(j=0;j<i;++j) {
2251  DisjointPattern *ipat = newlist[i].first;
2252  DisjointPattern *jpat = list[j].first;
2253  if (ipat->specializes(jpat))
2254  break;
2255  if (!jpat->specializes(ipat)) { // We have a potential conflict
2256  Constructor *iconst = newlist[i].second;
2257  Constructor *jconst = list[j].second;
2258  if (iconst == jconst) { // This is an OR in the pattern for ONE constructor
2259  // So there is no conflict
2260  }
2261  else { // A true conflict that needs to be resolved
2262  conflictlist.push_back(pair<DisjointPattern *,Constructor *>(ipat,iconst));
2263  conflictlist.push_back(pair<DisjointPattern *,Constructor *>(jpat,jconst));
2264  }
2265  }
2266  }
2267  for(k=i-1;k>=j;--k)
2268  list[k+1] = list[k];
2269  list[j] = newlist[i];
2270  }
2271 
2272  // Check if intersection patterns are present, which resolve conflicts
2273  for(i=0;i<conflictlist.size();i+=2) {
2274  DisjointPattern *pat1,*pat2;
2275  Constructor *const1,*const2;
2276  pat1 = conflictlist[i].first;
2277  const1 = conflictlist[i].second;
2278  pat2 = conflictlist[i+1].first;
2279  const2 = conflictlist[i+1].second;
2280  bool resolved = false;
2281  for(j=0;j<list.size();++j) {
2282  DisjointPattern *tpat = list[j].first;
2283  Constructor *tconst = list[j].second;
2284  if ((tpat == pat1)&&(tconst==const1)) break; // Ran out of possible specializations
2285  if ((tpat == pat2)&&(tconst==const2)) break;
2286  if (tpat->resolvesIntersect(pat1,pat2)) {

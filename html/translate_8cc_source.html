123 bool JoinRecord::operator<(const JoinRecord &op2) const
124 
125 {
126  // Some joins may have same piece but different unified size (floating point)
127  if (unified.size != op2.unified.size) // Compare size first
128  return (unified.size < op2.unified.size);
129  // Lexigraphic sort on pieces
130  int4 i=0;
131  for(;;) {
132  if (pieces.size()==i) {
133  return (op2.pieces.size()>i); // If more pieces in op2, it is bigger (return true), if same number this==op2, return false
134  }
135  if (op2.pieces.size()==i) return false; // More pieces in -this-, so it is bigger, return false
136  if (pieces[i] != op2.pieces[i])
137  return (pieces[i] < op2.pieces[i]);
138  i += 1;
139  }
140 }

244 void AddrSpaceManager::insertSpace(AddrSpace *spc)
245 
246 {
247  bool nametype_mismatch = false;
248  bool duplicatedefine = false;
249  switch(spc->getType()) {
250  case IPTR_CONSTANT:
251  if (spc->getName() != "const")
252  nametype_mismatch = true;
253  if (baselist.size()!=0)
254  throw LowlevelError("const space must be initialized first");
255  constantspace = spc;
256  break;
257  case IPTR_INTERNAL:
258  if (spc->getName() != "unique")
259  nametype_mismatch = true;
260  if (uniqspace != (AddrSpace *)0)
261  duplicatedefine = true;
262  uniqspace = spc;
263  break;
264  case IPTR_FSPEC:
265  if (spc->getName() != "fspec")
266  nametype_mismatch = true;
267  if (fspecspace != (AddrSpace *)0)
268  duplicatedefine = true;
269  fspecspace = spc;
270  break;
271  case IPTR_JOIN:
272  if (spc->getName() != "join")
273  nametype_mismatch = true;
274  if (joinspace != (AddrSpace *)0)
275  duplicatedefine = true;
276  joinspace = spc;
277  break;
278  case IPTR_IOP:
279  if (spc->getName() != "iop")
280  nametype_mismatch = true;
281  if (iopspace != (AddrSpace *)0)
282  duplicatedefine = true;
283  iopspace = spc;
284  break;
285  case IPTR_SPACEBASE:
286  if (spc->getName() == "stack") {
287  if (stackspace != (AddrSpace *)0)
288  duplicatedefine = true;
289  stackspace = spc;
290  }
291  // fallthru
292  case IPTR_PROCESSOR:
293  if (spc->isOverlay()) { // If this is a new overlay space
294  OverlaySpace *ospc = (OverlaySpace *)spc;
295  ospc->getBaseSpace()->setFlags(AddrSpace::overlaybase); // Mark the base as being overlayed
296  }
297  for(uint4 i=0;i<baselist.size();++i)
298  if (baselist[i]->getName() == spc->getName())
299  duplicatedefine = true;
300  break;
301  }
302  if (nametype_mismatch)
303  throw LowlevelError("Space "+spc->getName()+" was initialized with wrong type");
304  if (duplicatedefine)
305  throw LowlevelError("Space "+spc->getName()+" was initialized more than once");
306  if (baselist.size() != spc->getIndex())
307  throw LowlevelError("Space "+spc->getName()+" was initialized with a bad id");
308  baselist.push_back(spc);
309  spc->refcount += 1;
310 }

490 JoinRecord *AddrSpaceManager::findAddJoin(const vector<VarnodeData> &pieces,uint4 logicalsize)
491 
492 { // Find a pre-existing split record, or create a new one corresponding to the input -pieces-
493  // If -logicalsize- is 0, calculate logical size as sum of pieces
494  if (pieces.size() == 0)
495  throw LowlevelError("Cannot create a join without pieces");
496  if ((pieces.size()==1)&&(logicalsize==0))
497  throw LowlevelError("Cannot create a single piece join without a logical size");
498 
499  uint4 totalsize;
500  if (logicalsize != 0) {
501  if (pieces.size() != 1)
502  throw LowlevelError("Cannot specify logical size for multiple piece join");
503  totalsize = logicalsize;
504  }
505  else {
506  totalsize = 0;
507  for(int4 i=0;i<pieces.size();++i) // Calculate sum of the sizes of all pieces
508  totalsize += pieces[i].size;
509  if (totalsize == 0)
510  throw LowlevelError("Cannot create a zero size join");
511  }
512 
513  JoinRecord testnode;
514 
515  testnode.pieces = pieces;
516  testnode.unified.size = totalsize;
517  set<JoinRecord *,JoinRecordCompare>::const_iterator iter;
518  iter = splitset.find(&testnode);
519  if (iter != splitset.end()) // If already in the set
520  return *iter;
521 
522  JoinRecord *newjoin = new JoinRecord();
523  newjoin->pieces = pieces;
524 
525  uint4 roundsize = (totalsize + 15) & ~((uint4)0xf); // Next biggest multiple of 16
526 
527  newjoin->unified.space = joinspace;
528  newjoin->unified.offset = joinallocate;
529  joinallocate += roundsize;
530  newjoin->unified.size = totalsize;
531  splitset.insert(newjoin);
532  splitlist.push_back(newjoin);
533  return splitlist.back();
534 }

542 
543 { // Find a split record given the unified (join space) offset
544  int4 min=0;
545  int4 max=splitlist.size()-1;
546  while(min<=max) { // Binary search
547  int4 mid = (min+max)/2;
548  JoinRecord *rec = splitlist[mid];
549  uintb val = rec->unified.offset;
550  if (val == offset) return rec;
551  if (val < offset)
552  min = mid + 1;
553  else
554  max = mid - 1;
555  }
556  throw LowlevelError("Unlinked join address");

612 Address AddrSpaceManager::constructJoinAddress(const Translate *translate,
613  const Address &hiaddr,int4 hisz,
614  const Address &loaddr,int4 losz)
615 {
616  spacetype hitp = hiaddr.getSpace()->getType();
617  spacetype lotp = loaddr.getSpace()->getType();
618  bool usejoinspace = true;
619  if (((hitp != IPTR_SPACEBASE)&&(hitp != IPTR_PROCESSOR))||
620  ((lotp != IPTR_SPACEBASE)&&(lotp != IPTR_PROCESSOR)))
621  throw LowlevelError("Trying to join in appropriate locations");
622  if ((hitp == IPTR_SPACEBASE)||(lotp == IPTR_SPACEBASE)||
623  (hiaddr.getSpace() == getDefaultSpace())||
624  (loaddr.getSpace() == getDefaultSpace()))
625  usejoinspace = false;
626  if (hiaddr.isContiguous(hisz,loaddr,losz)) { // If we are contiguous
627  if (!usejoinspace) { // and in a mappable space, just return the earliest address
628  if (hiaddr.isBigEndian())
629  return hiaddr;
630  return loaddr;
631  }
632  else { // If we are in a non-mappable (register) space, check to see if a parent register exists
633  if (hiaddr.isBigEndian()) {
634  if (translate->getRegisterName(hiaddr.getSpace(),hiaddr.getOffset(),(hisz+losz)).size() != 0)
635  return hiaddr;
636  }
637  else {
638  if (translate->getRegisterName(loaddr.getSpace(),loaddr.getOffset(),(hisz+losz)).size() != 0)
639  return loaddr;
640  }
641  }
642  }
643  // Otherwise construct a formal JoinRecord
644  vector<VarnodeData> pieces;
645  pieces.push_back(VarnodeData());
646  pieces.push_back(VarnodeData());
647  pieces[0].space = hiaddr.getSpace();
648  pieces[0].offset = hiaddr.getOffset();
649  pieces[0].size = hisz;
650  pieces[1].space = loaddr.getSpace();
651  pieces[1].offset = loaddr.getOffset();
652  pieces[1].size = losz;
653  JoinRecord *join = findAddJoin(pieces,0);
654  return join->getUnified().getAddr();
655 }
